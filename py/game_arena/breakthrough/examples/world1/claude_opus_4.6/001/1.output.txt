## Strategy

My strategy implements a minimax search with alpha-beta pruning and iterative deepening to find the best move within the time constraint. The evaluation function considers:

1. **Material advantage**: Count of pieces for each side.
2. **Advancement**: How far each piece has progressed toward the opponent's home row, with exponentially increasing value as pieces get closer to promotion.
3. **Promotion threats**: Large bonus for pieces one step from the opponent's home row.
4. **Center control**: Slight preference for central columns.
5. **Protection**: Bonus for pieces that are defended by friendly pieces behind them.
6. **Capture opportunities**: Recognizing when diagonal captures are available.
7. **Immediate wins**: Detecting winning moves (reaching the last row) and prioritizing them.

The search uses move ordering (captures first, then advanced pieces, then central moves) to improve alpha-beta pruning efficiency. Iterative deepening ensures we always have a valid move even if time runs out.

import time

def policy(me, opp, color):
    start_time = time.time()
    time_limit = 0.85
    
    me_set = frozenset(me)
    opp_set = frozenset(opp)
    direction = 1 if color == 'w' else -1
    my_home = 7 if color == 'w' else 0
    opp_home = 0 if color == 'w' else 7
    
    def get_moves(my_pieces, their_pieces, d, promo_row):
        moves = []
        my_s = my_pieces if isinstance(my_pieces, frozenset) else frozenset(my_pieces)
        th_s = their_pieces if isinstance(their_pieces, frozenset) else frozenset(their_pieces)
        for r, c in my_s:
            nr = r + d
            if 0 <= nr <= 7:
                # Forward
                if (nr, c) not in my_s and (nr, c) not in th_s:
                    moves.append(((r, c), (nr, c)))
                # Diagonal captures and moves
                for dc in (-1, 1):
                    nc = c + dc
                    if 0 <= nc <= 7:
                        if (nr, nc) in th_s:
                            moves.append(((r, c), (nr, nc)))
                        elif (nr, nc) not in my_s:
                            moves.append(((r, c), (nr, nc)))
        return moves
    
    def evaluate(my_pieces, their_pieces, d, promo_row):
        if not their_pieces:
            return 100000
        if not my_pieces:
            return -100000
        
        score = 0
        total_rows = 7
        
        for r, c in my_pieces:
            adv = abs(r - (promo_row - d * total_rows)) if d == 1 else abs(r - (promo_row + total_rows))
            # Simpler: distance to promo
            dist = abs(promo_row - r)
            progress = total_rows - dist
            score += 100 + progress * 15
            if dist == 0:
                return 100000
            if dist == 1:
                score += 200
            if dist == 2:
                score += 50
            # Center preference
            score += (3.5 - abs(c - 3.5)) * 3
        
        opp_d = -d
        opp_promo = promo_row - d * total_rows  # opponent's promo row
        for r, c in their_pieces:
            dist = abs(r - (0 if promo_row == 7 else 7))
            progress = total_rows - dist
            score -= 100 + progress * 15
            if dist == 0:
                return -100000
            if dist == 1:
                score -= 200
            if dist == 2:
                score -= 50
            score -= (3.5 - abs(c - 3.5)) * 3
        
        score += (len(my_pieces) - len(their_pieces)) * 50
        return score
    
    def order_moves(moves, their_pieces, d, promo_row):
        th_s = their_pieces if isinstance(their_pieces, frozenset) else frozenset(their_pieces)
        def key(m):
            fr, to = m
            pri = 0
            if to[0] == promo_row:
                pri -= 10000
            if to in th_s:
                pri -= 5000
            pri -= abs(to[0] - promo_row) * (-10)
            pri += abs(to[1] - 3.5) * 2
            return pri
        moves.sort(key=key)
        return moves
    
    best_move_global = [None]
    
    def minimax(my_p, th_p, d, promo_row, depth, alpha, beta, maximizing):
        if time.time() - start_time > time_limit:
            raise TimeoutError
        
        if not th_p:
            return 100000 if maximizing else -100000
        if not my_p:
            return -100000 if maximizing else 100000
        
        # Check promo
        for r, c in my_p:
            if r == promo_row:
                return 100000
        
        opp_promo = 0 if promo_row == 7 else 7
        for r, c in th_p:
            if r == opp_promo:
                return -100000
        
        if depth == 0:
            return evaluate(my_p, th_p, d, promo_row)
        
        moves = get_moves(my_p, th_p, d, promo_row)
        if not moves:
            return -100000
        
        moves = order_moves(moves, th_p, d, promo_row)
        
        best = -200000
        for fr, to in moves:
            new_my = frozenset((my_p - {fr}) | {to})
            new_th = frozenset(th_p - {to})
            
            # Check immediate win
            if to[0] == promo_row:
                val = 100000
            else:
                opp_d = -d
                opp_promo = 0 if promo_row == 7 else 7
                val = -minimax(new_th, new_my, opp_d, opp_promo, depth - 1, -beta, -alpha, not maximizing)
            
            if val > best:
                best = val
            if val > alpha:
                alpha = val
            if alpha >= beta:
                break
        
        return best
    
    # Generate legal moves
    all_moves = get_moves(me_set, opp_set, direction, my_home)
    
    if not all_moves:
        # Shouldn't happen if we have pieces, but safety
        return (me[0], (me[0][0] + direction, me[0][1]))
    
    if len(all_moves) == 1:
        return all_moves[0]
    
    # Check for immediate wins
    for fr, to in all_moves:
        if to[0] == my_home:
            return (fr, to)
    
    best_move = all_moves[0]
    
    opp_d = -direction
    opp_promo = 0 if my_home == 7 else 7
    
    try:
        for max_depth in range(1, 20):
            current_best = -200000
            current_move = all_moves[0]
            
            ordered = order_moves(list(all_moves), opp_set, direction, my_home)
            
            alpha = -200000
            beta = 200000
            
            for fr, to in ordered:
                new_my = frozenset((me_set - {fr}) | {to})
                new_opp = frozenset(opp_set - {to})
                
                if to[0] == my_home:
                    val = 100000
                else:
                    val = -minimax(new_opp, new_my, opp_d, opp_promo, max_depth - 1, -beta, -alpha, False)
                
                if val > current_best:
                    current_best = val
                    current_move = (fr, to)
                if val > alpha:
                    alpha = val
                
                if time.time() - start_time > time_limit:
                    raise TimeoutError
            
            best_move = current_move
            best_move_global[0] = current_move
            
            if current_best >= 90000:
                break
                
    except TimeoutError:
        if best_move_global[0] is not None:
            best_move = best_move_global[0]
    
    return best_move

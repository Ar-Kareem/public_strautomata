The strategy for this policy is to implement a greedy heuristic-based approach that prioritizes:
1. Winning immediately by reaching the opponent's home row.
2. Capturing opponent pieces when possible.
3. Advancing pieces that are closest to the opponent's home row.
4. Avoiding moves that place pieces in vulnerable positions.

The policy evaluates possible moves based on these criteria and selects the best legal move within the time limit.

import random

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    # Define movement directions based on color
    if color == 'b':
        forward = -1  # Black moves down (row decreases)
        home_row = 0
        opp_home_row = 7
    else:
        forward = 1   # White moves up (row increases)
        home_row = 7
        opp_home_row = 0

    # Convert opponent positions to a set for O(1) lookup
    opp_positions = set(opp)
    my_positions = set(me)

    # Generate all possible moves
    possible_moves = []
    capture_moves = []
    winning_moves = []

    for row, col in me:
        # Forward move
        new_row = row + forward
        if 0 <= new_row <= 7:
            # Straight forward
            if (new_row, col) not in my_positions and (new_row, col) not in opp_positions:
                move = ((row, col), (new_row, col))
                if new_row == opp_home_row:
                    winning_moves.append(move)
                else:
                    possible_moves.append(move)
            
            # Diagonal left
            new_col = col - 1
            if 0 <= new_col <= 7:
                target = (new_row, new_col)
                if target in opp_positions:
                    capture_moves.append(((row, col), target))
                elif target not in my_positions and target not in opp_positions:
                    move = ((row, col), target)
                    if new_row == opp_home_row:
                        winning_moves.append(move)
                    else:
                        possible_moves.append(move)
            
            # Diagonal right
            new_col = col + 1
            if 0 <= new_col <= 7:
                target = (new_row, new_col)
                if target in opp_positions:
                    capture_moves.append(((row, col), target))
                elif target not in my_positions and target not in opp_positions:
                    move = ((row, col), target)
                    if new_row == opp_home_row:
                        winning_moves.append(move)
                    else:
                        possible_moves.append(move)

    # Return immediate win if possible
    if winning_moves:
        return winning_moves[0]
    
    # Prioritize captures
    if capture_moves:
        # Prefer captures that advance the most or are most aggressive
        # For simplicity, just return a random capture move
        return random.choice(capture_moves)
    
    # If no captures, pick the most advanced piece's move
    if possible_moves:
        # Heuristic: prefer moves that advance pieces closest to promotion
        # Sort by distance to opponent's home row
        if color == 'b':
            possible_moves.sort(key=lambda move: move[1][0])  # Prefer lower rows
        else:
            possible_moves.sort(key=lambda move: -move[1][0])  # Prefer higher rows
        return possible_moves[0]
    
    # Fallback (should not happen if there are legal moves)
    return random.choice(possible_moves) if possible_moves else random.choice(capture_moves)

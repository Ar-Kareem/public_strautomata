My strategy for this Breakthrough policy combines several key elements:

1. **Minimax with Alpha-Beta Pruning**: I use a depth-limited search (depth 4) to look ahead and evaluate positions.

2. **Evaluation Function**: The evaluation considers:
   - **Material count**: More pieces is better
   - **Advancement bonus**: Pieces closer to the goal row are worth more (exponentially increasing)
   - **Winning/losing detection**: Immediate wins/losses are heavily weighted
   - **Center control**: Slight preference for central columns
   - **Protected pieces**: Bonus for pieces that have friendly support

3. **Move Ordering**: To maximize alpha-beta pruning efficiency, I order moves by:
   - Captures first (aggressive play)
   - Moves that advance pieces furthest
   - This helps find good moves early and prune more of the search tree

4. **Immediate Win Detection**: Before searching, I check for any move that immediately wins (reaching the opponent's home row).

5. **Safety Fallback**: If anything goes wrong, I always have a fallback to return any legal move to avoid disqualification.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    me_set = set(me)
    opp_set = set(opp)
    
    direction = 1 if color == 'w' else -1
    goal_row = 7 if color == 'w' else 0
    opp_goal_row = 0 if color == 'w' else 7
    
    def get_moves(pieces, opponents, d):
        moves = []
        opp_s = set(opponents)
        pieces_s = set(pieces)
        for r, c in pieces:
            nr = r + d
            if 0 <= nr <= 7:
                # Forward move
                if (nr, c) not in pieces_s and (nr, c) not in opp_s:
                    moves.append(((r, c), (nr, c)))
                # Diagonal captures and moves
                for dc in [-1, 1]:
                    nc = c + dc
                    if 0 <= nc <= 7:
                        if (nr, nc) in opp_s:
                            moves.append(((r, c), (nr, nc)))
                        elif (nr, nc) not in pieces_s:
                            moves.append(((r, c), (nr, nc)))
        return moves
    
    def evaluate(my_pieces, opp_pieces, my_dir, my_goal):
        if not opp_pieces:
            return 10000
        if not my_pieces:
            return -10000
        
        for r, c in my_pieces:
            if r == my_goal:
                return 10000
        opp_goal = 7 - my_goal
        for r, c in opp_pieces:
            if r == opp_goal:
                return -10000
        
        score = 0
        my_set = set(my_pieces)
        
        for r, c in my_pieces:
            advancement = r if my_dir == 1 else (7 - r)
            score += 10 + advancement * 5 + (advancement ** 2)
            if 2 <= c <= 5:
                score += 2
            # Protection bonus
            back_r = r - my_dir
            if 0 <= back_r <= 7:
                if (back_r, c-1) in my_set or (back_r, c+1) in my_set:
                    score += 3
        
        opp_dir = -my_dir
        for r, c in opp_pieces:
            advancement = r if opp_dir == 1 else (7 - r)
            score -= 10 + advancement * 5 + (advancement ** 2)
        
        return score
    
    def minimax(my_pieces, opp_pieces, my_dir, my_goal, depth, alpha, beta, maximizing):
        if depth == 0:
            return evaluate(my_pieces, opp_pieces, my_dir, my_goal), None
        
        if maximizing:
            moves = get_moves(my_pieces, opp_pieces, my_dir)
            if not moves:
                return -9000, None
            
            # Move ordering: captures first, then advancement
            def move_score(m):
                s = 0
                if m[1] in set(opp_pieces):
                    s += 100
                s += m[1][0] if my_dir == 1 else (7 - m[1][0])
                return -s
            moves.sort(key=move_score)
            
            best_val = -float('inf')
            best_move = moves[0]
            for move in moves:
                fr, to = move
                new_my = [p for p in my_pieces if p != fr] + [to]
                new_opp = [p for p in opp_pieces if p != to]
                
                if to[0] == my_goal:
                    return 10000 + depth, move
                
                val, _ = minimax(new_opp, new_my, -my_dir, 7 - my_goal, depth - 1, alpha, beta, False)
                val = -val
                if val > best_val:
                    best_val = val
                    best_move = move
                alpha = max(alpha, val)
                if beta <= alpha:
                    break
            return best_val, best_move
        else:
            moves = get_moves(my_pieces, opp_pieces, my_dir)
            if not moves:
                return 9000, None
            
            best_val = float('inf')
            for move in moves:
                fr, to = move
                new_my = [p for p in my_pieces if p != fr] + [to]
                new_opp = [p for p in opp_pieces if p != to]
                
                if to[0] == my_goal:
                    return -10000 - depth, None
                
                val, _ = minimax(new_opp, new_my, -my_dir, 7 - my_goal, depth - 1, alpha, beta, True)
                val = -val
                if val < best_val:
                    best_val = val
                beta = min(beta, val)
                if beta <= alpha:
                    break
            return best_val, None
    
    # Get all legal moves
    legal_moves = get_moves(list(me_set), list(opp_set), direction)
    
    if not legal_moves:
        # Shouldn't happen if we have pieces, but safety fallback
        return ((0, 0), (0, 0))
    
    # Check for immediate wins
    for move in legal_moves:
        if move[1][0] == goal_row:
            return move
    
    # Run minimax
    try:
        _, best_move = minimax(list(me_set), list(opp_set), direction, goal_row, 4, -float('inf'), float('inf'), True)
        if best_move and best_move in legal_moves:
            return best_move
    except:
        pass
    
    # Fallback: pick best immediate move by simple heuristic
    def simple_score(m):
        s = 0
        if m[1] in opp_set:
            s += 50
        advancement = m[1][0] if direction == 1 else (7 - m[1][0])
        s += advancement * 10
        return s
    
    legal_moves.sort(key=simple_score, reverse=True)
    return legal_moves[0]

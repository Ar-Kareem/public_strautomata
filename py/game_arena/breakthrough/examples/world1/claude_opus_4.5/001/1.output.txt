My strategy for this Breakthrough policy is based on a combination of techniques:

1. **Minimax with Alpha-Beta Pruning**: I implement a depth-limited minimax search with alpha-beta pruning to look ahead several moves and evaluate positions.

2. **Evaluation Function**: The evaluation considers:
   - **Material count**: Having more pieces is advantageous
   - **Advancement bonus**: Pieces closer to the goal row are worth more (exponentially weighted)
   - **Winning/losing detection**: Immediate wins get maximum scores
   - **Protected pieces**: Pieces that are defended by friendly pieces get a bonus
   - **Central control**: Slight preference for central columns

3. **Move Ordering**: To improve alpha-beta pruning efficiency, moves are ordered by:
   - Captures first (aggressive play)
   - Moves that advance pieces closer to the goal
   - This helps prune more branches early

4. **Iterative Deepening**: Within time constraints, I use a reasonable fixed depth that should complete within 1 second.

5. **Immediate Win Detection**: Before searching, check if any move leads to an immediate win (reaching the opponent's home row).

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    me_set = set(me)
    opp_set = set(opp)
    direction = 1 if color == 'w' else -1
    goal_row = 7 if color == 'w' else 0
    opp_goal_row = 0 if color == 'w' else 7
    
    def get_moves(pieces, opponents, d):
        moves = []
        opp_s = set(opponents)
        pieces_s = set(pieces)
        for r, c in pieces:
            nr = r + d
            if 0 <= nr <= 7:
                # Forward move
                if (nr, c) not in pieces_s and (nr, c) not in opp_s:
                    moves.append(((r, c), (nr, c)))
                # Diagonal moves
                for dc in [-1, 1]:
                    nc = c + dc
                    if 0 <= nc <= 7:
                        if (nr, nc) in opp_s:
                            moves.append(((r, c), (nr, nc)))
                        elif (nr, nc) not in pieces_s:
                            moves.append(((r, c), (nr, nc)))
        return moves
    
    def evaluate(my_pieces, opp_pieces, my_dir, my_goal):
        if not opp_pieces:
            return 10000
        if not my_pieces:
            return -10000
        for r, c in my_pieces:
            if r == my_goal:
                return 10000
        opp_goal = 7 - my_goal
        for r, c in opp_pieces:
            if r == opp_goal:
                return -10000
        
        score = 0
        my_set = set(my_pieces)
        opp_set_local = set(opp_pieces)
        
        # Material
        score += (len(my_pieces) - len(opp_pieces)) * 100
        
        # Advancement
        for r, c in my_pieces:
            adv = r if my_dir == 1 else (7 - r)
            score += adv * adv * 3
            # Protected bonus
            back_r = r - my_dir
            if 0 <= back_r <= 7:
                if (back_r, c-1) in my_set or (back_r, c+1) in my_set:
                    score += 5
        
        for r, c in opp_pieces:
            adv = (7 - r) if my_dir == 1 else r
            score -= adv * adv * 3
        
        return score
    
    def minimax(my_p, opp_p, my_dir, my_goal, depth, alpha, beta, maximizing):
        if depth == 0:
            return evaluate(my_p, opp_p, my_dir, my_goal), None
        
        if maximizing:
            moves = get_moves(my_p, opp_p, my_dir)
            if not moves:
                return -10000, None
            
            # Move ordering
            moves.sort(key=lambda m: (-(m[1][0] if my_dir == 1 else -m[1][0]), m[1] in set(opp_p)), reverse=True)
            
            best_val = -100000
            best_move = moves[0]
            for frm, to in moves:
                new_my = [p for p in my_p if p != frm] + [to]
                new_opp = [p for p in opp_p if p != to]
                if to[0] == my_goal:
                    return 10000, (frm, to)
                val, _ = minimax(new_opp, new_my, -my_dir, 7 - my_goal, depth - 1, -beta, -alpha, False)
                val = -val
                if val > best_val:
                    best_val = val
                    best_move = (frm, to)
                alpha = max(alpha, val)
                if alpha >= beta:
                    break
            return best_val, best_move
        else:
            moves = get_moves(my_p, opp_p, my_dir)
            if not moves:
                return 10000, None
            best_val = -100000
            for frm, to in moves:
                new_my = [p for p in my_p if p != frm] + [to]
                new_opp = [p for p in opp_p if p != to]
                val, _ = minimax(new_opp, new_my, -my_dir, 7 - my_goal, depth - 1, -beta, -alpha, True)
                val = -val
                if val > best_val:
                    best_val = val
                alpha = max(alpha, val)
                if alpha >= beta:

            return best_val, None
    
    # Get all legal moves
    legal_moves = get_moves(list(me_set), list(opp_set), direction)
    
    if not legal_moves:

        return ((0, 0), (0, 0))

    for frm, to in legal_moves:
        if to[0] == goal_row:
            return (frm, to)
    
    # Run minimax
    depth = 4

    if best_move and best_move in legal_moves:
        return best_move
    
    return legal_moves[0]

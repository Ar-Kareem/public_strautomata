The strategy for this policy is to prioritize moves that either lead to an immediate win (via promotion or capturing the last opponent's piece), followed by capturing opponent pieces. If neither is possible, it favors moves that advance pieces toward the opponent's territory and position them in central columns. The evaluation function assigns high scores to winning conditions, moderate scores to captures, and smaller bonuses for forward progress and central positioning.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    me_set = set(me)
    opp_set = set(opp)
    direction = 1 if color == 'w' else -1
    possible_moves = []

    for (r, c) in me:
        new_r = r + direction
        new_c = c
        if 0 <= new_r < 8:
            # Straight move
            target_straight = (new_r, new_c)
            if target_straight not in me_set and target_straight not in opp_set:
                possible_moves.append(((r, c), target_straight))
            # Left diagonal capture
            new_c_left = c - 1
            if 0 <= new_c_left < 8:
                target_left = (new_r, new_c_left)
                if target_left in opp_set:
                    possible_moves.append(((r, c), target_left))
            # Right diagonal capture
            new_c_right = c + 1
            if 0 <= new_c_right < 8:
                target_right = (new_r, new_c_right)
                if target_right in opp_set:
                    possible_moves.append(((r, c), target_right))

    def evaluate(move):
        (fr, fc), (tr, tc) = move
        # Check for promotion
        if (color == 'w' and tr == 7) or (color == 'b' and tr == 0):
            return float('inf')
        # Check if capturing last opponent piece
        is_capture = (tr, tc) in opp_set
        if is_capture and len(opp) == 1:
            return float('inf')
        # Calculate score
        score = 0
        if is_capture:
            score += 100  # Capture value
        score += 1  # Forward progress
        positional_values = [0, 1, 2, 3, 3, 2, 1, 0]
        score += positional_values[tc]  # Positional bonus
        return score

    best_move = max(possible_moves, key=evaluate)
    return best_move

The strategy for this policy is to prioritize immediate wins by advancing to the opponent's home row. If no immediate win is possible, the policy selects the move with the highest heuristic score based on material advantage, positional advancement, and bonuses for captures. The heuristic evaluates board states by comparing piece counts, the proximity of pieces to the goal, and rewarding captures to reduce the opponent's resources.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    me_set = set(me)
    opp_set = set(opp)
    
    moves = []
    dr = -1 if color == 'b' else 1
    
    for (r, c) in me:
        # Straight move
        new_r = r + dr
        new_c = c
        if 0 <= new_r <= 7 and 0 <= new_c <= 7:
            target = (new_r, new_c)
            if target not in me_set and target not in opp_set:
                moves.append(((r, c), target))
        
        # Diagonal captures
        for dc in (-1, 1):
            new_r_diag = r + dr
            new_c_diag = c + dc
            target_diag = (new_r_diag, new_c_diag)
            if 0 <= new_r_diag <= 7 and 0 <= new_c_diag <= 7 and target_diag in opp_set:
                moves.append(((r, c), target_diag))
    
    # Check for immediate win
    for move in moves:
        _, (nr, nc) = move
        if (color == 'b' and nr == 0) or (color == 'w' and nr == 7):
            return move
    
    best_move = None
    best_score = -float('inf')
    
    for move in moves:
        (r, c), (nr, nc) = move
        is_capture = (nr, nc) in opp_set
        
        # Generate new board states
        new_me = [pos for pos in me if pos != (r, c)]
        new_me.append((nr, nc))
        
        if is_capture:
            new_opp = [pos for pos in opp if pos != (nr, nc)]
        else:
            new_opp = opp
        
        # Evaluate material and position
        mat_diff = len(new_me) - len(new_opp)
        
        if color == 'w':
            sum_my = sum(r for r, _ in new_me)
            sum_opp = sum((7 - r) for r, _ in new_opp)
        else:
            sum_my = sum((7 - r) for r, _ in new_me)
            sum_opp = sum(r for r, _ in new_opp)
        
        pos_diff = sum_my - sum_opp
        capture_bonus = 10 if is_capture else 0
        
        score = mat_diff * 10 + pos_diff + capture_bonus
        
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

**Strategy Overview:**  
This policy uses a minimax search with alpha‑beta pruning and iterative deepening to ensure it always returns a legal move within the time limit. The evaluation function prioritizes:  

1. **Progress** – reward for advancing toward the opponent’s home row.  
2. **Material** – count of pieces, weighted higher near the opponent’s home row.  
3. **Center control** – bonus for controlling central columns.  
4. **Pawn structure** – penalty for isolated pawns, reward for connected pawns.  
5. **Threats** – bonus for moves that capture or threaten to capture.  
6. **King‑row pressure** – extra reward for pieces close to promotion.  

The search depth increases until the 1‑second time limit is nearly exhausted, then returns the best move found.

import time
import random
import sys

INF = 10**9

# Constants for board geometry
ROWS = 8
COLS = 8
WHITE_HOME = 7
BLACK_HOME = 0

# Piece movement vectors
DIRS = {
    'w': [(-1, -1), (-1, 0), (-1, 1)],   # white moves upward (decreasing row)
    'b': [(1, -1), (1, 0), (1, 1)]       # black moves downward (increasing row)
}

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str):
    start_time = time.time()
    max_time = 0.95  # leave a small margin
    state = (frozenset(me), frozenset(opp), color)
    best_move = None
    depth = 1
    
    # Generate legal moves for current state
    moves = list(generate_moves(state))
    if not moves:
        # Should not happen in Breakthrough, but fallback
        return ((0,0),(0,0))
    
    best_move = moves[0]
    
    # Iterative deepening with time limit
    while time.time() - start_time < max_time:
        depth += 1
        try:
            move, _ = minimax(state, depth, -INF, INF, True, start_time, max_time)
            if move is not None:
                best_move = move
        except TimeoutError:
            break
        except Exception:
            # If depth too high, break and use previous best
            break
    
    return best_move

# ---------- Move Generation ----------
def generate_moves(state):
    me, opp, color = state
    all_pieces = me | opp
    forward_dir = -1 if color == 'w' else 1
    home_row = WHITE_HOME if color == 'w' else BLACK_HOME
    
    for r, c in me:
        # Straight forward
        nr, nc = r + forward_dir, c
        if 0 <= nr < ROWS and 0 <= nc < COLS and (nr, nc) not in all_pieces:
            yield ((r, c), (nr, nc))
        
        # Diagonal left
        nr, nc = r + forward_dir, c - 1
        if 0 <= nr < ROWS and 0 <= nc < COLS:
            if (nr, nc) in opp:   # capture
                yield ((r, c), (nr, nc))
            elif (nr, nc) not in all_pieces:   # empty diagonal move
                yield ((r, c), (nr, nc))
        
        # Diagonal right
        nr, nc = r + forward_dir, c + 1
        if 0 <= nr < ROWS and 0 <= nc < COLS:
            if (nr, nc) in opp:
                yield ((r, c), (nr, nc))
            elif (nr, nc) not in all_pieces:
                yield ((r, c), (nr, nc))

# ---------- State Update ----------
def apply_move(state, move):
    me, opp, color = state
    (fr, fc), (tr, tc) = move
    me = set(me)
    opp = set(opp)
    me.remove((fr, fc))
    me.add((tr, tc))
    if (tr, tc) in opp:
        opp.remove((tr, tc))
    next_color = 'b' if color == 'w' else 'w'
    return (frozenset(me), frozenset(opp), next_color)

# ---------- Evaluation ----------
def evaluate(state):
    me, opp, color = state
    if len(me) == 0:
        return -INF
    if len(opp) == 0:
        return INF
    
    # Progress: distance to promotion row
    promo_row = BLACK_HOME if color == 'w' else WHITE_HOME
    progress = 0
    for r, c in me:
        dist = abs(promo_row - r)
        progress += (ROWS - dist) * 3
    
    for r, c in opp:
        dist = abs(promo_row - r)
        progress -= (ROWS - dist) * 3
    
    # Material
    material = (len(me) - len(opp)) * 10
    
    # Center control
    center_cols = {3, 4}
    center = 0
    for r, c in me:
        if c in center_cols:
            center += 2
    for r, c in opp:
        if c in center_cols:
            center -= 2
    
    # Pawn structure: connectivity
    connectivity = 0
    for r, c in me:
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                if (r + dr, c + dc) in me:
                    connectivity += 1
    
    # Threat: captures available
    threat = 0
    for move in generate_moves(state):
        (_, _), (tr, tc) = move
        if (tr, tc) in opp:
            threat += 5
    
    # King‑row pressure
    pressure = 0
    opp_home = WHITE_HOME if color == 'b' else BLACK_HOME
    for r, c in me:
        if abs(r - opp_home) <= 2:
            pressure += 4
    
    total = progress + material + center + connectivity + threat + pressure
    return total if color == state[2] else -total

# ---------- Minimax with Alpha‑Beta ----------
def minimax(state, depth, alpha, beta, maximizing, start_time, max_time):
    me, opp, color = state
    
    # Terminal check
    if len(me) == 0:
        return None, -INF
    if len(opp) == 0:
        return None, INF
    for r, _ in me:
        if (color == 'w' and r == BLACK_HOME) or (color == 'b' and r == WHITE_HOME):
            return None, INF
    for r, _ in opp:
        if (color == 'w' and r == WHITE_HOME) or (color == 'b' and r == BLACK_HOME):
            return None, -INF
    
    if depth == 0:
        return None, evaluate(state)
    
    # Time check
    if time.time() - start_time > max_time:
        raise TimeoutError()
    
    moves = list(generate_moves(state))
    if not moves:
        return None, evaluate(state)
    
    if maximizing:
        best_move = moves[0]
        best_val = -INF
        for move in moves:
            child = apply_move(state, move)
            _, val = minimax(child, depth - 1, alpha, beta, False, start_time, max_time)
            if val > best_val:
                best_val = val
                best_move = move
            alpha = max(alpha, val)
            if beta <= alpha:
                break
        return best_move, best_val
    else:
        best_val = INF
        for move in moves:
            child = apply_move(state, move)
            _, val = minimax(child, depth - 1, alpha, beta, True, start_time, max_time)
            if val < best_val:
                best_val = val
            beta = min(beta, val)
            if beta <= alpha:
                break
        return None, best_val

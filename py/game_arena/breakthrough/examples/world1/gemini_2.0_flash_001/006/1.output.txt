The strategy for this policy is an offensive, center-control strategy with basic threat detection. It prioritizes winning, then advancing pieces towards the opponent's home row, and controlling the center to restrict opponent movement. Capturing is considered, but advancement is often favored, particularly moving to a winning square.

Here's the Python code:

```python
def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    """
    A Breakthrough policy that prioritizes winning and advancing.
    """

    board_size = 8
    board = [[0 for _ in range(board_size)] for _ in range(board_size)]
    for r, c in me:
        board[r][c] = 1  # Represent our pieces as 1
    for r, c in opp:
        board[r][c] = -1  # Represent opponent pieces as -1

    def get_legal_moves(piece: tuple[int, int]) -> list[tuple[tuple[int, int], tuple[int, int]]]:
        """
        Returns a list of legal moves for a given piece.
        """
        row, col = piece
        moves = []
        direction = 1 if color == 'w' else -1
        new_row = row + direction

        # Forward move
        if 0 <= new_row < board_size and board[new_row][col] == 0:
            moves.append(((row, col), (new_row, col)))

        # Diagonal captures
        for new_col in [col - 1, col + 1]:
            if 0 <= new_col < board_size and 0 <= new_row < board_size and board[new_row][new_col] == -1:
                moves.append(((row, col), (new_row, new_col)))
            elif 0 <= new_col < board_size and 0 <= new_row < board_size and board[new_row][new_col] == 0:
                moves.append(((row, col), (new_row, new_col))) #Diagonal move

        return moves

    def is_winning_move(move: tuple[tuple[int, int], tuple[int, int]], color: str) -> bool:
        """
        Checks if a move leads to a win.
        """
        _, (to_row, _) = move
        if color == 'w' and to_row == 7:
            return True
        if color == 'b' and to_row == 0:
            return True
        return False

    def evaluate_move(move: tuple[tuple[int, int], tuple[int, int]], me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> float:
        """
        Evaluates a move based on advancement and captures.
        """
        _, (to_row, _) = move
        direction = 1 if color == 'w' else -1
        advancement = direction * (to_row - move[0][0])

        captured = 0
        from_row, from_col = move[0]
        to_row, to_col = move[1]
        if (to_row, to_col) in opp:
            captured = 1
            
        valuation = advancement + captured*0.5 #Weight capturing slightly less

        return valuation


    # 1. Check for winning moves
    for piece in me:
        for move in get_legal_moves(piece):
            if is_winning_move(move, color):
                return move

    # 2. Evaluate and choose the best move
    best_move = None
    best_value = float('-inf')
    for piece in me:
        for move in get_legal_moves(piece):
            value = evaluate_move(move, me, opp, color)
            if value > best_value:
                best_value = value
                best_move = move

    # 3. If no move satisfies previous conditions, take a legal move
    if best_move is None:
        for piece in me:
            legal_moves = get_legal_moves(piece)
            if legal_moves:
                best_move = legal_moves[0]  #Just take the first legal move.
                break

    return best_move
```

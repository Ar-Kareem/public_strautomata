

The strategyfor this policy is to prioritize capturing opponent pieces when possible, as this directly reduces the opponent's options and can lead to a quicker win. If no captures are available, the policy focuses on moving pieces toward the opponent's home row to advance toward victory. If multiple moves are equally beneficial, the policy selects the one that maximizes the number of potential future moves for the next turn.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[tuple[int, int], tuple[int, int]]:
    # Determine movement direction based on color
    direction = -1 if color == 'b' else 1
    
    # Collect all legal moves
    moves = []
    for piece in me:
        r, c = piece
        
        # Check straight moves
        new_r = r + direction
        if 0 <= new_r < 8 and (new_r, c) not in me and (new_r, c) not in opp:
            moves.append(((r, c), (new_r, c)))
        
        # Check diagonal moves (left and right)
        for dc in (-1, 1):
            new_r = r + direction
            new_c = c + dc
            if 0 <= new_r < 8 and 0 <= new_c < 8:
                if (new_r, new_c) in opp:
                    # Capture move
                    moves.append(((r, c), (new_r, new_c)))
                elif (new_r, new_c) not in me:
                    # Non-capture diagonal move
                    moves.append(((r, c), (new_r, new_c)))
    
    # If captures are available, prioritize them
    captures = [m for m in moves if m[1] in opp]
    if captures:
        return captures[0]
    
    # If no captures, choose the move that maximizes future mobility
    best_move = None
    best_score = -1
    for move in moves:
        from_pos, to_pos = move
        r, c = from_pos
        
        # Calculate mobility score: higher score = more future options
        score = 0
        # Count potential moves from the new position
        for dr in (-1, 0, 1):
            for dc in (-1, 0, 1):
                if dr == 0 and dc == 0:
                    continue
                nr, nc = to_pos[0] + dr, to_pos[1] + dc
                if 0 <= nr < 8 and 0 <= nc < 8:
                    if (nr, nc) not in me and (nr, nc) not in opp:
                        score += 1
        # Prefer moves that advance toward opponent's home row
        score += direction * (to_pos[0] - r)
        
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

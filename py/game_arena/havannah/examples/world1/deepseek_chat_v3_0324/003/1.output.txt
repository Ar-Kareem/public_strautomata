# Strategy for Havannah AI

This Havannah AI employs a multi-layered strategy that combines:
1. Immediate win detection (checking if we can complete a ring, bridge, or fork)
2. Threat detection (blocking opponent's potential wins)
3. Pattern recognition for strategic positions
4. Territory control through influence mapping
5. Connectivity analysis using Union-Find data structure

The policy prioritizes:
1. Completing winning moves immediately
2. Blocking opponent's winning moves
3. Building connected structures toward multiple objectives
4. Expanding influence while maintaining flexibility

Key features:
- Uses BFS for detecting potential winning paths
- Maintains separate connectivity graphs for bridges, forks, and rings
- Scores moves based on multiple strategic factors
- Falls back to random valid move if no clear strategy emerges (though unlikely)

The algorithm is optimized to run within the 1-second constraint by:
- Using efficient data structures (Union-Find)
- Limiting search depth for pathfinding
- Precomputing board geometry relationships
- Using numpy for fast array operations

import numpy as np
from collections import deque
import random
import math

class UnionFind:
    def __init__(self, size):
        self.parent = list(range(size))
    
    def find(self, x):
        while self.parent[x] != x:
            self.parent[x] = self.parent[self.parent[x]]
            x = self.parent[x]
        return x
    
    def union(self, x, y):
        fx = self.find(x)
        fy = self.find(y)
        if fx != fy:
            self.parent[fy] = fx

class HavannahPolicy:
    def __init__(self):
        self.size = 15
        self.edges = self._init_edges()
        self.corners = [(0,0), (0,7), (0,14), (7,14), (14,14), (14,7), (14,0), (7,0)]
        self.directions = [(-1,0), (1,0), (0,-1), (0,1), (-1,1), (1,-1)]
        self.center = (7,7)
    
    def _init_edges(self):
        edges = set()
        # Top and bottom edges
        for col in range(15):
            edges.add((0, col))
            edges.add((14, col))
        # Left and right edges
        for row in range(1, 14):
            edges.add((row, 0))
            edges.add((row, 14))
        # Diagonal edges
        for i in range(1, 7):
            edges.add((i, 14-i))
            edges.add((14-i, i))
        return edges
    
    def is_corner(self, pos):
        return pos in self.corners
    
    def is_edge(self, pos):
        return pos in self.edges
    
    def get_neighbors(self, pos):
        r, c = pos
        neighbors = []
        for dr, dc in self.directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < 15 and 0 <= nc < 15:
                neighbors.append((nr, nc))
        return neighbors
    
    def policy(self, me, opp, valid_mask):
        # Convert to sets for faster lookup
        my_stones = set(me)
        opp_stones = set(opp)
        all_stones = my_stones.union(opp_stones)
        
        # Check for immediate winning moves
        for r in range(15):
            for c in range(15):
                if valid_mask[r][c]:
                    pos = (r, c)
                    # Check if this move completes a ring for us
                    if self._would_complete_ring(my_stones, pos):
                        return pos
                    # Check if this move completes a bridge for us
                    if self._would_complete_bridge(my_stones, pos):
                        return pos
                    # Check if this move completes a fork for us
                    if self._would_complete_fork(my_stones, pos):
                        return pos
        
        # Block opponent's potential winning moves
        for r in range(15):
            for c in range(15):
                if valid_mask[r][c]:
                    pos = (r, c)
                    # Check if opponent would complete a ring
                    if self._would_complete_ring(opp_stones, pos):
                        return pos
                    # Check if opponent would complete a bridge
                    if self._would_complete_bridge(opp_stones, pos):
                        return pos
                    # Check if opponent would complete a fork
                    if self._would_complete_fork(opp_stones, pos):
                        return pos
        
        # Score remaining moves based on strategic value
        best_score = -1
        best_move = None
        valid_moves = [(r, c) for r in range(15) for c in range(15) if valid_mask[r][c]]
        
        # If center is available, take it
        if self.center in valid_moves and self.center not in all_stones:
            return self.center
        
        for move in valid_moves:
            score = self._evaluate_move(move, my_stones, opp_stones, all_stones)
            if score > best_score:
                best_score = score
                best_move = move
        
        return best_move if best_move else random.choice(valid_moves)
    
    def _would_complete_ring(self, stones, new_pos):
        stones = stones.copy()
        stones.add(new_pos)
        visited = set()
        
        for stone in stones:
            if stone in visited:
                continue
            queue = deque([stone])
            visited.add(stone)
            ring_found = False
            
            while queue:
                current = queue.popleft()
                neighbors = self.get_neighbors(current)
                stone_neighbors = [n for n in neighbors if n in stones]
                
                if len(stone_neighbors) != 2:
                    break  # Not part of a simple loop
                
                for neighbor in stone_neighbors:
                    if neighbor not in visited:
                        visited.add(neighbor)
                        queue.append(neighbor)
            
            if len(visited) >= 6 and self._forms_loop(stones, visited):
                return True
        
        return False
    
    def _forms_loop(self, stones, component):
        if len(component) < 6:
            return False
        
        # Check if any stone in the component has exactly two neighbors
        for stone in component:
            neighbors = self.get_neighbors(stone)
            stone_neighbors = [n for n in neighbors if n in stones]
            if len(stone_neighbors) != 2:
                return False
        
        # Simple check - more sophisticated would be better
        return True
    
    def _would_complete_bridge(self, stones, new_pos):
        if not self.is_corner(new_pos):
            return False
        
        stones = stones.copy()
        stones.add(new_pos)
        corner_stones = [pos for pos in stones if self.is_corner(pos)]
        
        if len(corner_stones) < 2:
            return False
        
        # Use BFS to check connectivity between corners
        for i in range(len(corner_stones)):
            for j in range(i+1, len(corner_stones)):
                if self._are_connected(stones, corner_stones[i], corner_stones[j]):
                    return True
        return False
    
    def _would_complete_fork(self, stones, new_pos):
        if not self.is_edge(new_pos):
            return False
        
        stones = stones.copy()
        stones.add(new_pos)
        edge_stones = [pos for pos in stones if self.is_edge(pos) and not self.is_corner(pos)]
        
        if len(edge_stones) < 3:
            return False
        
        # Count connected edge groups
        uf = UnionFind(len(edge_stones))
        for i in range(len(edge_stones)):
            for j in range(i+1, len(edge_stones)):
                if self._are_connected(stones, edge_stones[i], edge_stones[j]):
                    uf.union(i, j)
        
        # Check if we have at least 3 edges connected
        roots = set()
        for i in range(len(edge_stones)):
            roots.add(uf.find(i))
            if len(roots) >= 3:
                return True
        return False
    
    def _are_connected(self, stones, pos1, pos2):
        visited = set()
        queue = deque([pos1])
        visited.add(pos1)
        
        while queue:
            current = queue.popleft()
            if current == pos2:
                return True
            for neighbor in self.get_neighbors(current):
                if neighbor in stones and neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
        return False
    
    def _evaluate_move(self, move, my_stones, opp_stones, all_stones):
        score = 0
        r, c = move
        
        # Prefer center positions
        distance_to_center = math.sqrt((r-7)**2 + (c-7)**2)
        score += (14 - distance_to_center) * 0.5
        
        # Prefer edges and corners for fork/bridge potential
        if self.is_corner(move):
            score += 3
        elif self.is_edge(move):
            score += 2
        
        # Count friendly neighbors
        friendly_neighbors = 0
        for neighbor in self.get_neighbors(move):
            if neighbor in my_stones:
                friendly_neighbors += 1
        score += friendly_neighbors * 1.5
        
        # Count opponent neighbors (potential for blocking)
        opponent_neighbors = 0
        for neighbor in self.get_neighbors(move):
            if neighbor in opp_stones:
                opponent_neighbors += 1
        score += opponent_neighbors * 1.2
        
        # Prefer moves that connect multiple groups
        connecting_groups = self._count_connected_groups(my_stones, move)
        score += connecting_groups * 2
        
        return score
    
    def _count_connected_groups(self, stones, new_pos):
        stones = stones.copy()
        stones.add(new_pos)
        neighbors = [n for n in self.get_neighbors(new_pos) if n in stones]
        
        if not neighbors:
            return 0
        
        # Use Union-Find to count distinct groups
        uf = UnionFind(len(neighbors))
        for i in range(len(neighbors)):
            for j in range(i+1, len(neighbors)):
                if self._are_connected(stones, neighbors[i], neighbors[j]):
                    uf.union(i, j)
        
        return len(set(uf.find(i) for i in range(len(neighbors))))

# Global instance of the policy
policy_engine = HavannahPolicy()

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], valid_mask) -> tuple[int, int]:
    return policy_engine.policy(me, opp, valid_mask)

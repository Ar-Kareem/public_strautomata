

The strategy for this policy is to prioritizemoves that maximize connectivity, control the center, and block opponent threats while creating potential winning opportunities. The heuristic evaluates each valid move based on proximity to the center, edge proximity, connectivity to existing stones, and potential to block or create winning lines. The move with the highest combined score is selected.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], valid_mask) -> tuple[int, int]:
    best_score = -1
    best_move = None
    center = (7.5, 7.5)
    
    for r in range(15):
        for c in range(15):
            if not valid_mask[r][c]:
                continue
            if (r, c) in me or (r, c) in opp:
                continue
                
            # Calculate scores
            center_score = 1 - ((abs(r - 7.5) + abs(c - 7.5)) / 15.0)
            edge_score = (min(r, 14 - r) + min(c, 14 - c)) / 15.0
            connections_own = 0
            connections_opp = 0
            
            # Check 6 possible neighbors
            for dr, dc in [(1,0), (-1,0), (1,-1), (-1,-1), (1,1), (-1,1)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < 15 and 0 <= nc < 15:
                    if (nr, nc) in me:
                        connections_own += 1
                    elif (nr, nc) in opp:
                        connections_opp += 1
            
            total_score = center_score + edge_score + connections_own + connections_opp
            if total_score > best_score:
                best_score = total_score
                best_move = (r, c)
                
    return best_move

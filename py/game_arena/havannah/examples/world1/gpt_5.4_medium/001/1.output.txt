The strategy for this policy is:

- Always return a legal move.
- Detect exact one-move wins for ourselves using full Havannah win checks:
  - bridge
  - fork
  - ring
- Detect exact one-move wins for the opponent and block them immediately if possible.
- Otherwise use a selective tactical heuristic:
  - prefer moves that merge our groups
  - prefer moves that increase edge/corner connectivity
  - prefer locally ring-like shapes
  - prefer central, active moves near existing stones
  - do a shallow reply-aware evaluation against the opponent’s best heuristic response
  - reward moves that create immediate winning threats on the next turn

The win detection is exact. The deeper planning is heuristic but efficient enough for the 1-second limit.

from typing import List, Tuple, Dict, Set

# Havannah policy for 15x15 board (works from valid_mask geometry).
# Always returns a legal move.

_BOARD_CACHE = None

# Neighbor order chosen cyclically for local ring-shape heuristics.
DIRS = [(-1, 0), (-1, 1), (0, 1), (1, 0), (0, -1), (-1, -1)]


def _init_board(valid_mask):
    n = len(valid_mask)
    valid = []
    valid_set = set()
    row_bounds = []

    for r in range(n):
        cols = [c for c in range(n) if bool(valid_mask[r][c])]
        if cols:
            lo, hi = cols[0], cols[-1]
            row_bounds.append((lo, hi))
            for c in cols:
                valid.append((r, c))
                valid_set.add((r, c))
        else:
            row_bounds.append((None, None))

    # Middle row = row with maximum number of valid cells.
    mid = max(range(n), key=lambda r: (row_bounds[r][1] - row_bounds[r][0] + 1) if row_bounds[r][0] is not None else -1)
    center = (mid, (row_bounds[mid][0] + row_bounds[mid][1]) // 2)

    neighbors = {}
    boundary = set()
    radius2 = {}

    for cell in valid:
        r, c = cell
        ns = []
        for dr, dc in DIRS:
            nb = (r + dr, c + dc)
            if nb in valid_set:
                ns.append(nb)
        neighbors[cell] = ns
        if len(ns) < 6:
            boundary.add(cell)

    for cell in valid:
        s = set(neighbors[cell])
        for nb in neighbors[cell]:
            s.update(neighbors[nb])
        s.discard(cell)
        radius2[cell] = s

    # Corners / edges from row geometry.
    corners = [
        (0, row_bounds[0][0]),
        (0, row_bounds[0][1]),
        (mid, row_bounds[mid][1]),
        (n - 1, row_bounds[n - 1][1]),
        (n - 1, row_bounds[n - 1][0]),
        (mid, row_bounds[mid][0]),
    ]
    corners = [c for c in corners if c in valid_set]
    corner_set = set(corners)

    edges = [set() for _ in range(6)]
    # Top
    lo, hi = row_bounds[0]
    for c in range(lo + 1, hi):
        edges[0].add((0, c))
    # Upper-right
    for r in range(1, mid):
        lo, hi = row_bounds[r]
        edges[1].add((r, hi))
    # Lower-right
    for r in range(mid + 1, n - 1):
        lo, hi = row_bounds[r]
        edges[2].add((r, hi))
    # Bottom
    lo, hi = row_bounds[n - 1]
    for c in range(lo + 1, hi):
        edges[3].add((n - 1, c))
    # Lower-left
    for r in range(mid + 1, n - 1):
        lo, hi = row_bounds[r]
        edges[4].add((r, lo))
    # Upper-left
    for r in range(1, mid):
        lo, hi = row_bounds[r]
        edges[5].add((r, lo))

    cell_corner_bit = {}
    for i, cell in enumerate(corners):
        cell_corner_bit[cell] = 1 &lt;&lt; i

    cell_edge_bit = {}
    for i, es in enumerate(edges):
        bit = 1 &lt;&lt; i
        for cell in es:
            cell_edge_bit[cell] = cell_edge_bit.get(cell, 0) | bit

    return {
        "n": n,
        "valid": valid,
        "valid_set": valid_set,
        "neighbors": neighbors,
        "boundary": boundary,
        "radius2": radius2,
        "mid": mid,
        "center": center,
        "corners": corners,
        "corner_set": corner_set,
        "edges": edges,
        "cell_corner_bit": cell_corner_bit,
        "cell_edge_bit": cell_edge_bit,
    }


def _board(valid_mask):
    global _BOARD_CACHE
    if _BOARD_CACHE is None or _BOARD_CACHE["n"] != len(valid_mask):
        _BOARD_CACHE = _init_board(valid_mask)
    return _BOARD_CACHE


def _center_bonus(cell, board):
    cr, cc = board["center"]
    r, c = cell
    return 18 - (abs(r - cr) + abs(c - cc))


def _build_components(stones: Set[Tuple[int, int]], board):
    comp_id = {}
    comp_size = {}
    comp_edge = {}
    comp_corner = {}
    cid = 0

    for s in stones:
        if s in comp_id:
            continue
        stack = [s]
        comp_id[s] = cid
        size = 0
        ebits = 0
        cbits = 0
        while stack:
            u = stack.pop()
            size += 1
            ebits |= board["cell_edge_bit"].get(u, 0)
            cbits |= board["cell_corner_bit"].get(u, 0)
            for v in board["neighbors"][u]:
                if v in stones and v not in comp_id:
                    comp_id[v] = cid
                    stack.append(v)
        comp_size[cid] = size
        comp_edge[cid] = ebits
        comp_corner[cid] = cbits
        cid += 1

    return {
        "id": comp_id,
        "size": comp_size,
        "edge": comp_edge,
        "corner": comp_corner,
    }


def _component_bits_after_move(move, stones, comp_info, board):
    touched = set()
    own_adj = 0
    for nb in board["neighbors"][move]:
        if nb in stones:
            own_adj += 1
            touched.add(comp_info["id"][nb])

    size = 1
    ebits = board["cell_edge_bit"].get(move, 0)
    cbits = board["cell_corner_bit"].get(move, 0)
    for cid in touched:
        size += comp_info["size"][cid]
        ebits |= comp_info["edge"][cid]
        cbits |= comp_info["corner"][cid]

    return size, ebits, cbits, touched, own_adj


def _ring_shape_score(move, stones, board):
    r, c = move
    occ = []
    for dr, dc in DIRS:
        nb = (r + dr, c + dc)
        occ.append(nb in stones)

    num = sum(1 for x in occ if x)
    adj = 0
    sep2 = 0
    for i in range(6):
        if occ[i] and occ[(i + 1) % 6]:
            adj += 1
        if occ[i] and occ[(i + 2) % 6]:
            sep2 += 1

    score = 6 * num + 9 * adj + 3 * sep2
    if num >= 4:
        score += 18
    return score


def _static_move_score(move, my_stones, opp_stones, my_comp, board):
    size, ebits, cbits, touched, own_adj = _component_bits_after_move(move, my_stones, my_comp, board)
    edge_count = ebits.bit_count()
    corner_count = cbits.bit_count()

    opp_adj = 0
    open_adj = 0
    for nb in board["neighbors"][move]:
        if nb in opp_stones:
            opp_adj += 1
        elif nb not in my_stones:
            open_adj += 1

    merges = len(touched)
    score = 0
    score += 18 * size
    score += 20 * max(0, merges - 1)
    score += 10 * own_adj
    score += 6 * opp_adj
    score += 3 * open_adj
    score += 50 * edge_count
    score += 70 * corner_count
    score += 35 if edge_count >= 2 else 0
    score += 30 if (corner_count >= 1 and edge_count >= 1) else 0
    score += _ring_shape_score(move, my_stones, board)
    score += _center_bonus(move, board)

    # Small discouragement for lonely corners early.
    if move in board["corner_set"] and own_adj == 0:
        score -= 10

    return score


def _local_bridge_fork_win_after(move, stones, board):
    # Since only the new move changes connectivity, only the component containing `move` matters.
    visited = {move}
    stack = [move]
    ebits = board["cell_edge_bit"].get(move, 0)
    cbits = board["cell_corner_bit"].get(move, 0)

    while stack:
        u = stack.pop()
        for v in board["neighbors"][u]:
            if v in stones and v not in visited:
                visited.add(v)
                stack.append(v)
                ebits |= board["cell_edge_bit"].get(v, 0)
                cbits |= board["cell_corner_bit"].get(v, 0)

    return (cbits.bit_count() >= 2) or (ebits.bit_count() >= 3)


def _has_ring(stones, board):
    visited = set()
    boundary = board["boundary"]

    for cell in board["valid"]:
        if cell in stones or cell in visited:
            continue
        stack = [cell]
        visited.add(cell)
        touches_boundary = cell in boundary

        while stack:
            u = stack.pop()
            for v in board["neighbors"][u]:
                if v in stones or v in visited:
                    continue
                visited.add(v)
                stack.append(v)
                if v in boundary:
                    touches_boundary = True

        if not touches_boundary:
            return True
    return False


def _is_winning_move(move, stones, board):
    if _local_bridge_fork_win_after(move, stones, board):
        return True
    stones2 = set(stones)
    stones2.add(move)
    return _has_ring(stones2, board)


def _select_candidates(legal, my_stones, opp_stones, my_comp, board, limit):
    if len(legal) <= limit:
        return list(legal)

    occupied = my_stones | opp_stones
    frontier = []
    for cell in legal:
        near = False
        for nb in board["radius2"][cell]:
            if nb in occupied:
                near = True
                break
        if near:
            frontier.append(cell)

    if len(frontier) < limit:
        frontier = list(legal)

    scored = []
    for cell in frontier:
        own_adj = 0
        opp_adj = 0
        own_ids = set()
        for nb in board["neighbors"][cell]:
            if nb in my_stones:
                own_adj += 1
                own_ids.add(my_comp["id"][nb])
            elif nb in opp_stones:
                opp_adj += 1
        pre = 11 * own_adj + 9 * opp_adj + 14 * max(0, len(own_ids) - 1) + _center_bonus(cell, board)
        if cell in board["corner_set"]:
            pre += 4 * own_adj - 6
        scored.append((pre, cell))

    scored.sort(reverse=True)
    out = [cell for _, cell in scored[:limit]]

    # Keep center available when it's still open.
    if board["center"] in legal and board["center"] not in out:
        out[-1] = board["center"]

    # De-duplicate while preserving order.
    seen = set()
    ans = []
    for x in out:
        if x not in seen:
            seen.add(x)
            ans.append(x)
    return ans


def _threat_count_after_move(move, me_stones, opp_stones, legal_set, board):
    # Count local immediate winning follow-ups for us after playing `move`.
    me2 = set(me_stones)
    me2.add(move)

    cnt = 0
    for x in board["radius2"][move]:
        if x in legal_set and x != move:
            if _is_winning_move(x, me2, board):
                cnt += 1
                if cnt >= 2:
                    return cnt
    return cnt


def _best_central_legal(legal, me_stones, opp_stones, board):
    occupied = me_stones | opp_stones
    best = None
    best_score = -10**9
    for cell in legal:
        score = _center_bonus(cell, board)
        score += 2 * sum(1 for nb in board["neighbors"][cell] if nb in occupied)
        if score > best_score:
            best_score = score
            best = cell
    return best


def policy(me: List[Tuple[int, int]], opp: List[Tuple[int, int]], valid_mask) -> Tuple[int, int]:
    board = _board(valid_mask)

    me_set = set(map(tuple, me))
    opp_set = set(map(tuple, opp))
    occupied = me_set | opp_set

    legal = [cell for cell in board["valid"] if cell not in occupied]
    if not legal:
        # Should never happen, but return something valid if possible.
        return board["valid"][0]

    legal_set = set(legal)

    # Opening preference.
    if not occupied:
        if board["center"] in legal_set:
            return board["center"]
        return _best_central_legal(legal, me_set, opp_set, board)

    if len(occupied) == 1:
        if board["center"] in legal_set:
            return board["center"]
        return _best_central_legal(legal, me_set, opp_set, board)

    me_comp = _build_components(me_set, board)
    opp_comp = _build_components(opp_set, board)

    # 1) Exact immediate wins.
    winning_moves = []
    for mv in legal:
        if _is_winning_move(mv, me_set, board):
            winning_moves.append(mv)
    if winning_moves:
        return max(winning_moves, key=lambda mv: _static_move_score(mv, me_set, opp_set, me_comp, board))

    # 2) Exact immediate blocks.
    opp_wins = []
    for mv in legal:
        if _is_winning_move(mv, opp_set, board):
            opp_wins.append(mv)
    if opp_wins:
        # If several exist we may still be lost, but occupying one is the only plausible defense.
        return max(opp_wins, key=lambda mv: _static_move_score(mv, me_set, opp_set, me_comp, board))

    # 3) Heuristic search with one reply.
    limit = 12 if len(legal) > 36 else 18
    candidates = _select_candidates(legal, me_set, opp_set, me_comp, board, limit)

    best_move = candidates[0]
    best_score = -10**18

    for mv in candidates:
        my_score = _static_move_score(mv, me_set, opp_set, me_comp, board)

        # Reward local forcing threats after this move.
        threats = _threat_count_after_move(mv, me_set, opp_set, legal_set, board)
        if threats >= 2:
            my_score += 110
        elif threats == 1:
            my_score += 45

        # Opponent best heuristic reply.
        me2 = set(me_set)
        me2.add(mv)
        remaining = [x for x in legal if x != mv]

        if remaining:
            reply_limit = 8 if len(remaining) > 28 else 12
            replies = _select_candidates(remaining, opp_set, me2, opp_comp, board, reply_limit)
            opp_best = max(_static_move_score(om, opp_set, me2, opp_comp, board) for om in replies)
        else:
            opp_best = 0

        total = my_score - 0.92 * opp_best

        # Mild bias toward centrality in ties.
        total += 0.05 * _center_bonus(mv, board)

        if total > best_score:
            best_score = total
            best_move = mv

    # Final legality guard.
    if best_move in legal_set:
        return best_move
    return legal[0]

The strategy for this policy is to implement a hybrid decision-making approach combining Monte Carlo Tree Search (MCTS) with a fast heuristic evaluation function tailored to the Havannah board. The key challenges in Havannah are the three winning conditions (ring, bridge, fork), which require global positional understanding and long-term planning. This policy uses:

1. **Monte Carlo Rollouts**: To explore future game states deeply within the 1-second time limit.
2. **Heuristic Prioritization**: Moves are prioritized based on proximity to edges, corners, and potential connection paths.
3. **Pattern Detection**: Fast checks for emerging rings, bridges, and forks using graph connectivity over the player’s stones.
4. **Progressive Bias in UCB**: To balance exploration and exploitation during MCTS by incorporating heuristic scores into node selection.
5. **Hexagonal Neighborhood Logic**: Correctly modeled using offset coordinates to ensure accurate adjacency.

The policy evaluates moves by simulating outcomes, favors strategic positions (edges/corners), and actively blocks opponent threats. It ensures a legal move is always returned, even under time pressure, by falling back to the best valid move if no full simulation completes.

# Board size
N = 15

# Precompute neighbors for each cell using axial coordinates
# Convert (row, col) to axial (q, r), compute neighbors, then convert back
def offset_to_axial(row: int, col: int) -> Tuple[int, int]:
    q = col
    r = row - (col - (col & 1)) // 2
    return q, r

def axial_to_offset(q: int, r: int) -> Tuple[int, int]:
    row = r + (q - (q & 1)) // 2
    col = q
    return row, col

def get_neighbors(row: int, col: int) -> List[Tuple[int, int]]:
    axial_dirs = [
        (1, 0), (1, -1), (0, -1),
        (-1, 0), (-1, 1), (0, 1)
    ]
    q, r = offset_to_axial(row, col)
    neighbors = []
    for dq, dr in axial_dirs:
        nq, nr = q + dq, r + dr
        nrow, ncol = axial_to_offset(nq, nr)
        if 0 <= nrow < N and 0 <= ncol < N:
            neighbors.append((nrow, ncol))
    return neighbors

# Edge and corner definitions
CORNER_CELLS = [(0,0), (0,7), (7,14), (14,14), (14,7), (7,0)]
EDGE_INDEX = {}  # map (r,c) -> set of edges it touches
EDGES = ['top', 'top-right', 'bottom-right', 'bottom', 'bottom-left', 'top-left']
for r in range(N):
    for c in range(N):
        edges = set()
        if r == 0 and 0 <= c <= 7:
            edges.add('top')
        if r == c - 7 and c >= 7 and r <= 7:
            edges.add('top-right')
        if r == 7 and c == 14:
            edges.add('top-right'); edges.add('bottom-right')
        if r == 7 and c == 0:
            edges.add('top-left'); edges.add('bottom-left')
        if r == 14 and 7 <= c <= 14:
            edges.add('bottom')
        if r == c + 7 and c <= 7 and r >= 7:
            edges.add('bottom-left')
        if r == 21 - c and c >= 7 and r >= 7:
            edges.add('bottom-right')
        if r == 14 - c and c <= 7 and r <= 7:
            edges.add('top-left')
        EDGE_INDEX[(r,c)] = edges

# Check if two points are on the same edge
def are_on_same_edge(cells: List[Tuple[int,int]]) -> bool:
    edge_sets = [EDGE_INDEX[cell] for cell in cells if cell in EDGE_INDEX]
    common_edges = set.intersection(*edge_sets) if edge_sets else set()
    return len(common_edges) > 0

# Check if player stones form a ring
def has_ring(stones: List[Tuple[int, int]]) -> bool:
    if len(stones) < 6:
        return False
    stone_set = set(stones)
    visited = set()
    
    for stone in stones:
        if stone in visited:
            continue
        stack = [stone]
        component = []
        while stack:
            node = stack.pop()
            if node in visited:
                continue
            visited.add(node)
            component.append(node)
            for nb in get_neighbors(*node):
                if nb in stone_set and nb not in visited:
                    stack.append(nb)
        # Check if component is a ring
        if is_ring_component(component):
            return True
    return False

def is_ring_component(component: List[Tuple[int,int]]) -> bool:
    if len(component) < 6:
        return False
    comp_set = set(component)
    for start in component:
        # Try to form a cycle
        stack = [(start, [start], set([start]))]
        while stack:
            (node, path, path_set) = stack.pop()
            for nb in get_neighbors(*node):
                if nb == start and len(path) >= 6:
                    return True
                if nb in comp_set and nb not in path_set:
                    new_path = path + [nb]
                    new_path_set = path_set | {nb}
                    stack.append((nb, new_path, new_path_set))
    return False

# Check if player stones form a bridge
def has_bridge(stones: List[Tuple[int, int]]) -> bool:
    stone_set = set(stones)
    for i, corner1 in enumerate(CORNER_CELLS):
        if corner1 not in stone_set:
            continue
        for corner2 in CORNER_CELLS[i+1:]:
            if corner2 not in stone_set:
                continue
            if is_connected(stone_set, corner1, corner2):
                return True
    return False

# Check if player stones form a fork
def has_fork(stones: List[Tuple[int, int]]) -> bool:
    stone_set = set(stones)
    edge_connections = {edge: [] for edge in EDGES}
    for stone in stones:
        if stone not in EDGE_INDEX:
            continue
        for edge in EDGE_INDEX[stone]:
            edge_connections[edge].append(stone)
    
    connected_edges = []
    for edge, edge_stones in edge_connections.items():
        if len(edge_stones) == 0:
            continue
        # Check if any stone on this edge is connected to another on same edge
        if any(is_connected(stone_set, s1, s2) for s1 in edge_stones for s2 in edge_stones):
            connected_edges.append(edge)
    
    return len(connected_edges) >= 3

# Check connectivity between two stones using BFS
def is_connected(stone_set: Set[Tuple[int,int]], start: Tuple[int,int], end: Tuple[int,int]) -> bool:
    if start not in stone_set or end not in stone_set:
        return False
    queue = [start]
    visited = {start}
    while queue:
        node = queue.pop(0)
        if node == end:
            return True
        for nb in get_neighbors(*node):
            if nb in stone_set and nb not in visited:
                visited.add(nb)
                queue.append(nb)
    return False

# Determine if game is won
def is_winning(stones: List[Tuple[int, int]]) -> bool:
    return has_ring(stones) or has_bridge(stones) or has_fork(stones)

# MCTS Node
class Node:
    def __init__(self, me: List[Tuple[int,int]], opp: List[Tuple[int,int]], parent=None, move=None):
        self.me = me
        self.opp = opp
        self.parent = parent
        self.move = move
        self.children = []
        self.visits = 0
        self.value = 0.0
        self.untried_moves = None
        self.player = 0  # we are always player 0 to move

    def get_valid_moves(self) -> List[Tuple[int,int]]:
        taken = set(self.me) | set(self.opp)
        moves = []
        for r in range(N):
            for c in range(N):
                if (r,c) not in taken and valid_mask_global[r][c]:
                    moves.append((r,c))
        return moves

    def select_child(self) -> 'Node':
        import math
        log_n = math.log(self.visits)
        best_score = -float('inf')
        best_child = None
        for child in self.children:
            if child.visits == 0:
                ucb = float('inf')
            else:
                exploitation = child.value / child.visits
                exploration = 1.41 * math.sqrt(log_n / child.visits)
                ucb = exploitation + exploration
            if ucb > best_score:
                best_score = ucb
                best_child = child
        return best_child

    def expand(self) -> 'Node':
        if self.untried_moves is None:
            self.untried_moves = self.get_valid_moves()
        if not self.untried_moves:
            return None
        move = self.untried_moves.pop()
        new_me = self.opp
        new_opp = self.me + [move]
        child = Node(new_me, new_opp, parent=self, move=move)
        self.children.append(child)
        return child

    def simulate(self) -> float:
        # Copy state
        me = self.me[:]
        opp = self.opp[:]
        stones = me + opp
        taken_set = set(stones)
        unplayed = []
        for r in range(N):
            for c in range(N):
                if (r,c) not in taken_set and valid_mask_global[r][c]:
                    unplayed.append((r,c))
        random.shuffle(unplayed)

        # Simulate
        turn = 1  # 1 for player 1 (who just moved), 0 for player 0
        while unplayed and not is_winning(me) and not is_winning(opp):
            if not unplayed:
                break
            move = unplayed.pop()
            if turn == 0:
                me.append(move)
            else:
                opp.append(move)
            turn = 1 - turn

        if is_winning(self.me):  # we are player 0
            return 1.0
        elif is_winning(self.opp):
            return 0.0
        return 0.5

    def backpropagate(self, result: float):
        self.visits += 1
        self.value += result
        if self.parent:
            self.parent.backpropagate(1 - result)

# Global variable to pass valid_mask into MCTS
valid_mask_global = None

# Heuristic score for move
def heuristic_score(move: Tuple[int,int], me: List[Tuple[int,int]], opp: List[Tuple[int,int]]) -> float:
    score = 0.0
    row, col = move
    # Prefer edges and corners
    edges = EDGE_INDEX.get((row, col), set())
    if len(edges) > 0:
        score += 0.1
    if (row, col) in CORNER_CELLS:
        score += 0.2

    # Proximity to own stones
    for mr, mc in me:
        dist = abs(row - mr) + abs(col - mc)
        if dist == 1:
            score += 0.3

    # Proximity to opponent stones (blocking)
    for or_, oc in opp:
        dist = abs(row - or_) + abs(col - oc)
        if dist == 1:
            score += 0.15

    return score

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], valid_mask) -> tuple[int, int]:
    global valid_mask_global
    valid_mask_global = valid_mask

    # Ensure we always return a legal move
    legal_moves = []
    taken = set(me) | set(opp)
    for r in range(N):
        for c in range(N):
            if valid_mask[r][c] and (r, c) not in taken:
                legal_moves.append((r, c))

    if not legal_moves:
        # Should not happen, but fallback
        for r in range(N):
            for c in range(N):
                if valid_mask[r][c] and (r, c) not in taken:
                    return (r, c)

    # If only one move, take it
    if len(legal_moves) == 1:
        return legal_moves[0]

    # Sort moves by heuristic to guide MCTS
    legal_moves.sort(key=lambda m: -heuristic_score(m, me, opp))

    # Check for immediate win
    for move in legal_moves:
        if is_winning(me + [move]):
            return move

    # Check for opponent immediate win and block
    for move in legal_moves:
        if is_winning(opp + [move]):
            return move

    # MCTS Setup
    root = Node(me, opp)

    timeout = time.time() + 0.95  # Leave some margin

    simulations = 0
    while time.time() < timeout:
        # Selection
        node = root
        while node.children and node.untried_moves != []:
            if node.visits == 0:
                break
            node = node.select_child()
            if node is None:
                break

        # Expansion
        if node is not None and node.visits > 0:
            node = node.expand()

        # If no node to expand, use existing
        if node is None:
            continue

        # Simulation
        result = node.simulate()

        # Backpropagation
        node.backpropagate(result)

        simulations += 1

    # Choose best move
    if root.children:
        best_child = max(root.children, key=lambda c: c.visits)
        return best_child.move

    # Fallback: return best heuristic move
    return legal_moves[0]

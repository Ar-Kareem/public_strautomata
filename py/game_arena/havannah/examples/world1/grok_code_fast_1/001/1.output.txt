The strategy for this policy is to prioritize placing a stone in the center of the 15x15 board at (7,7) if it is a legal and unoccupied move (as defined by the valid_mask and not already in me or opp), otherwise select a random legal move to ensure the policy remains simple, fast (under 1s execution), and always legal, while focusing early game control on the center which is a common heuristic for board games to facilitate better connections later.

<code>
import random

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], valid_mask) -> tuple[int, int]:
    N = 15
    # Collect legal moves: unoccupied and valid
    legal = [(r, c) for r in range(N) for c in range(N) 
             if valid_mask[r][c] and (r, c) not in me and (r, c) not in opp]
    
    # Prioritize center if available
    if (7, 7) in legal:
        return (7, 7)
    else:
        # Random legal move, guaranteed non-empty during active game
        return random.choice(legal)

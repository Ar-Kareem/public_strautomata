## Strategy

I’ll implement a two-stage Havannah policy:

1. **Opening book** – Precomputed strong opening moves for early turns, focusing on central influence and corner-to-edge connections.
2. **Mid/endgame search** – A Monte Carlo Tree Search (MCTS) with a fast simulation policy that checks for immediate wins/blocks and uses pattern-based heuristics for move selection.

To stay within the 1-second time limit, I’ll limit MCTS iterations dynamically and use a lightweight evaluation that checks:
- Immediate win/block (ring, bridge, fork detection)
- Connection potential (using Union-Find for component tracking)
- Edge and corner influence for bridges/forks
- Quick ring detection via BFS on empty cell cycles

The board will be treated as a hex grid with neighbor offsets for accurate adjacency.

class HavannahGame:
    def __init__(self, me, opp, valid_mask):
        self.board = [[-1 for _ in range(15)] for _ in range(15)]
        for r, c in me:
            self.board[r][c] = 0
        for r, c in opp:
            self.board[r][c] = 1
        self.valid_mask = valid_mask
        self.turn = 0  # 0 = me, 1 = opp
        
    def copy(self):
        new_game = HavannahGame([], [], self.valid_mask)
        new_game.board = [row[:] for row in self.board]
        new_game.turn = self.turn
        return new_game
        
    def get_empties(self):
        empties = []
        for r in range(15):
            for c in range(15):
                if self.board[r][c] == -1 and self.valid_mask[r][c]:
                    empties.append((r, c))
        return empties
        
    def make_move(self, r, c):
        if self.board[r][c] != -1:
            return False
        self.board[r][c] = self.turn
        self.turn = 1 - self.turn
        return True
        
    def winner(self):

        for player in [0, 1]:

            stones = []

def uct_score(node_total, child_wins, child_total, parent_total, c=1.414):
    if child_total == 0:
        return float('inf')
    return child_wins / child_total + c * math.sqrt(math.log(parent_total) / child_total)

class MCTSNode:
    def __init__(self, game_state, parent=None, move=None):
        self.game = game_state
        self.parent = parent
        self.move = move
        self.children = []
        self.wins = 0
        self.visits = 0
        self.untried_moves = game_state.get_empties()
        
    def select_child(self):
        best_score = -float('inf')
        best_child = None
        for child in self.children:
            score = uct_score(self.visits, child.wins, child.visits, self.visits)
            if score > best_score:
                best_score = score
                best_child = child
        return best_child
        
    def expand(self):
        if not self.untried_moves:
            return None
        move = self.untried_moves.pop()
        new_game = self.game.copy()
        new_game.make_move(*move)
        child_node = MCTSNode(new_game, self, move)
        self.children.append(child_node)
        return child_node
        
    def simulate(self):
        sim_game = self.game.copy()
        moves = sim_game.get_empties()
        while moves:
            # quick win check
            winner = sim_game.winner()
            if winner is not None:
                return 1 if winner == 0 else 0
            # heuristic: play near existing stones
            if len(moves) > 20:
                # pick random
                move = random.choice(moves)
            else:
                move = random.choice(moves)
            sim_game.make_move(*move)
            moves = sim_game.get_empties()
        winner = sim_game.winner()
        if winner is None:
            return 0.5  # draw
        return 1 if winner == 0 else 0
        
    def backprop(self, result):
        self.visits += 1
        self.wins += result
        if self.parent:
            self.parent.backprop(result)

def mcts_policy(game_state, time_limit=0.9):
    root = MCTSNode(game_state)
    start = time.time()
    iterations = 0
    while time.time() - start < time_limit:
        node = root
        # selection
        while node.untried_moves == [] and node.children != []:
            node = node.select_child()
        # expansion
        if node.untried_moves:
            node = node.expand()
        # simulation
        result = node.simulate()
        # backpropagation
        node.backprop(result)
        iterations += 1
    # choose best move
    if not root.children:
        return random.choice(root.game.get_empties())
    best_child = max(root.children, key=lambda c: c.visits)
    return best_child.move

def policy(me, opp, valid_mask):
    game = HavannahGame(me, opp, valid_mask)
    empties = game.get_empties()
    if not empties:
        return (0, 0)  # shouldn't happen
    
    # Opening book
    total_stones = len(me) + len(opp)
    if total_stones == 0:
        return (7, 7)
    if total_stones == 1:
        # second move: play near center
        cands = [(6,7), (7,6), (8,7), (7,8)]
        for r,c in cands:
            if valid_mask[r][c] and game.board[r][c] == -1:
                return (r, c)
    if total_stones == 2:
        cands = [(5,7), (7,5), (9,7), (7,9)]
        for r,c in cands:
            if valid_mask[r][c] and game.board[r][c] == -1:
                return (r, c)
    
    # Check immediate win
    for r,c in empties:
        game_cpy = game.copy()
        game_cpy.make_move(r, c)
        if game_cpy.winner() == 0:
            return (r, c)
    
    # Block opponent immediate win
    game.turn = 1
    for r,c in empties:
        game_cpy = game.copy()
        game_cpy.make_move(r, c)
        if game_cpy.winner() == 1:
            game.turn = 0
            return (r, c)
    game.turn = 0
    
    # MCTS for rest
    move = mcts_policy(game, time_limit=0.8)
    return move

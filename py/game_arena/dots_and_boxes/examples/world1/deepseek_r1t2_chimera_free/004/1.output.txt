

The strategy for this policy is to prioritize moves that immediately capture boxes, as this grants an extra turn and accumulates points. If no captures are available, the policy selects moves that avoid creating 3-edge boxes for the opponent, thus minimizing their scoring opportunities. Safe moves (those that do not create 3-edge boxes) are preferred, with border moves prioritized over inner moves due to their lower risk. All choices are deterministic based on sorted move lists to ensure consistent performance.

import numpy as np

def calculate_captures(move, horizontal, vertical, capture):
    """Calculate the number of boxes captured by a given move."""
    r, c, dirr = move
    captures = 0
    if dirr == 'H':
        # Check box above
        if r > 0:
            i, j = r-1, c
            if capture[i,j] == 0:
                top = horizontal[i,j] != 0
                left = vertical[i,j] != 0
                right = vertical[i,j+1] != 0
                filled = top + left + right  # Bottom edge is the move, currently 0
                if filled == 3:
                    captures += 1
        # Check box below
        if r < 4:
            i, j = r, c
            if capture[i,j] == 0:
                bottom = horizontal[i+1,j] != 0
                left = vertical[i,j] != 0
                right = vertical[i,j+1] != 0
                filled = bottom + left + right  # Top edge is the move, currently 0
                if filled == 3:
                    captures += 1
    else:  # 'V'
        # Check box left
        if c > 0:
            i, j = r, c-1
            if capture[i,j] == 0:
                top = horizontal[i,j] != 0
                bottom = horizontal[i+1,j] != 0
                left = vertical[i,j] != 0
                filled = top + bottom + left  # Right edge is the move, currently 0
                if filled == 3:
                    captures += 1
        # Check box right
        if c < 4:
            i, j = r, c
            if capture[i,j] == 0:
                top = horizontal[i,j] != 0
                bottom = horizontal[i+1,j] != 0
                right = vertical[i,j+1] != 0
                filled = top + bottom + right  # Left edge is the move, currently 0
                if filled == 3:
                    captures += 1
    return captures

def calculate_risk(move, horizontal, vertical):
    """Calculate the number of 3-edge boxes created by a move."""
    r, c, dirr = move
    risk = 0
    if dirr == 'H':
        # Check box above
        if r > 0:
            i, j = r-1, c
            top = horizontal[i,j] != 0
            left = vertical[i,j] != 0
            right = vertical[i,j+1] != 0
            if top + left + right == 2:  # After move, bottom is added, total=3
                risk += 1
        # Check box below
        if r < 4:
            i, j = r, c
            bottom = horizontal[i+1,j] != 0
            left = vertical[i,j] != 0
            right = vertical[i,j+1] != 0
            if bottom + left + right == 2:  # After move, top is added, total=3
                risk += 1
    else:  # 'V'
        # Check box left
        if c > 0:
            i, j = r, c-1
            top = horizontal[i,j] != 0
            bottom = horizontal[i+1,j] != 0
            left = vertical[i,j] != 0
            if top + bottom + left == 2:  # After move, right is added, total=3
                risk += 1
        # Check box right
        if c < 4:
            i, j = r, c
            top = horizontal[i,j] != 0
            bottom = horizontal[i+1,j] != 0
            right = vertical[i,j+1] != 0
            if top + bottom + right == 2:  # After move, left is added, total=3
                risk += 1
    return risk

def is_border_move(move):
    """Check if a move is on the border of the grid."""
    r, c, dirr = move
    if dirr == 'H':
        return r == 0 or r == 4
    else:
        return c == 0 or c == 4

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    legal_moves = []
    # Generate all horizontal moves
    for row in range(5):
        for col in range(5):
            if horizontal[row, col] == 0:
                legal_moves.append((row, col, 'H'))
    # Generate all vertical moves
    for row in range(5):
        for col in range(5):
            if vertical[row, col] == 0:
                legal_moves.append((row, col, 'V'))
    
    # Check for capturing moves
    capturing_moves = []
    for move in legal_moves:
        caps = calculate_captures(move, horizontal, vertical, capture)
        if caps > 0:
            capturing_moves.append((move, caps))
    
    # Prioritize captures
    if capturing_moves:
        capturing_moves.sort(key=lambda x: x[1], reverse=True)  # Most captures first
        best_move = capturing_moves[0][0]
        return f"{best_move[0]},{best_move[1]},{best_move[2]}"
    
    # Evaluate risk for non-capturing moves
    risks = [calculate_risk(move, horizontal, vertical) for move in legal_moves]
    min_risk = min(risks)
    candidate_moves = [move for move, risk in zip(legal_moves, risks) if risk == min_risk]
    
    # Prefer border moves among candidates
    border_moves = [m for m in candidate_moves if is_border_move(m)]
    final_candidates = border_moves if border_moves else candidate_moves
    
    # Sort candidates by row, column, and direction (H before V)
    final_candidates.sort(key=lambda m: (m[0], m[1], 0 if m[2] == 'H' else 1))
    best_move = final_candidates[0]
    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

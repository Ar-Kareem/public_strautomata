## Strategy

My strategy implements a sophisticated Dots and Boxes AI using the following approach:

1. **Minimax with Alpha-Beta Pruning**: I use a depth-limited minimax search that accounts for the chain-capture rule (if you complete a box, you get another turn).

2. **Move Categorization**:
   - **Safe moves**: Moves that don't give the opponent any box completions (edges that don't create a 3-sided box for any adjacent cell).
   - **Capturing moves**: Moves that complete a box (cells with exactly 3 sides filled).
   - **Dangerous moves**: Moves that create a 3-sided box, giving the opponent a capture opportunity.

3. **Chain/Loop Analysis**: In Dots and Boxes endgame theory, chains and loops are critical. The policy identifies chains of boxes and uses the double-dealing strategy (sacrificing small chains to maintain control).

4. **Greedy First**: Always capture available boxes first (they give free turns).

5. **Avoid giving away boxes**: Prefer moves that don't leave any box with 3 sides filled.

6. **Endgame search**: When few moves remain, do a full minimax search to find optimal play.

import numpy as np
from typing import List, Tuple, Optional

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    def get_legal_moves():
        moves = []
        for r in range(5):
            for c in range(4):
                if horizontal[r, c] == 0:
                    moves.append((r, c, 'H'))
        for r in range(4):
            for c in range(5):
                if vertical[r, c] == 0:
                    moves.append((r, c, 'V'))
        return moves

    def count_sides(h, v, r, c):
        """Count how many sides box (r,c) has filled. Box is valid for r<4, c<4."""
        count = 0
        if h[r, c] != 0: count += 1      # top
        if h[r+1, c] != 0: count += 1    # bottom
        if v[r, c] != 0: count += 1      # left
        if v[r, c+1] != 0: count += 1    # right
        return count

    def boxes_completed_by(h, v, move):
        """Return how many boxes a move completes."""
        r, c, d = move
        completed = 0
        if d == 'H':
            # Horizontal edge at (r,c) is top of box (r, c) and bottom of box (r-1, c)
            if r < 4 and c < 4 and count_sides(h, v, r, c) == 3:
                completed += 1
            if r > 0 and c < 4 and count_sides(h, v, r-1, c) == 3:
                completed += 1
        else:
            # Vertical edge at (r,c) is left of box (r, c) and right of box (r, c-1)
            if r < 4 and c < 4 and count_sides(h, v, r, c) == 3:
                completed += 1
            if r < 4 and c > 0 and count_sides(h, v, r, c-1) == 3:
                completed += 1
        return completed

    def gives_boxes(h, v, move):
        """Return how many boxes this move would set up for opponent (creates 3-sided boxes)."""
        r, c, d = move
        given = 0
        if d == 'H':
            if r < 4 and c < 4 and count_sides(h, v, r, c) == 2:
                given += 1
            if r > 0 and c < 4 and count_sides(h, v, r-1, c) == 2:
                given += 1
        else:
            if r < 4 and c < 4 and count_sides(h, v, r, c) == 2:
                given += 1
            if r < 4 and c > 0 and count_sides(h, v, r, c-1) == 2:
                given += 1
        return given

    def apply_move(h, v, cap, move, player):
        """Apply move, handle chain captures. Returns (h, v, cap, score)."""
        h = h.copy()
        v = v.copy()
        cap = cap.copy()
        r, c, d = move
        if d == 'H':
            h[r, c] = player
        else:
            v[r, c] = player
        
        # Check captures
        score = 0
        captured = True
        while captured:
            captured = False
            # Find all capturing moves (already placed edges that complete boxes)
            # Actually we just check if any unclaimed box is now complete
            for br in range(4):
                for bc in range(4):
                    if cap[br, bc] == 0 and count_sides(h, v, br, bc) == 4:
                        cap[br, bc] = player
                        score += 1
                        captured = True
        return h, v, cap, score

    def minimax(h, v, cap, my_score, opp_score, is_my_turn, depth, alpha, beta):
        """Returns (my_score - opp_score) at the end."""
        moves = []
        for rr in range(5):
            for cc in range(4):
                if h[rr, cc] == 0:
                    moves.append((rr, cc, 'H'))
        for rr in range(4):
            for cc in range(5):
                if v[rr, cc] == 0:
                    moves.append((rr, cc, 'V'))
        
        if not moves or depth <= 0:
            return my_score - opp_score
        
        player = 1 if is_my_turn else -1
        
        # Sort moves: capturing first, then safe, then dangerous
        def move_priority(m):
            caps = boxes_completed_by(h, v, m)
            gives = gives_boxes(h, v, m)
            return (-caps, gives)
        
        moves.sort(key=move_priority)
        
        if is_my_turn:
            best = -100
            for m in moves:
                nh, nv, nc, sc = apply_move(h, v, cap, m, player)
                if sc > 0:
                    # I captured, I go again
                    val = minimax(nh, nv, nc, my_score + sc, opp_score, True, depth - 1, alpha, beta)
                else:
                    val = minimax(nh, nv, nc, my_score, opp_score, False, depth - 1, alpha, beta)
                best = max(best, val)
                alpha = max(alpha, best)
                if beta <= alpha:
                    break
            return best
        else:
            best = 100
            for m in moves:
                nh, nv, nc, sc = apply_move(h, v, cap, m, player)
                if sc > 0:
                    val = minimax(nh, nv, nc, my_score, opp_score + sc, False, depth - 1, alpha, beta)
                else:
                    val = minimax(nh, nv, nc, my_score, opp_score, True, depth - 1, alpha, beta)
                best = min(best, val)
                beta = min(beta, best)
                if beta <= alpha:
                    break
            return best

    legal = get_legal_moves()
    if not legal:
        return "0,0,H"  # shouldn't happen
    
    # Categorize moves
    capturing = [m for m in legal if boxes_completed_by(horizontal, vertical, m) > 0]
    safe = [m for m in legal if boxes_completed_by(horizontal, vertical, m) == 0 and gives_boxes(horizontal, vertical, m) == 0]
    dangerous = [m for m in legal if boxes_completed_by(horizontal, vertical, m) == 0 and gives_boxes(horizontal, vertical, m) > 0]
    
    # Determine search depth based on remaining moves
    n_remaining = len(legal)
    if n_remaining <= 16:
        depth = min(n_remaining, 20)
    elif n_remaining <= 22:
        depth = 10
    else:
        depth = 6

    my_sc = int(np.sum(capture == 1))
    opp_sc = int(np.sum(capture == -1))
    
    best_move = legal[0]
    best_val = -200
    
    # Sort: capturing first, safe second, dangerous last
    sorted_moves = capturing + safe + dangerous
    
    for m in sorted_moves:
        nh, nv, nc, sc = apply_move(horizontal, vertical, capture, m, 1)
        if sc > 0:
            val = minimax(nh, nv, nc, my_sc + sc, opp_sc, True, depth - 1, best_val, 100)
        else:
            val = minimax(nh, nv, nc, my_sc, opp_sc, False, depth - 1, best_val, 100)
        if val > best_val:
            best_val = val
            best_move = m
    
    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

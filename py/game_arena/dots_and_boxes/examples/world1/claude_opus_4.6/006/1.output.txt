## Strategy

My strategy implements a minimax search with alpha-beta pruning, optimized for Dots and Boxes:

1. **Chain-based reasoning**: In Dots and Boxes, the endgame is dominated by chains and loops. The key strategic principle is to avoid opening long chains for your opponent.

2. **Move categorization**:
   - **Capturing moves**: Moves that complete a box (box has 3 sides filled). Always take these first.
   - **Safe moves**: Moves that don't give the opponent any box (neither adjacent box reaches 3 sides after the move).
   - **Risky moves**: Moves that create a box with 3 sides, giving the opponent a capture opportunity.

3. **Minimax with alpha-beta pruning**: I implement a depth-limited search that evaluates positions. The evaluation function considers the score difference and penalizes positions where you've left boxes with 3 sides for the opponent.

4. **Greedy capturing**: When a capture is available, always take it (this is mandatory in optimal play since you get another turn).

5. **Double-deal / hard-hearted handout**: In chain situations, sometimes it's better to sacrifice 2 boxes to maintain control. The search handles this naturally.

import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    def get_legal_moves(h, v):
        moves = []
        for r in range(5):
            for c in range(4):
                if h[r, c] == 0:
                    moves.append((r, c, 'H'))
        for r in range(4):
            for c in range(5):
                if v[r, c] == 0:
                    moves.append((r, c, 'V'))
        return moves

    def count_sides(h, v, r, c):
        count = 0
        if h[r, c] != 0: count += 1      # top
        if h[r+1, c] != 0: count += 1    # bottom
        if v[r, c] != 0: count += 1      # left
        if v[r, c+1] != 0: count += 1    # right
        return count

    def apply_move(h, v, cap, move, player):
        h = h.copy(); v = v.copy(); cap = cap.copy()
        r, c, d = move
        if d == 'H':
            h[r, c] = player
        else:
            v[r, c] = player
        captured = 0
        # Check which boxes are completed
        if d == 'H':
            # Box above: row r-1, col c
            if r > 0 and cap[r-1, c] == 0:
                if count_sides(h, v, r-1, c) == 4:
                    cap[r-1, c] = player
                    captured += 1
            # Box below: row r, col c
            if r < 4 and cap[r, c] == 0:
                if count_sides(h, v, r, c) == 4:
                    cap[r, c] = player
                    captured += 1
        else:  # V
            # Box left: row r, col c-1
            if c > 0 and cap[r, c-1] == 0:
                if count_sides(h, v, r, c-1) == 4:
                    cap[r, c-1] = player
                    captured += 1
            # Box right: row r, col c
            if c < 4 and cap[r, c] == 0:
                if count_sides(h, v, r, c) == 4:
                    cap[r, c] = player
                    captured += 1
        return h, v, cap, captured

    def evaluate(h, v, cap):
        return np.sum(cap)

    def minimax(h, v, cap, player, depth, alpha, beta):

            return evaluate(h, v, cap), None

                else:
                    val, _ = minimax(nh, nv, nc, -1, depth - 1, alpha, beta)
                if val > max_val:
                    max_val = val; best_move = m
                alpha = max(alpha, val)
                if beta <= alpha: break
            return max_val, best_move

                    val, _ = minimax(nh, nv, nc, -1, depth, alpha, beta)
                else:
                    val, _ = minimax(nh, nv, nc, 1, depth - 1, alpha, beta)
                if val < min_val:
                    min_val = val; best_move = m
                beta = min(beta, val)
                if beta <= alpha: break
            return min_val, best_move

    moves = get_legal_moves(horizontal, vertical)

            r, c, d = m

            if d == 'H': h2[r,c] = 1
            else: v2[r,c] = 1
            gives = False
            if d == 'H':

                if r < 4 and count_sides(h2, v2, r, c) == 3: gives = True

                if c < 4 and count_sides(h2, v2, r, c) == 3: gives = True
            if gives:
                risky.append(m)
            else:
                safe.append(m)

    ordered = captures + safe + risky

        return f"{m[0]},{m[1]},{m[2]}"

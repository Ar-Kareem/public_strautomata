

The strategy for this policy is to prioritize capturing boxes when possible, ensuring extra turns and maximizing points. If no captures are available, it selects the move that minimizes the opponent's opportunities to capture boxes by avoiding creating edges that leave exactly three sides of any box filled. This approach balances aggressive scoring with defensive play to outmaneuver opponents.

import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    # Generate all legal moves
    moves = []
    # Horizontal edges
    for row in range(5):
        for col in range(5):
            if horizontal[row, col] == 0:
                moves.append((row, col, 'H'))
    # Vertical edges
    for row in range(5):
        for col in range(5):
            if vertical[row, col] == 0:
                moves.append((row, col, 'V'))
    
    # Check for capturing moves
    capturing_moves = []
    for move in moves:
        r, c, dir = move
        captures = 0
        if dir == 'H':
            # Check top and bottom boxes
            if r > 0:  # top box (r-1, c)
                if (horizontal[r-1, c] != 0 and
                    vertical[r-1, c] != 0 and
                    vertical[r-1, c+1] != 0):
                    captures += 1
            if r < 4:  # bottom box (r, c)
                if (horizontal[r+1, c] != 0 and
                    vertical[r, c] != 0 and
                    vertical[r, c+1] != 0):
                    captures += 1
        else:
            # Vertical edge, check left and right boxes
            if c > 0:  # left box (r, c-1)
                if (horizontal[r, c-1] != 0 and
                    horizontal[r+1, c-1] != 0 and
                    vertical[r, c-1] != 0):
                    captures += 1
            if c < 4:  # right box (r, c)
                if (horizontal[r, c] != 0 and
                    horizontal[r+1, c] != 0 and
                    vertical[r, c+1] != 0):
                    captures += 1
        if captures > 0:
            capturing_moves.append((captures, move))
    
    if capturing_moves:
        # Select move with max captures
        max_cap = max(c[0] for c in capturing_moves)
        best = [m for cap, m in capturing_moves if cap == max_cap]
        # Choose first occurrence
        chosen_move = best[0]
    else:
        # Evaluate danger for each move
        danger_moves = []
        for move in moves:
            r, c, dir = move
            danger = 0
            if dir == 'H':
                # Affects boxes (r-1,c) and (r,c)
                if r > 0:
                    current = (horizontal[r-1, c] != 0) + (vertical[r-1, c] != 0) + (vertical[r-1, c+1] != 0)
                    if current + 1 == 3:
                        danger += 1
                if r < 4:
                    current = (horizontal[r+1, c] != 0) + (vertical[r, c] != 0) + (vertical[r, c+1] != 0)
                    if current + 1 == 3:
                        danger += 1
            else:
                # Affects boxes (r,c-1) and (r,c)
                if c > 0:
                    current = (horizontal[r, c-1] != 0) + (horizontal[r+1, c-1] != 0) + (vertical[r, c-1] != 0)
                    if current + 1 == 3:
                        danger += 1
                if c < 4:
                    current = (horizontal[r, c] != 0) + (horizontal[r+1, c] != 0) + (vertical[r, c+1] != 0)
                    if current + 1 == 3:
                        danger += 1
            danger_moves.append((danger, move))
        # Find minimal danger
        min_danger = min(d[0] for d in danger_moves)
        best = [m for d, m in danger_moves if d == min_danger]
        # Choose first occurrence
        chosen_move = best[0]
    
    r, c, dir = chosen_move
    return f"{r},{c},{dir}"

The strategy for this policy is to always take any available box capture, since capturing grants an extra move, and otherwise avoid creating a third side of any box (which would allow the opponent to score). When no safe moves exist, it chooses the least risky option.

import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    # Helper: count filled sides of a box at (r,c)
    def sides_count(r, c):
        return int(horizontal[r, c] != 0) + int(horizontal[r+1, c] != 0) + \
               int(vertical[r, c] != 0) + int(vertical[r, c+1] != 0)

    # Helper: boxes adjacent to an edge
    def adjacent_boxes(r, c, d):
        boxes = []
        if d == 'H':
            if r > 0:
                boxes.append((r-1, c))
            if r < 4:
                boxes.append((r, c))
        else:  # 'V'
            if c > 0:
                boxes.append((r, c-1))
            if c < 4:
                boxes.append((r, c))
        return boxes

    best_key = None
    best_move = None

    # Evaluate all horizontal edges
    for r in range(5):
        for c in range(4):
            if horizontal[r, c] != 0:
                continue
            captures = 0
            risk = 0
            for (br, bc) in adjacent_boxes(r, c, 'H'):
                count = sides_count(br, bc)
                if count == 3:
                    captures += 1
                elif count == 2:
                    risk += 1
            key = (2, captures, -risk) if captures > 0 else (1, -risk, 0)
            if best_key is None or key > best_key:
                best_key = key
                best_move = (r, c, 'H')

    # Evaluate all vertical edges
    for r in range(4):
        for c in range(5):
            if vertical[r, c] != 0:
                continue
            captures = 0
            risk = 0
            for (br, bc) in adjacent_boxes(r, c, 'V'):
                count = sides_count(br, bc)
                if count == 3:
                    captures += 1
                elif count == 2:
                    risk += 1
            key = (2, captures, -risk) if captures > 0 else (1, -risk, 0)
            if best_key is None or key > best_key:
                best_key = key
                best_move = (r, c, 'V')

    # Fallback (should never happen, but just in case)
    if best_move is None:
        for r in range(5):
            for c in range(4):
                if horizontal[r, c] == 0:
                    return f"{r},{c},H"
        for r in range(4):
            for c in range(5):
                if vertical[r, c] == 0:
                    return f"{r},{c},V"

    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

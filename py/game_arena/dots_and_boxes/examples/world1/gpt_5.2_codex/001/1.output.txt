The strategy for this policy is to always take any move that immediately captures one or more boxes. If no capturing move exists, it will prefer moves that do not create a box with three sides (which would allow the opponent to score). If forced, it chooses the move that creates the fewest such risky boxes. This simple heuristic avoids giving away points while still taking guaranteed scores.

import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    # Helper to count sides of a box
    def box_sides(r, c):
        return (
            (horizontal[r, c] != 0) +
            (horizontal[r+1, c] != 0) +
            (vertical[r, c] != 0) +
            (vertical[r, c+1] != 0)
        )

    moves = []

    # Gather all legal moves
    for r in range(5):
        for c in range(4):
            if horizontal[r, c] == 0:
                moves.append((r, c, 'H'))
    for r in range(4):
        for c in range(5):
            if vertical[r, c] == 0:
                moves.append((r, c, 'V'))

    if not moves:
        return "0,0,H"  # fallback (should never happen if legal)

    best_capture = -1
    best_moves = []

    # Evaluate all moves
    for r, c, d in moves:
        captures = 0
        risk = 0

        # Check affected boxes
        if d == 'H':
            # Box below
            if r < 4:
                sides = box_sides(r, c)
                if sides == 3:
                    captures += 1
                elif sides == 2:
                    risk += 1
            # Box above
            if r > 0:
                sides = box_sides(r-1, c)
                if sides == 3:
                    captures += 1
                elif sides == 2:
                    risk += 1
        else:  # 'V'
            # Box to right
            if c < 4:
                sides = box_sides(r, c)
                if sides == 3:
                    captures += 1
                elif sides == 2:
                    risk += 1
            # Box to left
            if c > 0:
                sides = box_sides(r, c-1)
                if sides == 3:
                    captures += 1
                elif sides == 2:
                    risk += 1

        moves_info = (r, c, d, captures, risk)

        if captures > best_capture:
            best_capture = captures
            best_moves = [moves_info]
        elif captures == best_capture:
            best_moves.append(moves_info)

    # If any capture available, choose among them with least risk
    if best_capture > 0:
        best_moves.sort(key=lambda x: x[4])  # minimal risk
        r, c, d, _, _ = best_moves[0]
        return f"{r},{c},{d}"

    # Otherwise choose move with minimal risk
    best_moves = sorted(best_moves, key=lambda x: x[4])
    r, c, d, _, _ = best_moves[0]
    return f"{r},{c},{d}"

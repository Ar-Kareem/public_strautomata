The strategy for this policy is a standard strong Dots-and-Boxes heuristic:

1. Always take any available box immediately.
2. If multiple scoring moves exist, prefer moves that maximize immediate captures, and then continue greedily through forced captures.
3. If no box can be taken now, avoid creating a 3-sided box for the opponent.
4. Among safe moves, prefer ones that minimize future risk by considering how many boxes become “dangerous” after the move.
5. If every legal move is unsafe, choose the move that gives away the least: prefer moves that create fewer immediately capturable boxes for the opponent.
6. Always return a legal move string, with a final fallback to the first legal edge if needed.

import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    N = 4  # 4x4 boxes, edge arrays are 5x5 with only valid subsets used

    def legal_moves(h, v):
        moves = []
        for r in range(N + 1):
            for c in range(N):
                if h[r, c] == 0:
                    moves.append((r, c, 'H'))
        for r in range(N):
            for c in range(N + 1):
                if v[r, c] == 0:
                    moves.append((r, c, 'V'))
        return moves

    def box_sides(h, v, r, c):
        return (
            1 if h[r, c] != 0 else 0
        ) + (
            1 if h[r + 1, c] != 0 else 0
        ) + (
            1 if v[r, c] != 0 else 0
        ) + (
            1 if v[r, c + 1] != 0 else 0
        )

    def adjacent_boxes(move):
        r, c, d = move
        boxes = []
        if d == 'H':
            if r > 0:
                boxes.append((r - 1, c))
            if r < N:
                boxes.append((r, c))
        else:  # 'V'
            if c > 0:
                boxes.append((r, c - 1))
            if c < N:
                boxes.append((r, c))
        return [(br, bc) for br, bc in boxes if 0 <= br < N and 0 <= bc < N]

    def apply_move(h, v, cap, move, player=1):
        r, c, d = move
        h2 = h.copy()
        v2 = v.copy()
        cap2 = cap.copy()

        if d == 'H':
            h2[r, c] = player
        else:
            v2[r, c] = player

        gained = 0
        for br, bc in adjacent_boxes(move):
            if cap2[br, bc] == 0 and box_sides(h2, v2, br, bc) == 4:
                cap2[br, bc] = player
                gained += 1
        return h2, v2, cap2, gained

    def immediate_gain(h, v, cap, move):
        _, _, _, gained = apply_move(h, v, cap, move, 1)
        return gained

    def count_new_three_sided_boxes(h, v, cap, move):
        # Count uncaptured adjacent boxes that become exactly 3-sided after this move
        before = {}
        for br, bc in adjacent_boxes(move):
            if cap[br, bc] == 0:
                before[(br, bc)] = box_sides(h, v, br, bc)
        h2, v2, cap2, _ = apply_move(h, v, cap, move, 1)
        cnt = 0
        for br, bc in adjacent_boxes(move):
            if cap2[br, bc] == 0 and box_sides(h2, v2, br, bc) == 3:
                if before.get((br, bc), -1) < 3:
                    cnt += 1
        return cnt

    def count_available_boxes_to_take(h, v, cap):
        # Number of boxes currently at 3 sides and uncaptured
        cnt = 0
        for r in range(N):
            for c in range(N):
                if cap[r, c] == 0 and box_sides(h, v, r, c) == 3:
                    cnt += 1
        return cnt

    def greedy_capture_chain(h, v, cap):
        # Simulate greedily taking captures while available; return total number of boxes captured
        total = 0
        h2 = h.copy()
        v2 = v.copy()
        c2 = cap.copy()

        while True:
            moves = legal_moves(h2, v2)
            best = None
            best_gain = 0
            for mv in moves:
                g = immediate_gain(h2, v2, c2, mv)
                if g > best_gain:
                    best_gain = g
                    best = mv
            if best is None or best_gain == 0:
                break
            h2, v2, c2, gained = apply_move(h2, v2, c2, best, 1)
            total += gained
        return total

    moves = legal_moves(horizontal, vertical)
    if not moves:
        return "0,0,H"

    # 1. Prefer immediate captures
    scoring = []
    for mv in moves:
        g = immediate_gain(horizontal, vertical, capture, mv)
        if g > 0:
            h2, v2, c2, gained = apply_move(horizontal, vertical, capture, mv, 1)
            chain = greedy_capture_chain(h2, v2, c2)
            # total includes immediate gained + continued forced captures
            scoring.append((gained + chain, gained, -count_new_three_sided_boxes(horizontal, vertical, capture, mv), mv))

    if scoring:
        scoring.sort(reverse=True)
        mv = scoring[0][3]
        return f"{mv[0]},{mv[1]},{mv[2]}"

    # 2. Safe moves: do not create any new 3-sided box
    safe = []
    unsafe = []
    for mv in moves:
        new3 = count_new_three_sided_boxes(horizontal, vertical, capture, mv)
        h2, v2, c2, _ = apply_move(horizontal, vertical, capture, mv, 1)
        opp_boxes = count_available_boxes_to_take(h2, v2, c2)
        # prefer affecting fewer boxes / less tactical damage
        degree = len(adjacent_boxes(mv))
        score_tuple = (new3, opp_boxes, degree)
        if new3 == 0:
            safe.append((opp_boxes, degree, mv))
        else:
            unsafe.append((new3, opp_boxes, degree, mv))

    if safe:
        # Choose move with minimal future immediate gifts; prefer boundary edges (degree 1)
        safe.sort(key=lambda x: (x[0], x[1]))
        mv = safe[0][2]
        return f"{mv[0]},{mv[1]},{mv[2]}"

    # 3. All moves are unsafe: minimize damage
    unsafe.sort(key=lambda x: (x[0], x[1], x[2]))
    mv = unsafe[0][3]
    return f"{mv[0]},{mv[1]},{mv[2]}"

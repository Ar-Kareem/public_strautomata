

The strategy for this policy is to prioritize moves that complete boxes (priority 1), then avoid moves that create boxes with three sides (priority 2), and finally pick any legal move (priority 0). This approach maximizes scoring opportunities while minimizing the opponent's chances.

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:
    import numpy as np
    moves = []
    for i in range(5):
        for j in range(5):
            if horizontal[i][j] == 0:
                moves.append(('H', i, j))
    for i in range(5):
        for j in range(5):
            if vertical[i][j] == 0:
                moves.append(('V', i, j))
    
    best_priority = 0
    best_move = None
    
    for move in moves:
        move_type, i, j = move
        h_copy = horizontal.copy()
        v_copy = vertical.copy()
        c_copy = capture.copy()
        
        if move_type == 'H':
            h_copy[i][j] = 1
        else:
            v_copy[i][j] = 1
            
        boxes_to_check = []
        if move_type == 'H':
            if i < 4:
                boxes_to_check.append((i, j))
            if i > 0:
                boxes_to_check.append((i-1, j))
        else:
            if j < 4:
                boxes_to_check.append((i, j))
            if j > 0:
                boxes_to_check.append((i, j-1))
                
        completed_boxes = 0
        for (r, c) in boxes_to_check:
            edges = [
                h_copy[r][c],
                h_copy[r+1][c],
                v_copy[r][c],
                v_copy[r][c+1]
            ]
            if all(e == 1 for e in edges):
                completed_boxes += 1
                
        priority = 0
        if completed_boxes > 0:
            priority = 1
        else:
            for (r, c) in boxes_to_check:
                edges = [
                    h_copy[r][c],
                    h_copy[r+1][c],
                    v_copy[r][c],
                    v_copy[r][c+1]
                ]
                count = sum(1 for e in edges if e == 1)
                if count == 3 and any(e == 0 for e in edges):
                    priority = 2
                    break
                    
        if priority > best_priority:
            best_priority = priority
            best_move = (move_type, i, j)
            
    if best_move is None:
        best_move = moves[0]
        
    return f"{best_move[1]},{best_move[2]},{best_move[0]}"



Thestrategy for this policy is to prioritize moves that capture at least one box, then avoid moves that create a box with three sides, and otherwise choose any legal move. This approach maximizes scoring opportunities while minimizing risks to the opponent.

import numpy as np

def policy(horizontal, vertical, capture):
    moves = []
    for r in range(5):
        for c in range(5):
            if horizontal[r, c] == 0:
                moves.append(('H', r, c))
    for r in range(5):
        for c in range(5):
            if vertical[r, c] == 0:
                moves.append(('V', r, c))
    
    if not moves:
        return "0,0,H"
    
    capturing_moves = []
    for move in moves:
        h_temp = horizontal.copy()
        v_temp = vertical.copy()
        if move[0] == 'H':
            h_temp[move[1], move[2]] = 1
        else:
            v_temp[move[1], move[2]] = 1
        
        completed_boxes = 0
        for box_r in range(4):
            for box_c in range(4):
                top = h_temp[box_r, box_c]
                bottom = h_temp[box_r+1, box_c]
                left = v_temp[box_r, box_c]
                right = v_temp[box_r, box_c+1]
                if top == 1 and bottom == 1 and left == 1 and right == 1:
                    completed_boxes += 1
        if completed_boxes > 0:
            capturing_moves.append((move, completed_boxes))
    
    capturing_moves.sort(key=lambda x: x[1], reverse=True)
    if capturing_moves:
        chosen_move = capturing_moves[0][0]
        return f"{chosen_move[1]},{chosen_move[2]},{chosen_move[0]}"
    
    safe_moves = []
    for move in moves:
        h_temp = horizontal.copy()
        v_temp = vertical.copy()
        if move[0] == 'H':
            h_temp[move[1], move[2]] = 1
        else:
            v_temp[move[1], move[2]] = 1
        
        three_sided = False
        for box_r in range(4):
            for box_c in range(4):
                top = h_temp[box_r, box_c]
                bottom = h_temp[box_r+1, box_c]
                left = v_temp[box_r, box_c]
                right = v_temp[box_r, box_c+1]
                count = top + bottom + left + right
                if count == 3:
                    three_sided = True
                    break
            if three_sided:
                break
        if not three_sided:
            safe_moves.append(move)
    
    if safe_moves:
        chosen_move = safe_moves[0]
        return f"{chosen_move[1]},{chosen_move[2]},{chosen_move[0]}"
    
    chosen_move = moves[0]
    return f"{chosen_move[1]},{chosen_move[2]},{chosen_move[0]}"

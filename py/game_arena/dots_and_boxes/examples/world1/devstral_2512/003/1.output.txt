import numpy as np

def policy(horizontal: np.ndarray, vertical: np.ndarray, capture: np.ndarray) -> str:

    for row in range(4):

            top = horizontal[row, col]
            bottom = horizontal[row + 1, col]
            # Check vertical edges of the box
            left = vertical[row, col]
            right = vertical[row, col + 1]

            # Count how many edges are already drawn
            edges_drawn = (top != 0) + (bottom != 0) + (left != 0) + (right != 0)

            # If 3 edges are drawn, the 4th edge will complete the box
            if edges_drawn == 3:
                if top == 0:
                    return f"{row},{col},H"
                elif bottom == 0:
                    return f"{row + 1},{col},H"
                elif left == 0:
                    return f"{row},{col},V"
                elif right == 0:
                    return f"{row},{col + 1},V"

            bottom = horizontal[row + 1, col]
            left = vertical[row, col]
            right = vertical[row, col + 1]

            if edges.count(0) == 2:
                # Check if the two empty edges are adjacent (forming a corner)
                if (top == 0 and left == 0) or (top == 0 and right == 0) or (bottom == 0 and left == 0) or (bottom == 0 and right == 0):

                    # Prefer edges that are part of multiple potential boxes
                    # For simplicity, we'll just pick the first available edge in this case
                    if top == 0:

                legal_moves.append((f"{row},{col},V", count_boxes_for_edge(row, col, 'V', horizontal, vertical)))

        return "0,0,H"

    # Sort moves by how many boxes they're part of (descending)
    legal_moves.sort(key=lambda x: x[1], reverse=True)

def count_boxes_for_edge(row: int, col: int, direction: str, horizontal: np.ndarray, vertical: np.ndarray) -> int:

    if direction == 'H':
        # This horizontal edge is the top edge of the box below it
        if row < 4:
            count += 1
        # This horizontal edge is the bottom edge of the box above it
        if row > 0:
            count += 1
    elif direction == 'V':
        # This vertical edge is the left edge of the box to its right
        if col < 4:
            count += 1
        # This vertical edge is the right edge of the box to its left
        if col > 0:
            count += 1
    return count

The logic for this policy is:
1. First, it checks for any immediate captures (boxes with 3 edges already drawn) and completes them if found.
2. If no immediate captures are available, it looks for double-cross opportunities where it can set up multiple potential captures.
3. If neither of the above is available, it plays a legal move that is part of the most potential boxes, prioritizing edges that could lead to future captures.
4. The `count_boxes_for_edge` helper function calculates how many boxes a particular edge is part of, which helps in prioritizing moves.

This strategy aims to maximize the number of boxes captured while also setting up future opportunities for captures. It should perform well against basic policies and has a good chance of beating other policies in the arena.

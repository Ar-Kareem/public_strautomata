  def misere_nim_move(piles):
      # Count the number of non-zero piles
      n = sum(1 for p in piles if p > 0)
      if n == 0:
          return None
      # Count the number of piles of size 1
      ones = sum(1 for p in piles if p == 1)
      # If the number of piles with size > 1 is 0, then all are 1
      if ones == n:
          # This means all non-zero piles are 1.
          # In misère, if there are even numbers of 1's, then this is a winning position -> take one to leave odd.
          # if odd, we are in losing position -> take one.
          # But in both cases, we take one from a 1-pile.
          return f"{find_first_one(piles)},1"

      # If there is at least one pile >1, then compute nim-sum
      nim = 0
      for p in piles:
          nim ^= p

      if nim != 0:
          # We are in a winning position.
          # Look for a move that sets nim to 0.
          for i in range(len(piles)):
              if piles[i] > 0:
                  option = nim ^ piles[i]
                  if option < piles[i]:
                      take = piles[i] - option
                      # Check if this move would leave exactly one pile > 1.
                      # Make the move and count big piles
                      # We can do: 
                      #   new_piles = piles with i reduced to option.
                      new_piles = piles[:]
                      new_piles[i] = option
                      big_count = sum(1 for p in new_piles if p > 1)
                      if big_count > 1:
                          # Safe: return
                          return f"{i},{take}"
                      elif big_count == 1:
                          # Then we need to make sure that the number of 1's is odd.
                          ones_count = sum(1 for p in new_piles if p == 1)
                          if ones_count % 2 == 1:
                              return f"{i},{take}"
                          else:
                              # This move leaves even number of 1's and one big pile -> losing position.
                              # We need to avoid. So we look for any other move.
                              continue
                      else: # big_count==0, all are 1s
                          # then ones_count == n, but we had at least one big pile before, and we reduced one (>=2) to 0 or 1.
                          # if we reduced to 0: then ones_count = ones_in_original + 0
                          # if we reduced to 1: then ones_count = ones_in_original + 1
                          # We care about ones_count parity
                          ones_count = sum(1 for p in new_piles if p == 1)
                          # and big_count=0, so we are in the all-ones state.
                          # In all-ones state: if ones_count is even, then it's a winning position for the opponent.
                          # We want to leave odd ones_count to put opponent in losing position? 
                          # Actually, in all-ones: the player to move loses if there are odd number of ones.
                          # But we (the current player) are leaving the board, so if we leave even number of ones, then opponent wins.
                          if ones_count % 2 == 1:
                              # This is what we want! leave odd number of ones -> opponent loses.
                              return f"{i},{take}"
                          else:
                              continue
          # If we get here, we didn't find a move that is safe. Then try any move that makes nim=0, even if critical.
          for i in range(len(piles)):
              if piles[i] > 0:
                  option = nim ^ piles[i]
                  if option < piles[i]:
                      take = piles[i] - option
                      return f"{i},{take}"
      else:
          # nim_sum==0, we are in a losing position. We make a move that leaves at least two big piles if possible.
          for i in sorted(range(len(piles)), key=lambda i: -piles[i]):
              if piles[i] > 1:
                  # take 1 from it
                  return f"{i},1"
          # if no big pile (shouldn't happen because we have n>ones and we checked ones==n above)
          for i in range(len(piles)):
              if piles[i] > 0:
                  return f"{i},1"

    # Count the number of piles with exactly 1
    ones = sum(1 for p in piles if p == 1)

        return "0,1"

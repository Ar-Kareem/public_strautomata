

The strategy for this policy is to use the standard Nim strategy for most cases, and adjust when all piles are of size 1. The nim-sum (XOR of all pile sizes) determines the optimal moves. If the nim-sum is non-zero, we make a move to nullify it, forcing the opponent into a losing position. When all piles are 1, we take 1 to leave an odd count, making the opponent take the last object. If the nim-sum is zero (and not all 1s), any legal move is taken, as it's a losing position.

def policy(piles: list[int]) -> str:
    # Check if all non-zero piles are 1
    all_ones = all(p <= 1 for p in piles)
    non_zero_piles = [p for p in piles if p > 0]
    if all_ones and len(non_zero_piles) > 0:
        count = len(non_zero_piles)
        # If even, take 1 to leave odd (opponent forced to take last)
        if count % 2 == 0:
            for i in range(4):
                if piles[i] >= 1:
                    return f"{i},1"
        else:
            # No winning move, take 1 from first available
            for i in range(4):
                if piles[i] >= 1:
                    return f"{i},1"
    else:
        # Compute nim-sum
        xor_sum = piles[0] ^ piles[1] ^ piles[2] ^ piles[3]
        if xor_sum == 0:
            # Losing position, take 1 from first available pile
            for i in range(4):
                if piles[i] > 0:
                    return f"{i},1"
        else:
            # Find a pile to modify to reach nim-sum zero
            for i in range(4):
                candidate = piles[i] ^ xor_sum
                if candidate < piles[i]:
                    take = piles[i] - candidate
                    if take <= 0 or take > piles[i]:
                        continue
                    return f"{i},{take}"
            # Fallback if no move found (unlikely)
            for i in range(4):
                if piles[i] > 0:
                    return f"{i},1"
    # Fallback for zero case (shouldn't happen as game ends)
    return "0,1"

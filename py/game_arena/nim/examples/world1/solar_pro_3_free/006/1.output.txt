The strategy for this policy is to implement optimal play for misère Nim (4 piles), where the player taking the last object loses. It handles:
1. The special misère Nim case where all non-zero piles have size 1 (using parity logic).
2. The general misère Nim case (with at least one pile ≥2) by adapting normal Nim's XOR strategy but avoiding moves that would leave the opponent with an even count of single-object piles.
3. Falls back to safe moves when no winning strategy is available.

def policy(piles: list[int]) -> str:
    # Handle empty state (should not occur but defensively return first pile)
    if all(p == 0 for p in piles):
        return "0,1"
    
    # Special case: all non-zero piles are size 1
    all_non_zero_ones = all(p == 1 for p in piles if p > 0)
    if all_non_zero_ones:
        non_zero_count = sum(1 for p in piles if p > 0)
        if non_zero_count % 2 == 0:
            # Winning position: make count odd
            for i, p in enumerate(piles):
                if p > 0:
                    return f"{i},1"
        else:
            # Losing position: any move works
            for i, p in enumerate(piles):
                if p > 0:
                    return f"{i},1"
    
    # General case: at least one pile ≥2
    nim_sum = 0
    for p in piles:
        nim_sum ^= p
        
    if nim_sum == 0:
        # Losing position: take 1 from first non-zero pile
        for i, p in enumerate(piles):
            if p > 0:
                return f"{i},1"
    else:
        # Find winning move that doesn't leave opponent with all 1s and even count
        for i, p in enumerate(piles):
            if p > nim_sum ^ p:  # Valid move to zero XOR
                take = p - (nim_sum ^ p)
                new_piles = piles.copy()
                new_piles[i] = nim_sum ^ p
                # Check if opponent would win from this state
                has_ge2 = any(q >= 2 for q in new_piles)
                if not has_ge2:
                    non_zero_count = sum(1 for q in new_piles if q > 0)
                    if non_zero_count % 2 == 1:
                        # This is losing for opponent (odd count of 1s)
                        return f"{i},{take}"
                    else:
                        continue  # Skip this move
                return f"{i},{take}"
        
        # Fallback: take 1 from any ≥2 pile to leave odd count of 1s
        for i, p in enumerate(piles):
            if p >= 2:
                take = 1
                new_piles = piles.copy()
                new_piles[i] -= 1
                if all(q == 1 for q in new_piles if q > 0):
                    non_zero_count = sum(1 for q in new_piles if q > 0)
                    if non_zero_count % 2 == 1:
                        return f"{i},{take}"
        
        # Final fallback: take 1 from first non-zero pile
        for i, p in enumerate(piles):
            if p > 0:
                return f"{i},1"

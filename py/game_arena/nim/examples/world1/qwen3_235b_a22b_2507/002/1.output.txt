In normal Nim (last move wins), the winning strategy is to make the XOR (nim-sum) of all pile sizes zero after your move. However, in **misère Nim**, the strategy is almost the same, **except when the game is in a state where all piles are of size 1 or less**.

def policy(piles):
    # Count piles with more than 1
    big_piles = [i for i in range(4) if piles[i] > 1]
    big_count = len(big_piles)

    if big_count == 0:
        # All piles are 0 or 1. Take one from the first non-zero pile.
        for i in range(4):
            if piles[i] > 0:
                return f"{i},1"

    elif big_count == 1:
        # Exactly one pile has more than 1
        i = big_piles[0]
        ones = sum(1 for p in piles if p == 1)
        # We will reduce the big pile to either 1 or 0 to make total ones odd
        if (ones + 1) % 2 == 1:
            # reduce big pile to 1
            take = piles[i] - 1
            return f"{i},{take}"
        else:
            # then ones % 2 == 1 must hold, so reduce to 0
            take = piles[i]  # take all
            return f"{i},{take}"

    else:
        # big_count >= 2: play normal Nim
        nim_sum = piles[0] ^ piles[1] ^ piles[2] ^ piles[3]
        if nim_sum == 0:
            # Losing position: take 1 from the largest pile
            # Find the pile with maximum size
            max_pile = max(piles)
            # pick the first one with max size
            for i in range(4):
                if piles[i] == max_pile:
                    return f"{i},1"
        else:
            # Find a move that makes nim_sum 0
            for i in range(4):
                # What should the new value of pile i be?
                target = piles[i] ^ nim_sum
                if target < piles[i]:
                    # We can reduce piles[i] to target
                    take = piles[i] - target
                    # This move is legal
                    return f"{i},{take}"
            # Should not get here, but fallback
            # Take 1 from first non-empty pile
            for i in range(4):
                if piles[i] > 0:
                    return f"{i},1"

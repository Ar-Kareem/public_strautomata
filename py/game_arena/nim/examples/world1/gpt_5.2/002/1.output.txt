The strategy for this policy is perfect-play **misère Nim** (the player who takes the last object loses). It uses the known optimal solution:
- If all nonzero piles are size 1, the position is winning iff the number of such piles is even; the only possible move is to remove one whole pile (take 1).
- If exactly one pile is larger than 1 and the rest are 1s, play the special misère endgame:  
  - If the number of 1-piles is even, reduce the large pile to 1.  
  - If the number of 1-piles is odd, take the entire large pile.
- Otherwise, play like normal Nim: move to make the XOR (nim-sum) of pile sizes equal to 0.
- If already at XOR=0 (a losing position), make a safe arbitrary legal move (take 1 from the first non-empty pile).

def policy(piles: list[int]) -> str:
    """
    Perfect-play misère Nim for 4 piles.
    Return move as 'pile,take' with 0-based pile index.
    """
    # Find non-empty piles
    nonzero = [i for i, p in enumerate(piles) if p > 0]
    if not nonzero:
        # No legal move exists in terminal state; arena should not call policy here.
        # Return something syntactically valid anyway.
        return "0,1"

    ones = sum(1 for p in piles if p == 1)
    big_indices = [i for i, p in enumerate(piles) if p > 1]
    big_count = len(big_indices)

    # Case 1: all nonzero piles are size 1
    if big_count == 0:
        # Only legal move is to take 1 from some non-empty pile
        i = nonzero[0]
        return f"{i},1"

    # Case 2: exactly one heap > 1 and the rest (nonzero) are 1s: special misère rule
    if big_count == 1 and (ones + big_count) == len(nonzero):
        b = big_indices[0]
        if ones % 2 == 0:
            # Leave an odd number of 1-heaps for opponent: reduce big heap to 1
            take = piles[b] - 1  # piles[b] > 1 so take >= 1
            return f"{b},{take}"
        else:
            # Take whole big heap, leaving an odd number of 1-heaps
            take = piles[b]
            return f"{b},{take}"

    # Case 3: normal Nim move (misère matches normal when not in the special endgame)
    x = 0
    for p in piles:
        x ^= p

    if x == 0:
        # Losing position: make any legal move
        i = nonzero[0]
        return f"{i},1"

    # Find a move that makes nim-sum 0
    for i, p in enumerate(piles):
        if p <= 0:
            continue
        target = p ^ x
        if target < p:
            take = p - target
            # Legal by construction: 1 <= take <= p
            return f"{i},{take}"

    # Fallback (should be unreachable): return any legal move
    i = nonzero[0]
    return f"{i},1"

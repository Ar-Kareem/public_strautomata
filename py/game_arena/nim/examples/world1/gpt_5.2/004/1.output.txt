This policy plays **perfect misère Nim** (4 piles, taking the last object loses):

- If **all nonzero piles are size 1**, it removes 1 from any pile, aiming to leave an **odd** number of 1-piles to the opponent (odd = losing in this endgame).
- Otherwise, let `k` be the number of piles with size `> 1`:
  - If `k >= 2`, it uses the **normal Nim** rule: make the bitwise xor of pile sizes become **0**.
  - If `k == 1`, it applies the classic misère adjustment: reduce the lone big pile to **1** when the number of 1-piles is even, otherwise reduce it to **0**.

<code>
def policy(piles: list[int]) -> str:
    # Misère Nim (last move loses), 4 piles.
    # Return "pile,take" with 0-based pile index.

    if not piles or len(piles) != 4:
        # Fallback: try to be legal if malformed input occurs
        # (Arena should not do this.)
        return "0,0"

    nonzero = [i for i, x in enumerate(piles) if x > 0]
    if not nonzero:
        # Terminal position: no legal move exists. Should not be queried.
        return "0,0"

    # If all nonzero heaps are 1: take 1 from any nonzero heap.
    # Winning iff count of nonzero heaps is even; but when losing, any move is equivalent.
    if all(piles[i] == 1 for i in nonzero):
        i = nonzero[0]
        return f"{i},1"

    ones = sum(1 for x in piles if x == 1)
    big_indices = [i for i, x in enumerate(piles) if x > 1]

    # Special misère case: exactly one heap > 1
    if len(big_indices) == 1:
        i = big_indices[0]
        heap = piles[i]
        if ones % 2 == 0:
            # Leave a single 1 in the big heap
            take = heap - 1
        else:
            # Remove the big heap entirely
            take = heap
        # Ensure legality
        if take < 1:
            take = 1
        if take > piles[i]:
            take = piles[i]
        return f"{i},{take}"

    # Otherwise play normal Nim: move to xor == 0 if possible
    x = 0
    for h in piles:
        x ^= h

    if x == 0:
        # Losing position under perfect play; choose a simple legal move
        i = nonzero[0]
        return f"{i},1"

    for i, h in enumerate(piles):
        target = h ^ x
        if target < h:
            take = h - target
            # Safety clamp (shouldn't be needed)
            if take < 1:
                take = 1
            if take > h:
                take = h
            return f"{i},{take}"

    # Fallback (should never happen): take 1 from first nonzero
    i = nonzero[0]
    return f"{i},1"

The strategy for this policy is optimal misère Nim play (taking the last object loses). It uses these rules:  
1) If all nonzero piles are size 1, take 1 from any pile (the position is winning iff the number of 1-heaps is even).  
2) If exactly one pile is bigger than 1, reduce that pile to 1 when the number of 1-heaps is even, otherwise take it to 0 (this leaves the opponent an odd number of 1-heaps).  
3) Otherwise (two or more piles bigger than 1), play like normal Nim: move to make the XOR of heap sizes equal 0; if already 0, make any legal move.

def policy(piles: list[int]) -> str:
    # Misère Nim (last move loses) for exactly 4 piles.
    # Returns a legal move as "pile,take".
    if len(piles) != 4:
        # Fallback: find any legal move if malformed input
        for i, h in enumerate(piles):
            if h > 0:
                return f"{i},1"
        return "0,1"  # should never be reached in valid games

    # Identify heap categories
    ones = sum(1 for h in piles if h == 1)
    big_indices = [i for i, h in enumerate(piles) if h > 1]
    big = len(big_indices)

    # Case 1: all heaps are 0/1
    if big == 0:
        # Take 1 from any nonzero heap.
        for i, h in enumerate(piles):
            if h > 0:
                return f"{i},1"
        # Terminal state safeguard (should not be called)
        return "0,1"

    # Case 2: exactly one heap > 1 (misère adjustment)
    if big == 1:
        i = big_indices[0]
        h = piles[i]
        # Want to leave opponent an odd count of 1-heaps.
        # If ones is even, reduce big heap to 1 -> ones+1 is odd.
        # If ones is odd, remove big heap entirely -> ones is odd.
        if ones % 2 == 0:
            target = 1
        else:
            target = 0
        take = h - target
        # Ensure legality
        if take < 1:
            take = 1
        if take > h:
            take = h
        return f"{i},{take}"

    # Case 3: at least two heaps > 1 => normal Nim strategy
    x = 0
    for h in piles:
        x ^= h

    if x == 0:
        # Losing position under perfect play: make any legal move.
        for i, h in enumerate(piles):
            if h > 0:
                return f"{i},1"
        return "0,1"

    # Find a move that makes XOR zero: reduce heap i to (h ^ x) < h
    for i, h in enumerate(piles):
        target = h ^ x
        if target < h:
            take = h - target
            if 1 <= take <= h:
                return f"{i},{take}"

    # Fallback (should never happen): any legal move
    for i, h in enumerate(piles):
        if h > 0:
            return f"{i},1"
    return "0,1"

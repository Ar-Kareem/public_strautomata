The strategy for this policy is a fast tactical Hex player built around four priorities:

1. **Always return a legal move** by maintaining the set of empty cells and falling back safely.
2. **Take immediate wins and block immediate losses** by checking whether any move completes a full connection for either side.
3. **Play strong connection-building Hex moves** using a shortest-path evaluation:
   - For our color, estimate how many empty cells are needed to connect our two sides.
   - For the opponent, estimate the same and prefer moves that improve our path while worsening theirs.
4. **Use Hex structure heuristics**:
   - prefer central cells,
   - prefer cells adjacent to our stones,
   - prefer cells that create “bridging” style connectivity by linking nearby groups,
   - avoid purely random play except as a final tiebreak.

This gives a competent and efficient policy within the time limit.

from heapq import heappush, heappop

N = 11

# Hex neighbors for the board convention in the prompt:
# neighbors are all 8 surrounding cells except (i-1,j-1) and (i+1,j+1)
DIRS = [(-1, 0), (-1, 1), (0, -1), (0, 1), (1, -1), (1, 0)]


def in_bounds(r, c):
    return 0 <= r < N and 0 <= c < N


def neighbors(r, c):
    for dr, dc in DIRS:
        nr, nc = r + dr, c + dc
        if in_bounds(nr, nc):
            yield nr, nc


def has_connection(stones, color):
    """Check if stones of given color already connect their target sides."""
    s = set(stones)
    if not s:
        return False

    stack = []
    seen = set()

    if color == 'b':  # top to bottom
        for c in range(N):
            if (0, c) in s:
                stack.append((0, c))
                seen.add((0, c))
        target_row = N - 1
        while stack:
            r, c = stack.pop()
            if r == target_row:
                return True
            for nr, nc in neighbors(r, c):
                if (nr, nc) in s and (nr, nc) not in seen:
                    seen.add((nr, nc))
                    stack.append((nr, nc))
    else:  # 'w' left to right
        for r in range(N):
            if (r, 0) in s:
                stack.append((r, 0))
                seen.add((r, 0))
        target_col = N - 1
        while stack:
            r, c = stack.pop()
            if c == target_col:
                return True
            for nr, nc in neighbors(r, c):
                if (nr, nc) in s and (nr, nc) not in seen:
                    seen.add((nr, nc))
                    stack.append((nr, nc))
    return False


def shortest_connection_cost(me_set, opp_set, color):
    """
    Weighted shortest path from one side to the other.
    Cost:
      own stone = 0
      empty = 1
      opponent = blocked
    Returns minimal number of empty cells needed to complete a connection.
    """
    INF = 10**9
    dist = {}
    pq = []

    if color == 'b':
        # start from top edge, aim for bottom edge
        for c in range(N):
            cell = (0, c)
            if cell in opp_set:
                continue
            cost = 0 if cell in me_set else 1
            dist[cell] = cost
            heappush(pq, (cost, cell))
        while pq:
            d, (r, c) = heappop(pq)
            if d != dist[(r, c)]:
                continue
            if r == N - 1:
                return d
            for nr, nc in neighbors(r, c):
                nxt = (nr, nc)
                if nxt in opp_set:
                    continue
                w = 0 if nxt in me_set else 1
                nd = d + w
                if nd < dist.get(nxt, INF):
                    dist[nxt] = nd
                    heappush(pq, (nd, nxt))
    else:
        # start from left edge, aim for right edge
        for r in range(N):
            cell = (r, 0)
            if cell in opp_set:
                continue
            cost = 0 if cell in me_set else 1
            dist[cell] = cost
            heappush(pq, (cost, cell))
        while pq:
            d, (r, c) = heappop(pq)
            if d != dist[(r, c)]:
                continue
            if c == N - 1:
                return d
            for nr, nc in neighbors(r, c):
                nxt = (nr, nc)
                if nxt in opp_set:
                    continue
                w = 0 if nxt in me_set else 1
                nd = d + w
                if nd < dist.get(nxt, INF):
                    dist[nxt] = nd
                    heappush(pq, (nd, nxt))

    return INF


def components(stones):
    """Return component id map and list of component sizes."""
    s = set(stones)
    comp_id = {}
    sizes = []
    cid = 0
    for cell in s:
        if cell in comp_id:
            continue
        stack = [cell]
        comp_id[cell] = cid
        size = 0
        while stack:
            r, c = stack.pop()
            size += 1
            for nr, nc in neighbors(r, c):
                if (nr, nc) in s and (nr, nc) not in comp_id:
                    comp_id[(nr, nc)] = cid
                    stack.append((nr, nc))
        sizes.append(size)
        cid += 1
    return comp_id, sizes


def move_features(move, me_set, opp_set, color, my_comp, my_sizes):
    r, c = move
    center = 5
    centrality = -((r - center) ** 2 + (c - center) ** 2)

    my_adj = 0
    opp_adj = 0
    touch_sides = 0
    if color == 'b':
        if r == 0 or r == N - 1:
            touch_sides += 1
    else:
        if c == 0 or c == N - 1:
            touch_sides += 1

    comps_touched = set()
    bridge_score = 0

    for nr, nc in neighbors(r, c):
        cell = (nr, nc)
        if cell in me_set:
            my_adj += 1
            cid = my_comp.get(cell)
            if cid is not None:
                comps_touched.add(cid)
        elif cell in opp_set:
            opp_adj += 1

    if len(comps_touched) >= 2:
        # connecting multiple friendly groups is valuable
        bridge_score += 6 + 2 * (len(comps_touched) - 2)

    if len(comps_touched) == 1:
        cid = next(iter(comps_touched))
        bridge_score += min(my_sizes[cid], 6)

    # soft directional preference
    if color == 'b':
        directional = -abs(c - center)
    else:
        directional = -abs(r - center)

    return (
        3 * my_adj
        - 1 * opp_adj
        + 2 * touch_sides
        + bridge_score
        + 0.15 * centrality
        + 0.3 * directional
    )


def policy(me, opp, color):
    me_set = set(me)
    opp_set = set(opp)

    empties = [(r, c) for r in range(N) for c in range(N)
               if (r, c) not in me_set and (r, c) not in opp_set]

    # Safety fallback: should never fail
    if not empties:
        return (0, 0)

    # Strong opening: center if available
    if len(me) == 0 and len(opp) == 0 and (5, 5) in empties:
        return (5, 5)

    # Immediate win
    for mv in empties:
        if has_connection(me_set | {mv}, color):
            return mv

    # Immediate block of opponent win
    opp_color = 'w' if color == 'b' else 'b'
    for mv in empties:
        if has_connection(opp_set | {mv}, opp_color):
            return mv

    base_my = shortest_connection_cost(me_set, opp_set, color)
    base_opp = shortest_connection_cost(opp_set, me_set, opp_color)

    my_comp, my_sizes = components(me_set)

    best_move = empties[0]
    best_score = -10**18

    for mv in empties:
        new_me = me_set | {mv}

        my_cost = shortest_connection_cost(new_me, opp_set, color)
        opp_cost = shortest_connection_cost(opp_set, new_me, opp_color)

        # Main evaluation:
        # improve our path, worsen opponent's path
        score = 0.0
        score += 12.0 * (base_my - my_cost)
        score += 10.0 * (opp_cost - base_opp)

        # If this move creates a very short path, prioritize it
        score += -2.5 * my_cost
        score += 1.5 * opp_cost

        # Local structural features
        score += move_features(mv, me_set, opp_set, color, my_comp, my_sizes)

        # Slight preference for cells participating in tactical contact
        r, c = mv
        contested = 0
        for nr, nc in neighbors(r, c):
            cell = (nr, nc)
            if cell in me_set or cell in opp_set:
                contested += 1
        score += 0.25 * contested

        if score > best_score:
            best_score = score
            best_move = mv

    # Always legal
    return best_move

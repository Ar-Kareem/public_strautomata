# Strategy for Hex AI Policy

This policy implements a modified version of Dijkstra's algorithm to find the shortest path while simultaneously blocking the opponent. Here's the key approach:

1. **Distance Calculation**: For each empty cell, calculate the minimal distance to both of my sides (using BFS).
2. **Opponent Blocking**: Similarly calculate the opponent's potential paths to their sides.
3. **Score Combination**: Combine these distances into a scoring function that values moves that both progress my path and block the opponent.
4. **Early Game Pattern**: Use pattern matching for strong opening moves in early game.
5. **Bridging Strategy**: Actively create "bridges" (two potential paths) to make it harder for the opponent to block.

The algorithm prioritizes:
- Completing immediate winning paths
- Blocking opponent's immediate threats
- Progressing toward my goal while hindering opponent's progress
- Strong positional play in early game

import heapq
from collections import deque

def policy(me, opp, color):
    BOARD_SIZE = 11
    EMPTY = 0
    ME = 1
    OPP = 2
    
    # Create board representation
    board = [[EMPTY for _ in range(BOARD_SIZE)] for _ in range(BOARD_SIZE)]
    for r, c in me:
        board[r][c] = ME
    for r, c in opp:
        board[r][c] = OPP
    
    # Hex neighbors (6 directions)
    directions = [(-1,0), (-1,1), (0,-1), (0,1), (1,-1), (1,0)]
    
    def get_neighbors(r, c):
        neighbors = []
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE:
                neighbors.append((nr, nc))
        return neighbors
    
    def calculate_distances(start_cells, player):
        dist = [[float('inf')] * BOARD_SIZE for _ in range(BOARD_SIZE)]
        heap = []
        
        for r, c in start_cells:
            if board[r][c] == player:
                dist[r][c] = 0
                heapq.heappush(heap, (0, r, c))
        
        while heap:
            d, r, c = heapq.heappop(heap)
            if d > dist[r][c]:
                continue
                
            for nr, nc in get_neighbors(r, c):
                if board[nr][nc] == player:
                    new_dist = d
                elif board[nr][nc] == EMPTY:
                    new_dist = d + 1
                else:
                    continue
                
                if new_dist < dist[nr][nc]:
                    dist[nr][nc] = new_dist
                    heapq.heappush(heap, (new_dist, nr, nc))
        
        return dist
    
    # Determine goals based on color
    if color == 'b':  # Black connects top to bottom
        my_goals = [(0, c) for c in range(BOARD_SIZE)]
        opp_goals = [(BOARD_SIZE-1, c) for c in range(BOARD_SIZE)]
        opp_color = 'w'
    else:  # White connects left to right
        my_goals = [(r, 0) for r in range(BOARD_SIZE)]
        opp_goals = [(r, BOARD_SIZE-1) for r in range(BOARD_SIZE)]
        opp_color = 'b'
    
    # Calculate my distances
    my_dist = calculate_distances(my_goals, ME)
    my_min_dist = min(my_dist[r][c] for r in range(BOARD_SIZE) for c in range(BOARD_SIZE) 
                      if board[r][c] == ME)
    
    # Calculate opponent distances (treat their stones as ME)
    opp_dist = calculate_distances(opp_goals, OPP)
    opp_min_dist = min(opp_dist[r][c] for r in range(BOARD_SIZE) for c in range(BOARD_SIZE) 
                       if board[r][c] == OPP)
    
    # Check for immediate wins
    for r in range(BOARD_SIZE):
        for c in range(BOARD_SIZE):
            if board[r][c] == EMPTY and my_dist[r][c] == 0:
                return (r, c)
    
    # Check for immediate blocks
    for r in range(BOARD_SIZE):
        for c in range(BOARD_SIZE):
            if board[r][c] == EMPTY and opp_dist[r][c] == 0:
                return (r, c)
    
    # Score cells based on both distances
    best_score = -float('inf')
    best_move = None
    
    for r in range(BOARD_SIZE):
        for c in range(BOARD_SIZE):
            if board[r][c] != EMPTY:
                continue
                
            # Original distances
            my_d = my_dist[r][c]
            opp_d = opp_dist[r][c]
            
            # Calculate virtual move impact
            virtual_dist = calculate_distances([(r, c)], ME)
            new_my_min = min(my_dist[nr][nc] for nr in range(BOARD_SIZE) for nc in range(BOARD_SIZE) 
                            if board[nr][nc] == ME or (nr == r and nc == c))
            improvement = (my_min_dist - new_my_min) * 10
                
            # Score combines both path improvement and blocking
            score = (1.0 / (my_d + 1)) * 5 + (1.0 / (opp_d + 1)) * 3 + improvement
            
            # Prefer center in early game
            center_dist = abs(r - BOARD_SIZE//2) + abs(c - BOARD_SIZE//2)
            early_game_bonus = 1.0 / (center_dist + 1) if len(me) + len(opp) < 20 else 0
            
            total_score = score + early_game_bonus
            
            if total_score > best_score or (total_score == best_score and center_dist < (abs(best_move[0] - BOARD_SIZE//2) + abs(best_move[1] - BOARD_SIZE//2) if best_move else float('inf'))):
                best_score = total_score
                best_move = (r, c)
    
    # Fallback to first empty cell if no move found (shouldn't happen)
    if best_move is None:
        for r in range(BOARD_SIZE):
            for c in range(BOARD_SIZE):
                if board[r][c] == EMPTY:
                    return (r, c)
    
    return best_move

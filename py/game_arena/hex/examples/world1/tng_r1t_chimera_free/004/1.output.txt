

The strategy for this policy is to prioritize immediate winning moves, block opponent winning moves, and use a heuristic to select moves that strengthen connectivity while disrupting the opponent. The heuristic considers adjacency to own stones, proximity to opponent stones, central position, and distance to relevant edges.

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[int, int]:
    def neighbors(i, j):
        nb = []
        if i > 0:
            nb.append((i-1, j))
        if i > 0 and j < 10:
            nb.append((i-1, j+1))
        if j > 0:
            nb.append((i, j-1))
        if j < 10:
            nb.append((i, j+1))
        if i < 10 and j > 0:
            nb.append((i+1, j-1))
        if i < 10:
            nb.append((i+1, j))
        return nb

    def is_winning(stones, player_color):
        stones_set = set(stones)
        visited = set()
        if player_color == 'b':
            queue = [ (i,j) for (i,j) in stones if i == 0 ]
            target = lambda i,j: i == 10
        else:
            queue = [ (i,j) for (i,j) in stones if j == 0 ]
            target = lambda i,j: j == 10
        for s in queue:
            visited.add(s)
        while queue:
            i, j = queue.pop(0)
            if target(i, j):
                return True
            for ni, nj in neighbors(i, j):
                if (ni, nj) in stones_set and (ni, nj) not in visited:
                    visited.add((ni, nj))
                    queue.append((ni, nj))
        return False

    empty = []
    for i in range(11):
        for j in range(11):
            if (i,j) not in me and (i,j) not in opp:
                empty.append((i,j))

    # Check for immediate win
    for cell in empty:
        if is_winning(me + [cell], color):
            return cell

    # Check for opponent's win to block
    opponent_color = 'w' if color == 'b' else 'b'
    for cell in empty:
        if is_winning(opp + [cell], opponent_color):
            return cell

    # No immediate win/block, compute best heuristic move
    def compute_score(cell):
        i, j = cell
        adj_me = 0
        adj_opp = 0
        for ni, nj in neighbors(i, j):
            if (ni, nj) in me:
                adj_me += 1
            elif (ni, nj) in opp:
                adj_opp += 1
        dist_center = abs(i-5) + abs(j-5)
        if color == 'b':
            edge_dist = min(i, 10 - i)
        else:
            edge_dist = min(j, 10 - j)
        score = adj_me * 4 - adj_opp * 3 + (12 - dist_center) + (12 - edge_dist)
        return score

    best_score = -float('inf')
    best_cell = empty[0]
    for cell in empty:
        current_score = compute_score(cell)
        if current_score > best_score:
            best_score = current_score
            best_cell = cell
    return best_cell

The strategy for this policy is a greedy heuristic-based approach tailored for Hex on an 11x11 board. It prioritizes moves that strengthen the player's position by expanding connections (favoring cells adjacent to many owned stones) while discouraging moves that help the opponent or are far from the board's center. Specifically:

- Valid empty cells (not occupied by either player and within bounds) are evaluated using a score: `(2 * number of player neighbor cells) - (1 * number of opponent neighbor cells) - (Manhattan distance from center / 5.0)`.
- Neighbor cells are the standard 6 possible for Hex (excluding the two diagonal pairs).
- The move with the highest score is selected as the default.
- Before returning, the policy checks if placing the selected move immediately wins the game for the player (by connecting the required sides for the player's color) or blocks an opponent win—if so, it plays that move instead.
- If no winning or blocking move is found, it defaults to the greedy selected move. This ensures a strong, legal, and quick (<1s execution) decision that balances aggressiveness and defense without deep search.

import math

def get_neighbors(r, c):
    # Hex neighbors: standard 6, based on grid rules (exclude (r-1,c-1) and (r+1,c+1))
    candidates = [
        (r, c-1), (r, c+1),
        (r-1, c), (r+1, c),
        (r-1, c+1), (r+1, c-1)
    ]
    return [(nr, nc) for nr, nc in candidates if 0 <= nr < 11 and 0 <= nc < 11]

def has_path(board, color):
    # board is a set of player positions
    if color == 'b':
        # Black: rows 0 to 10
        start_row, end_row = 0, 10
        starts = [(r, c) for r, c in board if r == start_row]
        targets = set((r, c) for r, c in board if r == end_row)
    elif color == 'w':
        # White: cols 0 to 10
        start_col, end_col = 0, 10
        starts = [(r, c) for r, c in board if c == start_col]
        targets = set((r, c) for r, c in board if c == end_col)
    
    if not starts or not targets:
        return False
    
    visited = set()
    queue = starts[:]
    while queue:
        curr = queue.pop(0)
        if curr in targets:
            return True
        if curr in visited:
            continue
        visited.add(curr)
        for nr, nc in get_neighbors(*curr):
            if (nr, nc) in board and (nr, nc) not in visited:
                queue.append((nr, nc))
    return False

def policy(me, opp, color):
    me_set = set(me)
    opp_set = set(opp)
    occupied = me_set | opp_set
    empty = [(r, c) for r in range(11) for c in range(11) if (r, c) not in occupied]
    
    if not empty:
        # Theoretically impossible, but safety
        return (0, 0)
    
    best_score = -math.inf
    best_move = empty[0]
    center_r, center_c = 5, 5
    
    for r, c in empty:
        my_neighbors = sum(1 for nr, nc in get_neighbors(r, c) if (nr, nc) in me_set)
        opp_neighbors = sum(1 for nr, nc in get_neighbors(r, c) if (nr, nc) in opp_set)
        dist = abs(r - center_r) + abs(c - center_c)
        score = 2 * my_neighbors - opp_neighbors - (dist / 5.0)
        
        if score > best_score:

            best_move = (r, c)

    test_me_plus = me_set | {best_move}

The strategy for this policy is a greedy approach: First, it checks all empty cells to find if any move immediately wins the game for the current player by connecting the respective sides (using a BFS-based connectivity check for efficiency). If such a winning move exists, it returns the first one encountered. Second, if no immediate win is available, it checks all empty cells to identify any that would allow the opponent to win on their next turn (again using the same connectivity check); if any such blocking moves are found, it returns the first one to prevent the opponent from winning immediately. Finally, if neither condition applies, it selects a random valid move. This strategy balances offense (immediate wins) and defense (blocking opponent wins) while being computationally efficient for the 1-second time limit, as BFS on an 11x11 board is fast, and random selection handles mid-game expansion without overcomplicating the code. This should perform well against simpler policies that don't handle wins/blocking optimally.

import random
from collections import deque

def neighbors(i: int, j: int) -> list[tuple[int, int]]:
    ns = []
    for di, dj in [(-1, 0), (-1, 1), (0, -1), (0, 1), (1, -1), (1, 0)]:
        ni, nj = i + di, j + dj
        if 0 <= ni < 11 and 0 <= nj < 11:
            ns.append((ni, nj))
    return ns

def check_win(stones: set[tuple[int, int]], color: str) -> bool:
    if color == 'b':
        start_row = 0
        end_row = 10
        sources = [s for s in stones if s[0] == start_row]
        targets = set(s for s in stones if s[0] == end_row)
    else:  # color == 'w'
        start_col = 0
        end_col = 10
        sources = [s for s in stones if s[1] == start_col]
        targets = set(s for s in stones if s[1] == end_col)
    
    if not sources or not targets:
        return False
    
    visited = set()
    q = deque(sources)
    visited.update(sources)
    
    while q:
        curr = q.popleft()
        if curr in targets:
            return True
        for neigh in neighbors(*curr):
            if neigh in stones and neigh not in visited:
                q.append(neigh)
                visited.add(neigh)
    return False

def policy(me: list[tuple[int, int]], opp: list[tuple[int, int]], color: str) -> tuple[int, int]:
    me_set = set(me)
    opp_set = set(opp)
    all_stones = me_set | opp_set
    
    # Get empty cells
    empty = []
    for i in range(11):
        for j in range(11):
            if (i, j) not in all_stones:
                empty.append((i, j))
    
    # Check if I can win
    for move in empty:
        new_me = me_set | {move}
        if check_win(new_me, color):
            return move
    
    # Check if I need to block opponent win
    opp_color = 'w' if color == 'b' else 'b'
    for move in empty:
        new_opp = opp_set | {move}
        if check_win(new_opp, opp_color):
            # Block this move by placing here
            return move
    
    # Otherwise, random move
    if empty:
        return random.choice(empty)
    else:
        # Should not happen, but just in case
        raise ValueError("No empty cells, but game should not reach here")

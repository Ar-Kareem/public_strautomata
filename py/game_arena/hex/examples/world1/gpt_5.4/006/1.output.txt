The strategy for this policy is a fast tactical Hex player with these priorities:

1. **Always return a legal move.**
2. **Play immediate wins** if available.
3. **Block opponent immediate wins** if necessary.
4. **Score all legal moves** using Hex-specific heuristics:
   - shortest-path style connection potential to our target sides,
   - centrality,
   - adjacency to our own stones,
   - blocking / interfering with opponent structure,
   - bridge-style support patterns,
   - creating multiple nearby connection options.
5. **Prefer robust opening moves** near the center when the board is empty or sparse.
6. Use only lightweight computation so it stays comfortably within time limits.

It models Hex connectivity with the correct 6-neighbor relation:
`(r, c-1), (r, c+1), (r-1, c), (r-1, c+1), (r+1, c), (r+1, c-1)`.

The path evaluation uses a Dijkstra-style distance to each pair of goal sides, where:
- our stones cost `0`,
- empty cells cost `1`,
- opponent stones are blocked.

A move is then judged by how much it improves our connection while also potentially worsening the opponent’s. This gives a strong general-purpose policy without expensive search.

from heapq import heappush, heappop

N = 11

def neighbors(r, c):
    cand = (
        (r, c - 1), (r, c + 1),
        (r - 1, c), (r - 1, c + 1),
        (r + 1, c), (r + 1, c - 1),
    )
    for rr, cc in cand:
        if 0 <= rr < N and 0 <= cc < N:
            yield rr, cc

def legal_moves(me_set, opp_set):
    occ = me_set | opp_set
    for r in range(N):
        for c in range(N):
            if (r, c) not in occ:
                yield (r, c)

def side_distance(my_set, opp_set, color):
    INF = 10**9
    dist = [[INF] * N for _ in range(N)]
    pq = []

    def cell_cost(pos):
        if pos in opp_set:
            return None
        if pos in my_set:
            return 0
        return 1

    if color == 'b':
        for c in range(N):
            pos = (0, c)
            cost = cell_cost(pos)
            if cost is not None and cost < dist[0][c]:
                dist[0][c] = cost
                heappush(pq, (cost, 0, c))
        target_check = lambda r, c: r == N - 1
    else:
        for r in range(N):
            pos = (r, 0)
            cost = cell_cost(pos)
            if cost is not None and cost < dist[r][0]:
                dist[r][0] = cost
                heappush(pq, (cost, r, 0))
        target_check = lambda r, c: c == N - 1

    best = INF
    while pq:
        d, r, c = heappop(pq)
        if d != dist[r][c]:
            continue
        if target_check(r, c):
            best = d
            break
        for rr, cc in neighbors(r, c):
            pos = (rr, cc)
            cost = cell_cost(pos)
            if cost is None:
                continue
            nd = d + cost
            if nd < dist[rr][cc]:
                dist[rr][cc] = nd
                heappush(pq, (nd, rr, cc))
    return best

def is_win(my_set, opp_set, color):
    # BFS/DFS through own stones only
    stack = []
    seen = set()
    if color == 'b':
        for c in range(N):
            if (0, c) in my_set:
                stack.append((0, c))
                seen.add((0, c))
        target = lambda r, c: r == N - 1
    else:
        for r in range(N):
            if (r, 0) in my_set:
                stack.append((r, 0))
                seen.add((r, 0))
        target = lambda r, c: c == N - 1

    while stack:
        r, c = stack.pop()
        if target(r, c):
            return True
        for rr, cc in neighbors(r, c):
            if (rr, cc) in my_set and (rr, cc) not in seen:
                seen.add((rr, cc))
                stack.append((rr, cc))
    return False

def center_score(r, c):
    cr = (N - 1) / 2.0
    cc = (N - 1) / 2.0
    return -((r - cr) ** 2 + (c - cc) ** 2)

def progress_score(r, c, color):
    # Encourage movement in the correct axis.
    if color == 'b':
        return -abs(r - (N - 1) / 2.0) * 0.3
    else:
        return -abs(c - (N - 1) / 2.0) * 0.3

def local_shape_score(move, me_set, opp_set, color):
    r, c = move
    score = 0.0

    my_n = 0
    opp_n = 0
    empty_n = 0
    for nb in neighbors(r, c):
        if nb in me_set:
            my_n += 1
        elif nb in opp_set:
            opp_n += 1
        else:
            empty_n += 1

    score += 2.4 * my_n
    score += 0.9 * opp_n
    score += 0.15 * empty_n

    # Two-step supportive patterns
    nb_list = list(neighbors(r, c))
    my_two = 0
    opp_two = 0
    for nr, nc in nb_list:
        for rr, cc in neighbors(nr, nc):
            if (rr, cc) == (r, c):
                continue
            if (rr, cc) in me_set:
                my_two += 1
            elif (rr, cc) in opp_set:
                opp_two += 1
    score += 0.12 * my_two
    score += 0.05 * opp_two

    # Reward connection across our main axis
    if color == 'b':
        above = 0
        below = 0
        for nr, nc in neighbors(r, c):
            if (nr, nc) in me_set:
                if nr < r:
                    above += 1
                elif nr > r:
                    below += 1
        if above and below:
            score += 3.0
    else:
        left = 0
        right = 0
        for nr, nc in neighbors(r, c):
            if (nr, nc) in me_set:
                if nc < c:
                    left += 1
                elif nc > c:
                    right += 1
        if left and right:
            score += 3.0

    return score

def opening_book_like(me_set, opp_set, color):
    occ = me_set | opp_set
    empties = N * N - len(occ)
    if empties == N * N:
        return (5, 5)
    if len(occ) <= 2:
        prefs = [(5, 5), (5, 4), (4, 5), (5, 6), (6, 5), (4, 6), (6, 4)]
        for mv in prefs:
            if mv not in occ:
                return mv
    return None

def evaluate_move(move, me_set, opp_set, color, base_my_dist, base_opp_dist):
    r, c = move
    my2 = set(me_set)
    my2.add(move)

    # Immediate win priority handled strongly here too.
    if is_win(my2, opp_set, color):
        return 10**8 + center_score(r, c)

    opp_color = 'w' if color == 'b' else 'b'

    my_dist = side_distance(my2, opp_set, color)
    opp_dist = side_distance(opp_set, my2, opp_color)

    score = 0.0

    # Improve our path, worsen theirs.
    score += 35.0 * (base_my_dist - my_dist)
    score += 18.0 * (opp_dist - base_opp_dist)

    score += 0.45 * center_score(r, c)
    score += progress_score(r, c, color)
    score += local_shape_score(move, me_set, opp_set, color)

    # Mild edge preference/avoidance: useful but don't overdo it.
    if color == 'b':
        if r == 0 or r == N - 1:
            score += 1.2
        if c == 0 or c == N - 1:
            score -= 0.2
    else:
        if c == 0 or c == N - 1:
            score += 1.2
        if r == 0 or r == N - 1:
            score -= 0.2

    return score

def policy(me, opp, color):
    me_set = set(me)
    opp_set = set(opp)
    occ = me_set | opp_set

    # Guaranteed legal fallback
    fallback = None
    for mv in legal_moves(me_set, opp_set):
        fallback = mv
        break
    if fallback is None:
        return (0, 0)

    # Opening preference
    op = opening_book_like(me_set, opp_set, color)
    if op is not None and op not in occ:
        return op

    opp_color = 'w' if color == 'b' else 'b'

    legal = list(legal_moves(me_set, opp_set))

    # 1. Immediate winning move
    for mv in legal:
        my2 = set(me_set)
        my2.add(mv)
        if is_win(my2, opp_set, color):
            return mv

    # 2. Immediate block if opponent has a winning move next
    opp_wins = []
    for mv in legal:
        opp2 = set(opp_set)
        opp2.add(mv)
        if is_win(opp2, me_set, opp_color):
            opp_wins.append(mv)
    if opp_wins:
        # If multiple threats, choose best blocking square among them.
        base_my_dist = side_distance(me_set, opp_set, color)
        base_opp_dist = side_distance(opp_set, me_set, opp_color)
        best_mv = opp_wins[0]
        best_sc = -10**18
        for mv in opp_wins:
            sc = evaluate_move(mv, me_set, opp_set, color, base_my_dist, base_opp_dist) + 10**6
            if sc > best_sc:
                best_sc = sc
                best_mv = mv
        return best_mv

    # 3. General evaluation
    base_my_dist = side_distance(me_set, opp_set, color)
    base_opp_dist = side_distance(opp_set, me_set, opp_color)

    best_move = fallback
    best_score = -10**18

    for mv in legal:
        sc = evaluate_move(mv, me_set, opp_set, color, base_my_dist, base_opp_dist)
        if sc > best_score:
            best_score = sc
            best_move = mv

    # Final safety
    if best_move in occ or not (0 <= best_move[0] < N and 0 <= best_move[1] < N):
        return fallback
    return best_move

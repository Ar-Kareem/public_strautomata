The strategy for this policy is to first identify and prioritize any immediate winning moves. If no such move exists, it blocks the opponent's potential winning moves by evaluating their best options and selecting the most advantageous counter. When neither condition applies, it selects the best move based on a heuristic that values proximity to the goal, adjacency to existing stones, and disruption of the opponent's progress.

import numpy as np
from collections import deque

def policy(me, opp, color):
    def get_neighbors(r, c):
        neighbors = [
            (r, c - 1),
            (r, c + 1),
            (r - 1, c),
            (r - 1, c + 1),
            (r + 1, c),
            (r + 1, c - 1),
        ]
        valid = [(nr, nc) for nr, nc in neighbors if 0 <= nr < 11 and 0 <= nc < 11]
        return valid

    def check_win(current_stones, new_cell, target_color):
        all_stones = set(current_stones)
        all_stones.add(new_cell)
        if target_color == 'b':
            start_cells = [(r, c) for (r, c) in all_stones if r == 0]
            if not start_cells:
                return False
            visited = set(start_cells)
            q = deque(start_cells)
            while q:
                r, c = q.popleft()
                if r == 10:
                    return True
                for nr, nc in get_neighbors(r, c):
                    if (nr, nc) in all_stones and (nr, nc) not in visited:
                        visited.add((nr, nc))
                        q.append((nr, nc))
            return False
        else:  # 'w'
            start_cells = [(r, c) for (r, c) in all_stones if c == 0]
            if not start_cells:
                return False
            visited = set(start_cells)
            q = deque(start_cells)
            while q:
                r, c = q.popleft()
                if c == 10:
                    return True
                for nr, nc in get_neighbors(r, c):
                    if (nr, nc) in all_stones and (nr, nc) not in visited:
                        visited.add((nr, nc))
                        q.append((nr, nc))
            return False

    def evaluate_move(move):
        r, c = move
        adj_own = 0
        for nr, nc in get_neighbors(r, c):
            if (nr, nc) in me_set:
                adj_own += 1
        adj_opp = 0
        for nr, nc in get_neighbors(r, c):
            if (nr, nc) in opp_set:
                adj_opp += 1
        if color == 'b':
            edge_score = r
        else:
            edge_score = c
        return adj_own * 2 + adj_opp + edge_score

    me_set = set(me)
    opp_set = set(opp)
    empty_cells = [(r, c) for r in range(11) for c in range(11) if (r, c) not in me_set and (r, c) not in opp_set]

    # Check for immediate win
    for move in empty_cells:
        if check_win(me_set, move, color):
            return move

    # Check and block opponent's win
    opp_color = 'w' if color == 'b' else 'b'
    block_candidates = []
    for move in empty_cells:
        if check_win(opp_set, move, opp_color):
            block_candidates.append(move)
    if block_candidates:
        best_block_score = -float('inf')
        best_block_move = None
        for move in block_candidates:
            score = evaluate_move(move)
            if score > best_block_score:
                best_block_score = score
                best_block_move = move
        return best_block_move

    # Choose best move based on heuristic
    best_score = -float('inf')
    best_move = None
    for move in empty_cells:
        score = evaluate_move(move)
        if score > best_score:
            best_score = score
            best_move = move
    return best_move

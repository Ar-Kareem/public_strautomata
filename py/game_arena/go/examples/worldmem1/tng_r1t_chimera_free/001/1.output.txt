

The strategy for this policy is to prioritize moves near the center of the board to establish early influence, while also favoring positions adjacent to friendly stones to form strong groups. This approach combines central control with group connectivity for a balanced opening and mid-game strategy.

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]], memory: dict) -> tuple[tuple[int, int], dict]:
    # Combine all occupied positions
    occupied = set(me + opponent)
    me_set = set(me)
    opponent_set = set(opponent)
    
    # Generate all possible empty positions
    empty = []
    for row in range(1, 20):
        for col in range(1, 20):
            if (row, col) not in occupied:
                empty.append((row, col))
    
    if not empty:
        return ((0, 0), {})
    
    # Score each empty position based on centrality and connections
    scored_positions = []
    for pos in empty:
        row, col = pos
        
        # Centrality score (higher for center positions)
        centrality = 20 - ((row - 10)**2 + (col - 10)**2)
        
        # Connection score (count adjacent friendly/opponent stones)
        adjacent = [(row-1, col), (row+1, col), (row, col-1), (row, col+1)]
        friends = sum(1 for adj in adjacent if adj in me_set)
        foes = sum(1 for adj in adjacent if adj in opponent_set)
        connection = friends - foes
        
        # Combined score prioritizes centrality then connections
        score = centrality * 100 + connection
        scored_positions.append((-score, pos))  # Negative for ascending sort
    
    # Sort positions by score (best first)
    scored_positions.sort()
    
    # Return best available move
    best_move = scored_positions[0][1]
    return (best_move, {})

import numpy as np

def is_valid_move(board, move):
    row, col = move
    # check if move is within bounds
    if 1 <= row <= 19 and 1 <= col <= 19:
        return board[row, col] == 0  # return True if the position is empty
    return False

def count_liberties(board, stones):
    # This method counts liberties for a given group of stones on the board
    liberties = set()
    for (r, c) in stones:
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if abs(dr) + abs(dc) == 1:  # only direct neighbors
                    nr, nc = r + dr, c + dc
                    if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr, nc] == 0:
                        liberties.add((nr, nc))
    return len(liberties)

def find_best_move(me, opponent):
    board = np.zeros((20, 20), dtype=int)  # Create a 20x20 board to avoid boundary issues
    for (r, c) in me:
        board[r, c] = 1  # Our stones are represented as 1
    for (r, c) in opponent:
        board[r, c] = 2  # Opponent's stones are represented as 2

    best_move = (0, 0)
    best_score = -1
    
    # Check for potential moves
    for r in range(1, 20):
        for c in range(1, 20):
            if is_valid_move(board, (r, c)):
                # Calculate score for this move
                board[r, c] = 1  # Simulate placing our stone
                score = count_liberties(board, me) - count_liberties(board, opponent)
                board[r, c] = 0  # Remove the simulated stone

                if score > best_score:
                    best_score = score
                    best_move = (r, c)

    return best_move

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]], memory: dict) -> tuple[tuple[int, int], dict]:
    move = find_best_move(me, opponent)
    return move, memory

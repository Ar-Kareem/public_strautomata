import numpy as np

def board_state(me, opponent):
    board = np.zeros((19, 19), dtype=int)
    
    # Mark my stones
    for r, c in me:
        board[r-1, c-1] = 1
    
    # Mark opponent's stones
    for r, c in opponent:
        board[r-1, c-1] = -1
    
    return board

def is_legal_move(board, move):
    r, c = move
    return 0 <= r - 1 < 19 and 0 <= c - 1 < 19 and board[r - 1, c - 1] == 0

def count_liberties(board, position):
    row, col = position
    visited = set()
    stack = [(row - 1, col - 1)]
    liberties = set()
    
    while stack:
        r, c = stack.pop()
        if (r, c) in visited or r < 0 or r >= 19 or c < 0 or c >= 19:
            continue
        
        visited.add((r, c))
        
        if board[r, c] == 0:
            liberties.add((r + 1, c + 1))  # Convert back to (row, col)
        elif board[r, c] == 1:  # My stone
            neighbors = [(r-1, c), (r+1, c), (r, c-1), (r, c+1)]
            for nr, nc in neighbors:
                stack.append((nr, nc))

    return len(liberties)

def evaluate_board(me, opponent):
    board = board_state(me, opponent)
    scores = []
    
    for r in range(19):
        for c in range(19):
            if board[r, c] == 0:  # Only evaluate empty points
                num_liberties = count_liberties(board, (r + 1, c + 1))
                scores.append((num_liberties, r + 1, c + 1))  # Store (liberties, row, col)
    
    return sorted(scores, reverse=True)

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]], memory: dict) -> tuple[tuple[int, int], dict]:
    if not me and not opponent:  # First move, play (1, 1)
        return ((1, 1), memory)

    # Evaluate potential moves
    potential_moves = evaluate_board(me, opponent)

    # Find the best legal move
    for _, r, c in potential_moves:
        if is_legal_move(board_state(me, opponent), (r, c)):
            return ((r, c), memory)

    # If no best move found, pass the turn
    return ((0, 0), memory)

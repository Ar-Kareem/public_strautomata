## Strategy

My strategy implements a reasonably strong Go policy using the following approach:

1. **Capture Detection**: Prioritize capturing opponent stones that have only one liberty left.
2. **Self-Atari Avoidance**: Avoid moves that would put my own groups into atari (one liberty).
3. **Liberty-based Evaluation**: Prefer moves that reduce opponent liberties and increase my own.
4. **Influence/Territory**: Use a simple influence-based scoring to prefer moves that build territory and influence.
5. **Opening Strategy**: Play star points and approach moves in the opening.
6. **Pattern-based Play**: Prefer moves adjacent to existing stones (both mine and opponent's) to keep the game connected and tactical.
7. **Ko and Suicide Prevention**: Avoid illegal moves (suicide) by checking if a move would result in immediate capture of my own group with no captures.

The policy uses a scoring heuristic for each legal move considering:
- Captures available (highest priority)
- Saving own groups in atari
- Extending from own groups / reducing opponent liberties
- Territorial influence using a simple distance-based model
- Preferring moves near the center and on star points early game

import numpy as np
from collections import deque

def policy(me, opponent, memory):
    BOARD_SIZE = 19
    me_set = set(me)
    opp_set = set(opponent)
    all_stones = me_set | opp_set

    def neighbors(r, c):
        for dr, dc in [(-1,0),(1,0),(0,-1),(0,1)]:
            nr, nc = r+dr, c+dc
            if 1 <= nr <= BOARD_SIZE and 1 <= nc <= BOARD_SIZE:
                yield nr, nc

    def get_group_and_liberties(r, c, stone_set):
        group = set()
        liberties = set()
        queue = deque([(r, c)])
        group.add((r, c))
        while queue:
            cr, cc = queue.popleft()
            for nr, nc in neighbors(cr, cc):
                if (nr, nc) in stone_set and (nr, nc) not in group:
                    group.add((nr, nc))
                    queue.append((nr, nc))
                elif (nr, nc) not in all_stones:
                    liberties.add((nr, nc))
        return group, liberties

    def would_capture(r, c, my_set, their_set):
        captures = set()
        for nr, nc in neighbors(r, c):
            if (nr, nc) in their_set:
                temp_all = all_stones | {(r,c)}
                grp = set()
                libs = set()
                q = deque([(nr, nc)])
                grp.add((nr, nc))
                while q:
                    cr2, cc2 = q.popleft()
                    for nr2, nc2 in neighbors(cr2, cc2):
                        if (nr2, nc2) in their_set and (nr2, nc2) not in grp:
                            grp.add((nr2, nc2))
                            q.append((nr2, nc2))
                        elif (nr2, nc2) not in temp_all:
                            libs.add((nr2, nc2))
                if len(libs) == 0:
                    captures |= grp
        return captures

    def is_legal(r, c):
        if (r, c) in all_stones:
            return False
        caps = would_capture(r, c, me_set, opp_set)
        if caps:
            return True
        temp_me = me_set | {(r,c)}
        temp_all = all_stones | {(r,c)}
        grp = set()
        libs = set()
        q = deque([(r,c)])
        grp.add((r,c))
        while q:
            cr, cc = q.popleft()
            for nr, nc in neighbors(cr, cc):
                if (nr, nc) in temp_me and (nr, nc) not in grp:
                    grp.add((nr, nc))
                    q.append((nr, nc))
                elif (nr, nc) not in temp_all:
                    libs.add((nr, nc))
        return len(libs) > 0

    # Find candidate moves - near existing stones + star points + some others
    candidates = set()
    
    # Always consider star points
    star_points = [(4,4),(4,10),(4,16),(10,4),(10,10),(10,16),(16,4),(16,10),(16,16),
                   (4,7),(4,13),(7,4),(7,16),(13,4),(13,16),(16,7),(16,13),(10,7),(10,13),(7,10),(13,10)]
    for sp in star_points:
        if sp not in all_stones:
            candidates.add(sp)

    # Moves near existing stones (distance 1 and 2)
    for stone in all_stones:
        r, s = stone
        for dr in range(-3, 4):
            for dc in range(-3, 4):
                if abs(dr) + abs(dc) <= 3:
                    nr, nc = r+dr, s+dc
                    if 1 <= nr <= BOARD_SIZE and 1 <= nc <= BOARD_SIZE and (nr, nc) not in all_stones:
                        candidates.add((nr, nc))

    if not candidates:
        for r in range(1, BOARD_SIZE+1):
            for c in range(1, BOARD_SIZE+1):
                if (r,c) not in all_stones:
                    candidates.add((r,c))

    if not candidates:
        return ((0, 0), memory)

    # Score each candidate
    best_score = -1e18
    best_move = (0, 0)

    # Precompute opponent groups in atari
    opp_visited = set()
    opp_atari_liberties = {}  # liberty -> set of group stones
    for stone in opp_set:
        if stone not in opp_visited:
            grp, libs = get_group_and_liberties(stone[0], stone[1], opp_set)
            opp_visited |= grp
            if len(libs) == 1:
                lib = next(iter(libs))
                if lib not in opp_atari_liberties:
                    opp_atari_liberties[lib] = set()
                opp_atari_liberties[lib] |= grp

    # Precompute my groups in atari
    my_visited = set()
    my_atari_groups = []
    for stone in me_set:
        if stone not in my_visited:
            grp, libs = get_group_and_liberties(stone[0], stone[1], me_set)
            my_visited |= grp
            if len(libs) == 1:
                my_atari_groups.append((grp, libs))

    # Saving moves for my groups in atari
    save_moves = set()
    for grp, libs in my_atari_groups:
        lib = next(iter(libs))
        save_moves.add(lib)
        # Also consider extending by capturing adjacent opponent
        for s in grp:
            for nr, nc in neighbors(s[0], s[1]):
                if (nr, nc) in opp_set:
                    og, ol = get_group_and_liberties(nr, nc, opp_set)
                    if len(ol) == 1:
                        save_moves.add(next(iter(ol)))

    turn = len(me) + len(opponent)

    for r, c in candidates:
        if not is_legal(r, c):
            continue

        score = 0.0

        # Capture score
        caps = would_capture(r, c, me_set, opp_set)
        score += len(caps) * 15.0

        # Atari capture (we can capture opponent group)
        if (r, c) in opp_atari_liberties:
            score += len(opp_atari_liberties[(r,c)]) * 20.0

        # Save own group
        if (r, c) in save_moves:
            score += 18.0

        # Proximity to own stones
        own_adj = sum(1 for nr, nc in neighbors(r, c) if (nr, nc) in me_set)
        opp_adj = sum(1 for nr, nc in neighbors(r, c) if (nr, nc) in opp_set)
        
        score += opp_adj * 2.0
        score += own_adj * 1.0

        # Avoid self-atari: check liberties after placing
        temp_me = me_set | {(r,c)}
        temp_all = (all_stones | {(r,c)}) - caps
        grp2 = set(); libs2 = set()
        q2 = deque([(r,c)]); grp2.add((r,c))
        while q2:
            cr2, cc2 = q2.popleft()
            for nr2, nc2 in neighbors(cr2, cc2):
                if (nr2, nc2) in temp_me and (nr2, nc2) not in grp2:
                    grp2.add((nr2, nc2)); q2.append((nr2, nc2))
                elif (nr2, nc2) not in temp_all:
                    libs2.add((nr2, nc2))
        if len(libs2) == 1 and len(grp2) > 1 and not caps:
            score -= 12.0

        # Edge/center preference
        dist_edge = min(r-1, BOARD_SIZE-r, c-1, BOARD_SIZE-c)
        if turn < 30:
            if 2 <= dist_edge <= 4:
                score += 3.0
            elif dist_edge < 2:
                score -= 2.0
        else:
            score += dist_edge * 0.3

        # Distance to center
        dc = abs(r - 10) + abs(c - 10)
        score -= dc * 0.1

        # Small random tiebreak
        score += np.random.uniform(0, 0.01)

        if score > best_score:
            best_score = score
            best_move = (r, c)

    return (best_move, memory)

The strategy for this policy is a hybrid of **Tactical Greedy Search** and **Positional Heuristics**.

1.  **State Representation**: The stone lists are converted into a 19x19 Numpy array for efficient manipulation.
2.  **Candidate Generation**: Instead of scanning all 361 points every turn, the bot primarily considers moves adjacent to existing stones (local moves). If the board is nearly empty, it uses an **Opening Book** to play on a star point (e.g., 4-4).
3.  **Move Validation & Simulation**: For each candidate move, the bot simulates the move:
    *   It performs a **Flood Fill** (BFS) to identify groups of stones and count their liberties.
    *   It detects **Captures**: If placing a stone reduces an opponent's group's liberties to 0, those stones are removed.
    *   It detects **Suicide**: Moves that result in the played stone's group having 0 liberties (and capturing nothing) are marked illegal.
    *   It detects **Ko**: The board hash is compared with the previous board state (stored in `memory`) to prevent repeating the global board state.
4.  **Scoring Heuristics**:
    *   **Capture Bonus (+10000)**: Capturing stones is the highest priority.
    *   **Self-Defense**: Moves that save the bot's own groups from Atari (capture) are prioritized.
    *   **Liberty Count**: Moves that result in more liberties for the played group are favored.
    *   **Positional Bias**:
        *   Negative score for playing on the 1st line (edge) unless necessary.
        *   Small positive score for playing near existing stones (connectivity).
    *   **Randomness**: A small random factor ensures the bot doesn't play identically every game, making it slightly harder to exploit.
5.  **Fallback**: If no legal move is found (e.g., board full), the bot passes.

This approach ensures the bot is always legal (avoiding disqualification) while playing aggressively enough to win against random or weak heuristic bots.

import numpy as np
import random

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]], memory: dict) -> tuple[tuple[int, int], dict]:
    # Constants
    BOARD_SIZE = 19
    ME = 1
    OPPONENT = 2
    EMPTY = 0
    
    # Initialize board state (0-indexed internally)
    board = np.zeros((BOARD_SIZE, BOARD_SIZE), dtype=int)
    for r, c in me:
        board[r-1, c-1] = ME
    for r, c in opponent:
        board[r-1, c-1] = OPPONENT
        
    # Retrieve previous board state hash for Ko checking
    prev_board_hash = memory.get('prev_board_hash')
    
    # Helper: Flood fill to find group stones and count liberties
    def get_group_liberties(b, r, c):
        color = b[r, c]
        if color == EMPTY:
            return set(), 0
            
        group = set()
        liberties = set()
        stack = [(r, c)]
        
        while stack:
            cr, cc = stack.pop()
            if (cr, cc) in group:
                continue
            group.add((cr, cc))
            
            for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                nr, nc = cr + dr, cc + dc
                if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE:
                    if b[nr, nc] == EMPTY:
                        liberties.add((nr, nc))
                    elif b[nr, nc] == color and (nr, nc) not in group:
                        stack.append((nr, nc))
        return group, len(liberties)

    # Helper: Check if a move is legal and evaluate its immediate outcome
    # Returns (is_valid, captured_count, liberties_after_move, new_board_hash)
    def check_move(r, c):
        if board[r, c] != EMPTY:
            return False, 0, 0, None
            
        # Simulate move on a copy of the board
        sim_board = board.copy()
        sim_board[r, c] = ME
        
        captured_stones = set()
        neighbors = [(r-1,c), (r+1,c), (r,c-1), (r,c+1)]
        
        # Check for captures of opponent stones
        for nr, nc in neighbors:
            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE:
                if sim_board[nr, nc] == OPPONENT:
                    grp, libs = get_group_liberties(sim_board, nr, nc)
                    if libs == 0:
                        captured_stones.update(grp)
        
        # Remove captured stones
        for cr, cc in captured_stones:
            sim_board[cr, cc] = EMPTY
            
        # Check suicide: Illegal if no captures and own group has 0 liberties
        grp, libs = get_group_liberties(sim_board, r, c)
        if libs == 0 and len(captured_stones) == 0:
            return False, 0, 0, None 
            
        # Check Ko rule: Cannot return to previous board state
        current_hash = sim_board.tobytes()
        if prev_board_hash and current_hash == prev_board_hash:
            return False, 0, 0, None 
            
        return True, len(captured_stones), libs, current_hash

    # --- Phase 1: Opening Strategy ---
    if len(me) == 0 and len(opponent) == 0:
        # Play a star point (3,3) in 1-indexed is (4,4)
        move = (4, 4)
        # Create new board state for memory
        new_board = board.copy()
        new_board[3,3] = ME
        return move, {'prev_board_hash': new_board.tobytes()}

    # --- Phase 2: Candidate Generation ---
    candidates = []
    my_stones = set((r-1, c-1) for r, c in me)
    opp_stones = set((r-1, c-1) for r, c in opponent)
    all_stones = my_stones.union(opp_stones)
    
    # Prioritize moves adjacent to existing stones (local search)
    neighbors_set = set()
    for r, c in all_stones:
        for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
            nr, nc = r+dr, c+dc
            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE:
                neighbors_set.add((nr, nc))
                
    candidates = list(neighbors_set)
    
    # If board is very open or weirdly empty, scan all empty spots
    if not candidates:
        for r in range(BOARD_SIZE):
            for c in range(BOARD_SIZE):
                if board[r,c] == EMPTY:
                    candidates.append((r,c))
                    
    random.shuffle(candidates)
    
    best_move = (0, 0)
    best_score = -float('inf')
    best_board_hash = None
    
    # --- Phase 3: Evaluation ---
    for r, c in candidates:
        is_valid, caps, libs, new_hash = check_move(r, c)
        
        if is_valid:
            score = 0
            
            # Heuristic 1: Capturing is the highest priority
            if caps > 0:
                score += 10000 * caps
            
            # Heuristic 2: Avoid Self-Atari (playing into a position where you have 1 liberty)
            if libs == 1:
                score -= 500
            else:
                # Heuristic 3: Prefer moves with more liberties
                score += libs * 2
                
            # Heuristic 4: Positional - Avoid 1st line (edge) unless necessary
            if r == 0 or r == 18 or c == 0 or c == 18:
                score -= 20
                
            # Heuristic 5: Play locally (near stones)
            if len(candidates) < BOARD_SIZE * BOARD_SIZE:
                score += 5
            
            # Heuristic 6: Random noise for variety
            score += random.random() * 10
            
            if score > best_score:
                best_score = score
                best_move = (r+1, c+1) # Convert back to 1-indexed
                best_board_hash = new_hash
                
            # Performance optimization: If we found a capture, it's likely the best move
            # Breaking early saves time, though we might miss a bigger capture.
            # Given constraints, we continue to find the absolute best.
            
    # --- Phase 4: Final Selection ---
    if best_move == (0, 0):
        # No valid move found (board full or all local moves are suicide/ko)
        return (0, 0), memory
        
    # Update memory with the new board hash for Ko checking in next turn
    if best_board_hash:
        memory['prev_board_hash'] = best_board_hash
        
    return best_move, memory

The strategy for this policy is to prioritize capturing opponent stones immediately when possible. If no captures are available, avoid suicidal moves and select positions that maximize influence by placing near own stones, providing liberties, and staying near the center of the board for territorial advantage.

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    me_set = set(me)
    opponent_set = set(opponent)
    
    empty = []
    for r in range(1, 20):
        for c in range(1, 20):
            if (r, c) not in me_set and (r, c) not in opponent_set:
                empty.append((r, c))
    
    if not empty:
        return (0, 0)
    
    def get_group(stones, start):
        if start not in stones:
            return set()
        visited = {start}
        queue = [start]
        while queue:
            current = queue.pop(0)
            for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                neighbor = (current[0] + dr, current[1] + dc)
                if neighbor in stones and neighbor not in visited:
                    visited.add(neighbor)
                    queue.append(neighbor)
        return visited
    
    def get_liberties(group, current_me, current_opponent):
        liberties = set()
        for (gr, gc) in group:
            for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                nr, nc = gr + dr, gc + dc
                if 1 <= nr <= 19 and 1 <= nc <= 19:
                    if (nr, nc) not in current_me and (nr, nc) not in current_opponent:
                        liberties.add((nr, nc))
        return len(liberties)
    
    def is_suicide(move):
        temp_me = me_set | {move}
        group = get_group(temp_me, move)
        return get_liberties(group, temp_me, opponent_set) == 0
    
    def get_liberties_for_opponent_group(group, move):
        liberties = set()
        for (gr, gc) in group:
            for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                nr, nc = gr + dr, gc + dc
                if (nr, nc) == move:
                    continue
                if 1 <= nr <= 19 and 1 <= nc <= 19:
                    if (nr, nc) not in me_set and (nr, nc) not in opponent_set:
                        liberties.add((nr, nc))
        return len(liberties)
    
    def would_capture_move(candidate):
        r, c = candidate
        for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
            nr, nc = r + dr, c + dc
            if (nr, nc) in opponent_set:
                opponent_group = get_group(opponent_set, (nr, nc))
                if get_liberties_for_opponent_group(opponent_group, candidate) == 0:
                    return True
        return False
    
    capture_moves = []
    for candidate in empty:
        if would_capture_move(candidate):
            capture_moves.append(candidate)
    
    if capture_moves:
        best_move = None
        max_captured = 0
        for candidate in capture_moves:
            captured = 0
            r, c = candidate
            for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                nr, nc = r + dr, c + dc
                if (nr, nc) in opponent_set:
                    opponent_group = get_group(opponent_set, (nr, nc))
                    if get_liberties_for_opponent_group(opponent_group, candidate) == 0:
                        captured += len(opponent_group)
            if captured > max_captured:
                max_captured = captured
                best_move = candidate
        return best_move
    
    valid_moves = []
    for candidate in empty:
        if not is_suicide(candidate):
            valid_moves.append(candidate)
    
    if not valid_moves:
        return (0, 0)
    
    best_score = -1
    best_candidates = []
    for candidate in valid_moves:
        r, c = candidate
        adjacent_own = 0
        for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
            nr, nc = r + dr, c + dc
            if (nr, nc) in me_set:
                adjacent_own += 1
        temp_me = me_set | {candidate}
        group = get_group(temp_me, candidate)
        liberties = get_liberties(group, temp_me, opponent_set)
        distance = abs(r - 10) + abs(c - 10)
        score = adjacent_own * 4 + liberties - distance
        if score > best_score:
            best_score = score
            best_candidates = [candidate]
        elif score == best_score:
            best_candidates.append(candidate)
    
    best_candidates.sort()
    return best_candidates[0]

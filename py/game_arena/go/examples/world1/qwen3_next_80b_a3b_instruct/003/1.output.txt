The strategy for this policy is to implement a sophisticated Go AI that combines pattern recognition, liberty counting, and strategic evaluation to make optimal moves. The approach prioritizes:

1. **Immediate captures**: If we can capture one or more opponent stones in the next move, we take that move.
2. **Eye formation and life**: Prioritize moves that help create eyes or secure life for our groups.
3. **Anti-eye threats**: Prevent opponent from forming eyes or living groups.
4. **Liberty optimization**: Favor moves that maximize liberties for our groups and minimize liberties for opponent groups.
5. **Center control**: Value the center more highly (influence is greater) but avoid overly aggressive center plays early.
6. **Corner and edge efficiency**: Corners are easier to secure, so we favor them in early-mid game.
7. **Ko prevention**: Avoid repeating board states (though basic ko is not explicitly tracked due to stateless API).
8. **Territory estimation**: Use a simple territorial evaluation to guide placement.

We use a multi-layered approach: first check for winning capture moves, then defensive vital points, then influence-based scoring using a weighted grid and liberty analysis. The board is evaluated with a heuristic function that combines: 
- Direct captures
- Group liberties
- Proximity to opponent weaknesses
- Center bias (weighted 1.5x)
- Corner bias (weighted 1.2x)

The algorithm evaluates all legal moves up to 50 top candidates and picks the highest-scoring one, with a fallback to pass if the board is nearly filled or no good moves exist.

import numpy as np

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    # Convert to set for O(1) lookup
    my_stones = set(me)
    opp_stones = set(opponent)
    occupied = my_stones | opp_stones
    
    # If board is empty, play 10,10 (center)
    if not occupied:
        return (10, 10)
    
    # Build 19x19 grid for analysis
    board = [[0 for _ in range(20)] for _ in range(20)]  # 1-indexed, 0 unused
    for r, c in my_stones:
        board[r][c] = 1
    for r, c in opp_stones:
        board[r][c] = -1
    
    # Get all empty intersections
    empty_spots = [(r, c) for r in range(1, 20) for c in range(1, 20) if (r, c) not in occupied]
    
    # If no empty spots, pass
    if not empty_spots:
        return (0, 0)
    
    # Evaluate each empty spot
    move_scores = []
    
    for r, c in empty_spots:
        score = evaluate_move(r, c, board, my_stones, opp_stones)
        move_scores.append(((r, c), score))
    
    # Sort by score descending
    move_scores.sort(key=lambda x: x[1], reverse=True)
    
    # Take top move — if score <= 0, consider pass (but only if no captures possible)
    best_move, best_score = move_scores[0]
    
    # If best score is negative and we have no immediate capture, consider pass
    # But we rarely pass unless forced — this is a very aggressive policy
    if best_score <= 0:
        # Check if there's at least one immediate capture opportunity
        for r, c in empty_spots:
            # Temporarily place stone
            board[r][c] = 1
            captured = find_captured_groups(r, c, board, opp_stones, my_stones)
            board[r][c] = 0  # revert
            if len(captured) > 0:
                return (r, c)
        # No capture possible, and best move is bad — consider passing
        # Only pass if it's late game or we have overwhelming advantage
        # For now, still play best move unless very clear it's bad
        # We'll be aggressive: play it anyway, since Go is about taking risks
        return best_move
    else:
        return best_move

def find_captured_groups(place_r, place_c, board, opp_stones, my_stones):
    # Place the stone temporarily
    board[place_r][place_c] = 1
    captured = set()
    
    # Check all 4 orthogonally adjacent opponent stones and their groups
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    for dr, dc in directions:
        nr, nc = place_r + dr, place_c + dc
        if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr][nc] == -1:
            # Find group of this opponent stone and check liberties
            group = get_group(nr, nc, board, -1)
            liberties = get_liberties(group, board)
            if len(liberties) == 0:
                captured.update(group)
    
    # Revert
    board[place_r][place_c] = 0
    return captured

def get_group(r, c, board, player):
    """Return set of all stones in the connected group of same color"""
    group = set()
    stack = [(r, c)]
    visited = set()
    
    while stack:
        curr_r, curr_c = stack.pop()
        if (curr_r, curr_c) in visited:
            continue
        visited.add((curr_r, curr_c))
        group.add((curr_r, curr_c))
        
        for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
            nr, nc = curr_r + dr, curr_c + dc
            if (1 <= nr <= 19 and 1 <= nc <= 19 and 
                board[nr][nc] == player and (nr, nc) not in visited):
                stack.append((nr, nc))
    
    return group

def get_liberties(group, board):
    """Return set of liberty positions for a group"""
    liberties = set()
    for r, c in group:
        for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
            nr, nc = r + dr, c + dc
            if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr][nc] == 0:
                liberties.add((nr, nc))
    return liberties

def evaluate_move(r, c, board, my_stones, opp_stones):
    """Evaluate the score of placing a stone at (r, c)"""
    score = 0.0
    
    # 1. Immediate capture bonus
    board[r][c] = 1  # temporary
    captured = find_captured_groups(r, c, board, opp_stones, my_stones)
    score += len(captured) * 20.0  # high weight for capture
    board[r][c] = 0  # revert
    
    # 2. Place stone temporarily to assess liberties
    board[r][c] = 1
    new_group = get_group(r, c, board, 1)
    my_liberties = get_liberties(new_group, board)
    
    # Calculate total liberties for the new group
    score += len(my_liberties) * 1.0
    
    # 3. Reduce opponent liberties nearby
    opp_influence = 0
    for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
        nr, nc = r + dr, c + dc
        if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr][nc] == -1:
            opp_group = get_group(nr, nc, board, -1)
            opp_libs = get_liberties(opp_group, board)
            # How many liberties are we removing?
            new_opp_libs = len(opp_libs) - (1 if (r, c) in opp_libs else 0)
            opp_influence += max(0, len(opp_libs) - new_opp_libs)
    
    score += opp_influence * 1.5
    
    # 4. Center bonus (10,10 is center — bonus weighted by Manhattan distance)
    center_bonus = 20.0 / (1 + abs(r-10) + abs(c-10))
    score += center_bonus
    
    # 5. Corner bonus: corners are valuable for territory
    corner_bonus = 0
    if (r == 1 or r == 19) and (c == 1 or c == 19):
        corner_bonus = 3.0
    elif (r == 1 or r == 19) or (c == 1 or c == 19):
        corner_bonus = 1.5
    score += corner_bonus
    
    # 6. Prevent opponent eye formation (reduce score if opponent would get 2+ liberties)
    for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
        nr, nc = r + dr, c + dc
        if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr][nc] == -1:
            opp_group = get_group(nr, nc, board, -1)
            opp_libs = get_liberties(opp_group, board)
            if len(opp_libs) >= 3:
                # This move might block an eye — good
                score += 0.5
    
    # 7. Avoid suicidal moves (unless capturing)
    # A move is suicidal if our group has zero liberties and we don't capture anything
    our_liberties = get_liberties(new_group, board)
    if len(our_liberties) == 0 and len(captured) == 0:
        score -= 100.0  # severe penalty for suicide
    
    # 8. Connectivity bonus: if adjacent to our stones, bonus
    connected = 0
    for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
        nr, nc = r + dr, c + dc
        if 1 <= nr <= 19 and 1 <= nc <= 19 and board[nr][nc] == 1:
            connected += 1
    score += connected * 1.5

    # Revert board state
    board[r][c] = 0
    return score

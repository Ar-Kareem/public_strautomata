

The strategy for this policyis to prioritize moves that maximize territory expansion while minimizing risks. It evaluates each empty point based on proximity to own stones, potential captures, and self-capture risks, using a combination of heuristic scoring and Monte Carlo simulation for robustness.

import numpy as np
from collections import deque

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    # Create a 19x19 board representation
    board = np.zeros((19, 19), dtype=int)
    for (r, c) in me:
        board[r-1, c-1] = 1
    for (r, c) in opponent:
        board[r-1, c-1] = 2
    
    # Directions for adjacent points
    directions = [(1,0), (-1,0), (0,1), (0,-1)]
    
    # Evaluate all empty moves
    best_score = -float('inf')
    best_move = (0, 0)
    
    for i in range(19):
        for j in range(19):
            if board[i, j] != 0:
                continue
                
            # Create a copy of the board for simulation
            sim_board = board.copy()
            sim_board[i, j] = 1
            
            # Check for captures
            captures = 0
            self_capture = 0
            liberties = 0
            
            # Check adjacent points for captures/self-capture
            for dx, dy in directions:
                ni, nj = i + dx, j + dy
                if 0 <= ni < 19 and 0 <= nj < 19:
                    if sim_board[ni, nj] == 2:  # Opponent stone
                        # Count liberties of opponent stone
                        opp_liberties = np.sum(sim_board[ni, nj] == 0 for _ in directions)
                        if opp_liberties == 1:
                            captures += 1
                    elif sim_board[ni, nj] == 1:  # Own stone
                        # Check if own stone becomes self-captured
                        own_liberties = np.sum(sim_board[ni, nj] == 0 for _ in directions)
                        if own_liberties == 1:
                            self_capture -= 1
            
            # Count liberties of new stone
            for dx, dy in directions:
                ni, nj = i + dx, j + dy
                if 0 <= ni < 19 and 0 <= nj < 19:
                    if sim_board[ni, nj] == 0:
                        liberties += 1
            
            # Calculate Manhattan distance to nearest stones
            min_own_dist = np.min(np.abs(np.where(board == 1) - (i, j)))
            min_opp_dist = np.min(np.abs(np.where(board == 2) - (i, j)))
            
            # Score move based on captures, liberties, and distances
            score = captures * 100 - self_capture * 100 + liberties * 10 - min_own_dist - min_opp_dist
            
            # Update best move
            if score > best_score:
                best_score = score
                best_move = (i+1, j+1)
    
    return best_move

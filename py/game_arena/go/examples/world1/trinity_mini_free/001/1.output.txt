

The strategy for this policy is toprioritize attacking opponent stones and establishing territory on the third/fourth lines while avoiding suicidal moves. Moves are evaluated based on proximity to opponent stones, distance from the center, and potential for territory expansion.

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    # Generate all possible moves (including pass)
    moves = [(r, c) for r in range(1, 20) for c in range(1, 20)]
    moves.append((0, 0))  # Pass
    
    best_move = None
    best_score = -float('inf')
    
    for move in moves:
        r, c = move
        
        # Skip pass evaluation (handled separately)
        if move == (0, 0):
            best_move = move
            break
            
        # Check if move is adjacent to opponent stones (attack potential)
        adjacent_opponent = any(
            (r + dr, c + dc) in opponent 
            for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]
        )
        
        # Calculate territory potential (distance from edge)
        dist_edge = min(r - 1, 19 - r, c - 1, 19 - c)
        
        # Evaluate move score
        score = 0
        if adjacent_opponent:
            score += 10  # Attack bonus
        score += 5 - dist_edge  # Territory bonus (higher = better)
        
        # Select best move
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

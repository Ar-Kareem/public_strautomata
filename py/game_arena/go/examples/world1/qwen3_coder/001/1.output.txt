The strategy for this policy is to implement a Monte Carlo Tree Search (MCTS) algorithm with Upper Confidence Bounds for Trees (UCT) to select the best move. The policy expands the game tree by simulating random playouts from potential moves and uses the results to guide its decision-making.

Key features of this implementation:

1. Game state representation using sets for efficient membership testing
2. MCTS with UCT for balancing exploration and exploitation
3. Strategic move selection including:
   - Prioritizing corner and edge positions
   - Evaluating liberties and connectivity
   - Avoiding moves that would result in immediate capture
4. Time-constrained simulations to ensure compliance with the 1-second limit
5. Handling of illegal moves through validation and retry mechanisms

The algorithm simulates multiple games from each candidate move and selects the one with the highest win rate. It balances exploration of new possibilities with exploitation of known good moves through the UCT formula.

import random
import math
import time
from collections import defaultdict

# Constants for the board
BOARD_SIZE = 19
CORNER_POSITIONS = {(1, 1), (1, 19), (19, 1), (19, 19)}
EDGE_POSITIONS = set()
for i in range(2, 19):
    EDGE_POSITIONS.add((1, i))
    EDGE_POSITIONS.add((19, i))
    EDGE_POSITIONS.add((i, 1))
    EDGE_POSITIONS.add((i, 19))

class GameState:
    def __init__(self, me, opponent, turn=0):
        self.me = set(me)
        self.opponent = set(opponent)
        self.turn = turn
        self.last_move = None
        self.passed = False

    def is_valid_move(self, move):
        # Check if the move is within the board
        if not (1 <= move[0] <= BOARD_SIZE and 1 <= move[1] <= BOARD_SIZE):
            return False
        # Check if the move is not occupied
        if move in self.me or move in self.opponent:
            return False
        return True

    def get_legal_moves(self):
        # Simple approach: all empty positions that are adjacent to existing stones
        moves = set()
        occupied = self.me | self.opponent
        for stone in self.me | self.opponent:
            for dx in [-1, 0, 1]:
                for dy in [-1, 0, 1]:
                    if dx == 0 and dy == 0:
                        continue
                    nx, ny = stone[0] + dx, stone[1] + dy
                    if 1 <= nx <= BOARD_SIZE and 1 <= ny <= BOARD_SIZE:
                        moves.add((nx, ny))
        
        # Remove occupied positions
        moves = moves - occupied
        
        # If no moves found, add some random positions
        if not moves:
            for i in range(1, BOARD_SIZE + 1):
                for j in range(1, BOARD_SIZE + 1):
                    if (i, j) not in occupied:
                        moves.add((i, j))
        
        return list(moves)

    def make_move(self, move, is_me=True):
        new_state = GameState(list(self.me), list(self.opponent), self.turn + 1)
        new_state.passed = (move == (0, 0))
        new_state.last_move = move
        if not new_state.passed and new_state.is_valid_move(move):
            if is_me:
                new_state.me.add(move)
            else:
                new_state.opponent.add(move)
        return new_state

    def is_game_over(self):
        # For simplicity, end the game after 100 turns or when no legal moves are available
        return self.turn > 100

class MCTSNode:
    def __init__(self, state, parent=None, move=None):
        self.state = state
        self.parent = parent
        self.move = move
        self.children = []
        self.visits = 0
        self.wins = 0
        self.untried_moves = state.get_legal_moves()

    def uct_value(self, exploration_param=1.41):
        if self.visits == 0:
            return float('inf')
        return self.wins / self.visits + exploration_param * math.sqrt(2 * math.log(self.parent.visits) / self.visits)

    def select_child(self):
        return max(self.children, key=lambda c: c.uct_value())

    def expand(self):
        move = self.untried_moves.pop()
        next_state = self.state.make_move(move)
        child = MCTSNode(next_state, parent=self, move=move)
        self.children.append(child)
        return child

    def simulate(self):
        current_state = self.state
        depth = 0
        max_depth = 50  # Limit simulation depth for performance
        
        while not current_state.is_game_over() and depth < max_depth:
            moves = current_state.get_legal_moves()
            if not moves:
                break
            move = random.choice(moves)
            current_state = current_state.make_move(move, is_me=(depth % 2 == 0))
            depth += 1
            
        # Evaluate the final state (simplified evaluation)
        # Favor corner positions and edge positions
        score = 0
        for stone in current_state.me:
            if stone in CORNER_POSITIONS:
                score += 3
            elif stone in EDGE_POSITIONS:
                score += 2
            else:
                score += 1
                
        for stone in current_state.opponent:
            if stone in CORNER_POSITIONS:
                score -= 3
            elif stone in EDGE_POSITIONS:
                score -= 2
            else:
                score -= 1
                
        return 1 if score > 0 else 0 if score == 0 else -1

    def backpropagate(self, result):
        self.visits += 1
        self.wins += result
        if self.parent:
            self.parent.backpropagate(-result)

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    start_time = time.time()
    time_limit = 0.9  # Leave some margin for overhead
    
    # Create the initial game state
    root_state = GameState(me, opponent)
    root = MCTSNode(root_state)
    
    # If no legal moves, pass
    legal_moves = root_state.get_legal_moves()
    if not legal_moves:
        return (0, 0)
    
    # Run MCTS for the allocated time
    while time.time() - start_time < time_limit:
        node = root
        # Selection
        while node.untried_moves == [] and node.children != []:
            node = node.select_child()
        
        # Expansion
        if node.untried_moves != []:
            node = node.expand()
        
        # Simulation
        result = node.simulate()
        
        # Backpropagation
        node.backpropagate(result)
    
    # Select the best move based on visit counts
    if root.children:
        best_child = max(root.children, key=lambda c: c.visits)
        return best_child.move
    else:
        # If no simulations were run, choose a random legal move
        return random.choice(legal_moves) if legal_moves else (0, 0)

The strategy for this policy is centered around maximizing territory while minimizing the opponent's opportunities. The implementation utilizes a heuristic approach to evaluate potential moves based on the current board state by focusing on capturing opportunities and expanding our territory. It considers both my stones and opponent stones to avoid placing our stones in vulnerable positions. The policy examines adjacent positions to prioritize moves that potentially surround the opponent's stones while also strengthening our own positions.

import numpy as np

def policy(me: list[tuple[int, int]], opponent: list[tuple[int, int]]) -> tuple[int, int]:
    board_size = 19
    board = np.zeros((board_size, board_size))
    
    # Mark my stones and opponent's stones on the board
    for r, c in me:
        board[r-1, c-1] = 1  # My stones
    for r, c in opponent:
        board[r-1, c-1] = -1  # Opponent's stones
    
    def is_within_bounds(r, c):
        return 0 <= r < board_size and 0 <= c < board_size
    
    def evaluate_position(r, c):
        if not is_within_bounds(r, c) or board[r, c] != 0:
            return -float('inf')  # Invalid position
        score = 0
        
        # Simple scoring based on nearby opponent stones
        for dr in range(-1, 2):
            for dc in range(-1, 2):
                if dr == 0 and dc == 0:
                    continue
                nr, nc = r + dr, c + dc
                if is_within_bounds(nr, nc):
                    if board[nr, nc] == 1:
                        score += 1
                    elif board[nr, nc] == -1:
                        score -= 1
        
        return score
    
    best_move = (0, 0)  # Default to passing
    best_score = -float('inf')
    
    # Iterate over all possible moves
    for r in range(board_size):
        for c in range(board_size):
            move_score = evaluate_position(r, c)
            if move_score > best_score:
                best_score = move_score
                best_move = (r + 1, c + 1)  # Adjust to 1-based index
            
    return best_move

My strategy for this policy is to implement a search-based evaluation using the Negamax algorithm with Alpha-Beta pruning, adhering to a fixed-depth limit to respect the 1-second execution time constraint.

**Strategy Breakdown:**
1.  **Board Representation:** I will map the input `pieces` dictionary into a 64-element array representing the 8x8 board. This allows for efficient move generation and evaluation using lookup tables.
2.  **Move Generation:** I will write a dedicated move generator that calculates pseudo-legal moves based on piece geometry (knight jumps, sliding vectors, pawn steps) and filter them for legality (checking if the king is left in check). I will handle en passant implicitly by detecting capture opportunities on empty squares (though without history, it's rare, but handled for completeness).
3.  **Evaluation Function:**
    -   **Material:** Standard piece values (Pawn=1, Knight=3, Bishop=3, Rook=5, Queen=9).
    -   **Positional Tables:** I will use Piece-Square Tables (PST) to reward centralization and proper development (e.g., Knights in the center, Pawns pushing).
    -   **Checkmate/Terminal Detection:** The highest priority is given to delivering checkmate (or forcing the opponent into a checkmate).
4.  **Search Algorithm (Negamax):**
    -   I will use Negamax with Alpha-Beta pruning.
    -   **Quiescence Search:** To mitigate the "horizon effect" (stopping search during a sequence of captures), I will extend the search on capture moves. This ensures we don't evaluate a position statically if a piece is about to be taken.
    -   **Iterative Deepening (Simulated):** Given the 1-second limit and the "complete file" requirement without external hooks, I will fix the search depth to a reasonable level (likely 3-4 ply with quiescence) that balances strength and performance within the time limit for standard Python execution.
    -   **Move Ordering:** I will sort moves to try "good" moves (captures, checks) first to maximize pruning efficiency.

This approach ensures a strong, tactical AI that prioritizes immediate threats and material gain while remaining within safe execution limits.

KNIGHT_OFFSETS = [-17, -15, -10, -6, 6, 10, 15, 17]

# Helper to convert algebraic to index
def sq_to_idx(sq):
    file = ord(sq[0]) - ord('a')
    rank = int(sq[1]) - 1
    return rank * 8 + file

# Helper to convert index to algebraic
def idx_to_sq(idx):
    rank = idx // 8
    file = idx % 8
    return chr(file + ord('a')) + str(rank + 1)

class ChessState:
    def __init__(self, pieces, to_play):
        self.board = [None] * 64
        self.turn = 1 if to_play == 'white' else -1
        self.kings = {'w': -1, 'b': -1}
        
        for sq, pc in pieces.items():
            idx = sq_to_idx(sq)
            self.board[idx] = pc
            if pc[1] == 'K':
                self.kings[pc[0]] = idx

    def is_square_attacked(self, square, color):
        """
        Check if 'square' is attacked by pieces of 'color'.
        'color' here is the color of the attacker (e.g., 'w' if checking if white attacks square).
        """
        # 1. Check attacks by Knights
        for offset in KNIGHT_OFFSETS:
            target = square + offset
            if 0 <= target < 64:
                # Check boundaries (wrapping prevention)
                s_file = square % 8
                t_file = target % 8
                if abs(s_file - t_file) > 2: continue
                
                p = self.board[target]
                if p and p[0] == color and p[1] == 'N':
                    return True

        # 2. Check attacks by Sliding pieces (B, R, Q) and Pawns
        # We scan rays from the square outwards
        # Diagonals (B, Q)
        for offset in BISHOP_OFFSETS:
            curr = square
            while True:
                prev = curr
                curr += offset
                if not (0 <= curr < 64): break
                # Boundary checks
                p_file, c_file = prev % 8, curr % 8
                if abs(p_file - c_file) != 1: break
                
                p = self.board[curr]
                if p:
                    if p[0] == color and (p[1] == 'B' or p[1] == 'Q'): return True
                    break
        
        # Orthogonals (R, Q)
        for offset in ROOK_OFFSETS:
            curr = square
            while True:
                prev = curr
                curr += offset
                if not (0 <= curr < 64): break
                # Boundary checks (horizontal wrap)
                if offset in [-1, 1]:
                    p_file, c_file = prev % 8, curr % 8
                    if abs(p_file - c_file) != 1: break
                
                p = self.board[curr]
                if p:
                    if p[0] == color and (p[1] == 'R' or p[1] == 'Q'): return True
                    break

        # 3. Check attacks by Pawns
        # White pawns attack up-left (7) and up-right (9)
        # Black pawns attack down-left (-9) and down-right (-7)
        # Note: Attacking 'square' means checking if a pawn is positioned such that it can move to 'square'
        # If we are checking if 'color' attacks 'square', we look for the pawn of 'color' at positions that attack 'square'.
        if color == 'w':
            # White pawn would be at square - 7 or square - 9
            targets = [square - 7, square - 9]
            for t in targets:
                if 0 <= t < 64:
                    p = self.board[t]
                    if p and p[0] == 'w' and p[1] == 'P':
                        # Check boundary wrap
                        if abs((t % 8) - (square % 8)) == 1:
                            return True
        else:
            # Black pawn would be at square + 7 or square + 9
            targets = [square + 7, square + 9]
            for t in targets:
                if 0 <= t < 64:
                    p = self.board[t]
                    if p and p[0] == 'b' and p[1] == 'P':
                        if abs((t % 8) - (square % 8)) == 1:
                            return True

        # 4. Check King attacks (adjacent squares)
        for offset in KING_OFFSETS:
            target = square + offset
            if 0 <= target < 64:
                s_file = square % 8
                t_file = target % 8
                if abs(s_file - t_file) > 1: continue
                
                p = self.board[target]
                if p and p[0] == color and p[1] == 'K':
                    return True

        return False

    def get_moves(self):

    def evaluate(self):
        # White perspective
        score = 0
        
        # Material and Position
        for i in range(64):
            p = self.board[i]
            if p:
                val = PIECE_VALUES[p[1]]
                pst_val = PST[p[1]][i if p[0] == 'w' else (63 - i)]
                
                if p[0] == 'w':
                    score += val + pst_val
                else:
                    score -= val + pst_val
        
        # Checkmate detection
        # If no legal moves, it's mate or stalemate
        # We can't easily detect that here without generating all moves.
        # Instead, we handle it in the search recursion.
        
        return score if self.turn == 1 else -score

    def make_move(self, move_str):
        # Create a new state with the move applied
        new_pieces = {}
        # Map current board to dict
        for i, p in enumerate(self.board):
            if p:
                new_pieces[idx_to_sq(i)] = p
        
        # Decode move
        src = move_str[:2]
        dst = move_str[2:4]
        
        # Handle promotion
        promo = None
        if len(move_str) == 5:
            promo = move_str[4].upper()
        
        # Update dict
        moving_piece = new_pieces.pop(src)
        if promo:
            moving_piece = moving_pieces[0] + promo # Color + Promotion piece
            
        new_pieces[dst] = moving_piece
        
        # En Passant removal (if applicable)
        # Implicitly handled if we assume captures are clean. 
        # However, if dst is empty in dict but we moved a pawn diagonally, it's EP.
        # Since we generate moves from board indices, let's just use board logic for the next state.
        # Actually, cleaner to just rebuild from board state.
        
        return ChessState(new_pieces, 'black' if self.turn == 1 else 'white')

# Negamax Search
MAX_DEPTH = 3 # Fixed depth for 1s execution time in Python
INF = 100000

def negamax(state, depth, alpha, beta):
    # Base case: depth limit
    # Quiescence search is implicitly handled by extending depth on captures? 
    # For simplicity and speed, we'll just stick to fixed depth + basic eval, 
    # but we will order moves to try captures first.
    
    if depth == 0:
        return state.evaluate()

    moves = state.get_moves()
    
    # Terminal node (Checkmate/Stalemate)
    if not moves:
        # Check if in check
        my_color = 'w' if state.turn == 1 else 'b'
        opp_color = 'b' if state.turn == 1 else 'w'
        king_pos = state.kings[my_color]
        
        if state.is_square_attacked(king_pos, opp_color):
            # Checkmate: very bad
            return -INF + (MAX_DEPTH - depth) # Prefer faster mates
        else:
            # Stalemate: draw
            return 0

    # Move Ordering: Prioritize captures
    def move_score(m):
        s_sq = sq_to_idx(m[:2])
        d_sq = sq_to_idx(m[2:4])
        # Score based on MVV-LVA (Most Valuable Victim - Least Valuable Aggressor) roughly
        target = state.board[d_sq]
        attacker = state.board[s_sq]
        score = 0
        if target:
            score = PIECE_VALUES[target[1]] * 10 - PIECE_VALUES[attacker[1]]
        return score

    moves.sort(key=move_score, reverse=True)
    
    max_score = -INF
    
    # Pruning
    for move in moves:
        # Simulate Move
        # We need a quick state transition or modify board in place
        # For simplicity, we use in-place modification and revert
        
        s_sq = sq_to_idx(move[:2])
        d_sq = sq_to_idx(move[2:4])
        promo = len(move) == 5
        
        old_src = state.board[s_sq]
        old_dst = state.board[d_sq]
        old_king = state.kings['w' if state.turn == 1 else 'b']
        
        # Apply
        state.board[s_sq] = None
        if promo:
            state.board[d_sq] = ('w' if state.turn == 1 else 'b') + move[4].upper()
        else:
            state.board[d_sq] = old_src
            
        # Update King
        if old_src[1] == 'K':
            state.kings['w' if state.turn == 1 else 'b'] = d_sq
            
        # Flip Turn
        state.turn = -state.turn
        
        score = -negamax(state, depth - 1, -beta, -alpha)
        
        # Revert
        state.turn = -state.turn
        state.board[s_sq] = old_src
        state.board[d_sq] = old_dst
        state.kings['w' if state.turn == 1 else 'b'] = old_king
        
        if score > max_score:
            max_score = score
        
        if max_score > alpha:
            alpha = max_score
            
        if alpha >= beta:
            break # Beta cutoff
            
    return max_score

def policy(pieces: dict[str, str], to_play: str) -> str:
    # 1. Initialize State
    state = ChessState(pieces, to_play)
    
    # 2. Get Legal Moves
    legal_moves = state.get_moves()
    
    if not legal_moves:
        # Should not happen in arena unless mate, but handle gracefully
        return "0000" 
        
    # 3. Search
    best_move = legal_moves[0]
    best_value = -INF
    
    # Move ordering for root
    def root_score(m):
        s_sq = sq_to_idx(m[:2])
        d_sq = sq_to_idx(m[2:4])
        target = state.board[d_sq]
        attacker = state.board[s_sq]
        score = 0
        if target:
            score = PIECE_VALUES[target[1]] * 10 - PIECE_VALUES[attacker[1]]
        return score
        
    legal_moves.sort(key=root_score, reverse=True)
    
    # Iterate
    alpha = -INF
    beta = INF
    
    for move in legal_moves:
        # Apply Move
        s_sq = sq_to_idx(move[:2])
        d_sq = sq_to_idx(move[2:4])
        promo = len(move) == 5
        
        old_src = state.board[s_sq]
        old_dst = state.board[d_sq]
        old_king = state.kings['w' if state.turn == 1 else 'b']
        
        state.board[s_sq] = None
        if promo:
            state.board[d_sq] = ('w' if state.turn == 1 else 'b') + move[4].upper()
        else:
            state.board[d_sq] = old_src
            
        if old_src[1] == 'K':
            state.kings['w' if state.turn == 1 else 'b'] = d_sq
            
        state.turn = -state.turn
        
        # Search
        val = -negamax(state, MAX_DEPTH - 1, -beta, -alpha)
        
        # Revert
        state.turn = -state.turn
        state.board[s_sq] = old_src
        state.board[d_sq] = old_dst
        state.kings['w' if state.turn == 1 else 'b'] = old_king
        
        if val > best_value:
            best_value = val
            best_move = move
            
        if val > alpha:
            alpha = val
            
    return best_move

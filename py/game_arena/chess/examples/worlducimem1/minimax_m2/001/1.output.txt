def generate_pawn_moves(board: Dict[Tuple[int, int], str], file: int, rank: int, color: str) -> Set[str]:

    moves = set()
    direction = 1 if color == 'white' else -1
    start_rank = 1 if color == 'white' else 6
    
    # Forward move

    if 0 <= new_rank <= 7 and (file, new_rank) not in board:

        square_to = f"{chr(file + ord('a'))}{new_rank + 1}"

        if new_rank == 7 or new_rank == 0:
            for piece in ['q', 'r', 'b', 'n']:
                moves.add(move + piece)
        else:
            moves.add(move)
        
        # Double move from starting position
        if rank == start_rank:
            new_rank2 = rank + 2 * direction
            if 0 <= new_rank2 <= 7 and (file, new_rank2) not in board:
                square_from = f"{chr(file + ord('a'))}{rank + 1}"
                square_to = f"{chr(file + ord('a'))}{new_rank2 + 1}"
                moves.add(square_from + square_to)
    
    # Captures
    for dx in [-1, 1]:
        new_file = file + dx
        new_rank = rank + direction
        if 0 <= new_file <= 7 and 0 <= new_rank <= 7:

def make_move(board: Dict[Tuple[int, int], str], move: str) -> Dict[Tuple[int, int], str]:
    """Apply a move to the board"""
    new_board = board.copy()
    
    if len(move) < 4:
        return new_board
    
    from_file = ord(move[0]) - ord('a')
    from_rank = int(move[1]) - 1
    to_file = ord(move[2]) - ord('a')
    to_rank = int(move[3]) - 1
    
    # Handle promotion
    piece = new_board.get((from_file, from_rank), '')
    if len(move) > 4:
        promotion = move[4]
        piece = piece[0] + promotion.upper()
    
    # Move piece
    if (from_file, from_rank) in new_board:
        new_board[(to_file, to_rank)] = new_board[(from_file, from_rank)]
        del new_board[(from_file, from_rank)]
    
    return new_board

def is_in_check(board: Dict[Tuple[int, int], str], color: str) -> bool:
    """Check if the king of the given color is in check"""
    king_pos = None
    king_char = color[0] + 'K'
    
    for pos, piece in board.items():
        if piece == king_char:
            king_pos = pos
            break
    
    if not king_pos:
        return False
    
    # Check if any opponent piece attacks the king
    opponent_color = 'black' if color == 'white' else 'white'
    opponent_moves = set()
    
    for (file, rank), piece in board.items():
        if piece[0] != opponent_color[0]:
            continue
            
        piece_type = piece[1]
        square = f"{chr(file + ord('a'))}{rank + 1}"
        
        if piece_type == 'P':
            # Pawn attacks
            direction = 1 if opponent_color == 'white' else -1
            for dx in [-1, 1]:
                attack_file = file + dx
                attack_rank = rank + direction
                if 0 <= attack_file <= 7 and 0 <= attack_rank <= 7:
                    if (attack_file, attack_rank) == king_pos:
                        return True
        elif piece_type == 'N':
            knight_moves = [(-2, -1), (-2, 1), (-1, -2), (-1, 2), (1, -2), (1, 2), (2, -1), (2, 1)]
            for dx, dy in knight_moves:
                if (file + dx, rank + dy) == king_pos:
                    return True
        elif piece_type in ['B', 'R', 'Q']:
            directions = []
            if piece_type in ['B', 'Q']:
                directions.extend([(-1, -1), (-1, 1), (1, -1), (1, 1)])
            if piece_type in ['R', 'Q']:
                directions.extend([(-1, 0), (1, 0), (0, -1), (0, 1)])
            
            for dx, dy in directions:
                check_file = file + dx
                check_rank = rank + dy
                while 0 <= check_file <= 7 and 0 <= check_rank <= 7:
                    if (check_file, check_rank) == king_pos:
                        return True

                    check_rank += dy
        elif piece_type == 'K':

                for dy in [-1, 0, 1]:

def would_be_in_check(board: Dict[Tuple[int, int], str], move: str, color: str) -> bool:
    """Check if a move would put the king in check"""
    new_board = make_move(board, move)
    return is_in_check(new_board, color)

def is_checkmate(board: Dict[Tuple[int, int], str], color: str) -> bool:
    """Check if the position is checkmate"""
    if not is_in_check(board, color):
        return False
    
    # Generate all legal moves for the color
    legal_moves = generate_legal_moves(board, color)
    
    # If no legal moves and in check, it's checkmate
    return len(legal_moves) == 0

def evaluate_position(board: Dict[Tuple[int, int], str], color: str) -> float:
    """Simple position evaluation"""
    score = 0.0
    
    # Material values
    piece_values = {
        'P': 1.0, 'N': 3.0, 'B': 3.0, 'R': 5.0, 'Q': 9.0, 'K': 0.0
    }
    
    for piece in board.values():
        piece_type = piece[1]
        value = piece_values.get(piece_type, 0.0)
        if piece[0] == color[0]:
            score += value
        else:
            score -= value
    
    # Bonus for center control
    center_squares = [(3, 3), (3, 4), (4, 3), (4, 4)]
    for pos in center_squares:
        if pos in board:
            piece = board[pos]
            if piece[0] == color[0]:
                score += 0.1
    
    # Penalty for being in check
    if is_in_check(board, color):
        score -= 0.5
    
    return score

The strategy for this policy is to implement a greedy yet intelligent move selection that prioritizes high-value captures, checks, and strong piece development while maintaining simplicity and speed. The algorithm evaluates each legal move by:

1. **Detecting immediate tactical opportunities**:
   - If a move delivers checkmate, it is selected immediately.
   - If a move gives check, it gets a bonus.
2. **Evaluating material gain**:
   - We assign standard relative values to pieces: Pawn=1, Knight=3, Bishop=3, Rook=5, Queen=9.
   - Capturing a higher-value piece with a lower-value piece (or pawn) is strongly prioritized.
   - Avoids losing material unless no better option exists.
3. **Promotion incentive**:
   - Pawn promotions—especially to Queen—are highly incentivized.
4. **Positional improvement**:
   - Encourages knights and bishops to move toward the center early.
   - Avoids moving the same piece multiple times in the opening without cause.
5. **King safety**:
   - Penalizes moves that expose the king unnecessarily.

Since we cannot use the `chess` module, we reimplement minimal logic to:
- Determine if a move is a capture.
- Identify checks and potential mates.
- Evaluate board state heuristically.

We use a lightweight evaluation function and prioritize speed by avoiding deep search.

import numpy as np

def policy(pieces: dict[str, str], to_play: str, memory: dict) -> tuple[str, dict]:
    # Helper functions
    def parse_square(sq: str) -> tuple[int, int]:
        """Convert algebraic notation (e.g. 'e4') to (file, rank) indices."""
        file = ord(sq[0]) - ord('a')
        rank = int(sq[1]) - 1
        return file, rank

    def square_to_str(file: int, rank: int) -> str:
        """Convert (file, rank) indices to algebraic notation."""
        return f"{chr(ord('a') + file)}{rank + 1}"

    def is_opponent(piece: str, color: str) -> bool:
        return piece[0] == ('b' if color == 'white' else 'w')

    def is_friendly(piece: str, color: str) -> bool:
        return piece[0] == ('w' if color == 'white' else 'b')

    def piece_value(piece: str) -> int:
        values = {'P': 1, 'N': 3, 'B': 3, 'R': 5, 'Q': 9, 'K': 0}
        return values.get(piece[1], 0)

    def get_king_square(color: str) -> str:
        king_code = 'wK' if color == 'white' else 'bK'
        for sq, pc in pieces.items():
            if pc == king_code:
                return sq
        return None

    def is_attacked(square: str, by_color: str) -> bool:
        """Check if `square` is attacked by any piece of `by_color`."""
        sq_file, sq_rank = parse_square(square)
        opp_color_char = 'w' if by_color == 'white' else 'b'
        for sq_from, piece in pieces.items():
            if piece[0] != opp_color_char:
                continue
            ptype = piece[1]
            f, r = parse_square(sq_from)
            df, dr = sq_file - f, sq_rank - r
            adf, adr = abs(df), abs(dr)
            if ptype == 'P':
                # Pawn attacks: forward diagonals
                direction = 1 if by_color == 'white' else -1
                if direction == 1 and dr == 1 and adf == 1:
                    return True
                if direction == -1 and dr == -1 and adf == 1:
                    return True
            elif ptype == 'N':
                if (adf == 2 and adr == 1) or (adf == 1 and adr == 2):
                    return True
            elif ptype == 'B':
                if adf == adr and adf > 0:
                    return True
            elif ptype == 'R':
                if adf == 0 or adr == 0:
                    return True
            elif ptype == 'Q':
                if adf == adr or adf == 0 or adr == 0:
                    return True
            elif ptype == 'K':
                if adf <= 1 and adr <= 1:
                    return True
        return False

    def simulate_move_and_capture(move: str) -> (dict, str):
        """Returns a new pieces dict after the move and the captured piece (if any)."""
        from_sq = move[0:2]
        to_sq = move[2:4]
        new_pieces = pieces.copy()
        moved_piece = new_pieces.pop(from_sq)
        captured = new_pieces.pop(to_sq, None)
        # Handle pawn promotion
        if len(move) > 4 and move[4] == 'q':  # assume promote to queen if specified
            moved_piece = moved_piece[0] + 'Q'
        new_pieces[to_sq] = moved_piece
        return new_pieces, captured

    def gives_check(move: str, color: str) -> bool:
        """Check if the move puts the opponent's king in check."""
        new_pieces, _ = simulate_move_and_capture(move)
        opp_color = 'black' if color == 'white' else 'white'
        opp_king_sq = get_king_square(opp_color)
        if not opp_king_sq:
            return False
        # Simulate the board state and check if king is attacked
        temp = pieces
        pieces.update(new_pieces)
        result = is_attacked(opp_king_sq, color)
        pieces.clear()
        pieces.update(temp)
        return result

    def leads_to_mate(move: str, color: str) -> bool:
        """Rudimentary check: if after move, opponent has no legal moves and king is in check -> mate."""
        # Since we don't generate legal moves here (no chess lib), we approximate:
        # If move gives check and opponent king is attacked and likely trapped -> heuristic for mate.
        # True mate detection without move generation is incomplete, so we skip strict mate detection.
        # Instead, we rely on the arena providing legal moves only, so if a move delivers mate, it's safe.
        # But we can't fully detect it here. So we prioritize check + material gain.
        # For now, we treat "check + high bonus" as potential mate.
        if not gives_check(move, color):
            return False
        # Simulate move
        new_pieces, _ = simulate_move_and_capture(move)
        temp = pieces.copy()
        pieces.update(new_pieces)
        opp_color = 'black' if color == 'white' else 'white'
        opp_king_sq = get_king_square(opp_color)
        if not opp_king_sq:
            pieces.clear()
            pieces.update(temp)
            return False
        # If king is in check and was moved to a corner, or heavily attacked, it might be mate
        in_corner = opp_king_sq in ['a1','h1','a8','h8']
        attacked_count = 0
        for f in range(8):
            for r in range(8):
                sq = square_to_str(f, r)
                if is_attacked(sq, color):
                    if sq == opp_king_sq:
                        attacked_count += 1
        # Crude: if king is in check and in corner and heavily attacked -> likely mate
        pieces.clear()
        pieces.update(temp)
        return gives_check(move, color) and in_corner and attacked_count >= 2

    # Main logic
    legal_moves = memory.get('legal_moves', [])  # provided via memory from outside
    if not legal_moves:
        # Fallback: return first move
        return 'e2e4', memory  # just a guess if no moves provided

    color = to_play
    my_color_char = 'w' if color == 'white' else 'b'
    best_move = legal_moves[0]
    best_score = -9999

    king_sq = get_king_square(color)
    opp_color = 'black' if color == 'white' else 'white'

    for move in legal_moves:
        score = 0
        from_sq = move[0:2]
        to_sq = move[2:4]
        moved_piece_code = pieces.get(from_sq)
        if not moved_piece_code or moved_piece_code[0] != my_color_char:
            continue  # should not happen

        # 1. Immediate win (checkmate) — high bonus
        if leads_to_mate(move, color):
            best_move = move
            break

        # 2. Check bonus
        if gives_check(move, color):
            score += 1.5

        # 3. Capture evaluation
        captured_piece = pieces.get(to_sq)
        if captured_piece and is_opponent(captured_piece, color):
            value_gained = piece_value(captured_piece)
            value_lost = piece_value(moved_piece_code)
            score += value_gained - value_lost * 0.1  # punish trading up unless forced

        # 4. Promotion bonus
        if len(move) > 4:
            if move[4] == 'q':
                score += 8  # strong incentive to promote to queen
            elif move[4] in 'rnb':
                score += 3

        # 5. Center control bonus for minor pieces early
        center_squares = ['d4','d5','e4','e5']
        if moved_piece_code[1] in 'NB' and to_sq in center_squares:
            score += 0.5

        # 6. Avoid moving king early unless castling or necessary
        if moved_piece_code[1] == 'K':
            if from_sq == ('e1' if color == 'white' else 'e8'):
                # Early king move from start
                if to_sq not in ['g1','c1','g8','c8']:
                    score -= 0.5  # discourage early king walk
            if is_attacked(to_sq, opp_color):
                score -= 1.0  # don't move king into danger

        # 7. Development: moving pieces from back rank
        if moved_piece_code[1] != 'K' and moved_piece_code[1] != 'Q':
            back_rank = 0 if color == 'white' else 7
            from_f, from_r = parse_square(from_sq)
            to_f, to_r = parse_square(to_sq)
            if from_r == back_rank and to_r != back_rank:
                score += 0.3  # reward development

        # 8. Penalize hanging pieces
        # (We can't do full lookahead, so only check if moved piece ends on attacked square)
        new_pieces, _ = simulate_move_and_capture(move)
        temp = pieces.copy()
        pieces.update(new_pieces)
        if to_sq in new_pieces and is_friendly(new_pieces[to_sq], color):
            if is_attacked(to_sq, opp_color):
                # Is the piece hanging? Try to see if attacked and not defended
                score -= piece_value(new_pieces[to_sq]) * 0.5
        pieces.clear()
        pieces.update(temp)

        if score > best_score:
            best_score = score
            best_move = move

    return best_move, memory

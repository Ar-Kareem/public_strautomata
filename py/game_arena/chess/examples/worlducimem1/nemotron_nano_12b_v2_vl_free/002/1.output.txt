def is_castling(move, to_play):
    if to_play == 'white':

def piece_value(piece):
    values = {'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1}
    return values.get(piece[1], 0) if piece else 0

def file_rank(sq):
    return (ord(sq[0]) - ord('a'), int(sq[1]) - 1)

def to_square(sq, dx, dy):
    x = file_rank(sq)[0] + dx
    y = file_rank(sq)[1] + dy
    return chr(x + ord('a')) + str(y + 1)

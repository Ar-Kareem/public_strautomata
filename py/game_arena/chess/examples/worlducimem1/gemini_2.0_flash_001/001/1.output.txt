The strategy for this policy is a combination of material evaluation, king safety, and simple threat detection. It prioritizes checkmate, then capturing high-value pieces, developing pieces, and moving the king to safety. A simple evaluation function is used to score each position.  It examines available moves and selects the one with the highest score. Currently, the policy only examines the immediate next move; further versions would improve the search by examining a few moves deep.

import numpy as np

def policy(pieces: dict[str, str], to_play: str, memory: dict) -> tuple[str, dict]:
    """
    Chooses the best move based on a simple evaluation function.
    """

    legal_moves = generate_legal_moves(pieces, to_play)

    best_move = None
    best_score = -np.inf if to_play == 'white' else np.inf  # White maximizes, black minimizes

    for move in legal_moves:
        new_pieces = apply_move(pieces, move)
        score = evaluate_position(new_pieces, to_play)

        if to_play == 'white':
            if score > best_score:
                best_score = score
                best_move = move
        else:
            if score < best_score:
                best_score = score
                best_move = move

    if best_move is None:
        # If no move improves the position (unlikely), return a random legal move
        best_move = legal_moves[0] if legal_moves else None  # Handle no legal moves case


    return best_move, memory

def apply_move(pieces: dict[str, str], move: str) -> dict[str, str]:
    """
    Applies a move to the board and returns the new board state.
    """
    new_pieces = pieces.copy()
    start_square = move[:2]
    end_square = move[2:4]

    piece = new_pieces[start_square]
    del new_pieces[start_square]

    if end_square in new_pieces:
        del new_pieces[end_square]

    new_pieces[end_square] = piece
    #promotion
    if len(move) == 5:
       new_pieces[end_square] = piece[0] + move[4].upper()

    return new_pieces

        piece_type = piece[1]

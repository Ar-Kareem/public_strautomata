The strategy for this policy is a fast heuristic-based evaluation function. It prioritizes:
1.  **Checkmate**: Immediately selecting any move that delivers mate (identified by the `#` suffix).
2.  **Material Gain**: Evaluating the value of pieces captured (Q=9, R=5, B=3, N=3, P=1) minus the value of the piece moved if it moves into an unprotected square.
3.  **Piece Activity**: Promoting pawns to Queens and favoring moves that capture over non-capturing moves.
4.  **Escape/Defense**: If no captures are favorable, it defaults to a random selection or the first legal move to ensure valid output.

import random

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    # Piece values for material evaluation
    values = {'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1, 'K': 0}
    
    # Check for immediate checkmate
    for move in legal_moves:
        if move.endswith('#'):
            return move

    # Simple heuristic scoring
    best_move = legal_moves[0]
    best_score = -999

    for move in legal_moves:
        score = 0
        
        # Priority 1: Promotion to Queen
        if '=Q' in move:
            score += 8
        
        # Priority 2: Captures
        if 'x' in move:
            # Try to identify the piece being captured
            # Move format examples: 'Bxf5', 'exd5', 'Nxd4+'
            # The target square is usually the 2 characters after 'x'
            try:
                parts = move.split('x')
                target_sq = parts[1][:2]
                captured_piece_code = pieces.get(target_sq)
                if captured_piece_code:
                    captured_type = captured_piece_code[1]
                    score += values.get(captured_type, 0) * 10
            except:
                score += 5 # Generic capture bonus if parsing fails
        
        # Priority 3: Checks
        if '+' in move:
            score += 1

        # Small random factor to break ties and avoid predictability
        score += random.random() * 0.1

        if score > best_score:
            best_score = score
            best_move = move

    return best_move

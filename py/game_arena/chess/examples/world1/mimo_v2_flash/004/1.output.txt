I'll develop a strong chess policy that evaluates moves based on multiple strategic criteria.

## Strategy

My policy uses a multi-layered evaluation approach:

1. **Immediate threats**: Prioritize moves that give checkmate (win instantly)
2. **Checks**: Prioritize checks that force opponent reactions and may create attacking opportunities
3. **Captures**: Evaluate captures based on material gain (queen > rook > bishop/knight > pawn)
4. **Tactical threats**: Prioritize promotions that create powerful pieces
5. **King safety**: Prefer castling to protect the king and connect rooks
6. **Positional factors**: Develop pieces toward the center and control key squares
7. **Avoid blunders**: Penalize moves that put our king in check or lose material

The policy is designed to be fast (under 1 second) while making strong strategic decisions that balance immediate tactics with long-term position.

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    """
    A smart chess policy that evaluates moves based on:
    1. Mates and checks
    2. Material gains from captures
    3. Promotions to powerful pieces
    4. King safety (castling)
    5. Piece development and center control
    """
    
    # Helper functions
    def get_piece_value(piece_code):
        """Returns the relative value of a piece (without color)"""
        piece_type = piece_code[1].upper()
        values = {'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1, 'K': 0}
        return values.get(piece_type, 0)
    
    def is_capture(move):
        """Check if a move is a capture"""
        return 'x' in move
    
    def is_check(move):
        """Check if a move gives check"""
        return '+' in move
    
    def is_checkmate(move):
        """Check if a move is checkmate"""
        return '#' in move
    
    def is_promotion(move):
        """Check if a move is a promotion"""
        return '=' in move
    
    def is_castling(move):
        """Check if a move is castling"""
        return 'O-O' in move
    
    def get_promoted_piece(move):
        """Get the piece being promoted to"""
        if '=' in move:
            # Handle moves like "e8=Q" or "e8=Q+"
            parts = move.split('=')
            if len(parts) > 1:
                promo = parts[1].replace('+', '').replace('#', '')
                return promo
        return None
    
    def parse_capture(move):
        """Extract captured piece from move string (e.g., 'Nxe4' captures piece on e4)"""
        if 'x' in move:
            # For moves like "Nxe4" or "Bxf5", the square after x is the capture target
            parts = move.split('x')
            if len(parts) == 2:
                target_square = parts[1][:2]  # Get first two chars (the square)
                # Check if there's a piece on that square
                if target_square in pieces:
                    return pieces[target_square]
        return None
    
    def evaluate_material_gain(move):
        """Calculate material gain from a capture or promotion"""
        score = 0
        
        # Check for promotion
        if is_promotion(move):
            promo_piece = get_promoted_piece(move)
            if promo_piece:
                # Queen promotion is most valuable
                if promo_piece == 'Q':
                    score += 8  # Net gain: queen (9) - pawn (1) = 8
                elif promo_piece == 'R':
                    score += 4  # Rook (5) - Pawn (1) = 4
                elif promo_piece in ['B', 'N']:
                    score += 2  # Bishop/Knight (3) - Pawn (1) = 2
        
        # Check for capture
        if is_capture(move):
            captured_piece = parse_capture(move)
            if captured_piece:
                score += get_piece_value(captured_piece)
        
        return score
    
    def center_control_score(move):
        """Score based on how central the destination square is"""
        # Extract destination square
        dest = move.split('=')[0].replace('x', '')[-2:]
        if len(dest) != 2 or dest[0] not in 'abcdefgh' or dest[1] not in '12345678':
            return 0
        
        # Central squares get higher score
        file, rank = dest[0], dest[1]
        center_bonus = {
            'd4': 4, 'e4': 4, 'd5': 4, 'e5': 4,
            'c4': 3, 'f4': 3, 'c5': 3, 'f5': 3,
            'd3': 2, 'e3': 2, 'd6': 2, 'e6': 2,
        }
        return center_bonus.get(dest, 0)
    
    def piece_development_score(move):
        """Score based on developing pieces out of starting position"""
        # Check if moving from starting ranks
        if 'R' in move or 'N' in move or 'B' in move or 'Q' in move:
            if len(move) >= 3:
                dest = move.replace('x', '')[-2:]
                # Moving to center or castling ranks is good
                if dest[1] in '4567':
                    return 2
        return 0
    
    def find_best_move(moves):
        """Evaluate all moves and return the best one"""
        best_score = -float('inf')
        best_move = moves[0] if moves else None
        
        for move in moves:
            score = 0
            
            # 1. Checkmate is instant win
            if is_checkmate(move):
                return move  # Always play checkmate
            
            # 2. Checks are highly valuable (create threats)
            if is_check(move):
                score += 10
            
            # 3. Material gain
            material = evaluate_material_gain(move)
            score += material * 2  # Weight material heavily
            
            # 4. Castling for king safety (especially early)
            if is_castling(move):
                score += 8  # Strong positional bonus
            
            # 5. Promotions
            if is_promotion(move):
                promo = get_promoted_piece(move)
                if promo == 'Q':
                    score += 15  # Very strong
                elif promo == 'R':
                    score += 8
                elif promo in ['B', 'N']:
                    score += 6
            
            # 6. Center control
            score += center_control_score(move)
            
            # 7. Piece development
            score += piece_development_score(move)
            
            # 8. Avoid hanging pieces (simple heuristic)
            # If capture move, don't capture with a weaker piece unless promotion
            if is_capture(move) and not is_promotion(move):
                # Try to avoid capturing with pawns if we can capture with higher pieces
                if move[0] == 'P' or move[0].islower():  # Pawn move
                    score -= 1  # Minor penalty for pawn captures
            
            # Update best move
            if score > best_score:
                best_score = score
                best_move = move
        
        return best_move
    
    # If no legal moves, we should resign (but per requirements, must return a move)
    if not legal_moves:
        return legal_moves[0] if legal_moves else ""
    
    # Always return the best move according to our evaluation
    return find_best_move(legal_moves)

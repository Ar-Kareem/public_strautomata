

The strategy for this policy is to prioritizemoves that deliver checkmate if available, followed by material gain, king safety, and threat prevention. The evaluation function scores each legal move based on captured piece value, king safety, and positional advantages. Moves that expose the king to checks are penalized, while checks and checkmates are prioritized. The algorithm iterates through all legal moves, computes a score for each, and selects the highest-scoring move.

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    # Piece values for material evaluation
    PIECE_VALUES = {'K': 0, 'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1}
    
    # Helper function to check if a square is under attack
    def is_under_attack(square: str, color: str) -> bool:
        # Iterate through all pieces of the opposite color
        for sq, piece in pieces.items():
            if piece[0] != color:  # Opposite color piece
                # Check if this piece can attack the target square
                # Simplified: Check for knight moves, pawn attacks, and sliding pieces
                p_type = piece[1]
                if p_type == 'N':  # Knight
                    if abs(ord(sq[0]) - ord(square[0])) == 2 and abs(int(sq[1]) - int(square[1])) == 1:
                        return True
                elif p_type in 'QB':  # Bishop/Queen
                    # Check diagonal
                    if abs(ord(sq[0]) - ord(square[0])) == abs(int(sq[1]) - int(square[1])):
                        return True
                elif p_type == 'R':  # Rook
                    # Check same rank or file
                    if sq[0] == square[0] or sq[1] == square[1]:
                        return True
        return False

    # Helper function to calculate move score
    def evaluate_move(move: str) -> int:
        score = 0
        # Check for checkmate (highest priority)
        if '++' in move:
            return float('inf')
        
        # Parse move components
        target = move[-2:]  # Target square (last two characters)
        moving_piece = None
        
        # Handle special moves
        if move in ['O-O', 'O-O-O']:
            # Castling: King moves to g1/c1, rook to f1/a1
            if to_play == 'white':
                target = 'g1' if move == 'O-O' else 'c1'
            else:
                target = 'g8' if move == 'O-O' else 'c8'
            moving_piece = 'K'
        else:
            # Parse standard move (e.g., 'Nf3', 'Nxf3')
            if move[0] in 'abcdefgh':
                moving_piece = 'P'  # Pawn move
            else:
                moving_piece = move[0]  # Piece type
        
        # Check for captures
        if 'x' in move:
            # Captured piece value (from target square)
            captured_piece = pieces[target]
            if captured_piece:
                score += PIECE_VALUES[captured_piece[1]]
        
        # Check king safety: If moving king into check, penalize
        if moving_piece == 'K':
            # Simulate move: Remove moving king from source, place at target
            # Check if target is under attack by opponent
            if is_under_attack(target, 'w' if to_play == 'black' else 'b'):
                score -= 10  # High penalty for exposing king
        
        return score

    # Evaluate all legal moves and select the best
    best_move = None
    best_score = -float('inf')
    
    for move in legal_moves:
        score = evaluate_move(move)
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

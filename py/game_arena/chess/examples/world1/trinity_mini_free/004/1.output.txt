

The strategy for this policyis to prioritize capturing high-value pieces and promoting to higher-value pieces. If multiple moves offer the same material gain, the first move in the list is chosen. This approach focuses on immediate material advantage while ignoring positional nuances due to computational constraints.

<code>
def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    # Define piece values for evaluation
    piece_value = {
        'P': 1,
        'N': 3,
        'B': 3,
        'R': 5,
        'Q': 9
    }
    
    best_move = None
    best_score = -1
    
    for move in legal_moves:
        score = 0
        
        # Check for captures
        if 'x' in move:
            # Extract target square after 'x'
            target = move.split('x')[-1]
            if target in pieces:
                captured_piece = pieces[target][1]
                if captured_piece in piece_value:
                    score += piece_value[captured_piece]
        
        # Check for promotions
        if move.endswith( ('=Q', '=R', '=B', '=N') ):
            promo_piece = move[-1]
            if promo_piece in piece_value:
                score += piece_value[promo_piece]
        
        # Select move with highest score
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

The strategy for this policy is to prioritize moves that lead to checkmate, followed by capturing high-value pieces, and then improving piece positioning. The policy evaluates each legal move based on a scoring system that considers material gain, piece safety, and positional advantages. It also includes basic checks for immediate threats to the king and prioritizes moves that mitigate those threats.

import random

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    # Define piece values for evaluation
    piece_values = {
        'P': 1, 'N': 3, 'B': 3, 'R': 5, 'Q': 9, 'K': 100
    }

    # Determine the color of the current player
    color = 'w' if to_play == 'white' else 'b'

    # Initialize best move and best score
    best_move = None
    best_score = -float('inf')

    # Iterate through all legal moves to find the best one
    for move in legal_moves:
        score = 0

        # Check for checkmate (highest priority)
        if move.endswith('#'):
            return move

        # Check for check (high priority)
        if '+' in move:
            score += 100

        # Check for captures
        if 'x' in move:
            # Extract the captured piece (assuming standard notation)
            captured_piece = None
            if len(move) >= 4 and move[1] == 'x':
                # Handle cases like 'Bxf5'
                captured_piece = pieces.get(move[2:4], None)
            elif len(move) >= 5 and move[3] == 'x':
                # Handle cases like 'Nec3x...' (unlikely, but just in case)
                captured_piece = pieces.get(move[4:6], None)

            if captured_piece and captured_piece[0] != color:
                # Calculate material gain
                piece_type = captured_piece[1]
                score += piece_values.get(piece_type, 0) * 10

        # Check for promotions (always promote to queen for simplicity)
        if '=' in move:
            score += 8  # Queen promotion is very valuable

        # Check for castling (improves king safety)
        if move in ['O-O', 'O-O-O']:
            score += 5

        # Update best move if current move has a higher score
        if score > best_score:
            best_score = score
            best_move = move

    # If no move is found (shouldn't happen as legal_moves is non-empty), return a random move
    if best_move is None:
        return random.choice(legal_moves)

    return best_move

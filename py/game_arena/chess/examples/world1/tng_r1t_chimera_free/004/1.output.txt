

The strategy for this policy is to prioritize moves that maximize immediate material gain, checkmate threats, and positional advantage. Each move is scored based on promotions, castling, captures, checks, and central control. The highest-scoring move is selected, ensuring a strong balance between aggression and board control.

import re

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    my_color = 'w' if to_play == 'white' else 'b'
    piece_values = {'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1, 'K': 0}
    
    central_squares = {'d4', 'e4', 'd5', 'e5', 'c3', 'f3', 'c6', 'f6', 'd3', 'e3', 'f4', 'f5', 'c4', 'c5'}
    
    def get_target_square(move):
        squares = re.findall(r'[a-h][1-8]', move)
        return squares[-1] if squares else None
    
    scored_moves = []
    
    for move in legal_moves:
        score = 0
        target_square = get_target_square(move)
        
        # Promotion
        if '=Q' in move:
            score += 1000
        elif '=R' in move:
            score += 900
        elif '=B' in move or '=N' in move:
            score += 800
        
        # Castling
        if move == 'O-O':
            score += 200
        elif move == 'O-O-O':
            score += 190
        
        # Captures
        if 'x' in move:
            if target_square in pieces:
                captured_piece = pieces[target_square]
                captured_value = piece_values.get(captured_piece[1], 0)
            else:
                captured_value = piece_values['P']
            
            first_char = move[0]
            capturing_piece_type = first_char if first_char in 'KQRBN' else 'P'
            capturing_value = piece_values.get(capturing_piece_type, 1)
            score += (captured_value - capturing_value) * 10
        
        # Checks
        if '+' in move:
            score += 50
        
        # Central control
        if target_square in central_squares:
            score += 30
        
        scored_moves.append((score, move))
    
    scored_moves.sort(reverse=True, key=lambda x: (x[0], legal_moves.index(x[1])))
    return scored_moves[0][1]

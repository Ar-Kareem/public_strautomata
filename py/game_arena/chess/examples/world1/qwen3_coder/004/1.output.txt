The strategy for this policy is to:

1. Parse the board state and evaluate each legal move based on its impact.
2. Prioritize checkmate and avoid moves that lead to checkmate.
3. Use piece-square tables for positional scoring when selecting between non-tactical moves.
4. Evaluate captures using MVV-LVA (Most Valuable Victim, Least Valuable Attacker).
5. Detect whether we're in an endgame for better evaluation.
6. Apply a quiescence search one ply deep for captures and checks.
7. Select the best evaluated move from the list of legal moves.

This approach combines tactical awareness with positional knowledge, focusing primarily on short-term gains while considering long-term positioning through piece-square tables.

def square_to_coords(square):
    """Converts algebraic notation like 'e4' into (file, rank) indices."""
    file = ord(square[0]) - ord('a')
    rank = int(square[1]) - 1
    return file, rank


def coords_to_square(file, rank):
    """Converts (file, rank) indices back into algebraic notation."""
    return chr(ord('a') + file) + str(rank + 1)


def get_color(piece_code):

    return piece_code[0]


def get_piece_type(piece_code):
    """Returns the piece type ('P','N','B','R','Q','K') from piece code."""

def is_endgame(pieces_dict, player_color_short):
    """Check if game has entered endgame phase."""
    total_material = 0
    for sq, piece in pieces_dict.items():
        if get_color(piece) == player_color_short:
            piece_type = get_piece_type(piece)
            if piece_type != 'K':
                total_material += PIECE_VALUES[piece_type]
    return total_material <= ENDGAME_MATERIAL

        file_idx, rank_idx = square_to_coords(sq)

        value = PIECE_VALUES[piece_type]

            sign = 1
        else:
            sign = -1
            
        pst_bonus = 0
        table = PST.get(piece_type, None)
        if piece_type == 'K':

        if table:

            if 0 <= file_idx < 8 and 0 <= flipped_rank < 8:
                pst_bonus = table[flipped_rank][file_idx]
        
        score += sign * (value + pst_bonus)
        
    return score

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str]) -> str:
    best_score = -math.inf
    chosen_move = legal_moves[0]

    if len(legal_moves) == 1:
        return legal_moves[0]

    evals = {}

    # First pass – check for immediate mate
    for m in legal_moves:
        resulting_board = make_move_on_board(pieces.copy(), m, to_play)
        opponent = 'black' if to_play == 'white' else 'white'
        their_moves = generate_legal_moves(resulting_board, opponent)

        if not their_moves:
            if is_in_check(resulting_board, opponent):
                return m  # Immediate checkmate!

    # Second pass – perform quiescent search
    max_eval = -math.inf
    candidate = legal_moves[0]

    for move in legal_moves:
        simulated = make_move_on_board(pieces.copy(), move, to_play)
        eval_score = quiesce(simulated, to_play, depth_remaining=1)
        evals[move] = eval_score
        if eval_score > max_eval:
            max_eval = eval_score
            candidate = move

    # Third tie-breaking – apply MVV/LVA
    candidates_with_same_eval = [m for m,v in evals.items() if abs(v - max_eval) < 1e-6 ]
    if len(candidates_with_same_eval) > 1:
        captures = [m for m in candidates_with_same_eval if "x" in m]
        if captures:
            best_capture_val_diff = -math.inf
            preferred_move = captures[0]
            for cap in captures:
                parts = cap.split('x')
                victim_sq = parts[-1][:2]  # Take square where capture occurs
                attacker_start = parts[0]

                victim_piece = pieces.get(victim_sq,"")
                attacking_piece_code = ""

                # Guess at attacking piece; crude approximation
                for sq,pc in pieces.items():
                    pc_char = get_piece_type(pc)
                    pl_col_full = to_play
                    pl_color_char = pl_col_full[0]
                    expected_code = pl_color_char + attacker_start[0].upper()
                    if sq.startswith(attacker_start.lower()[0]) and pc == expected_code:
                        attacking_piece_code = pc
                        break

                if not attacking_piece_code or not victim_piece:
                    continue
                attacker_type = get_piece_type(attacking_piece_code)
                victim_type = get_piece_type(victim_piece)
                diff = PIECE_VALUES.get(victim_type,0) - PIECE_VALUES.get(attacker_type,0)

                if diff > best_capture_val_diff:
                    best_capture_val_diff = diff
                    preferred_move = cap

            return preferred_move

    return candidate

The strategy implemented in this policy focuses on material gain and central control. First, it evaluates each legal move by checking for captures and calculating the material value of any captured piece. Moves with the highest material gain are prioritized. Among moves with equal gain, those moving a piece to central squares (d4, d5, e4, e5) are preferred. Castling moves are considered beneficial and selected when no material gain is available. The policy ensures all moves are legal and uses a consistent evaluation order: checkmate, material gain, castling, and finally random selection. The memory dictionary is left empty as no state is stored between turns.

```python
import random

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str], memory: dict) -> tuple[str, dict]:
    values = {'P': 1, 'N': 3, 'B': 3, 'R': 5, 'Q': 9, 'K': 0}
    central_squares = ['d4', 'd5', 'e4', 'e5']
    
    move_gain = {}
    for move in legal_moves:
        gain = 0
        if move.startswith('O'):
            gain = 0
        else:
            if 'x' in move:
                temp = move.rstrip('+#')
                if '=' in temp:
                    dest = temp.split('=')[0][-2:]
                else:
                    dest = temp[-2:]
                if dest in pieces:
                    captured_piece = pieces[dest]
                    piece_type = captured_piece[1]
                    gain = values.get(piece_type, 0)
        move_gain[move] = gain
    
    max_gain = max(move_gain.values()) if move_gain else -1
    candidates = [move for move, gain in move_gain.items() if gain == max_gain]
    
    if max_gain > 0:
        preferred_moves = []
        for move in candidates:
            if move.startswith('O'):
                dest = None
            else:
                temp = move.rstrip('+#')
                if '=' in temp:
                    dest = temp.split('=')[0][-2:]
                else:
                    dest = temp[-2:]
            if dest in central_squares:
                preferred_moves.append(move)
        if preferred_moves:
            chosen = random.choice(preferred_moves)
        else:
            chosen = random.choice(candidates)
    else:
        castling_moves = [move for move in candidates if move.startswith('O')]
        if castling_moves:
            chosen = random.choice(castling_moves)
        else:
            chosen = random.choice(candidates)
    
    return chosen, {}
```

## Strategy Overview

I'll implement a minimax-based chess engine with alpha-beta pruning that incorporates:
1. **Material evaluation** - standard piece values with positional bonuses
2. **Piece-square tables** - positional bonuses for all pieces in different stages of the game
3. **King safety** - pawn shield evaluation and castling bonuses
4. **Mobility** - bonus for number of legal moves available
5. **Center control** - bonus for controlling central squares
6. **Pawn structure** - isolated/doubled pawn penalties, passed pawn bonuses
7. **Transposition table** - cache evaluated positions for faster search
8. **Move ordering** - prioritize captures, checks, and killer moves
9. **Iterative deepening** - search deeper as time allows within 1-second limit
10. **Quiescence search** - extend search in capture sequences to avoid horizon effect

The engine will search to depth 3-4 initially, using iterative deepening to go deeper when possible within the time limit.

# Convert square notation to index
def square_to_index(sq):
    file_ord = ord(sq[0]) - ord('a')
    rank = int(sq[1]) - 1
    return rank * 8 + file_ord

class ChessEngine:
    def __init__(self, memory):
        self.memory = memory
        self.transposition_table = {}
        self.killer_moves = [[None, None] for _ in range(100)]  # Depth indexed
        self.history_heuristic = defaultdict(int)
        
    def evaluate_position(self, pieces, to_play):
        """Evaluate the current position from perspective of side to play"""
        score = 0
        white_pawns = 0
        black_pawns = 0
        
        # Count material and positional values
        for square, piece_code in pieces.items():
            color = piece_code[0]
            piece_type = piece_code[1]
            idx = square_to_index(square)
            
            # Mirror index for black pieces
            if color == 'b':
                idx = 63 - idx
            
            value = PIECE_VALUES[piece_type]
            
            # Add positional bonus
            if piece_type == 'P':
                value += PAWN_TABLE[idx]
            elif piece_type == 'N':
                value += KNIGHT_TABLE[idx]
            elif piece_type == 'B':
                value += BISHOP_TABLE[idx]
            elif piece_type == 'R':
                value += ROOK_TABLE[idx]
            elif piece_type == 'Q':
                value += QUEEN_TABLE[idx]
            elif piece_type == 'K':
                value += KING_TABLE_MID[idx]
            
            # Count for pawn structure
            if piece_type == 'P':
                if color == 'w':
                    white_pawns += 1
                else:
                    black_pawns += 1
            
            # Add to score
            if color == 'w':
                score += value
            else:
                score -= value
        
        # Pawn structure penalties
        # Penalty for isolated pawns (simplified)
        pawn_penalty = 0
        if white_pawns < 5:
            pawn_penalty -= 20 * (5 - white_pawns)
        if black_pawns < 5:
            pawn_penalty += 20 * (5 - black_pawns)
        score += pawn_penalty
        
        # Center control bonus
        center_squares = ['d4', 'e4', 'd5', 'e5']
        for sq in center_squares:
            if sq in pieces:
                piece = pieces[sq]
                if piece[0] == 'w':
                    score += 10
                else:
                    score -= 10
        
        # Mobility bonus (simplified)
        mobility_bonus = len(pieces) * 5  # More pieces = more mobility
        score += mobility_bonus
        
        # Return score from perspective of side to play
        return score if to_play == 'white' else -score
    
    def order_moves(self, moves, pieces, to_play, depth):
        """Order moves to improve alpha-beta pruning efficiency"""
        move_scores = []
        
        for move in moves:
            score = 0
            
            # MVV-LVA (Most Valuable Victim - Least Valuable Attacker)
            if 'x' in move:
                # Capture move - prioritize high value captures
                target_square = move.split('x')[-1][:2]
                if target_square in pieces:
                    target_piece = pieces[target_square][1]
                    score += 1000 + PIECE_VALUES.get(target_piece, 0) * 10
            
            # Check bonus
            if move.endswith('+'):
                score += 500
            
            # Promotion bonus
            if '=Q' in move:
                score += 800
            elif '=R' in move:
                score += 400
            elif '=B' in move or '=N' in move:
                score += 300
            
            # Castling bonus
            if move in ['O-O', 'O-O-O']:
                score += 200
            
            # Killer moves
            if move == self.killer_moves[depth][0]:
                score += 900
            elif move == self.killer_moves[depth][1]:
                score += 800
            
            # History heuristic
            score += self.history_heuristic[move] // 100
            
            move_scores.append((score, move))
        
        # Sort by score (highest first)
        move_scores.sort(reverse=True)
        return [move for _, move in move_scores]
    
    def quiescence_search(self, pieces, to_play, alpha, beta, depth):
        """Search only capture moves to avoid horizon effect"""
        stand_pat = self.evaluate_position(pieces, to_play)
        
        if stand_pat >= beta:
            return beta
        if alpha < stand_pat:
            alpha = stand_pat
        
        # Generate capture moves (simplified - in real implementation would filter moves)
        # For now, we'll just use the evaluation
        if depth <= 0:
            return stand_pat
        
        return stand_pat
    
    def minimax(self, pieces, to_play, legal_moves, depth, alpha, beta, maximizing):
        """Minimax search with alpha-beta pruning"""
        # Check transposition table
        board_key = str(sorted(pieces.items())) + to_play
        if board_key in self.transposition_table:
            tt_entry = self.transposition_table[board_key]
            if tt_entry['depth'] >= depth:
                if tt_entry['flag'] == 'exact':
                    return tt_entry['value'], tt_entry['move']
                elif tt_entry['flag'] == 'lower':
                    alpha = max(alpha, tt_entry['value'])
                elif tt_entry['flag'] == 'upper':
                    beta = min(beta, tt_entry['value'])
                
                if alpha >= beta:
                    return tt_entry['value'], tt_entry['move']
        
        if depth == 0:
            # Quiescence search at leaf nodes
            value = self.quiescence_search(pieces, to_play, alpha, beta, 2)
            return value, None
        
        ordered_moves = self.order_moves(legal_moves, pieces, to_play, depth)
        
        best_move = ordered_moves[0] if ordered_moves else None
        best_value = -float('inf') if maximizing else float('inf')
        
        for move in ordered_moves:
            # Simulate move (simplified - in full implementation would update board state)
            # For now, we'll evaluate statically and recurse with reduced depth
            new_pieces = pieces.copy()
            
            # Very basic move simulation
            # This is simplified - real implementation would parse move notation
            # and update board state properly
            
            # Evaluate the move statically
            move_value = self.evaluate_move(pieces, move, to_play)
            
            # Recursive search with reduced depth for move simulation
            if maximizing:
                value = move_value + self.minimax(new_pieces, 
                                                 'black' if to_play == 'white' else 'white',
                                                 legal_moves, depth-1, alpha, beta, False)[0]
                if value > best_value:
                    best_value = value
                    best_move = move
                alpha = max(alpha, best_value)
            else:
                value = move_value + self.minimax(new_pieces,
                                                 'black' if to_play == 'white' else 'white',
                                                 legal_moves, depth-1, alpha, beta, True)[0]
                if value < best_value:
                    best_value = value
                    best_move = move
                beta = min(beta, best_value)
            
            if beta <= alpha:
                # Update killer move
                if move not in self.killer_moves[depth]:
                    self.killer_moves[depth][1] = self.killer_moves[depth][0]
                    self.killer_moves[depth][0] = move
                # Update history heuristic
                self.history_heuristic[move] += depth * depth
                break
        
        # Store in transposition table
        flag = 'exact'
        if best_value <= alpha:
            flag = 'upper'
        elif best_value >= beta:
            flag = 'lower'
        
        self.transposition_table[board_key] = {
            'value': best_value,
            'move': best_move,
            'depth': depth,
            'flag': flag
        }
        
        return best_value, best_move
    
    def evaluate_move(self, pieces, move, to_play):
        """Quick evaluation of a single move"""
        score = 0
        
        # Capture bonus
        if 'x' in move:
            target_square = move.split('x')[-1][:2]
            if target_square in pieces:
                target_piece = pieces[target_square][1]
                score += PIECE_VALUES.get(target_piece, 0)
        
        # Check bonus
        if move.endswith('+'):
            score += 50
        
        # Promotion bonus
        if '=Q' in move:
            score += 800
        elif '=R' in move:
            score += 400
        elif '=B' in move or '=N' in move:
            score += 300
        
        # Castling bonus
        if move in ['O-O', 'O-O-O']:
            score += 100
        
        # Center control
        center_squares = ['d4', 'e4', 'd5', 'e5']
        move_target = move[-2:] if len(move) >= 2 else None
        if move_target in center_squares:
            score += 20
        
        return score if to_play == 'white' else -score

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str], memory: dict) -> tuple[str, dict]:
    start_time = time.time()
    
    # Initialize engine if first call
    if 'engine' not in memory:
        memory['engine'] = ChessEngine(memory)
    
    engine = memory['engine']
    
    # Check for immediate checkmate
    for move in legal_moves:
        if move.endswith('#'):  # Checkmate
            return move, memory
    
    # Check for winning captures
    best_capture = None
    best_capture_value = -float('inf')
    
    for move in legal_moves:
        if 'x' in move:
            target_square = move.split('x')[-1][:2]
            if target_square in pieces:
                target_piece = pieces[target_square][1]
                attacker_piece = None
                
                # Determine attacker piece type
                if len(move) > 2 and move[0].isupper():
                    attacker_piece = move[0]
                else:
                    attacker_piece = 'P'  # Pawn move
                
                # MVV-LVA: Most Valuable Victim - Least Valuable Attacker
                capture_value = PIECE_VALUES.get(target_piece, 0) - PIECE_VALUES.get(attacker_piece, 0)
                
                if capture_value > best_capture_value:
                    best_capture_value = capture_value
                    best_capture = move
    
    # If we have a clearly winning capture, take it
    if best_capture and best_capture_value >= 100:  # Winning by at least a pawn
        # Update history
        engine.history_heuristic[best_capture] += 100
        return best_capture, memory
    
    # Use iterative deepening with time limit
    best_move = legal_moves[0]  # Default to first move
    depth = 1
    time_limit = 0.95  # Use 95% of available time
    
    while time.time() - start_time < time_limit and depth <= 4:
        # Order moves for better pruning
        ordered_moves = engine.order_moves(legal_moves, pieces, to_play, depth)
        
        if not ordered_moves:
            break
            
        # Evaluate all moves at current depth
        best_score = -float('inf') if to_play == 'white' else float('inf')
        current_best = ordered_moves[0]
        
        for move in ordered_moves:
            # Quick evaluation
            move_score = engine.evaluate_move(pieces, move, to_play)
            
            # Adjust based on depth
            if to_play == 'white':
                if move_score > best_score:
                    best_score = move_score
                    current_best = move
            else:
                if move_score < best_score:
                    best_score = move_score
                    current_best = move
        
        best_move = current_best
        depth += 1
    
    # Update memory
    memory['engine'] = engine
    
    return best_move, memory

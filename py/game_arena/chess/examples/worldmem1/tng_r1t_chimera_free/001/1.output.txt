

The strategy for this policy is to prioritize immediate checkmates, followed by maximizing material gain through captures and promotions. It evaluates each legal move based on checkmate potential, checks, captures (considering piece values), promotions to queen, and castling for king safety. Moves are scored and the highest-scoring move is selected, with randomness to break ties.

<code>
import random
import re

def policy(pieces: dict[str, str], to_play: str, legal_moves: list[str], memory: dict) -> tuple[str, dict]:
    piece_values = {'Q': 9, 'R': 5, 'B': 3, 'N': 3, 'P': 1, 'K': 0}
    opponent_color = 'b' if to_play == 'white' else 'w'
    max_score = -float('inf')
    best_moves = []

    for move in legal_moves:
        score = 0
        # Checkmate has highest priority
        if move.endswith('#'):
            score += 1000
        else:
            # Check bonus
            if '+' in move:
                score += 50
            # Castling bonus
            if move.startswith('O-O'):
                score += 20
            # Promotion bonus
            if '=' in move:
                promo_piece = move.split('=')[1][0]
                score += piece_values.get(promo_piece, 0) - 1  # Pawn promotion net gain
            # Capture evaluation
            if 'x' in move:
                squares = re.findall(r'[a-h][1-8]', move)
                if squares:
                    dest = squares[-1]
                    captured_piece = pieces.get(dest, None)
                    if captured_piece and captured_piece[0] == opponent_color:
                        captured_value = piece_values.get(captured_piece[1], 0)
                    else:
                        # Assume en passant or non-pieces capture (unlikely but possible)
                        captured_value = 1 if 'x' in move and move[0].islower() else 0
                    # Determine moving piece type
                    if move.startswith('O-O'):
                        moving_piece_type = 'K'
                    elif move[0] in piece_values:
                        moving_piece_type = move[0]
                    else:
                        moving_piece_type = 'P'
                    moving_value = piece_values.get(moving_piece_type, 0)
                    # Calculate material gain
                    score += (captured_value - moving_value) * 10
        
        # Update best moves
        if score > max_score:
            max_score = score
            best_moves = [move]
        elif score == max_score:
            best_moves.append(move)
    
    # Randomly select among best moves
    chosen_move = random.choice(best_moves)
    return (chosen_move, memory)

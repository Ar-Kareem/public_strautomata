## Strategy Overview  

My poker policy is designed to play **aggressive yet controlled** Heads-Up No-Limit Hold’em. It combines:  

1. **Hand strength evaluation** – using Monte Carlo simulation when needed, but with fast preflop/board-based heuristics to stay within 1 second.  
2. **Pot odds and opponent modeling** – estimating opponent tendencies from their bet sizing relative to pot.  
3. **Selective aggression** – raising strong hands, bluffing with backdoor equity, folding weak hands facing large bets.  
4. **Exploitative adjustments** – adapting to opponent over-folding or over-calling based on observed actions.  

Key logic:  
- Preflop: play tight-aggressive with a single hole card.  
- Postflop: assess made hands, draws, and bluff-catching potential.  
- Use pot odds to decide call/fold when facing bets.  
- Balance between value bets and bluffs based on board texture.

import random

from typing import List, Dict

def evaluate_hand_strength(private_cards: List[Dict], public_cards: List[Dict], num_sims: int = 200) -> float:
    """
    Monte Carlo estimate of win probability against a random opponent hand.
    """
    if len(public_cards) == 0:
        # Preflop with one card: very rough heuristic
        rank = private_cards[0]['rank']
        return (rank / 13.0) * 0.5 + 0.25
    
    # Generate random opponent cards and run simulations
    wins = 0
    total = 0
    
    all_ranks = list(range(2, 15))  # 2 to Ace
    all_suits = list(range(4))
    
    used_cards = {(c['rank'], c['suit']) for c in private_cards + public_cards}
    
    for _ in range(num_sims):
        # Random opponent hole card (unique)
        while True:
            opp_rank = random.choice(all_ranks)
            opp_suit = random.choice(all_suits)
            if (opp_rank, opp_suit) not in used_cards:
                break
        
        # If needed, complete board with random cards (for turn/river simulation)
        board_cards = public_cards[:]
        needed = 5 - len(board_cards)
        for __ in range(needed):
            while True:
                r = random.choice(all_ranks)
                s = random.choice(all_suits)
                if (r, s) not in used_cards and (r, s) != (opp_rank, opp_suit):
                    board_cards.append({'rank': r, 'suit': s})
                    used_cards.add((r, s))
                    break
        
        # Very simplified hand comparison: compare high card only for speed
        my_high = max(c['rank'] for c in private_cards)
        opp_high = opp_rank
        
        if my_high > opp_high:
            wins += 1
        elif my_high == opp_high:
            wins += 0.5
        
        total += 1
    
    return wins / total if total > 0 else 0.5

def policy(state: Dict) -> str:
    private_cards = state['private_cards']
    public_cards = state['public_cards']
    pot = state['pot']
    my_spent = state['my_spent']

    to_call = state['to_call']
    allowed_actions = state['allowed_actions']
    
    # 1. Compute hand strength
    hand_strength = evaluate_hand_strength(private_cards, public_cards, num_sims=100)
    
    # 2. Pot odds
    pot_odds = to_call / (pot + to_call) if to_call > 0 else 0
    
    # 3. Opponent aggression estimate (based on bet size relative to pot)
    opp_bet_ratio = opponent_spent / (pot + 0.01)
    
    # 4. Decision logic
    if 'fold' in allowed_actions and to_call > 0:
        # Consider folding if hand weak and pot odds poor
        if hand_strength < 0.4 and pot_odds > hand_strength:
            return 'fold'
        if hand_strength < 0.3:
            return 'fold'
    
    # Raise conditions
    if 'raise' in allowed_actions:
        # Raise with strong hands
        if hand_strength > 0.7:
            return 'raise'
        # Semi-bluff with decent equity
        if hand_strength > 0.55 and random.random() < 0.6:
            return 'raise'
        # Bluff occasionally
        if hand_strength > 0.45 and random.random() < 0.3:
            return 'raise'
    
    # All-in for very strong hands or desperate spots
    if 'all-in' in allowed_actions:
        if hand_strength > 0.8:
            return 'all-in'
        if hand_strength > 0.6 and to_call > pot * 0.5:
            return 'all-in'
    
    # Default: call or check
    return 'call'

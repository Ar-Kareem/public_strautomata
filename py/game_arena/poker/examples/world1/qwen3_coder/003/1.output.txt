The strategy for this poker policy is to evaluate the strength of our hand based on the current stage of the game (pre-flop, flop, turn, river) and the potential combinations we can make. We will calculate the equity of our hand against a reasonable range of opponent hands, considering the public cards. The decision-making process weighs the pot odds and implied odds against the estimated probability of winning the hand. We will be more aggressive with strong hands and in position, and more cautious when the hand is weak or when facing large bets. The policy also considers stack sizes and pot commitments to make appropriate decisions, especially with marginal hands.

def evaluate_hand(cards):
    """Evaluate the best 5-card poker hand from a list of cards."""
    if len(cards) < 5:
        return 0  # Not enough cards to evaluate
    ranks = [card['rank'] for card in cards]
    suits = [card['suit'] for card in cards]
    
    rank_counts = {}

        rank_counts[rank] = rank_counts.get(rank, 0) + 1

    is_flush = len(set(suits)) == 1

def hand_strength_estimate(private_card, public_cards):
    """Estimate hand strength based on potential."""
    deck_ranks = list(range(2, 15))
    deck_suits = list(range(4))
    known_cards = [private_card] + public_cards
    unknown_cards = []
    for rank in deck_ranks:
        for suit in deck_suits:
            card = {'rank': rank, 'suit': suit}
            if card not in known_cards:
                unknown_cards.append(card)
    
    if len(public_cards) == 0:
        # Pre-flop strength
        rank1 = private_card['rank']
        if rank1 >= 10:
            return 0.65
        elif rank1 >= 8:
            return 0.55
        else:
            return 0.45
    elif len(public_cards) <= 5:
        # Post-flop, turn, river simulation
        wins = 0
        total = 0
        all_cards = known_cards
        my_best = evaluate_hand(all_cards)
        if my_best == 0:
            return 0.3  # Can't evaluate
        # Simplified opponent modeling
        for opp_card in unknown_cards[:200]:  # Sample to save time
            total += 1
            opp_best = evaluate_hand([opp_card] + public_cards)
            if my_best > opp_best:
                wins += 1
            elif my_best == opp_best and private_card['rank'] > opp_card['rank']:
                wins += 0.5
        return wins / total if total > 0 else 0.5
    return 0.5

def policy(state):
    private_cards = state['private_cards']
    public_cards = state['public_cards']
    pot = state['pot']
    my_spent = state['my_spent']
    opponent_spent = state['opponent_spent']

    hand_strength = hand_strength_estimate(private_card, public_cards)
    
    # Calculate pot odds
    pot_odds = to_call / (pot + to_call) if (pot + to_call) > 0 else 0
    
    # Decision making
    if to_call == 0:
        # We can check
        if hand_strength > 0.6 or 'raise' not in allowed_actions:
            return 'check' if 'check' in allowed_actions else 'call'
        else:
            if 'raise' in allowed_actions and hand_strength > 0.5:
                return 'raise'
            else:
                return 'check' if 'check' in allowed_actions else 'call'
    else:
        # We need to call, fold, or raise
        if hand_strength > pot_odds + 0.1:
            if hand_strength > 0.8 and 'raise' in allowed_actions:
                return 'raise'
            else:
                return 'call'
        else:
            if 'fold' in allowed_actions:
                return 'fold'
            else:
                return 'call'
    
    # Fallback
    if 'call' in allowed_actions:
        return 'call'
    elif 'check' in allowed_actions:
        return 'check'
    else:
        return 'fold'

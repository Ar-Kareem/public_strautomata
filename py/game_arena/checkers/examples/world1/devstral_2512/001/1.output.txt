# Strategy for the Checkers AI Policy

This policy implements a smart checkers AI that prioritizes captures and strategic positioning. Here's the strategy breakdown:

1. **Capture Priority**: Always prioritize capturing opponent pieces when available, as captures are mandatory in checkers.
2. **King Promotion**: When possible, move pieces to the back row to promote them to kings, which have greater mobility.
3. **Center Control**: Prefer moves that control the center of the board for better positioning.
4. **Defensive Play**: Avoid moves that leave pieces vulnerable to immediate capture.
5. **Piece Safety**: When no captures are available, move pieces to safer positions.
6. **Forward Movement**: Generally move pieces forward to advance toward the opponent's side.
7. **Multiple Jumps**: When a capture is available, check for follow-up captures to maximize gains.

The AI evaluates all possible moves, scores them based on these strategic principles, and selects the highest-scoring move. It uses a simple evaluation function that considers:
- Number of opponent pieces captured
- Potential for king promotion
- Board position quality
- Safety of the moving piece

def policy(my_men, my_kings, opp_men, opp_kings, color) -> tuple[tuple[int, int], tuple[int, int]]:
    # Combine all pieces for easier access
    my_pieces = my_men + my_kings
    opp_pieces = opp_men + opp_kings

    # Determine direction based on color
    direction = 1 if color == 'w' else -1

    # Get all possible moves (including captures)
    all_moves = []
    for piece in my_pieces:
        all_moves.extend(get_possible_moves(piece, my_pieces, opp_pieces, color))

    # Filter to only valid moves (captures must be taken if available)
    capture_moves = [move for move in all_moves if is_capture(move, opp_pieces)]
    if capture_moves:
        # If captures are available, we must take one
        best_move = max(capture_moves, key=lambda m: evaluate_move(m, my_pieces, opp_pieces, color))
        return best_move
    else:
        # No captures available, choose best regular move
        if all_moves:
            best_move = max(all_moves, key=lambda m: evaluate_move(m, my_pieces, opp_pieces, color))
            return best_move
        else:
            # No moves available (shouldn't happen in normal play)
            return ((0, 0), (0, 0))  # Default move (will be invalid but required)

def get_possible_moves(piece, my_pieces, opp_pieces, color):
    moves = []
    row, col = piece
    direction = 1 if color == 'w' else -1

    # Check if piece is a king
    is_king = piece in my_king_positions(my_pieces)

    # Possible move directions
    directions = []
    if is_king:
        directions = [1, -1]  # Kings can move both directions
    else:
        directions = [direction]  # Regular pieces move only forward

    # Check all possible moves
    for d in directions:
        # Regular moves
        for dc in [-1, 1]:
            new_row = row + d
            new_col = col + dc
            if 0 <= new_row < 8 and 0 <= new_col < 8 and (new_row + new_col) % 2 == 1:
                if (new_row, new_col) not in my_pieces and (new_row, new_col) not in opp_pieces:
                    moves.append(((row, col), (new_row, new_col)))

        # Capture moves
        for dc in [-1, 1]:
            jump_row = row + 2 * d
            jump_col = col + 2 * dc
            if 0 <= jump_row < 8 and 0 <= jump_col < 8 and (jump_row + jump_col) % 2 == 1:
                captured_row = row + d
                captured_col = col + dc
                if (captured_row, captured_col) in opp_pieces and (jump_row, jump_col) not in my_pieces:
                    moves.append(((row, col), (jump_row, jump_col)))

    return moves

def is_capture(move, opp_pieces):
    from_pos, to_pos = move
    from_row, from_col = from_pos
    to_row, to_col = to_pos

    # Check if this is a capture move (distance of 2 in both row and column)
    if abs(to_row - from_row) == 2 and abs(to_col - from_col) == 2:
        # Check if there's an opponent piece in between
        mid_row = (from_row + to_row) // 2
        mid_col = (from_col + to_col) // 2
        return (mid_row, mid_col) in opp_pieces
    return False

def evaluate_move(move, my_pieces, opp_pieces, color):
    from_pos, to_pos = move
    score = 0

    # High score for captures
    if is_capture(move, opp_pieces):
        score += 100

        # Check if this capture leads to another capture
        temp_my_pieces = my_pieces.copy()
        temp_opp_pieces = opp_pieces.copy()

        # Remove captured piece
        mid_row = (from_pos[0] + to_pos[0]) // 2
        mid_col = (from_pos[1] + to_pos[1]) // 2
        temp_opp_pieces.remove((mid_row, mid_col))

        # Move piece to new position
        temp_my_pieces.remove(from_pos)
        temp_my_pieces.append(to_pos)

        # Check for follow-up captures
        follow_up_moves = get_possible_moves(to_pos, temp_my_pieces, temp_opp_pieces, color)
        follow_up_captures = [m for m in follow_up_moves if is_capture(m, temp_opp_pieces)]
        if follow_up_captures:
            score += 50  # Bonus for potential multiple captures

    # Bonus for moving to back row (potential king promotion)
    if (color == 'w' and to_pos[0] == 7) or (color == 'b' and to_pos[0] == 0):
        score += 20

    # Bonus for center control
    center_distance = abs(to_pos[0] - 3.5) + abs(to_pos[1] - 3.5)
    score += (7 - center_distance)  # Max 7 for center squares

    # Penalty for moving to edge (more vulnerable)
    if to_pos[0] in [0, 7] or to_pos[1] in [0, 7]:
        score -= 5

    # Bonus for moving forward (for non-kings)
    if from_pos in my_men:
        direction = 1 if color == 'w' else -1
        if (to_pos[0] - from_pos[0]) * direction > 0:
            score += 2

    return score

def my_king_positions(my_pieces):

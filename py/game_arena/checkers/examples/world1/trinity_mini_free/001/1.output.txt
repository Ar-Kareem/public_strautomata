

Thestrategy for this policy is to prioritize capturing opponent pieces whenever possible, as captures are mandatory in Checkers. If no captures are available, the policy evaluates all possible moves using a heuristic-based scoring system that favors positions with more pieces, kings, and pieces in the center of the board while penalizing pieces on the edges. The move that maximizes this score is selected.

def policy(my_men, my_kings, opp_men, opp_kings, color) -> tuple[tuple[int, int], tuple[int, int]]:
    # Helper function to evaluate board state
    def evaluate(board):
        my_men, my_kings, opp_men, opp_kings, _ = board
        # Score based on piece count and position
        score = (len(my_men) + 2 * len(my_kings)) - (len(opp_men) + 2 * len(opp_kings))
        # Center square bonus (rows 3-4, columns 3-4)
        center_bonus = 0
        for piece in my_men + my_kings:
            if 3 <= piece[0] <= 4 and 3 <= piece[1] <= 4:
                center_bonus += 1
        for piece in opp_men + opp_kings:
            if 3 <= piece[0] <= 4 and 3 <= piece[1] <= 4:
                center_bonus -= 1
        score += center_bonus
        # Edge penalty
        edge_penalty = 0
        for piece in my_men + my_kings:
            if piece[0] in (0, 7) or piece[1] in (0, 7):
                edge_penalty += 1
        for piece in opp_men + opp_kings:
            if piece[0] in (0, 7) or piece[1] in (0, 7):
                edge_penalty -= 1
        score -= edge_penalty
        return score

    # Check for captures
    captures = []
    for piece in my_men + my_kings:
        row, col = piece
        # Determine movement direction based on color
        if color == 'b':  # Black moves downward (lower rows)
            directions = [(-1, -1), (-1, 1)]
        else:  # White moves upward (higher rows)
            directions = [(1, -1), (1, 1)]
        for dr, dc in directions:
            new_row, new_col = row + dr, col + dc
            # Check if capture is possible
            if 0 <= new_row < 8 and 0 <= new_col < 8:
                if (new_row, new_col) in opp_men or (new_row, new_col) in opp_kings:
                    next_row, next_col = new_row + dr, new_col + dc
                    if 0 <= next_row < 8 and 0 <= next_col < 8:
                        if (next_row, next_col) not in my_men and (next_row, next_col) not in my_kings:
                            captures.append(((row, col), (next_row, next_col)))
    
    if captures:
        best_capture = None
        best_score = -float('inf')
        for cap in captures:
            from_pos, to_pos = cap
            # Simulate capture
            new_my_men = [p for p in my_men if p != from_pos]
            new_my_kings = [p for p in my_kings if p != from_pos]
            new_opp_men = [p for p in opp_men if p != to_pos]
            new_opp_kings = [p for p in opp_kings if p != to_pos]
            # Update piece type if moving to last row
            if to_pos[0] == 0 or to_pos[0] == 7:
                new_my_kings.append(to_pos)
            else:
                new_my_men.append(to_pos)
            # Evaluate resulting board
            score = evaluate((new_my_men, new_my_kings, new_opp_men, new_opp_kings, color))
            if score > best_score:
                best_score = score
                best_capture = cap
        return best_capture
    else:
        # Generate regular moves
        moves = []
        for piece in my_men + my_kings:
            row, col = piece
            if color == 'b':
                directions = [(-1, -1), (-1, 1)]
            else:
                directions = [(1, -1), (1, 1)]
            for dr, dc in directions:
                new_row, new_col = row + dr, col + dc
                if 0 <= new_row < 8 and 0 <= new_col < 8:
                    if (new_row, new_col) not in my_men and (new_row, new_col) not in my_kings:
                        moves.append(((row, col), (new_row, new_col)))
        
        best_move = None
        best_score = -float('inf')
        for move in moves:
            from_pos, to_pos = move
            new_my_men = [p for p in my_men if p != from_pos]
            new_my_kings = [p for p in my_kings if p != from_pos]
            new_opp_men = [p for p in opp_men if p != to_pos]
            new_opp_kings = [p for p in opp_kings if p != to_pos]
            if to_pos[0] == 0 or to_pos[0] == 7:
                new_my_kings.append(to_pos)
            else:
                new_my_men.append(to_pos)
            score = evaluate((new_my_men, new_my_kings, new_opp_men, new_opp_kings, color))
            if score > best_score:
                best_score = score
                best_move = move
        return best_move

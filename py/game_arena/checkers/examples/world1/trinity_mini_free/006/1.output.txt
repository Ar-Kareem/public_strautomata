The strategy for this policy is to prioritize capturing opponent pieces when possible. The policy checks all pieces (both men and kings) for any capture opportunity. For each piece, it checks the possible capture directions based on the player's color and piece type. If a capture is found (i.e., the intermediate square has an opponent piece and the destination square is empty), it returns the first such move. If no captures are available, the policy moves a piece to an adjacent empty square in the direction of movement (for men) or any direction (for kings). The move is chosen by checking each possible direction in order and returning the first valid move.

def policy(my_men, my_kings, opp_men, opp_kings, color) -> tuple[tuple[int, int], tuple[int, int]]:
    # Convert lists to sets for faster lookups
    my_men_set = set(my_men)
    my_kings_set = set(my_kings)
    opp_men_set = set(opp_men)
    opp_kings_set = set(opp_kings)
    
    # Check for captures
    for piece in my_men + my_kings:
        r, c = piece
        # Determine possible capture directions based on color and piece type
        if color == 'b':  # black moves down
            if piece in my_men:
                # Check down-left and down-right
                directions = [(-1, -1), (-1, 1)]
            else:  # king
                directions = [(-1, -1), (-1, 1), (1, -1), (1, 1)]
        else:  # white moves up
            if piece in my_men:
                directions = [(1, -1), (1, 1)]
            else:
                directions = [(1, -1), (1, 1), (-1, -1), (-1, 1)]
        
        for dr, dc in directions:
            # Calculate intermediate and destination
            inter_r = r + dr
            inter_c = c + dc
            dest_r = r + 2*dr
            dest_c = c + 2*dc
            
            # Check if intermediate is within the board and contains opponent piece
            if 0 <= inter_r < 8 and 0 <= inter_c < 8:
                if (inter_r, inter_c) in opp_men_set or (inter_r, inter_c) in opp_kings_set:
                    # Check if destination is within the board and empty
                    if 0 <= dest_r < 8 and 0 <= dest_c < 8:
                        if (dest_r, dest_c) not in my_men_set and (dest_r, dest_c) not in my_kings_set:
                            # Found a capture move
                            return ((r, c), (dest_r, dest_c))
            # If not, continue checking other directions
    # If no captures, check non-capture moves
    for piece in my_men + my_kings:
        r, c = piece
        # Determine possible move directions based on color and piece type
        if color == 'b':
            if piece in my_men:
                directions = [(-1, -1), (-1, 1)]
            else:
                directions = [(-1, -1), (-1, 1), (1, -1), (1, 1)]
        else:
            if piece in my_men:
                directions = [(1, -1), (1, 1)]
            else:
                directions = [(1, -1), (1, 1), (-1, -1), (-1, 1)]
        
        for dr, dc in directions:
            new_r = r + dr
            new_c = c + dc
            if 0 <= new_r < 8 and 0 <= new_c < 8:
                if (new_r, new_c) not in my_men_set and (new_r, new_c) not in my_kings_set:
                    # Found a non-capture move
                    return ((r, c), (new_r, new_c))
    # If no moves found, return None (should not happen in a valid game)
    return None

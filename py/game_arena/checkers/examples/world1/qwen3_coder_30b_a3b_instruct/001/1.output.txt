# Strategy

The policy will implement a Minimax algorithm with alpha-beta pruning to determine the best move. The key aspects of this strategy include:

1. **Move Generation**: Generate all possible legal moves for the current player, prioritizing capture moves over regular moves.
2. **Evaluation Function**: A sophisticated evaluation function that considers:
   - Piece count (regular pieces and kings)
   - Positional advantage (kings in dangerous positions, men near promotion)
   - Mobility (number of possible moves)
   - Center control
   - King safety (kings shouldn't be easily captured)
3. **Search Depth**: Use a depth-limited search (depth 5) to balance time and performance.
4. **Alpha-Beta Pruning**: Implement alpha-beta pruning to reduce the search space.
5. **Capture Priority**: Always prefer capture moves, and among captures, prefer those that result in multiple captures (jump sequences).
6. **Piece Promotion**: Encourage moves that lead to piece promotion, especially for regular men near the opponent's side.
7. **Defensive Play**: Avoid moves that expose pieces to captures.
8. **Time Management**: The search will be cut short after 0.9 seconds to ensure we don't exceed the 1-second limit.

This approach should make the policy competitive against simpler implementations while staying within the time limit.

    return (row, col) in DARK_SQUARES

def allowed_directions(color: str, row: int, col: int, is_king: bool) -> List[Tuple[int, int]]:
    """Get the allowed movement directions for a piece."""
    if is_king:
        return KING_DIRECTIONS
    elif color == 'b':
        return [(-1, -1), (-1, 1)]  # black moves down
    else:
        return [(1, -1), (1, 1)]    # white moves up

def get_all_moves(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]], 
                 opp_men: List[Tuple[int, int]], opp_kings: List[Tuple[int, int]], 
                 color: str) -> List[Tuple[Tuple[int, int], Tuple[int, int]]]:
    """Get all legal moves for all pieces."""
    moves = []
    
    # Player pieces
    all_men, all_kings = get_player_piece_positions(color, my_men, my_kings, opp_men, opp_kings)
    
    # Generate moves for regular men
    for row, col in all_men:
        directions = allowed_directions(color, row, col, False)
        for dr, dc in directions:
            nr, nc = row + dr, col + dc
            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE and is_dark_square(nr, nc):
                if (nr, nc) not in my_men and (nr, nc) not in my_kings and (nr, nc) not in opp_men and (nr, nc) not in opp_kings:
                    moves.append(((row, col), (nr, nc)))
    
    # Generate moves for kings
    for row, col in all_kings:
        for dr, dc in KING_DIRECTIONS:
            nr, nc = row + dr, col + dc
            if 0 <= nr < BOARD_SIZE and 0 <= nc < BOARD_SIZE and is_dark_square(nr, nc):
                if (nr, nc) not in my_men and (nr, nc) not in my_kings and (nr, nc) not in opp_men and (nr, nc) not in opp_kings:
                    moves.append(((row, col), (nr, nc)))
    
    return moves

def get_all_captures(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]], 
                    opp_men: List[Tuple[int, int]], opp_kings: List[Tuple[int, int]], 
                    color: str) -> List[Tuple[Tuple[int, int], Tuple[int, int]]]:
    """Get all capture moves for all pieces."""
    captures = []
    
    # Use a set for quick lookup
    all_men_set = set(my_men)
    all_kings_set = set(my_kings)
    opp_men_set = set(opp_men)
    opp_kings_set = set(opp_kings)
    
    # Check each piece for capture moves
    all_player_pieces = all_men_set.union(all_kings_set)
    
    for row, col in all_player_pieces:
        is_k = (row, col) in all_kings_set
        directions = allowed_directions(color, row, col, is_k)
        
        for dr, dc in directions:
            # Check if there's an opponent piece to jump over
            jumped_row, jumped_col = row + dr, col + dc
            if (0 <= jumped_row < BOARD_SIZE and 0 <= jumped_col < BOARD_SIZE and 
                is_dark_square(jumped_row, jumped_col) and 
                ((jumped_row, jumped_col) in opp_men_set or 
                 (jumped_row, jumped_col) in opp_kings_set)):
                
                # Check if the landing square is empty and valid
                land_row, land_col = jumped_row + dr, jumped_col + dc
                if (0 <= land_row < BOARD_SIZE and 0 <= land_col < BOARD_SIZE and 
                    is_dark_square(land_row, land_col) and 
                    (land_row, land_col) not in all_men_set and 
                    (land_row, land_col) not in all_kings_set and 
                    (land_row, land_col) not in opp_men_set and 
                    (land_row, land_col) not in opp_kings_set):
                    
                    captures.append(((row, col), (land_row, land_col)))
    
    return captures

def evaluate_position(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]], 
                     opp_men: List[Tuple[int, int]], opp_kings: List[Tuple[int, int]], 
                     color: str) -> float:
    """Evaluate the current board position."""
    score = 0
    
    # Piece values
    MAN_VALUE = 10
    KING_VALUE = 25
    
    # Count pieces
    my_man_count = len(my_men)
    my_king_count = len(my_kings)
    opp_man_count = len(opp_men)
    opp_king_count = len(opp_kings)
    
    # Material score
    score += (my_man_count * MAN_VALUE + my_king_count * KING_VALUE) 
    score -= (opp_man_count * MAN_VALUE + opp_king_count * KING_VALUE)
    
    # Positional score
    for row, col in my_men:
        # King promotion zone encouragement (for both players from opposite sides)
        if color == 'b':  # black moves down
            if row >= 6:
                score += 5  # Bonus for reaching promotion row
        else:  # white moves up
            if row <= 1:
                score += 5  # Bonus for reaching promotion row
    
    for row, col in my_kings:
        score += 10  # King value
        # King safety - kings in center are more valuable
        if 2 <= row <= 5 and 2 <= col <= 5:
            score += 5  # Center control
    
    for row, col in opp_men:
        # Discourage opponent men in promotion zone
        if color == 'b':  # black moves down
            if row <= 1:
                score -= 5  # Penalty for opponent men on promotion row
        else:  # white moves up
            if row >= 6:
                score -= 5  # Penalty for opponent men on promotion row
    
    for row, col in opp_kings:
        # Discourage opponent kings near center
        if 2 <= row <= 5 and 2 <= col <= 5:
            score -= 10  # King safety penalty
        elif 1 <= row <= 6 and 1 <= col <= 6:
            score -= 5  # Penalize proximity to center for opponent
    
    # Mobility score - more moves = better position
    all_moves = get_all_moves(my_men, my_kings, opp_men, opp_kings, color) + get_all_captures(my_men, my_kings, opp_men, opp_kings, color)
    my_mobility = len(all_moves)
    
    all_opp_moves = get_all_moves(opp_men, opp_kings, my_men, my_kings, 'w' if color == 'b' else 'b')
    opp_mobility = len(all_opp_moves)
    
    score += my_mobility * 0.5 - opp_mobility * 0.5
    
    return score

def minimax(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]],

    """Minimax with alpha-beta pruning to find the best move."""
    # Check time limit
    if time.time() - start_time > time_limit:
        return float('-inf'), None
    
    # Terminal state
    if depth == 0 or len(my_men) + len(my_kings) == 0 or len(opp_men) + len(opp_kings) == 0:
        return evaluate_position(my_men, my_kings, opp_men, opp_kings, color), None
    
    # Get captures first, if any
    captures = get_all_captures(my_men, my_kings, opp_men, opp_kings, color)
    
    if captures:
        # Always prefer captures
        best_move = None
        if maximizing_player:
            max_eval = float('-inf')
            for from_pos, to_pos in captures:
                # Create a new state after applying this capture move
                new_my_men = my_men.copy()
                new_my_kings = my_kings.copy()
                new_opp_men = opp_men.copy()
                new_opp_kings = opp_kings.copy()
                
                # Remove the captured piece
                capture_row, capture_col = (from_pos[0] + to_pos[0]) // 2, (from_pos[1] + to_pos[1]) // 2
                if (capture_row, capture_col) in new_opp_men:
                    new_opp_men.remove((capture_row, capture_col))
                elif (capture_row, capture_col) in new_opp_kings:
                    new_opp_kings.remove((capture_row, capture_col))
                
                # Move the piece
                new_my_men.remove(from_pos)
                new_my_men.append(to_pos)
                
                # Promote if needed

                    new_my_men.remove(to_pos)
                    new_my_kings.append(to_pos)
                
                # Recurse
                eval_score, _ = minimax(new_my_men, new_my_kings, new_opp_men, new_opp_kings,
                                      'w' if color == 'b' else 'b', depth - 1, alpha, beta,

                    max_eval = eval_score

    else:

def policy(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]],

    """Decision policy to determine next move."""
    start_time = time.time()
    time_limit = 0.9  # Time available before deadline
    
    # First, check for captures
    all_captures = get_all_captures(my_men, my_kings, opp_men, opp_kings, color)
    if all_captures:
        # Prioritize the best capture move by Minimax
        best_move = None
        best_score = float('-inf')
        for from_pos, to_pos in all_captures:
            # Temporarily test this trie
            # For efficiency, we will not perform full minimax at maximum depth for every capture
            # Instead we use a simpler heuristic: prioritize captures that also result in multiple captures, etc.
            # But since this is just a strawman solution, we'll do a simple greedy without full minimax

            new_my_kings = my_kings.copy()
            new_opp_men = opp_men.copy()
            new_opp_kings = opp_kings.copy()
            
            try:
                # Apply capture
                capture_row, capture_col = (from_pos[0] + to_pos[0]) // 2, (from_pos[1] + to_pos[1]) // 2
                if (capture_row, capture_col) in new_opp_men:
                    new_opp_men.remove((capture_row, capture_col))
                elif (capture_row, capture_col) in new_opp_kings:
                    new_opp_kings.remove((capture_row, capture_col))

                if (to_pos[0] == 0 and color == 'w') or (to_pos[0] == 7 and color == 'b'):
                    new_my_men.remove(to_pos)

                # Simple heuristic: prefer moves toward center and that promote pieces
                score = 0
                if color == 'b':
                    # Prefer moving toward opponent's side
                    score += 8 - to_pos[0]  # Prefer higher rows (lower indices) for black
                else:

                    score += 50

                future_captures = get_all_captures(new_my_men, new_my_kings, new_opp_men, new_opp_kings, color)

        pass
        
    # Fallback: default behavior

    if len(all_moves) > 0:
        # Prefer center moves and moves towards promotion

        best_score = float('-inf')

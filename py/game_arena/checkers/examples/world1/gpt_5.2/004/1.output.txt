The strategy for this policy is to always generate **only legal moves** (respecting **mandatory captures**) and then choose among them using a **time-bounded iterative-deepening alpha-beta search** (minimax). The evaluation favors winning material (especially kings), safe positioning (back-rank guards), promotion progress, central control, and mobility. Captures and promotions are strongly prioritized via move ordering so the search is effective within the 1-second limit.

import time
import random
from typing import List, Tuple, Set, Optional

Move = Tuple[Tuple[int, int], Tuple[int, int]]  # ((from_r, from_c), (to_r, to_c))

BOARD_N = 8

# Dark squares assumption: (r+c)%2==0 if (0,0) is dark. We don't need to enforce explicitly
# as legal diagonal moves preserve parity, but we still ensure destination emptiness/bounds.

# Zobrist hashing for transposition table speed
_rng = random.Random(1337)
_Z = {
    "bm": [[_rng.getrandbits(64) for _ in range(BOARD_N)] for _ in range(BOARD_N)],
    "bk": [[_rng.getrandbits(64) for _ in range(BOARD_N)] for _ in range(BOARD_N)],
    "wm": [[_rng.getrandbits(64) for _ in range(BOARD_N)] for _ in range(BOARD_N)],
    "wk": [[_rng.getrandbits(64) for _ in range(BOARD_N)] for _ in range(BOARD_N)],
    "turn_b": _rng.getrandbits(64),
    "turn_w": _rng.getrandbits(64),
}


def _in_bounds(r: int, c: int) -> bool:
    return 0 <= r < BOARD_N and 0 <= c < BOARD_N


def _hash_state(bm: Set[Tuple[int, int]],
                bk: Set[Tuple[int, int]],
                wm: Set[Tuple[int, int]],
                wk: Set[Tuple[int, int]],
                turn: str) -> int:
    h = 0
    for (r, c) in bm:
        h ^= _Z["bm"][r][c]
    for (r, c) in bk:
        h ^= _Z["bk"][r][c]
    for (r, c) in wm:
        h ^= _Z["wm"][r][c]
    for (r, c) in wk:
        h ^= _Z["wk"][r][c]
    h ^= _Z["turn_b"] if turn == "b" else _Z["turn_w"]
    return h


def _dirs_for_man(color: str):
    # White moves upwards (increasing row). Black moves downwards (decreasing row).
    return [(1, -1), (1, 1)] if color == "w" else [(-1, -1), (-1, 1)]


def _dirs_for_king():
    return [(-1, -1), (-1, 1), (1, -1), (1, 1)]


def _opponent(color: str) -> str:
    return "w" if color == "b" else "b"


def _get_sets_for(color: str,
                  bm: Set[Tuple[int, int]],
                  bk: Set[Tuple[int, int]],
                  wm: Set[Tuple[int, int]],
                  wk: Set[Tuple[int, int]]):
    if color == "b":
        return bm, bk, wm, wk
    else:
        return wm, wk, bm, bk


def _generate_moves(bm: Set[Tuple[int, int]],
                    bk: Set[Tuple[int, int]],
                    wm: Set[Tuple[int, int]],
                    wk: Set[Tuple[int, int]],
                    turn: str) -> List[Move]:
    """Generate legal moves for 'turn' side. Enforces mandatory captures."""
    mym, myk, oppm, oppk = _get_sets_for(turn, bm, bk, wm, wk)
    occupied = bm | bk | wm | wk
    opp_occ = oppm | oppk

    captures: List[Move] = []
    quiets: List[Move] = []

    # Men
    for (r, c) in mym:
        for dr, dc in _dirs_for_man(turn):
            r1, c1 = r + dr, c + dc
            r2, c2 = r + 2 * dr, c + 2 * dc
            # capture
            if _in_bounds(r2, c2) and (r1, c1) in opp_occ and (r2, c2) not in occupied:
                captures.append(((r, c), (r2, c2)))
            # quiet
            if _in_bounds(r1, c1) and (r1, c1) not in occupied:
                quiets.append(((r, c), (r1, c1)))

    # Kings
    for (r, c) in myk:
        for dr, dc in _dirs_for_king():
            r1, c1 = r + dr, c + dc
            r2, c2 = r + 2 * dr, c + 2 * dc
            if _in_bounds(r2, c2) and (r1, c1) in opp_occ and (r2, c2) not in occupied:
                captures.append(((r, c), (r2, c2)))
            if _in_bounds(r1, c1) and (r1, c1) not in occupied:
                quiets.append(((r, c), (r1, c1)))

    return captures if captures else quiets


def _apply_move(bm: Set[Tuple[int, int]],
                bk: Set[Tuple[int, int]],
                wm: Set[Tuple[int, int]],
                wk: Set[Tuple[int, int]],
                turn: str,
                mv: Move):
    """Return new sets after applying move for 'turn'."""
    (fr, fc), (tr, tc) = mv
    nbm, nbk, nwm, nwk = set(bm), set(bk), set(wm), set(wk)

    mym, myk, oppm, oppk = _get_sets_for(turn, nbm, nbk, nwm, nwk)

    # Determine moving piece type
    is_king = (fr, fc) in myk
    if is_king:
        myk.remove((fr, fc))
    else:
        mym.remove((fr, fc))

    # Capture?
    if abs(tr - fr) == 2 and abs(tc - fc) == 2:
        mr, mc = (fr + tr) // 2, (fc + tc) // 2
        if (mr, mc) in oppm:
            oppm.remove((mr, mc))
        elif (mr, mc) in oppk:
            oppk.remove((mr, mc))

    # Place piece, with promotion if man reaches back rank
    if not is_king:
        if (turn == "w" and tr == 7) or (turn == "b" and tr == 0):
            myk.add((tr, tc))
        else:
            mym.add((tr, tc))
    else:
        myk.add((tr, tc))

    return nbm, nbk, nwm, nwk


def _terminal_score(bm: Set[Tuple[int, int]],
                    bk: Set[Tuple[int, int]],
                    wm: Set[Tuple[int, int]],
                    wk: Set[Tuple[int, int]],
                    root: str) -> Optional[float]:
    """If terminal, return huge score from root perspective, else None."""
    bcount = len(bm) + len(bk)
    wcount = len(wm) + len(wk)
    if bcount == 0:
        return -1e9 if root == "b" else 1e9
    if wcount == 0:
        return 1e9 if root == "b" else -1e9
    return None


def _evaluate(bm: Set[Tuple[int, int]],
              bk: Set[Tuple[int, int]],
              wm: Set[Tuple[int, int]],
              wk: Set[Tuple[int, int]],
              root: str) -> float:
    """Heuristic evaluation from 'root' perspective."""
    term = _terminal_score(bm, bk, wm, wk, root)
    if term is not None:
        return term

    # Material
    men_wt = 1.0
    king_wt = 1.85

    b_mat = men_wt * len(bm) + king_wt * len(bk)
    w_mat = men_wt * len(wm) + king_wt * len(wk)
    score = (b_mat - w_mat) if root == "b" else (w_mat - b_mat)

    # Advancement (men closer to promotion are better)
    # For white men: higher row is better; for black men: lower row is better (i.e., 7-row)
    b_adv = sum((7 - r) for (r, c) in bm) / 7.0 if bm else 0.0
    w_adv = sum(r for (r, c) in wm) / 7.0 if wm else 0.0
    adv_term = (b_adv - w_adv) if root == "b" else (w_adv - b_adv)
    score += 0.15 * adv_term

    # Center control
    def center_bonus(pieces: Set[Tuple[int, int]]) -> float:
        s = 0.0
        for (r, c) in pieces:
            # distance from center (3.5,3.5)
            dr = abs(r - 3.5)
            dc = abs(c - 3.5)
            s += (3.5 - (dr + dc) / 2.0)  # higher is better
        return s

    b_center = center_bonus(bm | bk)
    w_center = center_bonus(wm | wk)
    cen_term = (b_center - w_center) if root == "b" else (w_center - b_center)
    score += 0.05 * cen_term

    # Back-rank safety (keeping some pieces on home row helps prevent opponent promotion)
    b_back = sum(1 for (r, c) in (bm | bk) if r == 7)
    w_back = sum(1 for (r, c) in (wm | wk) if r == 0)
    back_term = (b_back - w_back) if root == "b" else (w_back - b_back)
    score += 0.08 * back_term

    return score


def _move_ordering_key(mv: Move, turn: str) -> float:
    (fr, fc), (tr, tc) = mv
    is_capture = (abs(tr - fr) == 2)
    promo = 0
    if not is_capture:
        # promotion possibility on quiet move still matters
        if turn == "w" and tr == 7:
            promo = 1
        if turn == "b" and tr == 0:
            promo = 1
    else:
        # captures are prioritized heavily
        promo = 2
        if turn == "w" and tr == 7:
            promo = 3
        if turn == "b" and tr == 0:
            promo = 3

    # prefer central landings
    center = - (abs(tr - 3.5) + abs(tc - 3.5))
    return promo * 100.0 + (50.0 if is_capture else 0.0) + center


class _Searcher:
    def __init__(self, time_limit_s: float):
        self.time_limit_s = time_limit_s
        self.t0 = time.perf_counter()
        self.tt = {}  # (hash, depth, turn, root) -> score

    def _time_up(self) -> bool:
        return (time.perf_counter() - self.t0) >= self.time_limit_s

    def minimax(self,
                bm: Set[Tuple[int, int]],
                bk: Set[Tuple[int, int]],
                wm: Set[Tuple[int, int]],
                wk: Set[Tuple[int, int]],
                turn: str,
                root: str,
                depth: int,
                alpha: float,
                beta: float) -> float:
        if self._time_up():
            # Return a static eval when time is up
            return _evaluate(bm, bk, wm, wk, root)

        term = _terminal_score(bm, bk, wm, wk, root)
        if term is not None:
            return term
        if depth <= 0:
            return _evaluate(bm, bk, wm, wk, root)

        h = _hash_state(bm, bk, wm, wk, turn)
        key = (h, depth, turn, root)
        if key in self.tt:
            return self.tt[key]

        moves = _generate_moves(bm, bk, wm, wk, turn)
        if not moves:
            # No legal moves => loss for side to move
            val = (-1e9 if turn == root else 1e9)
            self.tt[key] = val
            return val

        # Order moves to help alpha-beta
        moves.sort(key=lambda mv: _move_ordering_key(mv, turn), reverse=True)

        if turn == root:
            best = -1e18
            for mv in moves:
                nbm, nbk, nwm, nwk = _apply_move(bm, bk, wm, wk, turn, mv)
                val = self.minimax(nbm, nbk, nwm, nwk, _opponent(turn), root, depth - 1, alpha, beta)
                if val > best:
                    best = val
                if best > alpha:
                    alpha = best
                if alpha >= beta or self._time_up():
                    break
        else:
            best = 1e18
            for mv in moves:
                nbm, nbk, nwm, nwk = _apply_move(bm, bk, wm, wk, turn, mv)
                val = self.minimax(nbm, nbk, nwm, nwk, _opponent(turn), root, depth - 1, alpha, beta)
                if val < best:
                    best = val
                if best < beta:
                    beta = best
                if alpha >= beta or self._time_up():
                    break

        self.tt[key] = best
        return best

    def best_move(self,
                  bm: Set[Tuple[int, int]],
                  bk: Set[Tuple[int, int]],
                  wm: Set[Tuple[int, int]],
                  wk: Set[Tuple[int, int]],
                  turn: str) -> Move:
        moves = _generate_moves(bm, bk, wm, wk, turn)
        # Must always return a legal move string/tuple; if none exist, fall back (shouldn't happen in valid play)
        if not moves:
            # Desperation: return a dummy that is at least well-formed (arena likely won't call policy in terminal)
            # but we try to avoid disqualification by picking something within board.
            return ((0, 0), (0, 0))

        # If only one legal move, take it quickly
        if len(moves) == 1:
            return moves[0]

        # Iterative deepening
        root = turn
        best_mv = moves[0]
        best_score = -1e18

        # Move ordering at root
        moves.sort(key=lambda mv: _move_ordering_key(mv, turn), reverse=True)

        depth = 1
        while depth <= 10 and not self._time_up():
            local_best_mv = best_mv
            local_best_score = -1e18
            alpha, beta = -1e18, 1e18

            for mv in moves:
                if self._time_up():
                    break
                nbm, nbk, nwm, nwk = _apply_move(bm, bk, wm, wk, turn, mv)
                val = self.minimax(nbm, nbk, nwm, nwk, _opponent(turn), root, depth - 1, alpha, beta)
                if val > local_best_score:
                    local_best_score = val
                    local_best_mv = mv
                if val > alpha:
                    alpha = val

            # If we completed the depth, accept results
            if not self._time_up():
                best_mv = local_best_mv
                best_score = local_best_score
                # Reorder moves with principal variation first for next depth
                moves.sort(key=lambda mv: 1 if mv == best_mv else 0, reverse=True)

            depth += 1

        return best_mv


def policy(my_men, my_kings, opp_men, opp_kings, color) -> Move:
    """
    Return a legal move ((from_row, from_col), (to_row, to_col)).
    Captures are mandatory when available.
    """
    my_men_s = set(map(tuple, my_men))
    my_kings_s = set(map(tuple, my_kings))
    opp_men_s = set(map(tuple, opp_men))
    opp_kings_s = set(map(tuple, opp_kings))

    # Map to black/white global sets
    if color == "b":
        bm, bk = my_men_s, my_kings_s
        wm, wk = opp_men_s, opp_kings_s
    else:
        wm, wk = my_men_s, my_kings_s
        bm, bk = opp_men_s, opp_kings_s

    # Ensure we always return a legal move quickly if needed
    # Hard time limit: keep margin for Python overhead
    searcher = _Searcher(time_limit_s=0.93)
    mv = searcher.best_move(bm, bk, wm, wk, color)

    # Safety net: verify legality; if something went wrong, pick the first legal move.
    legal = _generate_moves(bm, bk, wm, wk, color)
    if legal and mv not in legal:
        mv = legal[0]
    elif not legal:
        # Terminal position; return well-formed fallback
        mv = ((0, 0), (0, 0))

    return mv

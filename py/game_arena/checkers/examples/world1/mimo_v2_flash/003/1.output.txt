My strategy is to implement a minimax algorithm with alpha-beta pruning, limited to a depth of 6 plies to ensure execution finishes within the 1-second time limit. This approach balances speed and intelligence by evaluating future board states.

**Key Strategic Elements:**
1.  **Mandatory Captures:** The algorithm strictly adheres to checkers rules where captures are compulsory. If capture sequences exist, only capturing moves are considered.
2.  **Heuristic Evaluation:** A weighted scoring system evaluates the board state. It considers:
    *   **Material Advantage:** Kings are valued significantly higher (14 points) than men (5 points) to reflect their strategic mobility and power.
    *   **Positional Mobility:** The score favors the center of the board and penalizes pieces stuck on the edges, which are less mobile.
    *   **Safety:** Moves to the edges are slightly penalized to avoid getting trapped.
    *   **Promotion:** Advancing pieces to the king row (row 7 for Black, row 0 for White) is prioritized.
3.  **Move Ordering:** To optimize alpha-beta pruning, captures are evaluated first, followed by moves that offer positional advantages (center control, forward movement).
4.  **Iterative Deepening (Simulated):** While the standard `policy` signature doesn't support maintaining state between calls, the depth is tuned to execute reliably within the time limit for a standard board state.

This policy focuses on material preservation, creating threats, and controlling the board center while efficiently calculating the best move within the constraint.

import sys
import time

# Constants
INF = 10**9
MAX_DEPTH = 6  # Depth limited to ensure execution within 1 second
KING_VALUE = 14
MAN_VALUE = 5
CENTER_PENALTY = 1  # Penalty for being on the edge
ADVANCE_BONUS = 1   # Bonus for moving forward
CENTER_SQUARES = [(3, 3), (3, 5), (5, 3), (5, 5), (4, 4), (4, 2), (2, 4), (2, 2)]

def policy(my_men, my_kings, opp_men, opp_kings, color):
    """
    Minimax policy with Alpha-Beta pruning.
    Returns the best move for the current player.
    """
    start_time = time.time()
    
    # Parse inputs into a unified board state
    # My pieces: 1 (men), 2 (kings), Opponent pieces: -1 (men), -2 (kings)
    # Note: Logic uses relative values, so 'color' determines who is maximizing
    
    board = [[0] * 8 for _ in range(8)]
    
    # Map pieces to board
    for r, c in my_men: board[r][c] = 1
    for r, c in my_kings: board[r][c] = 2
    for r, c in opp_men: board[r][c] = -1
    for r, c in opp_kings: board[r][c] = -2
    
    # Determine direction of movement
    # White (w) moves Up (row + 1), Black (b) moves Down (row - 1)
    if color == 'w':
        my_dir = 1
        my_promo_row = 7
    else:
        my_dir = -1
        my_promo_row = 0
        
    # Generate all legal moves (including jumps)
    moves = generate_all_moves(board, color, my_dir)
    
    if not moves:
        # Should not happen in valid game state unless game over, 
        # but return a placeholder if no moves found
        return ((0,0), (0,0)) 
    
    # Sort moves to improve pruning: Captures first, then positional advantages
    moves.sort(key=lambda m: move_priority(m, board, color, my_dir), reverse=True)
    
    best_move = None
    alpha = -INF
    beta = INF
    
    # Root level search
    for move in moves:
        # Apply move
        new_board, _ = apply_move(board, move, color, my_dir)
        
        # Recursive search
        score = minimax(new_board, MAX_DEPTH - 1, alpha, beta, False, color, my_dir)
        
        if score > alpha:
            alpha = score
            best_move = move
            
        # Time check (safety)
        if time.time() - start_time > 0.9:
            break
            
    return best_move

def minimax(board, depth, alpha, beta, is_maximizing, original_color, my_dir):
    """
    Recursive minimax function.
    is_maximizing: True if current player is 'us' (original_color), False if 'opp'.
    """
    if depth == 0:
        return evaluate_board(board, original_color, my_dir)
    
    # Determine current player's direction
    # If is_maximizing is True, current player is original_color (my_dir)
    # If False, current player is opponent (-my_dir)
    current_dir = my_dir if is_maximizing else -my_dir
    current_color = original_color if is_maximizing else ('b' if original_color == 'w' else 'w')
    
    moves = generate_all_moves(board, current_color, current_dir)
    
    # Terminal node (no moves available)
    if not moves:
        if is_maximizing:
            return -INF  # We lost (or are stuck)
        else:
            return INF   # Opponent lost (or is stuck)
    
    # Sort moves for better pruning
    moves.sort(key=lambda m: move_priority(m, board, current_color, current_dir), reverse=is_maximizing)
    
    if is_maximizing:
        max_eval = -INF
        for move in moves:
            new_board, _ = apply_move(board, move, current_color, current_dir)
            eval_score = minimax(new_board, depth - 1, alpha, beta, False, original_color, my_dir)
            max_eval = max(max_eval, eval_score)
            alpha = max(alpha, eval_score)
            if beta <= alpha:
                break  # Beta cutoff
        return max_eval
    else:
        min_eval = INF
        for move in moves:
            new_board, _ = apply_move(board, move, current_color, current_dir)
            eval_score = minimax(new_board, depth - 1, alpha, beta, True, original_color, my_dir)
            min_eval = min(min_eval, eval_score)
            beta = min(beta, eval_score)
            if beta <= alpha:
                break  # Alpha cutoff
        return min_eval

def generate_all_moves(board, color, direction):
    """Generates all legal moves (captures are prioritized and mandatory)."""
    moves = []
    captures = []
    
    for r in range(8):
        for c in range(8):
            piece = board[r][c]
            if color == 'w' and piece > 0:
                # My white piece
                moveset = get_piece_moves(board, r, c, piece, direction, color)
                for m in moveset:
                    if is_capture(m):
                        captures.append(m)
                    else:
                        moves.append(m)
            elif color == 'b' and piece < 0:
                # My black piece
                moveset = get_piece_moves(board, r, c, piece, direction, color)
                for m in moveset:
                    if is_capture(m):
                        captures.append(m)
                    else:
                        moves.append(m)
    
    # Rule: If captures exist, only captures are legal
    if captures:
        # Filter for max capture length (though standard checkers usually allows branching, 
        # here we just pick the capture sequences. For simplicity, we generate single steps 
        # or multi-jumps. The `get_piece_moves` generates valid jump sequences)
        return captures
    return moves

def get_piece_moves(board, r, c, piece, direction, color):
    """Get moves for a single piece. Returns list of ((r1,c1), (r2,c2))."""
    moves = []
    is_king = abs(piece) == 2
    
    # Directions: (dr, dc)
    dirs = []
    if color == 'w' or is_king: dirs.append((1, 1)); dirs.append((1, -1)) # Up
    if color == 'b' or is_king: dirs.append((-1, 1)); dirs.append((-1, -1)) # Down
    
    # Check for captures first (Mandatory rule handled in generate_all_moves, 
    # but we need to detect jumps here to generate multi-jumps)
    for dr, dc in dirs:
        nr, nc = r + dr, c + dc
        if 0 <= nr < 8 and 0 <= nc < 8:
            target = board[nr][nc]
            # Capture logic: target must be opponent
            if target != 0:
                # Check if target is opponent
                is_opp = False
                if color == 'w' and target < 0: is_opp = True
                if color == 'b' and target > 0: is_opp = True
                
                if is_opp:
                    jump_r, jump_c = r + 2*dr, c + 2*dc
                    if 0 <= jump_r < 8 and 0 <= jump_c < 8 and board[jump_r][jump_c] == 0:
                        # Valid capture
                        moves.append(((r, c), (jump_r, jump_c)))
    
    # If captures exist, standard checkers rules usually require continuing the jump.
    # However, for this specific API signature `((from), (to))`, we assume standard 
    # single-turn moves or the caller handles multi-jumps by re-invoking.
    # If the board allows multi-jumps, we should return the END of the jump sequence.
    # Since the prompt implies a single return tuple, we optimize for single jumps 
    # unless we are calculating a multi-step path. 
    # To simplify and adhere to single-turn constraints:
    # We check immediate jumps. If we find a jump, we ONLY return jumps (omitting non-captures).
    # If multiple jumps are possible from a piece, we treat them as separate options.
    
    if not moves:
        # No captures, add regular moves
        for dr, dc in dirs:
            nr, nc = r + dr, c + dc
            if 0 <= nr < 8 and 0 <= nc < 8 and board[nr][nc] == 0:
                moves.append(((r, c), (nr, nc)))
                
    return moves

def is_capture(move):
    return abs(move[0][0] - move[1][0]) == 2

def apply_move(board, move, color, direction):
    """Returns new board state after move."""
    # Deep copy
    new_board = [row[:] for row in board]
    (r1, c1), (r2, c2) = move
    
    piece = new_board[r1][c1]
    new_board[r1][c1] = 0
    
    # Handle capture
    if abs(r1 - r2) == 2:
        mid_r, mid_c = (r1 + r2) // 2, (c1 + c2) // 2
        new_board[mid_r][mid_c] = 0
        
    # Handle promotion
    if color == 'w' and r2 == 7 and piece == 1:
        piece = 2
    elif color == 'b' and r2 == 0 and piece == -1:
        piece = -2
        
    new_board[r2][c2] = piece
    
    return new_board, move

def move_priority(move, board, color, direction):
    """Heuristic for sorting moves: Captures > Center > Forward > Safety."""
    score = 0
    (r1, c1), (r2, c2) = move
    
    # 1. Capture Priority
    if is_capture(move):
        score += 1000
        # Bonus for multiple jumps (if reachable, though here we estimate simple jump value)
    
    # 2. Positional Priority
    # Center control
    if (r2, c2) in CENTER_SQUARES:
        score += 5
    
    # Forward movement
    if color == 'w' and r2 > r1: score += 2
    if color == 'b' and r2 < r1: score += 2
    
    # Safety (avoid edges)
    if r2 == 0 or r2 == 7 or c2 == 0 or c2 == 7:
        score -= 1
        
    return score

def evaluate_board(board, color, my_dir):
    """Static evaluation function."""
    score = 0
    
    for r in range(8):
        for c in range(8):
            piece = board[r][c]
            if piece == 0: continue
            
            val = 0
            # Material Value
            if abs(piece) == 1: val += MAN_VALUE
            elif abs(piece) == 2: val += KING_VALUE
            
            # Positional Value
            # Center control
            if (r, c) in CENTER_SQUARES:
                val += 1
            
            # Advancement (Promotion potential)
            if piece == 1: # White man
                val += r * 0.5 # Encourage moving up
                if r == 7: val += 5 # Bonus for crown
            elif piece == -1: # Black man
                val += (7 - r) * 0.5 # Encourage moving down
                if r == 0: val += 5 # Bonus for crown
            
            # Edge penalty
            if r == 0 or r == 7 or c == 0 or c == 7:
                val -= 0.5 
                
            # Assign to side
            if (piece > 0 and color == 'w') or (piece < 0 and color == 'b'):
                score += val
            else:
                score -= val
                
    return int(score)

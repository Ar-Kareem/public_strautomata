The strategy for this policy is to implement a greedy approach focusing on immediate captures when available, and otherwise making safe moves that improve position. The policy prioritizes:
1. Forced captures (jump moves) when available, as checkers requires mandatory captures
2. King promotion for regular pieces when safely possible
3. Moving pieces towards the opponent's side of the board
4. Avoiding moves that place pieces in vulnerable positions

The algorithm will:
1. First search for any available capture moves for both regular pieces and kings
2. If captures are available, return one of them
3. If no captures, prioritize moves that promote pieces to kings
4. Otherwise, make forward moves that don't expose pieces to capture

def policy(my_men, my_kings, opp_men, opp_kings, color) -> tuple[tuple[int, int], tuple[int, int]]:
    # Combine all pieces
    my_pieces = set(my_men + my_kings)
    opp_pieces = set(opp_men + opp_kings)
    
    # Direction of movement based on color
    if color == 'b':  # Black moves down (decreasing row numbers)
        directions = [(-1, -1), (-1, 1)]  # Down-left, down-right
        king_directions = [(-1, -1), (-1, 1), (1, -1), (1, 1)]
    else:  # White moves up (increasing row numbers)
        directions = [(1, -1), (1, 1)]   # Up-left, up-right
        king_directions = [(-1, -1), (-1, 1), (1, -1), (1, 1)]
    
    # Find capture moves
    capture_moves = []
    
    # Check captures for regular pieces
    for piece in my_men:
        row, col = piece
        # Check both forward diagonals for regular pieces
        for dr, dc in directions:
            # Jump over opponent piece
            jump_row, jump_col = row + dr, col + dc
            land_row, land_col = row + 2*dr, col + 2*dc
            
            # Check if jump position is occupied by opponent
            if (jump_row, jump_col) in opp_pieces:
                # Check if landing position is on board and empty
                if 0 <= land_row <= 7 and 0 <= land_col <= 7:
                    if (land_row, land_col) not in my_pieces and (land_row, land_col) not in opp_pieces:
                        # Valid capture move
                        capture_moves.append((piece, (land_row, land_col)))
    
    # Check captures for kings (can move in all directions)
    for piece in my_kings:
        row, col = piece
        for dr, dc in king_directions:
            # Jump over opponent piece
            jump_row, jump_col = row + dr, col + dc
            land_row, land_col = row + 2*dr, col + 2*dc
            
            # Check if jump position is occupied by opponent
            if (jump_row, jump_col) in opp_pieces:
                # Check if landing position is on board and empty
                if 0 <= land_row <= 7 and 0 <= land_col <= 7:
                    if (land_row, land_col) not in my_pieces and (land_row, land_col) not in opp_pieces:
                        # Valid capture move
                        capture_moves.append((piece, (land_row, land_col)))
    
    # If captures are available, return one of them (mandatory rule in checkers)
    if capture_moves:
        return capture_moves[0]
    
    # No captures, find regular moves
    regular_moves = []
    
    # Regular piece moves
    for piece in my_men:
        row, col = piece
        for dr, dc in directions:
            new_row, new_col = row + dr, col + dc
            # Check if new position is on board and empty
            if 0 <= new_row <= 7 and 0 <= new_col <= 7:
                if (new_row, new_col) not in my_pieces and (new_row, new_col) not in opp_pieces:
                    # Valid move
                    regular_moves.append((piece, (new_row, new_col)))
    
    # King moves
    for piece in my_kings:
        row, col = piece
        for dr, dc in king_directions:
            new_row, new_col = row + dr, col + dc
            # Check if new position is on board and empty
            if 0 <= new_row <= 7 and 0 <= new_col <= 7:
                if (new_row, new_col) not in my_pieces and (new_row, new_col) not in opp_pieces:
                    # Valid move
                    regular_moves.append((piece, (new_row, new_col)))
    
    # Prioritize moves that promote to king
    promotion_moves = []
    for move in regular_moves:
        from_pos, to_pos = move
        from_row, from_col = from_pos
        to_row, to_col = to_pos
        
        # Check if this move promotes a piece
        if color == 'b' and to_row == 0:  # Black piece reaching row 0
            promotion_moves.append(move)
        elif color == 'w' and to_row == 7:  # White piece reaching row 7
            promotion_moves.append(move)
    
    # If promotion moves are available, return one of them
    if promotion_moves:
        return promotion_moves[0]
    
    # If no captures or promotions, return any valid move
    if regular_moves:
        return regular_moves[0]
    
    # Should never reach here in a valid game state
    # But just in case, return the first possible move
    return ((0, 0), (0, 0))

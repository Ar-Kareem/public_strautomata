The strategy for this policy is a combination of maximizing captures, prioritizing king safety, advancing pieces towards the opponent's side, and preferring moves that lead to central positions or protect other pieces. The policy first checks for available captures and executes the longest available capture sequence. If no captures are available, it evaluates non-capture moves. The evaluation takes into account piece advancement (closer to king row), centralization (closer to the center of the board), king protection (protecting kings), and protection of other pieces. The highest-scoring move is then returned. If multiple moves have the same high score, a random choice is made among them. Handling of edge cases and illegal moves is implicit in the capture/legal move generation.

def policy(my_men, my_kings, opp_men, opp_kings, color):
    """
    Selects the next move for a Checkers game.

    Args:
        my_men: A list of squares containing your regular pieces.
        my_kings: A list of squares containing your kings.
        opp_men: A list of squares containing opponent regular pieces.
        opp_kings: A list of squares containing opponent kings.

        board[piece] = ('W' if color == 'b' else 'B')

        Finds all legal moves for the given color.

        for piece in all_pieces:
            captures = find_captures(board, piece, color)

                must_capture = True

                     legal_moves.append(capture)

        if not must_capture:
            for piece in all_pieces:
                moves = find_moves(board, piece, color)
                legal_moves.extend(moves)
        return legal_moves

    def find_moves(board, piece, color):

        """
        row, col = piece
        moves = []

            if 0 <= new_row <= 7 and 0 <= new_col <= 7 and 0 <= jump_row <= 7 and 0 <= jump_col <= 7:
                if board.get((jump_row, jump_col)) in opponent_colors and board.get((new_row, new_col)) is None:
                    return True
            return False
        
        if color == 'b' or king:

            if row > 1:

                    captures.append(((row, col), (row + 2, col + 2), (row + 1, col + 1)))
        return captures
    
    def execute_move(board, move, color):
          """Executes a move and returns the updated board."""
          from_row, from_col = move[0]
          to_row, to_col = move[1]
          original_piece = board[(from_row, from_col)]
          board[(to_row, to_col)] = original_piece
          board[(from_row, from_col)] = None

          # Handle promotion to king
          if color == 'b' and to_row == 0 and original_piece == 'b':
              board[(to_row, to_col)] = 'B'
          elif color == 'w' and to_row == 7 and original_piece == 'w':
              board[(to_row, to_col)] = 'W'

          # Remove captured piece in capture moves
          if len(move) == 3:
              captured_row, captured_col = move[2]
              board[(captured_row, captured_col)] = None

          return board

        if (color == 'b' and to_row == 0 and piece == 'b') or (color == 'w' and to_row == 7 and piece == 'w'):

        if (from_col != 0 and from_col != 7) and (to_col == 0 or to_col == 7):

            dummy_moves = find_moves(board, piece, color)
            if dummy_moves:
               return dummy_moves[0]
        return ((0,1),(1,0)) # failsafe


    # If captures are available, find the longest sequence
    capture_moves = [move for move in legal_moves if len(move) == 3]
    if capture_moves:

      def find_all_capture_sequences(board, move, color, sequence):
          """Recursively finds all possible capture sequences starting from a move."""

          row, col = move[1]  # The destination of the move becomes the new starting position
          new_piece = new_board[(row, col)] #Piece at destination after move

          further_captures = find_captures(new_board, (row, col), color)

              for capture in further_captures:

      max_length = 0

      #If there are several equally longest sequences, select based on evaluation

          for sequence in longest_sequences:
              sequence_score = 0

              if sequence_score > best_score:

          return longest_sequences[0][0][0], longest_sequences[0][0][1]

        # If no captures are available, evaluate each move
        best_move = None
        best_score = float('-inf')

        for move in legal_moves:

            if score > best_score:
                best_score = score
                best_move = move
        
        if best_move is not None:

            # If no move was better than -inf, then return a random move in legal_moves
            return random.choice(legal_moves)[0], random.choice(legal_moves)[1]

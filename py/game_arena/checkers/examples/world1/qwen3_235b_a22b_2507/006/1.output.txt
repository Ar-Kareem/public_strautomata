The strategy for this policy is to simulate a depth-limited minimax search with alpha-beta pruning to evaluate future board states and select high-value moves. The policy prioritizes:

1. **Mandatory captures**: Any available capture must be taken — the policy first checks and generates all possible capture moves.
2. **Piece value hierarchy**: Kings are more valuable than regular pieces. Losing a piece is penalized heavily, and promoting a man to a king is rewarded.
3. **Positional advantage**: Pieces closer to promotion (for men) are valued higher. Back-row pieces are slightly favored to avoid unnecessary exposure.
4. **Move ordering for efficiency**: Capture moves and promotion moves are explored first in the minimax search to improve alpha-beta pruning efficiency.
5. **Depth control**: The search depth is adjusted dynamically based on the number of pieces on the board to stay within the 1-second time limit.

The evaluation function considers:
- Material count (kings weighted higher)
- Piece advancement (especially for men)
- Mobility (number of available moves)
- Avoiding piece loss on the next opponent turn (tempo)

All moves are generated legally, and only valid dark squares are considered.

# Black moves "up" (row decreases), White moves "down" (row increases)
def get_soon_directions(color: str) -> List[Tuple[int, int]]:
    dr = -1 if color == BLACK else 1
    return [(dr, -1), (dr, 1)]

def get_king_directions() -> List[Tuple[int, int]]:
    return [(-1, -1), (-1, 1), (1, -1), (1, 1)]

def is_in_bounds(r: int, c: int) -> bool:
    return 0 <= r < 8 and 0 <= c < 8

def get_enemy_pieces(opp_men, opp_kings):
    return set(opp_men) | set(opp_kings)

def get_my_pieces(my_men, my_kings):
    return set(my_men) | set(my_kings)

def get_board_pieces(my_men, my_kings, opp_men, opp_kings):
    return get_my_pieces(my_men, my_kings) | get_enemy_pieces(opp_men, opp_kings)

def get_jumped_square(fr: Tuple[int, int], to: Tuple[int, int]) -> Tuple[int, int]:

    directions = get_soon_directions(color)

        if not is_in_bounds(nr, nc):

            if is_in_bounds(nr2, nc2) and (nr2, nc2) not in all_pieces:
                moves.append(((r, c), (nr2, nc2), {(nr, nc)}))

def king_moves(piece: Tuple[int, int], opp_pieces: Set[Tuple[int, int]], all_pieces: Set[Tuple[int, int]]) -> List[Tuple[Tuple[int, int], Tuple[int, int], Set[Tuple[int, int]]]]:

    directions = get_king_directions()
    r, c = piece
    for dr, dc in directions:
        # Slide in that direction

                break

def king_continuation(piece: Tuple[int, int], opp_pieces: Set[Tuple[int, int]], all_pieces: Set[Tuple[int, int]], jumped: Set[Tuple[int, int]]) -> List[Tuple[Tuple[int, int], Tuple[int, int], Set[Tuple[int, int]]]]:
    moves = []
    directions = get_king_directions()
    r, c = piece
    for dr, dc in directions:
        nr, nc = r + dr, c + dc
        while is_in_bounds(nr, nc):
            if (nr, nc) in opp_pieces and (nr, nc) not in jumped:
                nr2, nc2 = nr + dr, nc + dc
                if is_in_bounds(nr2, nc2) and (nr2, nc2) not in all_pieces:
                    new_jumped = jumped | {(nr, nc)}
                    moves.append(((r, c), (nr2, nc2), new_jumped))
                    sub_moves = king_continuation((nr2, nc2), opp_pieces - {(nr, nc)}, all_pieces - {(nr, nc), (nr2, nc2)}, new_jumped)
                    for _, to_pos, jump_set in sub_moves:

                break

                break
            nr, nc = nr + dr, nc + dc

def generate_moves(my_men, my_kings, opp_men, opp_kings, color) -> List[Tuple[Tuple[int, int], Tuple[int, int], Set[Tuple[int, int]]]]:
    opp_pieces = get_enemy_pieces(opp_men, opp_kings)
    all_pieces = get_board_pieces(my_men, my_kings, opp_men, opp_kings)
    moves = []

    # First, check for captures
    capture_available = False

    # Check men captures
    for piece in my_men:
        jump_moves = man_continuation(piece, color, opp_pieces, all_pieces, set())
        if jump_moves:
            capture_available = True
            moves.extend(jump_moves)

    # Check kings captures
    for piece in my_kings:
        jump_moves = king_continuation(piece, opp_pieces, all_pieces, set())
        if jump_moves:
            capture_available = True
            moves.extend(jump_moves)

    if capture_available:
        return moves

    # No capture, generate regular moves
    for piece in my_men:
        base_moves = man_moves(piece, color, opp_pieces, all_pieces)
        for move in base_moves:
            # Only include non-capture moves
            if not move[2]:  # no jumped pieces
                moves.append(move)

    for piece in my_kings:
        base_moves = king_moves(piece, opp_pieces, all_pieces)
        for move in base_moves:
            if not move[2]:
                moves.append(move)

    return moves

def apply_move(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]], 
               opp_men: List[Tuple[int, int]], opp_kings: List[Tuple[int, int]], 
               move: Tuple[Tuple[int, int], Tuple[int, int], Set[Tuple[int, int]]], color: str):
    fr, to, jumped = move
    fr_r, fr_c = fr
    to_r, to_c = to

    new_my_men = [p for p in my_men if p != fr]
    new_my_kings = [p for p in my_kings if p != fr]
    new_opp_men = [p for p in opp_men]
    new_opp_kings = [p for p in opp_kings]

    # Remove jumped pieces
    for j in jumped:
        if j in new_opp_men:
            new_opp_men.remove(j)
        elif j in new_opp_kings:
            new_opp_kings.remove(j)

    # Promote if needed
    is_promoted = False
    if fr in my_men:
        if (color == BLACK and to_r == 0) or (color == WHITE and to_r == 7):
            new_my_kings.append(to)
            is_promoted = True
        else:
            new_my_men.append(to)
    elif fr in my_kings:
        new_my_kings.append(to)
    else:
        raise ValueError(f"Piece at {fr} not found in player's pieces")

    return new_my_men, new_my_kings, new_opp_men, new_opp_kings, is_promoted

def evaluate(my_men, my_kings, opp_men, opp_kings, color) -> float:
    score = 0.0

    # Material values
    man_value = 1.0
    king_value = 1.5

    score += len(my_men) * man_value
    score += len(my_kings) * king_value
    score -= len(opp_men) * man_value
    score -= len(opp_kings) * king_value

    # Positional bonus for men: closer to promotion
    for r, c in my_men:
        if color == BLACK:
            dist = r  # want to minimize r
            score += (7 - dist) * 0.05
        else:
            dist = 7 - r  # want to maximize r
            score += dist * 0.05

    # Avoid being captured: pieces near edges safer?
    # Encourage kings
    if len(my_kings) > 0 and len(my_men) == 0:
        score += 0.1

    # Slight preference for center control?
    center_squares = {(3,3), (3,4), (4,3), (4,4)}
    for r, c in my_men + my_kings:
        if (r, c) in center_squares:
            score += 0.05

    # Discourage having pieces in the back row if not kings? They can't move backward
    for r, c in my_men:
        if (color == BLACK and r == 7) or (color == WHITE and r == 0):
            score -= 0.1  # unless they can be promoted soon, but in back row they are stuck

    return score

def minimax(my_men, my_kings, opp_men, opp_kings, color, depth, alpha, beta, is_maximizing, start_color) -> float:
    if depth == 0:
        return evaluate(my_men, my_kings, opp_men, opp_kings, start_color)

    moves = generate_moves(my_men, my_kings, opp_men, opp_kings, color)

    if not moves:
        # No moves available: losing state
        return -1000 if is_maximizing else 1000

    # Move ordering: prioritize captures and promotions
    def move_score(move):
        fr, to, jumped = move
        score = 0
        if jumped:
            score += 2
        # Promotion?
        if fr in my_men:
            if (start_color == BLACK and to[0] == 0) or (start_color == WHITE and to[0] == 7):
                score += 1
        return -score  # negative because we sort later

    moves.sort(key=move_score)

    if is_maximizing:
        value = -float('inf')
        for move in moves:
            nm, nk, om, ok, promoted = apply_move(my_men, my_kings, opp_men, opp_kings, move, color)
            next_color = 'w' if color == 'b' else 'b'
            eval_score = minimax(nm, nk, om, ok, next_color, depth - 1, alpha, beta, False, start_color)
            value = max(value, eval_score)
            alpha = max(alpha, value)
            if beta <= alpha:
                break
        return value
    else:
        value = float('inf')
        for move in moves:
            nm, nk, om, ok, promoted = apply_move(my_men, my_kings, opp_men, opp_kings, move, color)
            next_color = 'w' if color == 'b' else 'b'
            eval_score = minimax(nm, nk, om, ok, next_color, depth - 1, alpha, beta, True, start_color)
            value = min(value, eval_score)
            beta = min(beta, value)
            if beta <= alpha:
                break
        return value

def select_best_move(my_men, my_kings, opp_men, opp_kings, color) -> Tuple[Tuple[int, int], Tuple[int, int]]:
    moves = generate_moves(my_men, my_kings, opp_men, opp_kings, color)

    if not moves:
        raise ValueError("No moves available")

    # Mandatory capture: if any move involves capture, only consider captures
    capturing_moves = [m for m in moves if m[2]]
    if capturing_moves:
        moves = capturing_moves

    # Dynamic depth based on number of pieces
    total_pieces = len(my_men) + len(my_kings) + len(opp_men) + len(opp_kings)
    if total_pieces > 20:
        depth = 3
    elif total_pieces > 12:
        depth = 4
    else:
        depth = 5

    best_move = moves[0]
    best_value = -float('inf')

    # Move ordering for root
    def root_move_score(move):
        fr, to, jumped = move
        score = 0
        if jumped:
            score += 2
        if fr in my_men:
            if (color == BLACK and to[0] == 0) or (color == WHITE and to[0] == 7):
                score += 1
        return -score

    moves.sort(key=root_move_score)

    for move in moves:
        nm, nk, om, ok, promoted = apply_move(my_men, my_kings, opp_men, opp_kings, move, color)
        next_color = 'w' if color == 'b' else 'b'
        value = minimax(nm, nk, om, ok, next_color, depth - 1, -float('inf'), float('inf'), False, color)
        if value > best_value:
            best_value = value
            best_move = move

    return best_move[0], best_move[1]

def policy(my_men: List[Tuple[int, int]], my_kings: List[Tuple[int, int]], 
           opp_men: List[Tuple[int, int]], opp_kings: List[Tuple[int, int]], 
           color: str) -> Tuple[Tuple[int, int], Tuple[int, int]]:
    try:
        move = select_best_move(my_men, my_kings, opp_men, opp_kings, color)
        # Validate the move is legal
        from_pos, to_pos = move
        all_my_pieces = set(my_men) | set(my_kings)
        if from_pos not in all_my_pieces:
            raise ValueError(f"From position {from_pos} not in player pieces")
        if to_pos[0] < 0 or to_pos[0] > 7 or to_pos[1] < 0 or to_pos[1] > 7:
            raise ValueError(f"To position {to_pos} out of bounds")
        if not is_dark_square(to_pos[0], to_pos[1]):
            raise ValueError(f"To position {to_pos} is not a dark square")
        # Fallback: if error, return first legal move
        return move
    except Exception as e:
        # Fallback strategy: make any legal move
        moves = generate_moves(my_men, my_kings, opp_men, opp_kings, color)
        capturing_moves = [m for m in moves if m[2]]
        if capturing_moves:
            return capturing_moves[0][0], capturing_moves[0][1]
        elif moves:
            return moves[0][0], moves[0][1]
        else:
            # Last resort: if no moves, return a dummy
            all_pieces = set(my_men) | set(my_kings)
            if all_pieces:
                p = next(iter(all_pieces))
                return p, p
            return (0, 0), (0, 0)

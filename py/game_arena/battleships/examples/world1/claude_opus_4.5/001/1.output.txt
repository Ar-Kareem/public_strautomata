Key observations:
1. Ships have lengths 5, 4, 3, 3, 2 (total 17 cells)
2. I need to find all ship cells efficiently
3. Two main phases: hunting (finding ships) and targeting (sinking found ships)

Strategy:
1. **Target mode**: If there are hits (value 1) that likely belong to unsunk ships, I should target adjacent cells to finish off the ship.
2. **Hunt mode**: When no active hits to follow up on, I should search efficiently. A checkerboard pattern is good because ships must occupy at least 2 cells, so every ship will have at least one cell on the checkerboard.

For targeting:
- Find all hits (1s) that are adjacent to unknowns (0s)
- Prioritize cells that are adjacent to multiple hits (likely along the ship's direction)
- If we have two adjacent hits, continue in that line

For hunting:
- Use a probability-based approach or a simple checkerboard
- The checkerboard can be refined - larger ships can only fit in certain places
- Prioritize cells that could contain more possible ship placements

def policy(board: list[list[int]]) -> tuple[int, int]:
    import random
    
    # Find all cells of each type
    unknown = []
    hits = []
    
    for r in range(10):
        for c in range(10):
            if board[r][c] == 0:
                unknown.append((r, c))
            elif board[r][c] == 1:
                hits.append((r, c))
    
    unknown_set = set(unknown)
    hits_set = set(hits)
    
    # Helper to get adjacent cells
    def adjacent(r, c):
        adj = []
        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < 10 and 0 <= nc < 10:
                adj.append((nr, nc))
        return adj
    
    # Find hits that have unknown neighbors (active hits)
    active_hits = []
    for r, c in hits:
        if any((nr, nc) in unknown_set for nr, nc in adjacent(r, c)):
            active_hits.append((r, c))
    
    if active_hits:
        # Target mode - try to sink ships
        # Look for lines of hits
        target_scores = {}
        
        for r, c in active_hits:
            # Check if this hit is part of a line
            # Horizontal check
            h_line = [(r, c)]
            # Go left
            nc = c - 1
            while nc >= 0 and (r, nc) in hits_set:
                h_line.append((r, nc))
                nc -= 1
            # Go right
            nc = c + 1

                h_line.append((r, nc))
                nc += 1
            
            # Vertical check
            v_line = [(r, c)]

                v_line.append((nr, c))
                nr += 1

                min_c = min(cc for _, cc in h_line)

                    if (nr, nc) in unknown_set:
                        target_scores[(nr, nc)] = target_scores.get((nr, nc), 0) + 5

            best = max(target_scores, key=target_scores.get)
            return best
    
    # Hunt mode - use checkerboard pattern with probability weighting
    # Prefer cells where more ships could fit
    ship_lengths = [5, 4, 3, 3, 2]

                    score += 1
        
        cell_scores[(r, c)] = score

        max_score = max(cell_scores.values())
        best_cells = [cell for cell, score in cell_scores.items() if score == max_score]
        return random.choice(best_cells)
    
    # Fallback
    return unknown[0]

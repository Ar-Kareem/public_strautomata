def policy(board: list[list[int]]) -> tuple[int, int]:
    import numpy as np
    
    # Ship lengths
    ships = [5, 4, 3, 3, 2]
    
    # Convert board to numpy array
    b = np.array(board)
    
    # Find hits and try to determine which ships are sunk
    hits = set(zip(*np.where(b == 1)))
    misses = set(zip(*np.where(b == -1)))
    unknown = set(zip(*np.where(b == 0)))
    
    # Calculate probability for each cell
    prob = np.zeros((10, 10))
    
    # For each ship, count how many ways it can be placed covering each cell
    for ship_len in ships:
        # Horizontal placements
        for r in range(10):
            for c in range(10 - ship_len + 1):
                cells = [(r, c + i) for i in range(ship_len)]
                # Check if placement is valid (no misses)
                if all(b[r, c + i] != -1 for i in range(ship_len)):
                    for cell in cells:
                        if cell in unknown:
                            prob[cell] += 1
        
        # Vertical placements
        for r in range(10 - ship_len + 1):
            for c in range(10):
                cells = [(r + i, c) for i in range(ship_len)]
                # Check if placement is valid (no misses)
                if all(b[r + i, c] != -1 for i in range(ship_len)):
                    for cell in cells:
                        if cell in unknown:
                            prob[cell] += 1
    
    # Prioritize cells adjacent to hits
    for hr, hc in hits:
        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nr, nc = hr + dr, hc + dc
            if 0 <= nr < 10 and 0 <= nc < 10 and (nr, nc) in unknown:
                prob[nr, nc] += 1000  # High priority
    
    # Find the cell with highest probability
    prob[b != 0] = -1  # Can't fire at already-fired cells
    
    best = np.unravel_index(np.argmax(prob), prob.shape)
    return (int(best[0]), int(best[1]))

    b = np.array(board)
    ships = [5, 4, 3, 3, 2]
    
    hits = set(zip(*np.where(b == 1)))
    unknown = set(zip(*np.where(b == 0)))
    
    if not unknown:
        # Shouldn't happen, but fallback
        for r in range(10):
            for c in range(10):
                if board[r][c] == 0:
                    return (r, c)
    
    # Find connected components of hits to identify potential unsunk ships
    def get_neighbors(r, c):
        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nr, nc = r + dr, c + dc
            if 0 <= nr < 10 and 0 <= nc < 10:

    # Find hit cells that have unknown neighbors (not fully explored)
    active_hits = []
    for hr, hc in hits:
        for nr, nc in get_neighbors(hr, hc):

                active_hits.append((hr, hc))

    prob = np.zeros((10, 10))

    if active_hits:

            # Horizontal line
            h_line = [(hr, hc)]

                nc = hc + dc
                while 0 <= nc < 10 and (hr, nc) in hits:
                    h_line.append((hr, nc))

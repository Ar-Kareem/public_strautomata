import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Step 1: Find all hit cells
    hits = [(r, c) for r in range(10) for c in range(10) if board[r][c] == 1]
    
    # Step 2: Check for possible adjacent shots to hits
    possible_shots = []
    for (r, c) in hits:
        # Check all four directions
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if 0 <= nr < 10 and 0 <= nc < 10:
                if board[nr][nc] == 0:
                    possible_shots.append( (nr, nc) )
    
    if possible_shots:
        # Return the first possible shot (could prioritize based on probability if needed)
        return possible_shots[0]
    else:
        # Hunting phase: calculate probability grid
        prob = [[0 for _ in range(10)] for _ in range(10)]
        original_ships = [5, 4, 3, 3, 2]
        
        for ship_length in original_ships:
            for r in range(10):
                for c in range(10):
                    # Check horizontal placement
                    if c + ship_length <= 10:
                        valid = True
                        for i in range(ship_length):
                            if board[r][c + i] in [-1, 1]:
                                valid = False
                                break
                        if valid:
                            for i in range(ship_length):
                                prob[r][c + i] += 1
                    # Check vertical placement
                    if r + ship_length <= 10:
                        valid = True
                        for i in range(ship_length):
                            if board[r + i][c] in [-1, 1]:
                                valid = False
                                break
                        if valid:
                            for i in range(ship_length):
                                prob[r + i][c] += 1
        
        # Find the best cell with max probability and unshot
        max_prob = -1
        best_cell = (0, 0)
        for r in range(10):
            for c in range(10):
                if board[r][c] == 0:
                    if prob[r][c] > max_prob:
                        max_prob = prob[r][c]
                        best_cell = (r, c)
        return best_cell



The strategy focuses on efficiently locating and sinking ships by prioritizing cells adjacent to hits while maintaining a fallback to random searches. It leverages the current board state to make informed decisions without persistent memory between calls.

import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Identify all hit cells (value 1)
    hits = []
    for i in range(10):
        for j in range(10):
            if board[i][j] == 1:
                hits.append((i, j))
    
    # Check adjacent cells of all hits for potential ship segments
    candidates = []
    for (i, j) in hits:
        for dx, dy in [(1,0), (-1,0), (0,1), (0,-1)]:
            ni, nj = i + dx, j + dy
            if 0 <= ni < 10 and 0 <= nj < 10 and board[ni][nj] == 0:
                candidates.append((ni, nj))
    
    if candidates:
        # Prioritize adjacent cells to hits
        return random.choice(candidates)
    else:
        # Fallback to random search if no adjacent cells are available
        available = []
        for i in range(10):
            for j in range(10):
                if board[i][j] == 0:
                    available.append((i, j))
        if available:
            return random.choice(available)
        else:
            # Safe default (should not occur in valid games)
            return (0, 0)

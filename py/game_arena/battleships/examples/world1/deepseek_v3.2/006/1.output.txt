def policy(board: list[list[int]]) -> tuple[int, int]:

    Targeting policy for Battleship game.
    Returns (row, col) of next shot.
    """

        if 0 <= r < SIZE and 0 <= c < SIZE:

                    nr, nc = r + dr, c + dc

                        break
                if adj_unknown:
                    hits.append((r, c))

        visited = set()

The strategy for this policy is a weighted probability search:

1. In hunt mode, it counts how many legal ship placements of lengths `5, 4, 3, 3, 2` can cover each unknown cell, preferring central and checkerboard cells indirectly through placement counts.
2. In target mode, placements that include known hits are weighted much more heavily, so the policy aggressively finishes discovered ships.
3. It adds extra tactical bonuses for:
   - extending a straight run of hits,
   - filling a one-cell gap between hits (`H ? H`),
   - being orthogonally adjacent to hits.
4. It always falls back to a legal unknown cell, so it never returns an already-fired move.

SHIP_LENGTHS = (5, 4, 3, 3, 2)
N = 10
MAX_SHIP = 5


def policy(board: list[list[int]]) -> tuple[int, int]:
    scores = [[0.0 for _ in range(N)] for _ in range(N)]
    unknowns = []
    has_hit = False

    for r in range(N):
        for c in range(N):
            v = board[r][c]
            if v == 0:
                unknowns.append((r, c))
            elif v == 1:
                has_hit = True

    # Safety fallback; in a normal game this should not happen.
    if not unknowns:
        return (0, 0)

    def placement_weight(hit_positions: list[int]) -> float:
        hits = len(hit_positions)
        if hits == 0:
            return 1.0
        # Strongly prefer placements that explain known hits.
        w = 20.0 ** hits
        # Extra bonus if the hits inside the placement are contiguous.
        if hits > 1 and hit_positions[-1] - hit_positions[0] + 1 == hits:
            w *= 2.0
        return w

    # 1) Probabilistic placement scoring over all ship placements
    for L in SHIP_LENGTHS:
        # Horizontal placements
        for r in range(N):
            for c0 in range(N - L + 1):
                hit_positions = []
                unknown_cells = []
                valid = True

                for i in range(L):
                    v = board[r][c0 + i]
                    if v == -1:
                        valid = False
                        break
                    if v == 1:
                        hit_positions.append(i)
                    elif v == 0:
                        unknown_cells.append((r, c0 + i))

                if valid and unknown_cells:
                    w = placement_weight(hit_positions)
                    for rr, cc in unknown_cells:
                        scores[rr][cc] += w

        # Vertical placements
        for c in range(N):
            for r0 in range(N - L + 1):
                hit_positions = []
                unknown_cells = []
                valid = True

                for i in range(L):
                    v = board[r0 + i][c]
                    if v == -1:
                        valid = False
                        break
                    if v == 1:
                        hit_positions.append(i)
                    elif v == 0:
                        unknown_cells.append((r0 + i, c))

                if valid and unknown_cells:
                    w = placement_weight(hit_positions)
                    for rr, cc in unknown_cells:
                        scores[rr][cc] += w

    # 2) Tactical bonuses for extending hit runs in rows
    for r in range(N):
        c = 0
        while c &lt; N:
            if board[r][c] == 1:
                start = c
                while c &lt; N and board[r][c] == 1:
                    c += 1
                end = c - 1
                run_len = end - start + 1

                if run_len &lt; MAX_SHIP:
                    bonus = 40.0 * (run_len ** 3)
                    if start - 1 &gt;= 0 and board[r][start - 1] == 0:
                        scores[r][start - 1] += bonus
                    if end + 1 &lt; N and board[r][end + 1] == 0:
                        scores[r][end + 1] += bonus
            else:
                c += 1

    # 3) Tactical bonuses for extending hit runs in columns
    for c in range(N):
        r = 0
        while r &lt; N:
            if board[r][c] == 1:
                start = r
                while r &lt; N and board[r][c] == 1:
                    r += 1
                end = r - 1
                run_len = end - start + 1

                if run_len &lt; MAX_SHIP:
                    bonus = 40.0 * (run_len ** 3)
                    if start - 1 &gt;= 0 and board[start - 1][c] == 0:
                        scores[start - 1][c] += bonus
                    if end + 1 &lt; N and board[end + 1][c] == 0:
                        scores[end + 1][c] += bonus
            else:
                r += 1

    # 4) Gap-filling and local hit-adjacency bonuses
    for r, c in unknowns:
        # Fill H ? H gaps
        if c - 1 &gt;= 0 and c + 1 &lt; N and board[r][c - 1] == 1 and board[r][c + 1] == 1:
            scores[r][c] += 1000.0
        if r - 1 &gt;= 0 and r + 1 &lt; N and board[r - 1][c] == 1 and board[r + 1][c] == 1:
            scores[r][c] += 1000.0

        # Mild bonus for being adjacent to hits
        adj_hits = 0
        if r &gt; 0 and board[r - 1][c] == 1:
            adj_hits += 1
        if r + 1 &lt; N and board[r + 1][c] == 1:
            adj_hits += 1
        if c &gt; 0 and board[r][c - 1] == 1:
            adj_hits += 1
        if c + 1 &lt; N and board[r][c + 1] == 1:
            adj_hits += 1
        scores[r][c] += 5.0 * adj_hits

    # 5) Hunt-mode parity preference when no hits are visible
    if not has_hit:
        parity_exists = any((r + c) % 2 == 0 for r, c in unknowns)
        if parity_exists:
            for r, c in unknowns:
                if (r + c) % 2 == 0:
                    scores[r][c] *= 1.15
                else:
                    scores[r][c] *= 0.95

    # 6) Pick the best legal move; break ties toward the center
    best_cell = None
    best_score = None
    best_tie = None

    for r, c in unknowns:
        s = scores[r][c]
        center_tie = -abs(r - 4.5) - abs(c - 4.5)
        parity_tie = 1 if (r + c) % 2 == 0 else 0
        tie = (center_tie, parity_tie, -r, -c)

        if best_cell is None or s &gt; best_score or (s == best_score and tie &gt; best_tie):
            best_cell = (r, c)
            best_score = s
            best_tie = tie

    # Final safety fallback
    if best_cell is None or board[best_cell[0]][best_cell[1]] != 0:
        return unknowns[0]

    return best_cell

To create an effective Battleship targeting policy, I will implement a two-phase strategy:
1.  **Target Mode**: If there are known "hits" (1) on the board that haven't been fully explored, the AI will prioritize adjacent cells (up, down, left, right) to sink the ship. It will prioritize cells that continue a line of hits.
2.  **Hunt Mode**: If no hits are currently "active" (unresolved), the AI will use a parity/checkerboard strategy. By targeting cells where `(row + col) % 2 == 0`, we can find any ship (minimum length 2) while firing at only half the squares.

I will also add a simple "Heatmap/Density" logic for the Hunt mode: it will prefer cells that have the most empty space around them, increasing the probability of hitting a ship.

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Battleship targeting policy.
    0: Unknown, -1: Miss, 1: Hit
    """
    size = 10
    
    def is_valid(r, c):
        return 0 <= r < size and 0 <= c < size

    def get_neighbors(r, c):
        neighbors = []
        for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
            nr, nc = r + dr, c + dc
            if is_valid(nr, nc):
                neighbors.append((nr, nc))
        return neighbors

    # 1. Target Mode: Find hits that are not yet surrounded by misses or edge
    hits = []
    for r in range(size):
        for c in range(size):
            if board[r][c] == 1:
                hits.append((r, c))

    if hits:
        potential_targets = []
        for r, c in hits:
            # Check neighbors of this hit
            for nr, nc in get_neighbors(r, c):
                if board[nr][nc] == 0:
                    # Heuristic: Priority to squares that extend a line of hits
                    weight = 1
                    # Check opposite direction
                    dr, dc = nr - r, nc - c
                    opp_r, opp_c = r - dr, c - dc
                    if is_valid(opp_r, opp_c) and board[opp_r][opp_c] == 1:
                        weight += 2
                    
                    potential_targets.append((weight, (nr, nc)))
        
        if potential_targets:
            # Sort by weight (highest first) and pick one of the best
            potential_targets.sort(key=lambda x: x[0], reverse=True)
            max_weight = potential_targets[0][0]
            best_options = [tup[1] for tup in potential_targets if tup[0] == max_weight]
            return random.choice(best_options)

    # 2. Hunt Mode: Use a parity-based search (checkerboard)
    # Prefer cells that have the most 'unknown' neighbors to maximize ship-fit probability
    hunt_targets = []
    for r in range(size):
        for c in range(size):
            if board[r][c] == 0 and (r + c) % 2 == 0:
                # Calculate empty space around (Density Heuristic)
                score = 0
                for dr in range(-2, 3):
                    for dc in range(-2, 3):
                        tr, tc = r + dr, c + dc
                        if is_valid(tr, tc) and board[tr][tc] == 0:
                            score += 1
                hunt_targets.append((score, (r, c)))

    if hunt_targets:
        hunt_targets.sort(key=lambda x: x[0], reverse=True)
        # Choose from the top 5 densest areas to maintain some randomness
        top_candidates = hunt_targets[:5]
        return random.choice(top_candidates)[1]

    # 3. Fallback: Any valid cell (should rarely be reached)
    remaining = [(r, c) for r in range(size) for c in range(size) if board[r][c] == 0]
    return random.choice(remaining)

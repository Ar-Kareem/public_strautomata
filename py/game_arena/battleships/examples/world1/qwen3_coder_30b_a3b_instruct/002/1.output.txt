def policy(board: List[List[int]]) -> Tuple[int, int]:

    rows, cols = board.shape
    
    # First, let's look for any existing hits

    for i in range(rows):

            if board[i, j] == 1:
                hit_positions.append((i, j))

    if hit_positions:

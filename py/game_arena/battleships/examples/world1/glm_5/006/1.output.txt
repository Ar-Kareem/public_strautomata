The strategy for this policy combines Probability Density Map calculations with Target Mode logic to efficiently locate and destroy opponent ships.

1.  **Target Mode**: When hits (`1`) exist on the board, the policy identifies "clusters" of hits. It prioritizes extending existing lines of hits (e.g., turning a hit sequence of 3 into 4) over exploring neighbors of single isolated hits. This ensures we finish sinking ships we've already found before searching for new ones. It checks for valid ship placements relative to the remaining ships to avoid shooting into dead ends.

2.  **Hunt Mode**: When no active hits are found, the policy switches to a Hunt Mode using a Probability Density Map. It calculates the likelihood of each unknown cell (`0`) containing a ship by counting how many valid placements of the remaining ships pass through that cell. It prioritizes cells with the highest probability.

3.  **State Inference**: The policy infers which ships have been sunk by identifying clusters of hits that are fully surrounded by misses or board edges. It removes these ships from the list of possibilities, making subsequent probability calculations more accurate.

4.  **Parity**: In Hunt Mode, a checkerboard parity heuristic is used as a tie-breaker to minimize the search space, as ships of length 2 or greater must occupy at least one cell of a specific parity.

import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    N = 10
    SHIPS = [5, 4, 3, 3, 2]
    
    # Convert to numpy array for easier slicing if needed, but list of lists is fine.
    # board: 0=unknown, -1=miss, 1=hit
    
    # 1. Identify all hits and misses
    hits = []
    misses = []
    unknowns = []
    for r in range(N):
        for c in range(N):
            if board[r][c] == 1:
                hits.append((r, c))
            elif board[r][c] == -1:
                misses.append((r, c))
            else:
                unknowns.append((r, c))

    # If no history (start of game), return a high probability center point
    if not hits and not misses:
        return (5, 5)

    # 2. Cluster hits into groups (connected components)
    visited = set()
    clusters = []
    for h in hits:
        if h in visited:
            continue
        cluster = []
        q = [h]
        while q:
            curr = q.pop(0)
            if curr in visited:
                continue
            visited.add(curr)
            cluster.append(curr)
            r, c = curr
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < N and 0 <= nc < N and board[nr][nc] == 1:
                    if (nr, nc) not in visited:
                        q.append((nr, nc))
        clusters.append(cluster)

    # 3. Identify remaining ships (Sunk detection)
    remaining_ships = list(SHIPS)
    for cluster in clusters:
        # Check if cluster is surrounded by misses or edges
        is_closed = True
        for r, c in cluster:
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < N and 0 <= nc < N:
                    if board[nr][nc] == 0:
                        is_closed = False
                        break
            if not is_closed:
                break
        
        if is_closed:
            size = len(cluster)
            if size in remaining_ships:
                remaining_ships.remove(size)

    # 4. Target Mode: Prioritize extending existing hits
    active_clusters = []
    for cluster in clusters:
        has_unknown_neighbor = False
        for r, c in cluster:
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < N and 0 <= nc < N and board[nr][nc] == 0:
                    has_unknown_neighbor = True
                    break
            if has_unknown_neighbor:
                break
        if has_unknown_neighbor:
            active_clusters.append(cluster)

    # Sort clusters by size descending (prioritize finishing larger found ships)
    active_clusters.sort(key=len, reverse=True)

    for cluster in active_clusters:
        if len(cluster) > 1:
            # Determine orientation and get ends
            rows = [p[0] for p in cluster]
            cols = [p[1] for p in cluster]
            candidates = []
            
            if len(set(rows)) == 1: # Horizontal
                r = rows[0]
                min_c = min(cols)
                max_c = max(cols)
                # Check left
                if min_c > 0 and board[r][min_c-1] == 0:
                    candidates.append((r, min_c-1))
                # Check right
                if max_c < N-1 and board[r][max_c+1] == 0:
                    candidates.append((r, max_c+1))
            else: # Vertical
                c = cols[0]
                min_r = min(rows)
                max_r = max(rows)
                # Check up
                if min_r > 0 and board[min_r-1][c] == 0:
                    candidates.append((min_r-1, c))
                # Check down
                if max_r < N-1 and board[max_r+1][c] == 0:
                    candidates.append((max_r+1, c))
            
            if candidates:
                # Filter candidates that are valid extensions for remaining ships
                valid_candidates = []
                curr_len = len(cluster)
                for cand in candidates:
                    # Can any remaining ship fit this extended length?
                    # The current cluster is part of a ship. 
                    # If we extend, new_len = curr_len + 1.
                    # We need a ship of size >= new_len.
                    # Also, we should check if the specific geometry allows it (e.g. not hitting wall)
                    # But candidates are generated by checking 0, so they are geometrically valid.
                    if any(s >= curr_len + 1 for s in remaining_ships):
                        valid_candidates.append(cand)
                
                if valid_candidates:
                    return valid_candidates[0]
        
        else:
            # Single hit, check all 4 neighbors
            r, c = cluster[0]
            candidates = []
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < N and 0 <= nc < N and board[nr][nc] == 0:
                    candidates.append((nr, nc))
            
            if candidates:
                # Score candidates based on local probability (number of valid placements passing through)
                best_score = -1
                best_move = candidates[0]
                for cr, cc in candidates:
                    score = 0
                    for s_len in remaining_ships:
                        # Horizontal check
                        # ship at [cr, cc-k] ... [cr, cc + (s_len-1-k)]
                        for k in range(s_len):
                            sc = cc - k
                            if 0 <= sc <= N - s_len:
                                valid = True
                                covers_target = False
                                for i in range(s_len):
                                    cell_r, cell_c = cr, sc + i
                                    val = board[cell_r][cell_c]
                                    if val == -1: valid = False; break
                                    if (cell_r, cell_c) == (r, c): covers_target = True
                                    # Check if overlaps other hits (illegal in standard, but handle carefully)
                                    if val == 1 and (cell_r, cell_c) != (r, c): 
                                        valid = False; break
                                if valid and covers_target:
                                    score += 1
                        
                        # Vertical check
                        for k in range(s_len):
                            sr = cr - k
                            if 0 <= sr <= N - s_len:
                                valid = True
                                covers_target = False
                                for i in range(s_len):
                                    cell_r, cell_c = sr + i, cc
                                    val = board[cell_r][cell_c]
                                    if val == -1: valid = False; break
                                    if (cell_r, cell_c) == (r, c): covers_target = True
                                    if val == 1 and (cell_r, cell_c) != (r, c):
                                        valid = False; break
                                if valid and covers_target:
                                    score += 1
                    if score > best_score:
                        best_score = score
                        best_move = (cr, cc)
                return best_move

    # 5. Hunt Mode: Probability Density Map
    # Calculate probability for every 0 cell
    density = np.zeros((N, N))
    
    for s_len in remaining_ships:
        # Horizontal
        for r in range(N):
            for c in range(N - s_len + 1):
                # Check if placement is valid (no misses, no hits from sunk ships)
                # In Hunt Mode, hits should be processed or ignored. 
                # If we are in Hunt Mode, active_clusters is empty. 
                # So all 1s are sunk ships. We treat them as obstacles (-1 logic).
                valid = True
                for i in range(s_len):
                    if board[r][c+i] == -1 or board[r][c+i] == 1:
                        valid = False
                        break
                if valid:
                    for i in range(s_len):
                        density[r][c+i] += 1
        # Vertical
        for r in range(N - s_len + 1):
            for c in range(N):
                valid = True
                for i in range(s_len):
                    if board[r+i][c] == -1 or board[r+i][c] == 1:
                        valid = False
                        break
                if valid:
                    for i in range(s_len):
                        density[r+i][c] += 1

    # Find max density
    best_moves = []
    max_d = -1
    for r in range(N):
        for c in range(N):
            if board[r][c] == 0:
                if density[r][c] > max_d:
                    max_d = density[r][c]
                    best_moves = [(r, c)]
                elif density[r][c] == max_d:
                    best_moves.append((r, c))
    
    if not best_moves:
        # Fallback if everything somehow exhausted or logic error
        for r in range(N):
            for c in range(N):
                if board[r][c] == 0:
                    return (r, c)
        return (0, 0)

    # Tie-breaking: Parity
    # Determine which parity is "rarer" or just alternate.
    # Standard checkerboard parity
    # Count existing shots
    even_shots = 0
    odd_shots = 0
    for r in range(N):
        for c in range(N):
            if board[r][c] != 0:
                if (r + c) % 2 == 0:
                    even_shots += 1
                else:
                    odd_shots += 1
    
    target_parity = 0 if even_shots <= odd_shots else 1
    
    parity_moves = [m for m in best_moves if (m[0] + m[1]) % 2 == target_parity]
    if parity_moves:
        return parity_moves[0]
    
    return best_moves[0]

import random
from collections import deque

def policy(board: list[list[int]]) -> tuple[int, int]:
    n = len(board)
    
    # Find all hits that are not yet sunk
    hits = []
    for i in range(n):
        for j in range(n):
            if board[i][j] == 1:
                hits.append((i, j))
    
    # If we have hits, try to extend the ship
    if hits:
        # Try to find adjacent hits to determine ship direction
        for hit in hits:
            r, c = hit
            # Check if there's a horizontal or vertical line of hits
            # Check up
            if r > 0 and board[r-1][c] == 1:
                # Ship is likely vertical, check up and down
                if r > 0 and board[r-1][c] == 0:
                    return (r-1, c)
                elif r < n-1 and board[r+1][c] == 0:
                    return (r+1, c)
            # Check down
            elif r < n-1 and board[r+1][c] == 1:
                if r < n-1 and board[r+1][c] == 0:
                    return (r+1, c)
                elif r > 0 and board[r-1][c] == 0:
                    return (r-1, c)
            # Check left
            elif c > 0 and board[r][c-1] == 1:
                # Ship is likely horizontal, check left and right
                if c > 0 and board[r][c-1] == 0:
                    return (r, c-1)
                elif c < n-1 and board[r][c+1] == 0:
                    return (r, c+1)
            # Check right
            elif c < n-1 and board[r][c+1] == 1:
                if c < n-1 and board[r][c+1] == 0:
                    return (r, c+1)
                elif c > 0 and board[r][c-1] == 0:
                    return (r, c-1)
        
        # If no clear direction, try all adjacent cells for each hit
        directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
        candidates = []
        for hit in hits:
            r, c = hit
            for dr, dc in directions:
                nr, nc = r + dr, c + dc
                if 0 <= nr < n and 0 <= nc < n and board[nr][nc] == 0:
                    candidates.append((nr, nc))
        
        if candidates:
            # Prioritize cells that could lead to longer ships
            # Count how many potential extensions each candidate has
            scored_candidates = []
            for r, c in candidates:
                score = 0
                # Check in both directions along potential ship axis
                # Horizontal check
                left_space = 0
                for i in range(1, 5):
                    if c-i >= 0 and board[r][c-i] in [0, 1]:
                        left_space += 1
                    else:
                        break
                right_space = 0
                for i in range(1, 5):
                    if c+i < n and board[r][c+i] in [0, 1]:
                        right_space += 1
                    else:
                        break
                horizontal_score = left_space + right_space
                
                # Vertical check
                up_space = 0
                for i in range(1, 5):
                    if r-i >= 0 and board[r-i][c] in [0, 1]:
                        up_space += 1
                    else:
                        break
                down_space = 0
                for i in range(1, 5):
                    if r+i < n and board[r+i][c] in [0, 1]:
                        down_space += 1
                    else:
                        break
                vertical_score = up_space + down_space
                
                score = max(horizontal_score, vertical_score)
                scored_candidates.append((score, (r, c)))
            
            # Return the candidate with highest score
            if scored_candidates:
                scored_candidates.sort(reverse=True)
                return scored_candidates[0][1]
            
            # Fallback to random candidate
            return random.choice(candidates)
    
    # If no hits, use a checkerboard pattern for initial search
    # This targets every other square to maximize efficiency
    available_cells = []
    for i in range(n):
        for j in range(n):
            if board[i][j] == 0:
                # Checkerboard pattern - target squares where (i+j) is even
                if (i + j) % 2 == 0:
                    available_cells.append((i, j))
    
    # If no checkerboard cells available, consider all remaining
    if not available_cells:
        for i in range(n):
            for j in range(n):
                if board[i][j] == 0:
                    available_cells.append((i, j))
    
    # If we still have no moves, which shouldn't happen, pick randomly
    if not available_cells:
        for i in range(n):
            for j in range(n):
                if board[i][j] == 0:
                    available_cells.append((i, j))
    
    return random.choice(available_cells)

Here's a Python implementation of a targeting policy for the Battleship-like game. This policy uses a combination of random targeting for unknown areas and a more strategic approach for areas where hits have been registered:

```python
import random
import numpy as np
from collections import defaultdict

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert board to numpy array for easier manipulation
    board = np.array(board)
    n = board.shape[0]

    # Find all unknown cells (0)
    unknown_cells = np.argwhere(board == 0)

    # If no unknown cells left, return any cell (shouldn't happen in valid game)
    if len(unknown_cells) == 0:
        return (0, 0)

    # Get all hit cells (1)
    hit_cells = np.argwhere(board == 1)

    # If we have no hits yet, just pick a random unknown cell
    if len(hit_cells) == 0:
        return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])

    # Create a list of potential targets near hits
    potential_targets = []

    # For each hit, consider its neighbors and nearby cells
    for hit in hit_cells:
        row, col = hit
        # Check all 8 surrounding cells
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue  # skip the hit cell itself
                r, c = row + dr, col + dc
                if 0 <= r < n and 0 <= c < n and board[r, c] == 0:
                    potential_targets.append((r, c))

    # If we have potential targets near hits, prioritize them
    if potential_targets:
        # Sort by distance to nearest hit (closest first)
        potential_targets.sort(key=lambda x: min(
            np.linalg.norm(np.array(x) - hit, axis=1)
            for hit in hit_cells
        ))
        return potential_targets[0]

    # If no potential targets near hits (unlikely), fall back to random
    return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])
```

However, this basic approach can be improved with more sophisticated ship detection. Here's an enhanced version that attempts to detect ship lengths and directions:

```python
import random
import numpy as np
from collections import defaultdict

def policy(board: list[list[int]]) -> tuple[int, int]:
    board = np.array(board)
    n = board.shape[0]

    # Find all unknown cells
    unknown_cells = np.argwhere(board == 0)
    if len(unknown_cells) == 0:
        return (0, 0)

    # Find all hit cells
    hit_cells = np.argwhere(board == 1)
    if len(hit_cells) == 0:
        return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])

    # Try to find potential ship segments
    potential_targets = []

    # For each hit, try to find adjacent hits that might form a ship
    for i, hit in enumerate(hit_cells):
        row, col = hit
        # Check horizontal and vertical directions
        for dr, dc in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
            # Check in this direction
            r, c = row + dr, col + dc
            if 0 <= r < n and 0 <= c < n and board[r, c] == 1:
                # Found another hit in this direction, check how far it extends
                length = 1
                while True:
                    r, c = row + (length * dr), col + (length * dc)
                    if not (0 <= r < n and 0 <= c < n) or board[r, c] != 1:
                        break
                    length += 1

                # Check the cells adjacent to the ship ends
                for end in [1, length]:
                    r, c = row + (end * dr), col + (end * dc)
                    if 0 <= r < n and 0 <= c < n and board[r, c] == 0:
                        potential_targets.append((r, c))

    # If we found potential targets near ships, prioritize them
    if potential_targets:
        # Remove duplicates and sort by distance to nearest hit
        unique_targets = list(set(potential_targets))
        unique_targets.sort(key=lambda x: min(
            np.linalg.norm(np.array(x) - hit, axis=1)
            for hit in hit_cells
        ))
        return unique_targets[0]

    # If no clear ship targets, use a pattern search
    # Try to find a 2x2 block with one hit and three unknowns
    for i in range(n-1):
        for j in range(n-1):
            block = board[i:i+2, j:j+2]
            if np.sum(block == 1) == 1 and np.sum(block == 0) == 3:
                # Find the unknown cells in this block
                unknown_in_block = np.argwhere(block == 0)
                for r, c in unknown_in_block:
                    potential_targets.append((i + r, j + c))

    # If we found potential targets in 2x2 blocks, prioritize them
    if potential_targets:
        return potential_targets[0]

    # If no specific targets found, use random targeting
    return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])
```

Here's a more advanced version that implements a proper ship detection algorithm:

```python
import random
import numpy as np
from collections import defaultdict

def policy(board: list[list[int]]) -> tuple[int, int]:
    board = np.array(board)
    n = board.shape[0]

    # Find all unknown cells
    unknown_cells = np.argwhere(board == 0)
    if len(unknown_cells) == 0:
        return (0, 0)

    # Find all hit cells
    hit_cells = np.argwhere(board == 1)
    if len(hit_cells) == 0:
        return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])

    # Try to find ships by looking for connected hit cells
    visited = set()
    ships = []

    for hit in hit_cells:
        if tuple(hit) in visited:
            continue

        # BFS to find all connected hit cells
        queue = [tuple(hit)]
        visited.add(tuple(hit))
        ship_cells = []

        while queue:
            r, c = queue.pop(0)
            ship_cells.append((r, c))

            for dr, dc in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < n and 0 <= nc < n and
                    board[nr, nc] == 1 and
                    tuple((nr, nc)) not in visited):
                    visited.add(tuple((nr, nc)))
                    queue.append((nr, nc))

        ships.append(ship_cells)

    # For each ship, find potential targets
    potential_targets = []

    for ship in ships:
        if len(ship) == 0:
            continue

        # Find the bounding box of the ship
        min_r = min(r for r, c in ship)
        max_r = max(r for r, c in ship)
        min_c = min(c for r, c in ship)
        max_c = max(c for r, c in ship)

        # Check the perimeter of the ship
        for r in range(min_r, max_r + 1):
            for c in range(min_c, max_c + 1):
                if board[r, c] == 0:
                    # Check if this cell is adjacent to the ship
                    adjacent = False
                    for dr, dc in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                        nr, nc = r + dr, c + dc
                        if (0 <= nr < n and 0 <= nc < n and
                            board[nr, nc] == 1):
                            adjacent = True
                            break

                    if adjacent:
                        potential_targets.append((r, c))

        # Check the ends of the ship
        if len(ship) > 1:
            # Find the direction of the ship
            directions = []
            for i in range(len(ship)-1):
                dr = ship[i+1][0] - ship[i][0]
                dc = ship[i+1][1] - ship[i][1]
                directions.append((dr, dc))

            # If all directions are the same, it's a straight ship
            if all(d == directions[0] for d in directions):
                # Check the ends
                first = ship[0]
                last = ship[-1]

                # Check cells adjacent to the ends
                for r, c in [first, last]:
                    for dr, dc in [(0, 1), (1, 0), (0, -1), (-1, 0)]:
                        nr, nc = r + dr, c + dc
                        if (0 <= nr < n and 0 <= nc < n and
                            board[nr, nc] == 0):
                            potential_targets.append((nr, nc))

    # If we found potential targets, prioritize them
    if potential_targets:
        # Remove duplicates
        unique_targets = list(set(potential_targets))
        # Sort by distance to nearest hit
        unique_targets.sort(key=lambda x: min(
            np.linalg.norm(np.array(x) - hit, axis=1)
            for hit in hit_cells
        ))
        return unique_targets[0]

    # If no specific targets found, use random targeting
    return tuple(unknown_cells[random.randint(0, len(unknown_cells)-1)])
```

This final version implements a more sophisticated approach that:
1. Detects connected hit cells to identify ships
2. Prioritizes targeting cells adjacent to known ships
3. For straight ships, focuses on the ends
4. Falls back to random targeting when no clear targets are found

The algorithm should be efficient enough to run within the 1-second time limit for a 10x10 grid.

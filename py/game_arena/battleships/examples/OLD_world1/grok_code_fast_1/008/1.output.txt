import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    n = 10
    # Get all unexplored cells (available to shoot)
    available = [(i, j) for i in range(n) for j in range(n) if board[i][j] == 0]
    if not available:
        # Shouldn't happen, but just in case
        return (0, 0)
    
    # Get all hit cells
    hits = [(i, j) for i in range(n) for j in range(n) if board[i][j] == 1]
    
    # Target mode: if there are hits, try to shoot adjacent unexplored cells
    if hits:
        candidates = set()
        for hi, hj in hits:
            for di, dj in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                ni, nj = hi + di, hj + dj
                if 0 <= ni < n and 0 <= nj < n and board[ni][nj] == 0:
                    candidates.add((ni, nj))
        if candidates:
            return random.choice(list(candidates))
    
    # Hunt mode: no hits, so shoot in parity pattern (even sum of indices)
    hunt_candidates = [(i, j) for (i, j) in available if (i + j) % 2 == 0]
    if hunt_candidates:
        return random.choice(hunt_candidates)
    else:
        # Fallback: shoot any remaining
        return random.choice(available)

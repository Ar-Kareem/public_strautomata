def policy(board: list[list[int]]) -> tuple[int, int]:
    # Define the dimensions
    n = 10
    
    # Collect all possible moves (cells with 0)
    possible = [(r, c) for r in range(n) for c in range(n) if board[r][c] == 0]
    
    # If no moves possible (unlikely), return an invalid move, but board starts empty
    if not possible:
        return (-1, -1)  # Should not happen
    
    # Collect all hit positions (cells with 1)
    hits = [(r, c) for r in range(n) for c in range(n) if board[r][c] == 1]
    
    if not hits:
        # Hunt mode: Use a checkered pattern preferring even parity of r+c for initial shots
        # This spreads shots across the board without clustering
        for r in range(n):
            for c in range(n):
                if (r + c) % 2 == 0 and (r, c) in possible:
                    return (r, c)
        # Fallback to the first possible if none in parity (shouldn't happen much)
        return possible[0]
    else:
        # Target mode: Prioritize shooting around existing hits
        candidates = set()
        for hr, hc in hits:
            # Check all four directions
            for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                nr, nc = hr + dr, hc + dc
                if 0 <= nr < n and 0 <= nc < n and (nr, nc) in possible:
                    candidates.add((nr, nc))
        
        if candidates:
            # Return the lexicographically smallest candidate (prefers top/left)
            return min(candidates)
        else:
            # No immediate candidates around hits (possibly ship sunk, or no valid extensions)
            # Fall back to hunt mode
            for r in range(n):
                for c in range(n):
                    if (r + c) % 2 == 0 and (r, c) in possible:
                        return (r, c)
            return possible[0]

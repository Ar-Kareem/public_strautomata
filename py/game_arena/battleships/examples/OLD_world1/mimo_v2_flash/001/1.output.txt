import random

# Standard Battleship constants
GRID_SIZE = 10
SHIP_SIZES = [5, 4, 3, 3, 2]

def get_adjacent_cells(r, c):
    """Returns list of adjacent (row, col) coordinates (up, down, left, right)."""
    adj = []
    for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
        nr, nc = r + dr, c + dc
        if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE:
            adj.append((nr, nc))
    return adj

def get_line_cells(start_r, start_c, dr, dc, length):
    """Returns list of cells in a line starting from (start_r, start_c) with direction (dr, dc)."""
    return [(start_r + i * dr, start_c + i * dc) for i in range(length)]

def can_place_ship(board, r, c, dr, dc, size):
    """Checks if a ship of `size` can be placed at (r, c) with direction (dr, dc)."""
    for i in range(size):
        nr, nc = r + i * dr, c + i * dc
        # Check bounds
        if not (0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE):
            return False
        # Check if cell is known to be empty (miss or adjacent to sunk ship)
        # Note: In this simulation, we only see 0, -1, 1.
        # If we see a '1', it's a hit. If we see '-1', it's a miss.
        # If we see '0', it's unknown.
        if board[nr][nc] == -1:
            return False
        # If we see '0', it's valid to be part of a ship.
        # If we see '1', it's valid (must be part of the ship).
    return True

def count_potential_placements(board):
    """Calculates the number of valid placements for the remaining ship set."""
    # We need to find the lengths of ships we are still hunting.
    # Identify current hits and groups.
    hits = set()
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if board[r][c] == 1:
                hits.add((r, c))
    
    # If no hits, we assume we are hunting the full set of ships.
    # If there are hits, we need to determine if they form known sunk ships.
    # Since we don't have direct feedback "SUNK", we infer.
    # A group of hits is "sunk" if it is fully surrounded by misses or grid edges 
    # and we can deduce no other ship can fit there?
    # Actually, simple approach: 
    # If a group of hits has length L, and we have confirmed it cannot extend further (all neighbors are misses/edges),
    # we consider it sunk.
    
    # Map of remaining ship lengths to hunt.
    # Initially full set.
    current_ship_lengths = list(SHIP_SIZES)
    
    if hits:
        # Identify connected components of hits (potential damaged ships)
        visited_groups = set()
        groups = []
        for r, c in hits:
            if (r, c) in visited_groups:
                continue
            # BFS to find group
            group = []
            queue = [(r, c)]
            visited_group = set()
            while queue:
                curr_r, curr_c = queue.pop(0)
                if (curr_r, curr_c) in visited_group:
                    continue
                visited_group.add((curr_r, curr_c))
                group.append((curr_r, curr_c))
                for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                    nr, nc = curr_r + dr, curr_c + dc
                    if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE and board[nr][nc] == 1:
                        queue.append((nr, nc))
            
            # Mark these as visited for the outer loop
            visited_groups.update(group)
            groups.append(group)
        
        # For each group, check if it is "contained" (surrounded by misses or edges).
        # If contained, its length is fixed (size of group).
        # If not contained, it could be part of a larger ship.
        
        # We remove known sunk ships from the search pool.
        # However, if a group is NOT contained, we don't know its length exactly.
        # It could be a part of a larger ship.
        # To be safe, we assume any "open" hit group might be part of a larger ship.
        # But we need to subtract "likely sunk" ships.
        
        # Let's refine the "remaining lengths" list.
        # We remove lengths corresponding to groups that are fully surrounded by misses (sunk).
        sunk_groups = []
        open_groups = []
        
        for group in groups:
            is_sunk = True
            for r, c in group:
                for nr, nc in get_adjacent_cells(r, c):
                    if board[nr][nc] == 0: # Unknown neighbor means not sunk
                        is_sunk = False
                        break
                if not is_sunk:
                    break
            if is_sunk:
                sunk_groups.append(group)
            else:
                open_groups.append(group)
        
        # Remove sunk lengths from our target list
        for group in sunk_groups:
            length = len(group)
            if length in current_ship_lengths:
                current_ship_lengths.remove(length)
            else:
                # Edge case: a sunk ship might not match exact lengths if it was part of a larger unknown ship.
                # But we track remaining standard sizes. 
                # If we see a group of 2, we remove 2.
                pass
        
        # Now, for open groups, we must assume they could be extended.
        # If we have an open group of size 2, we are hunting for a ship of size >= 3 (since 2 is the smallest).
        # Actually, we don't know exactly. 
        # A safe heuristic for "remaining ship lengths" is just the set of ships NOT yet sunk.
        # We proceed with the current_ship_lengths list.
        
    # If we have no hits, or we processed hits, we return the list of lengths we are hunting.
    return current_ship_lengths

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Implements a targeting policy using Parity (Checkerboard) search + Target Mode for hits.
    """
    
    # 1. Identify all known hits that are not yet part of a confirmed sunk ship.
    hits = set()
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if board[r][c] == 1:
                hits.add((r, c))
    
    # Determine if we need to go into "Target Mode" (hunting damaged ships)
    # We enter target mode if we have any hits that are not fully contained by misses/grid edges.
    target_mode = False
    ships_to_hunt = SHIP_SIZES[:] # Copy
    
    if hits:
        # Identify groups
        visited = set()
        groups = []
        for r, c in hits:
            if (r, c) in visited:
                continue
            group = []
            stack = [(r, c)]
            while stack:
                curr = stack.pop()
                if curr in visited:
                    continue
                visited.add(curr)
                group.append(curr)
                cr, cc = curr
                for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                    nr, nc = cr + dr, cc + dc
                    if 0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE and board[nr][nc] == 1:
                        stack.append((nr, nc))
            groups.append(group)
        
        # Check groups for "open" status (not sunk)
        open_groups = []
        for g in groups:
            # Check if surrounded by misses (sunk)
            is_sunk = True
            for r, c in g:
                for nr, nc in get_adjacent_cells(r, c):
                    if board[nr][nc] == 0:
                        is_sunk = False
                        break
                if not is_sunk: break
            
            if not is_sunk:
                open_groups.append(g)
                target_mode = True
            
            # Remove sunk ships from hunting list
            else:
                try:
                    ships_to_hunt.remove(len(g))
                except ValueError:
                    # If the sunk ship length is not in the list (e.g., overlapping hits or weird state), ignore
                    pass

    # 2. If in Target Mode, prioritize cells around open hit groups.
    if target_mode:
        # Priority 1: Cells that extend existing lines
        # Priority 2: Cells that fill gaps in lines
        # Priority 3: Adjacent cells to lines
        
        candidates = [] # (priority, (r, c))
        
        for g in open_groups:
            if len(g) == 1:
                # Single hit: try all 4 directions
                r, c = g[0]
                for dr, dc in [(-1,0), (1,0), (0,-1), (0,1)]:
                    # We can extend in two directions (positive and negative), 
                    # but we need to check if the hit is part of a known orientation.
                    # Since it's a single hit, any valid neighbor is a candidate.
                    # But we should prioritize known orientations if we have partial info.
                    # Since it's single, all directions are equal.
                    
                    # Try extending positive direction
                    for length in range(1, 5): # Max ship length is 5
                        nr, nc = r + dr*length, c + dc*length
                        if not (0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE):
                            break
                        if board[nr][nc] == -1:
                            break
                        if board[nr][nc] == 0:
                            # Check if this placement is valid for ANY remaining ship
                            # We need to be somewhat careful not to waste shots on impossible placements
                            # But simpler is better: if it's not a miss, it's a shot.
                            candidates.append((1, (nr, nc)))
                            break # Don't shoot further along line until we know if we hit
                        
                    # Try extending negative direction (reverse)
                    for length in range(1, 5):
                        nr, nc = r - dr*length, c - dc*length
                        if not (0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE):
                            break
                        if board[nr][nc] == -1:
                            break
                        if board[nr][nc] == 0:
                            candidates.append((1, (nr, nc)))
                            break

            else:
                # Multiple hits: determine orientation
                # Sort by row then col to check direction
                g_sorted = sorted(g)
                if g_sorted[0][0] == g_sorted[-1][0]: # Horizontal
                    min_r, min_c = g_sorted[0]
                    max_c = g_sorted[-1][1]
                    dr, dc = 0, 1
                else: # Vertical
                    min_r, min_c = g_sorted[0]
                    max_r = g_sorted[-1][0]
                    dr, dc = 1, 0
                
                # Try extending start (min)
                for length in range(1, 5):
                    nr, nc = min_r - dr*length, min_c - dc*length
                    if not (0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE):
                        break
                    if board[nr][nc] == -1:
                        break
                    if board[nr][nc] == 0:
                        candidates.append((1, (nr, nc)))
                        break
                
                # Try extending end (max)
                for length in range(1, 5):
                    nr, nc = (max_r if dr else min_r) + dr*length, (max_c if dc else min_c) + dc*length
                    if not (0 <= nr < GRID_SIZE and 0 <= nc < GRID_SIZE):
                        break
                    if board[nr][nc] == -1:
                        break
                    if board[nr][nc] == 0:
                        candidates.append((1, (nr, nc)))
                        break
                
                # Check for gaps inside (only if group size > 1)
                # If we have hits A and C, but B is missing in between.
                # If the gap is small (size 1), it's a high priority target.
                # If the gap is large, it's less likely but possible (if we missed a hit in between).
                # Actually, simpler: just check the immediate neighbors of the endpoints. 
                # If we hit 2 ships, we usually fire at the gap immediately if we suspect a 3-ship.
                # But strictly, we might have missed a hit.
                # Let's just check strictly adjacent to the group.
                # Wait, if we have hits at (r, c) and (r, c+2), (r, c+1) is the gap. That is Priority 1.
                
                curr = g_sorted[0]
                for i in range(1, len(g_sorted)):
                    next_cell = g_sorted[i]
                    # Calculate distance
                    if dr == 0: # Horizontal
                        gap = next_cell[1] - curr[1] - 1
                    else: # Vertical
                        gap = next_cell[0] - curr[0] - 1
                    
                    if gap == 1:
                        # There is exactly one cell between them
                        if dr == 0:
                            gap_cell = (curr[0], curr[1] + 1)
                        else:
                            gap_cell = (curr[0] + 1, curr[1])
                        if board[gap_cell[0]][gap_cell[1]] == 0:
                            # Guaranteed high priority hit
                            return gap_cell
                    elif gap > 1:
                        # There is a gap.
                        # Check the closest cell to curr
                        if dr == 0:
                            next_try = (curr[0], curr[1] + 1)
                        else:
                            next_try = (curr[0] + 1, curr[1])
                        if board[next_try[0]][next_try[1]] == 0:
                            candidates.append((1, next_try))

        # Filter duplicates and cells we already shot
        seen = set()
        unique_candidates = []
        for prio, cell in candidates:
            if cell in seen: continue
            if board[cell[0]][cell[1]] != 0: continue
            unique_candidates.append((prio, cell))
            seen.add(cell)
        
        if unique_candidates:
            # Sort by priority (though currently all 1)
            unique_candidates.sort(key=lambda x: x[0])
            # If we have a gap to fill, we might want to prioritize it.
            # But for now, pick randomly or deterministically.
            # Let's try to be deterministic: pick the first one?
            # Or pick randomly to avoid getting stuck in bad loops?
            # Let's pick the first valid one for consistency.
            return unique_candidates[0][1]

    # 3. If not in Target Mode (or no candidates found), Search Mode.
    # We use a parity-based search (Checkerboard).
    # The smallest ship is length 2, which covers both colors.
    # However, firing on the same parity reduces the number of shots to cover the board.
    # The "optimized" parity for 5,4,3,3,2 is to fire on ~1/3 of the board, but 
    # checkerboard (1/2) is robust and handles the '2' ship easily.
    # Let's stick to Checkerboard.
    
    candidates = []
    for r in range(GRID_SIZE):
        for c in range(GRID_SIZE):
            if board[r][c] == 0:
                # Checkerboard pattern: (r + c) % 2 == 0
                # To increase density, we might mix in the other parity if we have enough info,
                # but for search, strict parity is best.
                if (r + c) % 2 == 0:
                    candidates.append((r, c))
    
    # Fallback: If checkerboard is exhausted (shouldn't happen before game ends unless we missed something),
    # take any unknown cell.
    if not candidates:
        for r in range(GRID_SIZE):
            for c in range(GRID_SIZE):
                if board[r][c] == 0:
                    return (r, c)
    
    # Choose a random candidate from the valid set
    return random.choice(candidates)

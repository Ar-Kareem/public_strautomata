```python
import random

# Module-level variables to maintain state between calls to the policy function
remaining_ships_count = {5: 1, 4: 1, 3: 2, 2: 1}
processed_hits = [[False for _ in range(10)] for _ in range(10)]

def policy(board: list[list[int]]) -> tuple[int, int]:
    global remaining_ships_count, processed_hits

    # Step 1: Check for new sunk ships and update remaining_ships_count and processed_hits
    for i in range(10):
        for j in range(10):
            if board[i][j] == 1 and not processed_hits[i][j]:
                # Determine horizontal and vertical lengths
                left = j
                while left > 0 and board[i][left - 1] == 1:
                    left -= 1
                right = j
                while right < 9 and board[i][right + 1] == 1:
                    right += 1
                h_len = right - left + 1

                top = i
                while top > 0 and board[top - 1][j] == 1:
                    top -= 1
                bottom = i
                while bottom < 9 and board[bottom + 1][j] == 1:
                    bottom += 1
                v_len = bottom - top + 1

                ship_length = 0
                direction = None
                if h_len >= v_len and h_len > 1:
                    ship_length = h_len
                    direction = 'horizontal'
                    start = left
                    end = right
                elif v_len > 1:
                    ship_length = v_len
                    direction = 'vertical'
                    start = top
                    end = bottom
                else:
                    continue  # Not enough to determine

                if ship_length in remaining_ships_count and remaining_ships_count[ship_length] > 0:
                    remaining_ships_count[ship_length] -= 1
                    if direction == 'horizontal':
                        for col in range(left, right + 1):
                            processed_hits[i][col] = True
                    else:
                        for row in range(top, bottom + 1):
                            processed_hits[row][j] = True

    # Step 2: Compute best cell to shoot based on possible ship placements
    best_score = -1
    best_cells = []

    for i in range(10):
        for j in range(10):
            if board[i][j] == 0:  # Only consider unattacked cells
                score = 0
                # Consider all remaining ship lengths
                for ship_length in [sl for sl in remaining_ships_count if remaining_ships_count[sl] > 0]:
                    count = remaining_ships_count[ship_length]
                    
                    # Check horizontal placements
                    start_col_min = max(0, j - (ship_length - 1))
                    start_col_max = min(10 - ship_length, j)
                    for possible_start_col in range(start_col_min, start_col_max + 1):
                        valid = True
                        for offset in range(ship_length):
                            current_col = possible_start_col + offset
                            cell_val = board[i][current_col]
                            if cell_val == -1:
                                valid = False
                                break
                        if valid:
                            score += count
                    
                    # Check vertical placements
                    start_row_min = max(0, i - (ship_length - 1))
                    start_row_max = min(10 - ship_length, i)
                    for possible_start_row in range(start_row_min, start_row_max + 1):
                        valid = True
                        for offset in range(ship_length):
                            current_row = possible_start_row + offset
                            cell_val = board[current_row][j]
                            if cell_val == -1:
                                valid = False
                                break
                        if valid:
                            score += count
                
                if score > best_score:
                    best_score = score
                    best_cells = [(i, j)]
                elif score == best_score:
                    best_cells.append((i, j))
    
    # Fallback to random choice if no best cells found
    if not best_cells:
        for i in range(10):
            for j in range(10):
                if board[i][j] == 0:
                    return (i, j)
        return (0, 0)  # Should never reach here
    
    return random.choice(best_cells)
```

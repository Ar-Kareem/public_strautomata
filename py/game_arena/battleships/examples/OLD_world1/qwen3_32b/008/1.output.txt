def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the next move for the Battleship game using a hybrid approach:
    - **Targeting Phase**: Focuses on adjacent cells to known hits.
    - **Hunting Phase**: Uses a fixed checkerboard pattern for initial exploration.

    Args:
        board: A 10x10 grid representing the opponent's board from the player's perspective.
               - 0: Unknown (never fired here)
               - 1: Hit (fired here and hit a ship)
               - -1: Miss (fired here and hit water)

    Returns:
        A tuple (row, col) indicating the next cell to fire upon.
    """
    
    # --- TARGETING PHASE ---
    # Generate candidate cells adjacent to all current hits
    candidates = []
    for i in range(10):
        for j in range(10):
            if board[i][j] == 1:  # Current cell is a hit
                # Check all four neighboring directions (up, down, left, right)
                for di, dj in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < 10 and 0 <= nj < 10:
                        if board[ni][nj] == 0:  # Only add unexplored cells
                            candidates.append((ni, nj))
    
    # Eliminate duplicate candidates using a set, then sort them
    unique_candidates = list(set(candidates))
    if unique_candidates:
        # Sort by row then column for deterministic behavior (top-left to bottom-right)
        unique_candidates.sort()
        # Return the first candidate from the sorted list
        return unique_candidates[0]
    
    # --- HUNTING PHASE ---
    # If no adjacent hits are found, use a checkerboard pattern to explore
    # to guarantee at least one cell of any ship of length ≥2 is hit eventually
    priority_order = []
    for i in range(10):
        for j in range(10):
            if (i + j) % 2 == 0:  # First, fire at all cells with even (i + j)
                priority_order.append((i, j))
    for i in range(10):
        for j in range(10):
            if (i + j) % 2 == 1:  # Then, handle the rest
                priority_order.append((i, j))
    
    # Find the first unexplored cell in this priority order
    for move in priority_order:
        if board[move[0]][move[1]] == 0:
            return move
    
    # If all cells were fired upon (highly unlikely in a 10x10 board),
    # return a default position for safety
    return (0, 0)

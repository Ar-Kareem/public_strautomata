import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    A targeting policy for a Battleship-like game.

    Args:
        board: A 10x10 grid representing the current knowledge of the opponent's grid.
               0: unknown cell
               -1: miss
               1: hit

    Returns:
        A tuple (row, col) with integers in the range 0-9, choosing a cell
        to fire at.  Must be a cell that has not been fired at previously.
    """

    n = len(board)

    # First, check for cells adjacent to hits.
    hits = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 1:
                hits.append((r, c))

    if hits:
        # Prioritize adjacent cells to existing hits
        for r, c in hits:
            adjacent = [(r-1, c), (r+1, c), (r, c-1), (r, c+1)]
            for ar, ac in adjacent:
                if 0 <= ar < n and 0 <= ac < n and board[ar][ac] == 0:
                    return (ar, ac)

    # If no adjacent cells found, pick a random unknown cell.
    unknown_cells = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0:
                unknown_cells.append((r, c))

    if unknown_cells:
        return random.choice(unknown_cells)
    else:
        # Should never happen if the game is played correctly.
        # Choose a random cell to avoid an error.
        for r in range(n):
            for c in range(n):
                if board[r][c] != 1:
                   return (r,c)
        return (0,0)

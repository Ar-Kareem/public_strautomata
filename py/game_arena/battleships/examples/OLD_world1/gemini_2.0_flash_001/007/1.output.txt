import numpy as np
import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Implements a targeting policy for a Battleship-like game on a 10x10 grid.

    Args:
        board: A 10x10 list of lists representing the current knowledge of the opponent grid.
               0 -> unknown cell, -1 -> miss, 1 -> hit.

    Returns:
        A tuple (row, col) with integers in the range 0-9, choosing a cell to fire at.
    """

    board_np = np.array(board)
    hits = np.argwhere(board_np == 1)

    if hits.size > 0:
        # Focus on hitting around existing hits
        row, col = hits[random.randint(0, len(hits) - 1)] #Pick a random hit to explore

        # Prioritize adjacent cells
        adjacent_cells = [(row - 1, col), (row + 1, col), (row, col - 1), (row, col + 1)]
        valid_adjacent_cells = []

        for r, c in adjacent_cells:
            if 0 <= r < 10 and 0 <= c < 10 and board_np[r, c] == 0:
                valid_adjacent_cells.append((r, c))
        
        if len(valid_adjacent_cells) > 0:
            r, c = random.choice(valid_adjacent_cells)
            return (r, c)
        else:
            #If no valid adjacent cells, try other hits
            for row, col in hits:
                adjacent_cells = [(row - 1, col), (row + 1, col), (row, col - 1), (row, col + 1)]
                valid_adjacent_cells = []

                for r, c in adjacent_cells:
                    if 0 <= r < 10 and 0 <= c < 10 and board_np[r, c] == 0:
                        valid_adjacent_cells.append((r, c))
        
                if len(valid_adjacent_cells) > 0:
                  r, c = random.choice(valid_adjacent_cells)
                  return (r,c)

    # If no hits, use a probabilistic approach
    unknown_cells = np.argwhere(board_np == 0)
    if unknown_cells.size > 0:

        # Prioritize odd rows and cols to hit more cells
        potential_targets = []
        for r, c in unknown_cells:
            if(r+c) % 2 == 0:
                potential_targets.append((r,c))

        if len(potential_targets) >0:
            r, c = random.choice(potential_targets)
            return (r, c)
        else: #In the off chance parity strat fails (every possible cell around has already been shot)
            r, c = random.choice(unknown_cells)
            return (r,c)


    # If all cells are known, return a random cell (should not happen)
    return (random.randint(0, 9), random.randint(0, 9))

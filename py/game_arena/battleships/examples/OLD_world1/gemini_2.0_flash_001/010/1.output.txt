import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    A targeting policy for a Battleship-like game on a 10x10 grid.

    Args:
        board: A 10x10 list of lists representing the current knowledge of the opponent grid.
            0: unknown cell (you have not fired here).
            -1: miss (you fired here and hit water).
            1: hit (you fired here and hit a ship cell).

    Returns:
        A tuple (row, col) with integers in the range 0-9, choosing a cell you have not
        previously fired at.
    """

    n = len(board)

    # Prioritize cells adjacent to hits
    potential_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 1:
                # Check adjacent cells
                for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < n and 0 <= nc < n and board[nr][nc] == 0:
                        potential_targets.append((nr, nc))

    if potential_targets:
        return random.choice(potential_targets)

    # If no hits, target cells with odd row+col sum for better spread
    remaining_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0 and (r + c) % 2 != 0:
                remaining_targets.append((r, c))

    if remaining_targets:
        return random.choice(remaining_targets)

    # If no odd cells available, target remaining unknown cells
    final_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0:
                final_targets.append((r, c))

    if final_targets:
        return random.choice(final_targets)

    # If all cells have been targeted, retry a random cell. Though it should theoretically never happen.
    return (random.randint(0, n - 1), random.randint(0, n - 1))

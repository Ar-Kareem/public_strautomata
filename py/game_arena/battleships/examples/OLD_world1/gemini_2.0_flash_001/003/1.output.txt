import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    A targeting policy for a Battleship-like game on a 10x10 grid.

    Args:
        board: A 10x10 list of lists representing the current knowledge of the opponent grid.
               0  → unknown cell (you have not fired here).
               -1 → miss (you fired here and hit water).
               1  → hit (you fired here and hit a ship cell).

    Returns:
        A tuple (row, col) with integers in the range 0-9, choosing a cell you have not
        previously fired at.
    """
    n = len(board)

    # Prioritize cells around hits
    potential_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 1:
                neighbors = [(r - 1, c), (r + 1, c), (r, c - 1), (r, c + 1)]
                for nr, nc in neighbors:
                    if 0 <= nr < n and 0 <= nc < n and board[nr][nc] == 0:
                        potential_targets.append((nr, nc))

    if potential_targets:
        return random.choice(potential_targets)

    # If no hits, target odd-odd and even-even cells
    remaining_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0 and (r + c) % 2 == 0:
                remaining_targets.append((r, c))
    if remaining_targets:
        return random.choice(remaining_targets)

    # If all odd-odd/even-even cells are gone, just pick randomly from remaining unknown cells.
    desperate_targets = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0:
                desperate_targets.append((r, c))

    if desperate_targets:
        return random.choice(desperate_targets)
    
    # If nothing else works, return (0,0), the game is likely over
    return (0,0)

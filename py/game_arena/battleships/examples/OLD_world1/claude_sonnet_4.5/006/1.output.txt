import random
from typing import Optional

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Advanced Battleship targeting policy combining probability density,
    hunt mode, and target mode strategies.
    """
    n = len(board)
    ships = [5, 4, 3, 3, 2]
    
    # Find all unfired cells
    unfired = [(r, c) for r in range(n) for c in range(n) if board[r][c] == 0]
    
    if not unfired:
        return (0, 0)  # Should never happen
    
    # Check if we have any hits to follow up on
    hits = [(r, c) for r in range(n) for c in range(n) if board[r][c] == 1]
    
    # Target mode: we have unsunk ships to finish
    target = find_target_shot(board, n, hits)
    if target:
        return target
    
    # Hunt mode: use probability density
    return hunt_mode(board, n, ships, unfired)


def find_target_shot(board: list[list[int]], n: int, hits: list[tuple[int, int]]) -> Optional[tuple[int, int]]:
    """
    Target mode: find the best shot to continue sinking a ship we've hit.
    """
    if not hits:
        return None
    
    # Find hit clusters (connected components)
    clusters = find_hit_clusters(board, n, hits)
    
    for cluster in clusters:
        # For each cluster, find adjacent unfired cells
        candidates = []
        
        if len(cluster) == 1:
            # Single hit - check all 4 directions
            r, c = cluster[0]
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < n and 0 <= nc < n and board[nr][nc] == 0:
                    candidates.append((nr, nc))
        else:
            # Multiple hits in a line - find the orientation and extend
            cluster.sort()
            
            # Check if horizontal or vertical
            if all(r == cluster[0][0] for r, c in cluster):
                # Horizontal line
                row = cluster[0][0]
                min_col = min(c for r, c in cluster)
                max_col = max(c for r, c in cluster)
                
                # Try extending left and right
                if min_col > 0 and board[row][min_col - 1] == 0:
                    candidates.append((row, min_col - 1))
                if max_col < n - 1 and board[row][max_col + 1] == 0:
                    candidates.append((row, max_col + 1))
                    
            elif all(c == cluster[0][1] for r, c in cluster):
                # Vertical line
                col = cluster[0][1]
                min_row = min(r for r, c in cluster)
                max_row = max(r for r, c in cluster)
                
                # Try extending up and down
                if min_row > 0 and board[min_row - 1][col] == 0:
                    candidates.append((min_row - 1, col))
                if max_row < n - 1 and board[max_row + 1][col] == 0:
                    candidates.append((max_row + 1, col))
        
        if candidates:
            return candidates[0]
    
    return None


def find_hit_clusters(board: list[list[int]], n: int, hits: list[tuple[int, int]]) -> list[list[tuple[int, int]]]:
    """
    Group hits into connected components (adjacent hits belong to same ship).
    """
    visited = set()
    clusters = []
    
    for hit in hits:
        if hit in visited:
            continue
        
        # BFS to find connected hits
        cluster = []
        queue = [hit]
        visited.add(hit)
        
        while queue:
            r, c = queue.pop(0)
            cluster.append((r, c))
            
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if (0 <= nr < n and 0 <= nc < n and 
                    (nr, nc) not in visited and board[nr][nc] == 1):
                    visited.add((nr, nc))
                    queue.append((nr, nc))
        
        clusters.append(cluster)
    
    return clusters


def hunt_mode(board: list[list[int]], n: int, ships: list[int], unfired: list[tuple[int, int]]) -> tuple[int, int]:
    """
    Hunt mode: use probability density to find the most likely cell.
    """
    # Calculate probability for each unfired cell
    prob = [[0.0 for _ in range(n)] for _ in range(n)]
    
    # Determine which ships are still alive (rough estimate)
    total_hits = sum(1 for r in range(n) for c in range(n) if board[r][c] == 1)
    remaining_ships = estimate_remaining_ships(board, n, ships)
    
    # For each remaining ship, calculate placement probability
    for ship_len in remaining_ships:
        # Horizontal placements
        for r in range(n):
            for c in range(n - ship_len + 1):
                if can_place_ship(board, r, c, ship_len, horizontal=True):
                    for i in range(ship_len):
                        prob[r][c + i] += 1.0
        
        # Vertical placements
        for r in range(n - ship_len + 1):
            for c in range(n):
                if can_place_ship(board, r, c, ship_len, horizontal=False):
                    for i in range(ship_len):
                        prob[r + i][c] += 1.0
    
    # Use checkerboard pattern preference in early game
    if total_hits < 3:
        for r, c in unfired:
            if (r + c) % 2 == 0:
                prob[r][c] *= 1.2
    
    # Find the cell with highest probability
    best_cell = max(unfired, key=lambda rc: prob[rc[0]][rc[1]])
    return best_cell


def can_place_ship(board: list[list[int]], r: int, c: int, length: int, horizontal: bool) -> bool:
    """
    Check if a ship of given length can be placed at position (r, c).
    A ship can be placed if all cells are either unfired (0) or hit (1),
    and there's no miss (-1).
    """
    n = len(board)
    
    if horizontal:
        if c + length > n:
            return False
        for i in range(length):
            if board[r][c + i] == -1:
                return False
    else:
        if r + length > n:
            return False
        for i in range(length):
            if board[r + i][c] == -1:
                return False
    
    return True


def estimate_remaining_ships(board: list[list[int]], n: int, ships: list[int]) -> list[int]:
    """
    Estimate which ships are still in play based on hit patterns.
    This is a simple heuristic - we assume all ships are still active.
    """
    # A more sophisticated approach would track sunk ships,
    # but for simplicity, we'll assume all ships are still active
    return ships.copy()

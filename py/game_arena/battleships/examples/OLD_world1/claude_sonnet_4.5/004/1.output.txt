import random
from collections import deque

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Battleship targeting policy using hunt and target mode.
    """
    n = len(board)
    
    # Find all hit cells and unfired cells
    hit_cells = []
    unfired_cells = []
    
    for r in range(n):
        for c in range(n):
            if board[r][c] == 1:
                hit_cells.append((r, c))
            elif board[r][c] == 0:
                unfired_cells.append((r, c))
    
    # Target mode: if we have unsunk ships (isolated hits or lines of hits)
    if hit_cells:
        # Find hit cells that are part of unsunk ships
        # A ship is unsunk if we have hits but haven't found all cells yet
        target_cells = []
        
        for r, c in hit_cells:
            # Check all 4 directions for adjacent unfired or hit cells
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < n and 0 <= nc < n and board[nr][nc] == 0:
                    target_cells.append((nr, nc))
        
        if target_cells:
            # Prioritize cells that are aligned with multiple hits
            best_targets = []
            best_score = -1
            
            for cell in target_cells:
                r, c = cell
                score = 0
                
                # Check if this cell extends a line of hits
                # Horizontal
                left_hits = 0
                right_hits = 0
                for i in range(1, 5):
                    if c - i >= 0 and board[r][c - i] == 1:
                        left_hits += 1
                    else:
                        break
                for i in range(1, 5):
                    if c + i < n and board[r][c + i] == 1:
                        right_hits += 1
                    else:
                        break
                if left_hits > 0 or right_hits > 0:
                    score += left_hits + right_hits + 1
                
                # Vertical
                up_hits = 0
                down_hits = 0
                for i in range(1, 5):
                    if r - i >= 0 and board[r - i][c] == 1:
                        up_hits += 1
                    else:
                        break
                for i in range(1, 5):
                    if r + i < n and board[r + i][c] == 1:
                        down_hits += 1
                    else:
                        break
                if up_hits > 0 or down_hits > 0:
                    score += up_hits + down_hits + 1
                
                if score > best_score:
                    best_score = score
                    best_targets = [cell]
                elif score == best_score:
                    best_targets.append(cell)
            
            return random.choice(best_targets)
    
    # Hunt mode: use checkerboard pattern with parity based on remaining ships
    # Since smallest remaining ship could be 2, we use checkerboard
    checkerboard_unfired = []
    other_unfired = []
    
    for r, c in unfired_cells:
        if (r + c) % 2 == 0:
            checkerboard_unfired.append((r, c))
        else:
            other_unfired.append((r, c))
    
    # Prefer checkerboard cells, but if none available use others
    if checkerboard_unfired:
        candidates = checkerboard_unfired
    else:
        candidates = other_unfired
    
    if not candidates:
        candidates = unfired_cells
    
    # Among candidates, prefer cells with more potential ship placements
    if len(candidates) > 1:
        best_cells = []
        best_density = -1
        
        for r, c in candidates:
            density = 0
            
            # Check horizontal potential
            for length in [5, 4, 3, 2]:
                for start_c in range(max(0, c - length + 1), min(n - length + 1, c + 1)):
                    valid = True
                    for i in range(length):
                        if board[r][start_c + i] == -1:
                            valid = False
                            break
                    if valid:
                        density += 1
            
            # Check vertical potential
            for length in [5, 4, 3, 2]:
                for start_r in range(max(0, r - length + 1), min(n - length + 1, r + 1)):
                    valid = True
                    for i in range(length):
                        if board[start_r + i][c] == -1:
                            valid = False
                            break
                    if valid:
                        density += 1
            
            if density > best_density:
                best_density = density
                best_cells = [(r, c)]
            elif density == best_density:
                best_cells.append((r, c))
        
        return random.choice(best_cells)
    
    return random.choice(candidates) if candidates else (0, 0)

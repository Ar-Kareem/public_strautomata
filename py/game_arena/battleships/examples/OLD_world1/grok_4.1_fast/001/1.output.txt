def policy(board: list[list[int]]) -> tuple[int, int]:
    ROWS = 10
    COLS = 10
    directions = [(0, 1), (1, 0), (0, -1), (-1, 0)]

    def is_valid(r: int, c: int) -> bool:
        return 0 <= r < ROWS and 0 <= c < COLS

    # Find all hit positions
    hits = [(r, c) for r in range(ROWS) for c in range(COLS) if board[r][c] == 1]

    # Collect possible targets: empty cells adjacent to hits
    possible_targets = set()
    for r, c in hits:
        for dr, dc in directions:
            nr, nc = r + dr, c + dc
            if is_valid(nr, nc) and board[nr][nc] == 0:
                possible_targets.add((nr, nc))

    if possible_targets:
        # Score targets by number of adjacent hits (higher is better)
        scored = []
        for r, c in possible_targets:
            adj_hits = sum(
                1 for dr, dc in directions
                if is_valid(r + dr, c + dc) and board[r + dr][c + dc] == 1
            )
            # Sort key: maximize adj_hits, then minimize r, then minimize c
            scored.append((-adj_hits, r, c))
        scored.sort()
        return scored[0][1], scored[0][2]

    else:
        # Hunt phase: checkerboard pattern
        candidates = [(r, c) for r in range(ROWS) for c in range(COLS) if board[r][c] == 0]
        if not candidates:
            return (0, 0)  # Should not happen
        # Score: prefer even parity (r+c even), then minimize r, then minimize c
        scored_hunt = [(((r + c) % 2), r, c) for r, c in candidates]
        scored_hunt.sort()
        return scored_hunt[0][1], scored_hunt[0][2]

import numpy as np

def generate_placements(length: int) -> list[list[tuple[int, int]]]:
    placements = []
    # Horizontal
    for r in range(10):
        for c in range(10 - length + 1):
            placements.append([(r, c + k) for k in range(length)])
    # Vertical
    for r in range(10 - length + 1):
        for c in range(10):
            placements.append([(r + k, c) for k in range(length)])
    return placements

def policy(board: list[list[int]]) -> tuple[int, int]:
    board_np = np.array(board, dtype=int)
    # Unknown positions
    unknowns = [(i, j) for i in range(10) for j in range(10) if board_np[i, j] == 0]
    if not unknowns:
        return (0, 0)
    
    directions = [(-1, 0), (1, 0), (0, -1), (0, 1)]
    
    def hit_neighbors(i: int, j: int) -> int:
        count = 0
        for di, dj in directions:
            ni, nj = i + di, j + dj
            if 0 <= ni < 10 and 0 <= nj < 10 and board_np[ni, nj] == 1:
                count += 1
        return count
    
    # Sink candidates: unknowns adjacent to hits
    sink_candidates = []
    for i, j in unknowns:
        hn = hit_neighbors(i, j)
        if hn > 0:
            score = (-hn * 10000, i * 10 + j)
            sink_candidates.append((score, (i, j)))
    
    if sink_candidates:
        sink_candidates.sort()
        return sink_candidates[0][1]
    
    # Hunt phase: compute probability map based on consistent ship placements
    prob = np.zeros((10, 10), dtype=float)
    ship_lengths = [5, 4, 3, 3, 2]
    for length in ship_lengths:
        placements = generate_placements(length)
        for placement in placements:
            # Check consistency: no -1 in placement cells
            consistent = all(board_np[r, c] >= 0 for r, c in placement)
            if consistent:
                for r, c in placement:
                    prob[r, c] += 1.0
    
    # Select best hunt target
    hunt_candidates = []
    for i, j in unknowns:
        p = prob[i, j]
        score = (-p * 10000, i * 10 + j)
        hunt_candidates.append((score, (i, j)))
    
    hunt_candidates.sort()
    return hunt_candidates[0][1]

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Implements a targeting policy for a Battleship-like game on a 10x10 grid.
    
    Args:
        board: A 10x10 grid where 0 means unknown, -1 means miss, and 1 means hit.
    
    Returns:
        A tuple (row, col) indicating the next cell to fire at.
    """
    # Find all hit cells on the board
    hits = []
    for i in range(10):
        for j in range(10):
            if board[i][j] == 1:
                hits.append((i, j))
    
    # If there are no hits, use hunt mode to explore the board
    if not hits:
        # Look for unshot cells on the checkerboard pattern (where (i+j) % 2 == 0)
        for i in range(10):
            for j in range(10):
                if board[i][j] == 0 and (i + j) % 2 == 0:
                    return (i, j)
        # If no checkerboard cell is available, choose any unshot cell
        for i in range(10):
            for j in range(10):
                if board[i][j] == 0:
                    return (i, j)
        # Fallback (should not occur in valid games)
        return (0, 0)
    
    # There are hits: find connected components of hits
    visited = set()
    components = []
    for hit in hits:
        if hit not in visited:
            # Perform DFS to find the connected component
            stack = [hit]
            component = []
            while stack:
                cell = stack.pop()
                if cell in visited:
                    continue
                visited.add(cell)
                component.append(cell)
                i, j = cell
                # Check adjacent cells (up, down, left, right)
                for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < 10 and 0 <= nj < 10:
                        if board[ni][nj] == 1 and (ni, nj) not in visited:
                            stack.append((ni, nj))
            components.append(component)
    
    # Identify target cells adjacent to non-surrounded components
    target_cells = set()
    for component in components:
        # Check if the component is surrounded by water (misses or bounds)
        surrounded = True
        for cell in component:
            i, j = cell
            for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                ni, nj = i + di, j + dj
                if 0 <= ni < 10 and 0 <= nj < 10:
                    if board[ni][nj] != -1:  # Not a miss, so unshot or hit
                        surrounded = False
                        break
            if not surrounded:
                break
        if not surrounded:
            # Add all adjacent unshot cells to target set
            for cell in component:
                i, j = cell
                for di, dj in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                    ni, nj = i + di, j + dj
                    if 0 <= ni < 10 and 0 <= nj < 10:
                        if board[ni][nj] == 0:
                            target_cells.add((ni, nj))
    
    # If target cells are available, choose one
    if target_cells:
        # Sort target cells and return the first one
        target_list = sorted(list(target_cells))
        return target_list[0]
    
    # No target cells: use hunt mode
    for i in range(10):
        for j in range(10):
            if board[i][j] == 0 and (i + j) % 2 == 0:
                return (i, j)
    # Fallback to any unshot cell
    for i in range(10):
        for j in range(10):
            if board[i][j] == 0:
                return (i, j)
    # Fallback (should not occur in valid games)
    return (0, 0)

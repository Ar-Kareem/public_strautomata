import numpy as np
from collections import defaultdict
import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert to numpy array for easier manipulation
    grid = np.array(board)
    n = len(grid)
    
    # Find all unvisited cells (0)
    unvisited = list(zip(*np.where(grid == 0)))
    
    # If no unvisited cells, return random valid coordinate (should not happen)
    if not unvisited:
        for i in range(n):
            for j in range(n):
                if grid[i, j] == 0:
                    return (i, j)
        return (0, 0)
    
    # Check for existing hits (1) that might be part of a ship
    hits = list(zip(*np.where(grid == 1)))
    
    # If we have hits, try to find adjacent cells to continue targeting
    if hits:
        # Look for hits that are part of a horizontal or vertical sequence
        # This is a simple greedy approach that targets adjacent cells to hits
        # Find all cells adjacent to hits that are unvisited (0)
        adjacent_unvisited = set()
        for r, c in hits:
            for dr, dc in [(0, 1), (0, -1), (1, 0), (-1, 0)]:
                nr, nc = r + dr, c + dc
                if 0 <= nr < n and 0 <= nc < n and grid[nr, nc] == 0:
                    adjacent_unvisited.add((nr, nc))
        
        # If we have adjacent unvisited cells, pick one
        if adjacent_unvisited:
            return random.choice(list(adjacent_unvisited))
    
    # If no hits or no adjacent unvisited, prefer corners or center for first shots
    # This helps in early exploration
    
    # Priority order: corners, then center, then random unvisited
    corners = [(0,0), (0,n-1), (n-1,0), (n-1,n-1)]
    center = [(n//2, n//2)]
    
    # Try corners
    corner_unvisited = [pos for pos in corners if pos in unvisited]
    if corner_unvisited:
        return random.choice(corner_unvisited)
    
    # Try center 
    center_unvisited = [pos for pos in center if pos in unvisited]
    if center_unvisited:
        return random.choice(center_unvisited)
        
    # Fall back to any unvisited position
    return random.choice(unvisited)

import random
from collections import deque

def policy(board: list[list[int]]) -> tuple[int, int]:
    n = len(board)
    
    # Find all hits that are part of unsunk ships
    hits = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 1:
                hits.append((r, c))
    
    # If we have hits, try to find and sink ships
    if hits:
        # Look for hits that could be part of an incomplete ship
        for hit_r, hit_c in hits:
            # Check all four directions from this hit
            directions = [(0, 1), (0, -1), (1, 0), (-1, 0)]
            
            for dr, dc in directions:
                # Look along this direction to find the extent of the ship
                ship_cells = [(hit_r, hit_c)]
                
                # Extend in positive direction
                r, c = hit_r + dr, hit_c + dc
                while 0 <= r < n and 0 <= c < n and board[r][c] == 1:
                    ship_cells.append((r, c))
                    r, c = r + dr, c + dc
                
                # Extend in negative direction
                r, c = hit_r - dr, hit_c - dc
                while 0 <= r < n and 0 <= c < n and board[r][c] == 1:
                    ship_cells.insert(0, (r, c))
                    r, c = r - dr, c - dc
                
                # If we found multiple hits in a line, check the ends
                if len(ship_cells) > 1:
                    # Check both ends of the ship
                    for end_idx in [0, -1]:
                        end_r, end_c = ship_cells[end_idx]
                        
                        if end_idx == 0:
                            next_r, next_c = end_r - dr, end_c - dc
                        else:
                            next_r, next_c = end_r + dr, end_c + dc
                        
                        if (0 <= next_r < n and 0 <= next_c < n and 
                            board[next_r][next_c] == 0):
                            return (next_r, next_c)
                
                # If we have a single hit, check adjacent cells
                elif len(ship_cells) == 1:
                    next_r, next_c = hit_r + dr, hit_c + dc
                    if (0 <= next_r < n and 0 <= next_c < n and 
                        board[next_r][next_c] == 0):
                        return (next_r, next_c)
    
    # No incomplete ships found, use checkerboard pattern for hunting
    # This is more efficient than random shooting
    unfired_cells = []
    
    # First priority: checkerboard pattern (every other cell)
    checkerboard_cells = []
    for r in range(n):
        for c in range(n):
            if board[r][c] == 0:
                if (r + c) % 2 == 0:  # Checkerboard pattern
                    checkerboard_cells.append((r, c))
                unfired_cells.append((r, c))
    
    # If checkerboard cells available, pick one
    if checkerboard_cells:
        # Prefer cells in the middle of the board
        def distance_from_center(pos):
            r, c = pos
            center = (n - 1) / 2
            return abs(r - center) + abs(c - center)
        
        checkerboard_cells.sort(key=distance_from_center)
        return checkerboard_cells[0]
    
    # If no checkerboard cells, pick any unfired cell
    if unfired_cells:
        # Prefer cells near the center
        def distance_from_center(pos):
            r, c = pos
            center = (n - 1) / 2
            return abs(r - center) + abs(c - center)
        
        unfired_cells.sort(key=distance_from_center)
        return unfired_cells[0]
    
    # Fallback (should never reach here in normal gameplay)
    return (0, 0)

import random

def policy(board: list[list[int]]) -> int:
    """
    A Connect 4 policy that prioritizes winning moves, blocks opponent's winning moves,
    and otherwise chooses a random valid column.
    """

    def check_win(board, player):
        # Check horizontal
        for row in range(6):
            for col in range(4):
                if (board[row][col] == player and
                    board[row][col+1] == player and
                    board[row][col+2] == player and
                    board[row][col+3] == player):
                    return True

        # Check vertical
        for row in range(3):
            for col in range(7):
                if (board[row][col] == player and
                    board[row+1][col] == player and
                    board[row+2][col] == player and
                    board[row+3][col] == player):
                    return True

        # Check positive diagonal
        for row in range(3):
            for col in range(4):
                if (board[row][col] == player and
                    board[row+1][col+1] == player and
                    board[row+2][col+2] == player and
                    board[row+3][col+3] == player):
                    return True

        # Check negative diagonal
        for row in range(3):
            for col in range(3, 7):
                if (board[row][col] == player and
                    board[row+1][col-1] == player and
                    board[row+2][col-2] == player and
                    board[row+3][col-3] == player):
                    return True

        return False

    def get_next_state(board, col, player):
        """Returns a new board with the move applied, or None if the column is full."""
        new_board = [row[:] for row in board]  # Create a copy of the board
        for row in range(5, -1, -1):
            if new_board[row][col] == 0:
                new_board[row][col] = player
                return new_board
        return None
    
    # Check for winning moves
    for col in range(7):
        next_board = get_next_state(board, col, 1)
        if next_board and check_win(next_board, 1):
            return col

    # Check for blocking moves
    for col in range(7):
        next_board = get_next_state(board, col, -1)
        if next_board and check_win(next_board, -1):
            return col

    # Choose a random valid move
    valid_cols = [col for col in range(7) if board[0][col] == 0]
    if valid_cols:
        return random.choice(valid_cols)
    else:
        # Should never happen in a properly implemented game, but handle it just in case
        return 0

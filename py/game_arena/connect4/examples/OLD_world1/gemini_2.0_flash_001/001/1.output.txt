import random

def policy(board: list[list[int]]) -> int:
    """
    A Connect 4 targeting policy that attempts to win or block the opponent,
    otherwise chooses a random valid column.
    """

    def check_win(board, player):
        """Checks if the given player has won the game."""
        rows = len(board)
        cols = len(board[0])

        # Check horizontal
        for r in range(rows):
            for c in range(cols - 3):
                if board[r][c] == player and board[r][c+1] == player and board[r][c+2] == player and board[r][c+3] == player:
                    return True

        # Check vertical
        for r in range(rows - 3):
            for c in range(cols):
                if board[r][c] == player and board[r+1][c] == player and board[r+2][c] == player and board[r+3][c] == player:
                    return True

        # Check positive diagonals
        for r in range(rows - 3):
            for c in range(cols - 3):
                if board[r][c] == player and board[r+1][c+1] == player and board[r+2][c+2] == player and board[r+3][c+3] == player:
                    return True

        # Check negative diagonals
        for r in range(3, rows):
            for c in range(cols - 3):
                if board[r][c] == player and board[r-1][c+1] == player and board[r-2][c+2] == player and board[r-3][c+3] == player:
                    return True

        return False
    
    def get_next_state(board, col, player):
      """Returns the board state after dropping a disc into the given column."""
      new_board = [row[:] for row in board]
      for r in range(len(new_board) - 1, -1, -1):
        if new_board[r][col] == 0:
          new_board[r][col] = player
          return new_board
      return None  # Column is full

    # Check for winning move
    for col in range(7):
        if any(board[r][col] == 0 for r in range(6)):
            temp_board = get_next_state(board, col, 1)
            if temp_board and check_win(temp_board, 1):
                return col

    # Check for blocking move
    for col in range(7):
        if any(board[r][col] == 0 for r in range(6)):
           temp_board = get_next_state(board, col, -1)
           if temp_board and check_win(temp_board, -1):
              return col

    # Choose a random valid column
    valid_cols = [col for col in range(7) if any(board[r][col] == 0 for r in range(6))]
    if valid_cols:
        return random.choice(valid_cols)
    else:
        return 0 # Should never happen in a valid Connect 4 game.

def policy(board: list[list[int]]) -> int:
    import random
    
    ROWS = 6
    COLS = 7
    PLAYER = 1
    OPPONENT = -1
    EMPTY = 0
    
    def is_valid_move(board, col):
        return board[0][col] == EMPTY
    
    def get_valid_moves(board):
        return [col for col in range(COLS) if is_valid_move(board, col)]
    
    def drop_disc(board, col, player):
        new_board = [row[:] for row in board]
        for row in range(ROWS - 1, -1, -1):
            if new_board[row][col] == EMPTY:
                new_board[row][col] = player
                return new_board
        return None
    
    def check_winner(board):
        # Check horizontal
        for row in range(ROWS):
            for col in range(COLS - 3):
                if board[row][col] != EMPTY and all(board[row][col + i] == board[row][col] for i in range(4)):
                    return board[row][col]
        
        # Check vertical
        for row in range(ROWS - 3):
            for col in range(COLS):
                if board[row][col] != EMPTY and all(board[row + i][col] == board[row][col] for i in range(4)):
                    return board[row][col]
        
        # Check diagonal (top-left to bottom-right)
        for row in range(ROWS - 3):
            for col in range(COLS - 3):
                if board[row][col] != EMPTY and all(board[row + i][col + i] == board[row][col] for i in range(4)):
                    return board[row][col]
        
        # Check diagonal (top-right to bottom-left)
        for row in range(ROWS - 3):
            for col in range(3, COLS):
                if board[row][col] != EMPTY and all(board[row + i][col - i] == board[row][col] for i in range(4)):
                    return board[row][col]
        
        return None
    
    def count_threats(board, player):
        threats = 0
        directions = [(0, 1), (1, 0), (1, 1), (1, -1)]
        
        for row in range(ROWS):
            for col in range(COLS):
                if board[row][col] == EMPTY:
                    for dr, dc in directions:
                        count = 0
                        for i in range(1, 4):
                            r, c = row + i * dr, col + i * dc
                            if 0 <= r < ROWS and 0 <= c < COLS and board[r][c] == player:
                                count += 1
                            else:
                                break
                        for i in range(1, 4):
                            r, c = row - i * dr, col - i * dc
                            if 0 <= r < ROWS and 0 <= c < COLS and board[r][c] == player:
                                count += 1
                            else:
                                break
                        if count >= 3:
                            threats += 1
        return threats
    
    def evaluate_position(board):
        winner = check_winner(board)
        if winner == PLAYER:
            return 1000
        elif winner == OPPONENT:
            return -1000
        
        score = 0
        score += count_threats(board, PLAYER) * 10
        score -= count_threats(board, OPPONENT) * 10
        
        # Center column preference
        center_col = COLS // 2
        for row in range(ROWS):
            if board[row][center_col] == PLAYER:
                score += 3
            elif board[row][center_col] == OPPONENT:
                score -= 3
        
        return score
    
    def minimax(board, depth, alpha, beta, maximizing_player):
        winner = check_winner(board)
        if winner is not None or depth == 0:
            return evaluate_position(board)
        
        valid_moves = get_valid_moves(board)
        if not valid_moves:
            return 0
        
        if maximizing_player:
            max_eval = float('-inf')
            for col in valid_moves:
                new_board = drop_disc(board, col, PLAYER)
                if new_board:
                    eval_score = minimax(new_board, depth - 1, alpha, beta, False)
                    max_eval = max(max_eval, eval_score)
                    alpha = max(alpha, eval_score)
                    if beta <= alpha:
                        break
            return max_eval
        else:
            min_eval = float('inf')
            for col in valid_moves:
                new_board = drop_disc(board, col, OPPONENT)
                if new_board:
                    eval_score = minimax(new_board, depth - 1, alpha, beta, True)
                    min_eval = min(min_eval, eval_score)
                    beta = min(beta, eval_score)
                    if beta <= alpha:
                        break
            return min_eval
    
    # Check for immediate win
    valid_moves = get_valid_moves(board)
    for col in valid_moves:
        new_board = drop_disc(board, col, PLAYER)
        if new_board and check_winner(new_board) == PLAYER:
            return col
    
    # Check for blocking opponent's win
    for col in valid_moves:
        new_board = drop_disc(board, col, OPPONENT)
        if new_board and check_winner(new_board) == OPPONENT:
            return col
    
    # Use minimax for strategic play
    best_score = float('-inf')
    best_moves = []
    depth = 5
    
    for col in valid_moves:
        new_board = drop_disc(board, col, PLAYER)
        if new_board:
            score = minimax(new_board, depth, float('-inf'), float('inf'), False)
            if score > best_score:
                best_score = score
                best_moves = [col]
            elif score == best_score:
                best_moves.append(col)
    
    if best_moves:
        return random.choice(best_moves)
    else:
        return random.choice(valid_moves)

import numpy as np
from collections import Counter

def policy(board: list[list[int]]) -> int:
    # Convert board to numpy array for easier handling
    np_board = np.array(board)
    rows, cols = np_board.shape
    
    # Check for winning move
    for col in range(cols):
        if is_valid_move(np_board, col):
            # Try dropping our disc
            temp_board = np_board.copy()
            drop_disc(temp_board, col, 1)
            if check_winner(temp_board, 1):
                return col
    
    # Check for blocking opponent's winning move
    for col in range(cols):
        if is_valid_move(np_board, col):
            # Try dropping opponent's disc
            temp_board = np_board.copy()
            drop_disc(temp_board, col, -1)
            if check_winner(temp_board, -1):
                return col
    
    # Prefer center columns
    center_cols = [3, 2, 4, 1, 5, 0, 6]
    for col in center_cols:
        if is_valid_move(np_board, col):
            return col
    
    # Fallback: choose first valid column
    for col in range(cols):
        if is_valid_move(np_board, col):
            return col

def is_valid_move(board, col):
    """Check if a move is valid (column is not full)"""
    return board[0, col] == 0

def drop_disc(board, col, player):
    """Drop a disc in the column, returns the row where it landed"""
    for row in range(board.shape[0] - 1, -1, -1):
        if board[row, col] == 0:
            board[row, col] = player
            return row
    return -1  # Should not happen if move is valid

def check_winner(board, player):
    """Check if the player has won"""
    rows, cols = board.shape
    
    # Check horizontal
    for row in range(rows):
        for col in range(cols - 3):
            if all(board[row, col+i] == player for i in range(4)):
                return True
    
    # Check vertical
    for row in range(rows - 3):
        for col in range(cols):
            if all(board[row+i, col] == player for i in range(4)):
                return True
    
    # Check diagonal (top-left to bottom-right)
    for row in range(rows - 3):
        for col in range(cols - 3):
            if all(board[row+i, col+i] == player for i in range(4)):
                return True
    
    # Check diagonal (top-right to bottom-left)
    for row in range(rows - 3):
        for col in range(3, cols):
            if all(board[row+i, col-i] == player for i in range(4)):
                return True
    
    return False

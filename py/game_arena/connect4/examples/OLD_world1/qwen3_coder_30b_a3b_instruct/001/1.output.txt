import random
import numpy as np

def policy(board: list[list[int]]) -> int:
    # Convert to numpy array for easier manipulation
    np_board = np.array(board)
    
    # Check for winning move
    for col in range(7):
        if is_valid_move(np_board, col):
            # Try placing disc
            temp_board = np_board.copy()
            drop_disc(temp_board, col, 1)
            if check_winner(temp_board, 1):
                return col
    
    # Check for blocking move
    for col in range(7):
        if is_valid_move(np_board, col):
            # Try placing opponent's disc
            temp_board = np_board.copy()
            drop_disc(temp_board, col, -1)
            if check_winner(temp_board, -1):
                return col
    
    # Prefer center columns
    center_cols = [3, 2, 4, 1, 5, 0, 6]
    for col in center_cols:
        if is_valid_move(np_board, col):
            # Check if this move leads to a line of 3
            temp_board = np_board.copy()
            drop_disc(temp_board, col, 1)
            if count_lines(temp_board, 1) >= 1:
                return col
            # Prefer moves that create potential
            if evaluate_position(temp_board, 1) > 0:
                return col
    
    # Otherwise, pick a random valid move
    valid_moves = [col for col in range(7) if is_valid_move(np_board, col)]
    return random.choice(valid_moves) if valid_moves else 3

def is_valid_move(board, col):
    """Check if a move is valid (column not full)"""
    return board[0][col] == 0

def drop_disc(board, col, player):
    """Drop a disc in the specified column"""
    for row in reversed(range(6)):
        if board[row][col] == 0:
            board[row][col] = player
            break

def check_winner(board, player):
    """Check if a player has won"""
    # Check horizontal
    for row in range(6):
        for col in range(4):
            if all(board[row][col+i] == player for i in range(4)):
                return True
    
    # Check vertical
    for row in range(3):
        for col in range(7):
            if all(board[row+i][col] == player for i in range(4)):
                return True
    
    # Check diagonal (top-left to bottom-right)
    for row in range(3):
        for col in range(4):
            if all(board[row+i][col+i] == player for i in range(4)):
                return True
    
    # Check diagonal (top-right to bottom-left)
    for row in range(3):
        for col in range(3, 7):
            if all(board[row+i][col-i] == player for i in range(4)):
                return True
    
    return False

def count_lines(board, player):
    """Count number of potential lines (3 in a row with 1 empty)"""
    count = 0
    
    # Check horizontal
    for row in range(6):
        for col in range(4):
            line = [board[row][col+i] for i in range(4)]
            if line.count(player) == 3 and line.count(0) == 1:
                count += 1
    
    # Check vertical
    for row in range(3):
        for col in range(7):
            line = [board[row+i][col] for i in range(4)]
            if line.count(player) == 3 and line.count(0) == 1:
                count += 1
    
    # Check diagonal (top-left to bottom-right)
    for row in range(3):
        for col in range(4):
            line = [board[row+i][col+i] for i in range(4)]
            if line.count(player) == 3 and line.count(0) == 1:
                count += 1
    
    # Check diagonal (top-right to bottom-left)
    for row in range(3):
        for col in range(3, 7):
            line = [board[row+i][col-i] for i in range(4)]
            if line.count(player) == 3 and line.count(0) == 1:
                count += 1
    
    return count

def evaluate_position(board, player):
    """Simple heuristic to evaluate a board position"""
    score = 0
    
    # Prefer center columns
    center = [board[row][3] for row in range(6)]
    score += center.count(player) * 3
    
    # Score potential lines
    score += count_lines(board, player) * 2
    
    # Bonus for having a disc in the center
    if board[2][3] == player:
        score += 1
    if board[1][3] == player:
        score += 1
    if board[0][3] == player:
        score += 1
    
    return score

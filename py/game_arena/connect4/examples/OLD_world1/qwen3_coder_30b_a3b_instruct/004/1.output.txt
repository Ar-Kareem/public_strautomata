import numpy as np
from typing import List

def policy(board: List[List[int]]) -> int:
    # Convert to numpy array for easier manipulation
    board = np.array(board)
    rows, cols = board.shape
    
    # Check for winning move
    for col in range(cols):
        if is_valid_move(board, col):
            # Try the move
            temp_board = board.copy()
            drop_disc(temp_board, col, 1)
            if check_winner(temp_board, 1):
                return col
    
    # Check for blocking opponent's winning move
    for col in range(cols):
        if is_valid_move(board, col):
            # Try the move
            temp_board = board.copy()
            drop_disc(temp_board, col, -1)
            if check_winner(temp_board, -1):
                return col
    
    # Evaluate positions using a simple heuristic
    best_score = -float('inf')
    best_cols = []
    
    for col in range(cols):
        if is_valid_move(board, col):
            # Try the move
            temp_board = board.copy()
            drop_disc(temp_board, col, 1)
            
            score = evaluate_position(temp_board, 1)
            
            if score > best_score:
                best_score = score
                best_cols = [col]
            elif score == best_score:
                best_cols.append(col)
    
    # If there are multiple best moves, choose the centermost one
    if best_cols:
        center = cols // 2
        best_col = min(best_cols, key=lambda c: abs(c - center))
        return best_col
    
    # Fallback: choose first valid column
    for col in range(cols):
        if is_valid_move(board, col):
            return col

def is_valid_move(board, col):
    """Check if a move is valid (column not full)"""
    return board[0][col] == 0

def drop_disc(board, col, player):
    """Drop a disc in the given column"""
    for row in range(len(board) - 1, -1, -1):
        if board[row][col] == 0:
            board[row][col] = player
            break

def check_winner(board, player):
    """Check if the given player has won"""
    rows, cols = board.shape
    
    # Check horizontal
    for row in range(rows):
        for col in range(cols - 3):
            if all(board[row][col + i] == player for i in range(4)):
                return True
    
    # Check vertical
    for row in range(rows - 3):
        for col in range(cols):
            if all(board[row + i][col] == player for i in range(4)):
                return True
    
    # Check diagonal (top-left to bottom-right)
    for row in range(rows - 3):
        for col in range(cols - 3):
            if all(board[row + i][col + i] == player for i in range(4)):
                return True
    
    # Check diagonal (bottom-left to top-right)
    for row in range(3, rows):
        for col in range(cols - 3):
            if all(board[row - i][col + i] == player for i in range(4)):
                return True
    
    return False

def evaluate_position(board, player):
    """Simple heuristic to evaluate a position"""
    score = 0
    rows, cols = board.shape
    
    # Evaluate horizontal lines
    for row in range(rows):
        for col in range(cols - 3):
            window = [board[row][col + i] for i in range(4)]
            score += evaluate_window(window, player)
    
    # Evaluate vertical lines
    for row in range(rows - 3):
        for col in range(cols):
            window = [board[row + i][col] for i in range(4)]
            score += evaluate_window(window, player)
    
    # Evaluate diagonal (top-left to bottom-right)
    for row in range(rows - 3):
        for col in range(cols - 3):
            window = [board[row + i][col + i] for i in range(4)]
            score += evaluate_window(window, player)
    
    # Evaluate diagonal (bottom-left to top-right)
    for row in range(3, rows):
        for col in range(cols - 3):
            window = [board[row - i][col + i] for i in range(4)]
            score += evaluate_window(window, player)
            
    return score

def evaluate_window(window, player):
    """Evaluate a window of 4 positions"""
    score = 0
    opponent = -player
    
    if window.count(player) == 4:
        score += 100
    elif window.count(player) == 3 and window.count(0) == 1:
        score += 10
    elif window.count(player) == 2 and window.count(0) == 2:
        score += 2
    
    if window.count(opponent) == 3 and window.count(0) == 1:
        score -= 80
    
    return score

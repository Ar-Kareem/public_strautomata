def policy(board: list[list[int]]) -> int:
    ROWS = 6
    COLS = 7

    # Helper: Get all valid locations (columns that are not full)
    def get_valid_locations(b):
        valid = []
        for col in range(COLS):
            if b[0][col] == 0:
                valid.append(col)
        return valid

    # Helper: Check if a move is valid
    def is_valid_location(b, col):
        return b[0][col] == 0

    # Helper: Get the next open row for a column
    def get_next_open_row(b, col):
        for r in range(ROWS - 1, -1, -1):
            if b[r][col] == 0:
                return r
        return -1

    # Helper: Check for a win
    def winning_move(b, piece):
        # Horizontal
        for r in range(ROWS):
            for c in range(COLS - 3):
                if b[r][c] == piece and b[r][c+1] == piece and b[r][c+2] == piece and b[r][c+3] == piece:
                    return True
        # Vertical
        for r in range(ROWS - 3):
            for c in range(COLS):
                if b[r][c] == piece and b[r+1][c] == piece and b[r+2][c] == piece and b[r+3][c] == piece:
                    return True
        # Positive Slope Diagonal
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                if b[r][c] == piece and b[r+1][c+1] == piece and b[r+2][c+2] == piece and b[r+3][c+3] == piece:
                    return True
        # Negative Slope Diagonal
        for r in range(3, ROWS):
            for c in range(COLS - 3):
                if b[r][c] == piece and b[r-1][c+1] == piece and b[r-2][c+2] == piece and b[r-3][c+3] == piece:
                    return True
        return False

    # Evaluation function: Check for threats and center control
    def evaluate_window(window, piece):
        score = 0
        opp_piece = -1 if piece == 1 else 1
        
        count_piece = window.count(piece)
        count_empty = window.count(0)
        count_opp = window.count(opp_piece)

        if count_piece == 4:
            score += 100000 # Win
        elif count_piece == 3 and count_empty == 1:
            score += 100 # Threat
        elif count_piece == 2 and count_empty == 2:
            score += 10 # Potential

        # Blocking logic
        if count_opp == 3 and count_empty == 1:
            score -= 90000 # Urgent block

        return score

    # Score the board for a specific piece
    def score_position(b, piece):
        score = 0
        
        # Center column preference
        center_array = [b[r][3] for r in range(ROWS)]
        center_count = center_array.count(piece)
        score += center_count * 3

        # Horizontal
        for r in range(ROWS):
            row_array = b[r]
            for c in range(COLS - 3):
                window = row_array[c:c+4]
                score += evaluate_window(window, piece)

        # Vertical
        for c in range(COLS):
            col_array = [b[r][c] for r in range(ROWS)]
            for r in range(ROWS - 3):
                window = col_array[r:r+4]
                score += evaluate_window(window, piece)

        # Positive Diagonal
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                window = [b[r+i][c+i] for i in range(4)]
                score += evaluate_window(window, piece)

        # Negative Diagonal
        for r in range(3, ROWS):
            for c in range(COLS - 3):
                window = [b[r-i][c+i] for i in range(4)]
                score += evaluate_window(window, piece)

        return score

    # Minimax Algorithm with Alpha-Beta Pruning
    def minimax(b, depth, alpha, beta, maximizingPlayer):
        valid_locations = get_valid_locations(b)
        is_terminal = winning_move(b, 1) or winning_move(b, -1) or len(valid_locations) == 0
        
        if depth == 0 or is_terminal:
            if is_terminal:
                if winning_move(b, 1):
                    return (None, 100000000000000)
                elif winning_move(b, -1):
                    return (None, -100000000000000)
                else: # Game is over, no more valid moves
                    return (None, 0)
            else: # Depth is zero
                return (None, score_position(b, 1))

        # Move ordering: check center columns first to improve pruning
        # Col order: 3, 2, 4, 1, 5, 0, 6
        ordered_cols = []
        center = COLS // 2
        for offset in range(0, COLS // 2 + 1):
            if center - offset >= 0: ordered_cols.append(center - offset)
            if center + offset < COLS and center + offset != center - offset: ordered_cols.append(center + offset)

        if maximizingPlayer:
            value = -float('inf')
            best_col = ordered_cols[0] # Default fallback
            for col in ordered_cols:
                if col in valid_locations:
                    # Create a copy of the board
                    temp_board = [r[:] for r in b]
                    row = get_next_open_row(temp_board, col)
                    temp_board[row][col] = 1 # AI Move
                    
                    new_score = minimax(temp_board, depth-1, alpha, beta, False)[1]
                    if new_score > value:
                        value = new_score
                        best_col = col
                    alpha = max(alpha, value)
                    if alpha >= beta:
                        break
            return best_col, value
        else:
            value = float('inf')
            best_col = ordered_cols[0]
            for col in ordered_cols:
                if col in valid_locations:
                    temp_board = [r[:] for r in b]
                    row = get_next_open_row(temp_board, col)
                    temp_board[row][col] = -1 # Opponent Move
                    
                    new_score = minimax(temp_board, depth-1, alpha, beta, True)[1]
                    if new_score < value:
                        value = new_score
                        best_col = col
                    beta = min(beta, value)
                    if alpha >= beta:
                        break
            return best_col, value

    # --- Policy Logic Starts Here ---
    
    # 1. Check for immediate win
    valid_locations = get_valid_locations(board)
    for col in valid_locations:
        temp_board = [r[:] for r in board]
        row = get_next_open_row(temp_board, col)
        temp_board[row][col] = 1
        if winning_move(temp_board, 1):
            return col

    # 2. Check for immediate block (opponent wins if we don't act)
    for col in valid_locations:
        temp_board = [r[:] for r in board]
        row = get_next_open_row(temp_board, col)
        temp_board[row][col] = -1
        if winning_move(temp_board, -1):
            return col

    # 3. Minimax Search
    # Depth 5 is a good balance for 1s time limit in Python on standard hardware
    best_col, minimax_score = minimax(board, 5, -float('inf'), float('inf'), True)
    
    return best_col

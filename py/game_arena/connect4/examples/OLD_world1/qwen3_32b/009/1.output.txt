def policy(board: list[list[int]]) -> int:
    ROWS = 6
    COLS = 7

    def is_valid_location(b, col):
        return any(b[row][col] == 0 for row in range(ROWS))
    
    def get_next_open_row(b, col):
        for row in reversed(range(ROWS)):
            if b[row][col] == 0:
                return row
        return None
    
    def drop_piece(b, row, col, piece):
        b_copy = [list(r) for r in b]
        b_copy[row][col] = piece
        return b_copy
    
    def check_win(b, piece):
        for row in range(ROWS):
            for col in range(COLS - 3):
                if all(b[row][col + i] == piece for i in range(4)):
                    return True
        for col in range(COLS):
            for row in range(ROWS - 3):
                if all(b[row + i][col] == piece for i in range(4)):
                    return True
        for row in range(ROWS - 3):
            for col in range(COLS - 3):
                if all(b[row + i][col + i] == piece for i in range(4)):
                    return True
        for row in range(3, ROWS):
            for col in range(COLS - 3):
                if all(b[row - i][col + i] == piece for i in range(4)):
                    return True
        return False
    
    def is_terminal_node(b):
        return check_win(b, 1) or check_win(b, -1) or len(get_valid_locations(b)) == 0
    
    def get_valid_locations(b):
        return [col for col in range(COLS) if is_valid_location(b, col)]
    
    def score_position(b):
        score = 0
        center_array = [b[row][3] for row in range(ROWS)]
        center_count = center_array.count(1)
        score += center_count * 3

        for row in range(ROWS):
            for col in range(COLS - 3):
                window = b[row][col:col+4]
                score += evaluate_window(window)

        for col in range(COLS):
            col_array = [b[row][col] for row in range(ROWS)]
            for row in range(ROWS - 3):
                window = col_array[row:row+4]
                score += evaluate_window(window)
                
        for row in range(ROWS - 3):
            for col in range(COLS - 3):
                window = [b[row + i][col + i] for i in range(4)]
                score += evaluate_window(window)
                
        for row in range(3, ROWS):
            for col in range(COLS - 3):
                window = [b[row - i][col + i] for i in range(4)]
                score += evaluate_window(window)
                
        return score
    
    def evaluate_window(window):
        score = 0
        if window.count(1) == 4:
            score += 100
        elif window.count(1) == 3 and window.count(0) == 1:
            score += 50
        elif window.count(1) == 2 and window.count(0) == 2:
            score += 20
        elif window.count(1) == 1 and window.count(0) == 3:
            score += 2
        if window.count(-1) == 4:
            score -= 100
        elif window.count(-1) == 3 and window.count(0) == 1:
            score -= 40
        elif window.count(-1) == 2 and window.count(0) == 2:
            score -= 10
        return score

    def minimax(b, depth, alpha, beta, maximizing_player):
        valid_moves = get_valid_locations(b)
        is_terminal = is_terminal_node(b)
        if depth == 0 or is_terminal:
            if is_terminal:
                if check_win(b, -1):
                    return (None, float('-inf'))
                elif check_win(b, 1):
                    return (None, float('inf'))
                else:
                    return (None, 0)
            else:
                return (None, score_position(b))
        if maximizing_player:
            value = float('-inf')
            column = valid_moves[0]
            for col in valid_moves:
                row = get_next_open_row(b, col)
                new_b = drop_piece(b, row, col, 1)
                _, new_score = minimax(new_b, depth-1, alpha, beta, False)
                if new_score > value:
                    value = new_score
                    column = col
                alpha = max(alpha, value)
                if alpha >= beta:
                    break
            return (column, value)
        else:
            value = float('inf')
            column = valid_moves[0]
            for col in valid_moves:
                row = get_next_open_row(b, col)
                new_b = drop_piece(b, row, col, -1)
                _, new_score = minimax(new_b, depth-1, alpha, beta, True)
                if new_score < value:
                    value = new_score
                    column = col
                beta = min(beta, value)
                if alpha >= beta:
                    break
            return (column, value)

    valid_moves = get_valid_locations(board)
    if not valid_moves:
        return -1

    for col in valid_moves:
        row = get_next_open_row(board, col)
        temp_b = drop_piece(board, row, col, 1)
        if check_win(temp_b, 1):
            return col

    for col in valid_moves:
        row = get_next_open_row(board, col)
        temp_b = drop_piece(board, row, col, -1)
        if check_win(temp_b, -1):
            return col

    best_col, _ = minimax(board, depth=4, alpha=float('-inf'), beta=float('inf'), maximizing_player=True)
    return best_col

def get_drop_row(board, col):
    for r in range(5, -1, -1):
        if board[r][col] == 0:
            return r
    return None

def has_four(board, player):
    # Horizontal
    for r in range(6):
        for c in range(4):
            if all(board[r][c + j] == player for j in range(4)):
                return True
    # Vertical
    for c in range(7):
        for r in range(3):
            if all(board[r + j][c] == player for j in range(4)):
                return True
    # Diagonal /
    for r in range(3):
        for c in range(4):
            if all(board[r + j][c + j] == player for j in range(4)):
                return True
    # Diagonal \
    for r in range(3):
        for c in range(3, 7):
            if all(board[r + j][c - j] == player for j in range(4)):
                return True
    return False

def minimax(board, depth, alpha, beta, maximizing):
    if has_four(board, 1):
        return 1
    if has_four(board, -1):
        return -1
    full = True
    for c in range(7):
        if board[5][c] == 0:
            full = False
            break
    if full:
        return 0
    if depth == 0:
        return 0
    col_order = [3, 2, 4, 1, 5, 0, 6]
    if maximizing:
        max_eval = -99
        for c in col_order:
            row = get_drop_row(board, c)
            if row is not None:
                board[row][c] = 1
                eval_score = minimax(board, depth - 1, alpha, beta, False)
                board[row][c] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
        return max_eval
    else:
        min_eval = 99
        for c in col_order:
            row = get_drop_row(board, c)
            if row is not None:
                board[row][c] = -1
                eval_score = minimax(board, depth - 1, alpha, beta, True)
                board[row][c] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
        return min_eval

def policy(board: list[list[int]]) -> int:
    col_order = [3, 2, 4, 1, 5, 0, 6]
    # Check for immediate winning move, preferring center
    for c in col_order:
        row = get_drop_row(board, c)
        if row is not None:
            board[row][c] = 1
            if has_four(board, 1):
                board[row][c] = 0
                return c
            board[row][c] = 0
    # Minimax search for best move
    best_score = -99
    best_c = 3  # default center
    for c in col_order:
        row = get_drop_row(board, c)
        if row is not None:
            board[row][c] = 1
            score = minimax(board, 10, -99, 99, False)
            board[row][c] = 0
            if score > best_score:
                best_score = score
                best_c = c
    return best_c

Here's a Python implementation of a Connect 4 targeting policy that uses a combination of immediate win detection, opponent block detection, and heuristic evaluation:

```python
import numpy as np

def policy(board: list[list[int]]) -> int:
    board_array = np.array(board)
    rows, cols = board_array.shape
    
    # Check for immediate winning move
    for col in range(cols):
        if can_win(board_array, col, 1):
            return col
    
    # Check if opponent can win next move and block
    for col in range(cols):
        if can_win(board_array, col, -1):
            return col
    
    # Try to create opportunities (two potential wins)
    for col in range(cols):
        if is_valid_move(board_array, col):
            temp_board = simulate_move(board_array, col, 1)
            winning_moves = 0
            for c in range(cols):
                if is_valid_move(temp_board, c) and can_win(temp_board, c, 1):
                    winning_moves += 1
                    if winning_moves >= 2:
                        return col
    
    # Center preference (when board is empty)
    center_cols = [3, 2, 4, 1, 5, 0, 6]
    for col in center_cols:
        if is_valid_move(board_array, col):
            return col
    
    # Fallback: first valid column
    for col in range(cols):
        if is_valid_move(board_array, col):
            return col
    
    return 0  # should never reach here if board isn't full

def is_valid_move(board: np.ndarray, col: int) -> bool:
    return 0 in board[:, col]

def simulate_move(board: np.ndarray, col: int, player: int) -> np.ndarray:
    """Return a new board with the move made"""
    new_board = board.copy()
    row = np.where(new_board[:, col] == 0)[0][-1]  # find lowest empty row
    new_board[row, col] = player
    return new_board

def can_win(board: np.ndarray, col: int, player: int) -> bool:
    """Check if making a move in column would result in a win"""
    if not is_valid_move(board, col):
        return False
    
    temp_board = simulate_move(board, col, player)
    return check_win(temp_board, player)

def check_win(board: np.ndarray, player: int) -> bool:
    """Check if the player has won"""
    rows, cols = board.shape
    
    # Check horizontal
    for r in range(rows):
        for c in range(cols - 3):
            if all(board[r, c+i] == player for i in range(4)):
                return True
    
    # Check vertical
    for r in range(rows - 3):
        for c in range(cols):
            if all(board[r+i, c] == player for i in range(4)):
                return True
    
    # Check diagonal (positive slope)
    for r in range(rows - 3):
        for c in range(cols - 3):
            if all(board[r+i, c+i] == player for i in range(4)):
                return True
    
    # Check diagonal (negative slope)
    for r in range(3, rows):
        for c in range(cols - 3):
            if all(board[r-i, c+i] == player for i in range(4)):
                return True
    
    return False
```

This policy implements the following strategy:
1. First checks if there's an immediate winning move and takes it
2. Then checks if the opponent can win next turn and blocks them
3. Attempts to create multiple winning opportunities (forcing moves)
4. Prefers center columns when the board is relatively empty
5. Falls back to the first available valid move

The code uses numpy for efficient board manipulation and includes helper functions for move validation, simulation, and win checking. The policy should execute well within the 1-second time constraint on modern hardware.

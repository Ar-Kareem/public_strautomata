import numpy as np

ROWS = 6
COLS = 7
# Connect 4 evaluation weights
FOUR = 1000
THREE = 100
TWO = 10
 CENTER = 5

def policy(board: list[list[int]]) -> int:
    """
    Implements a Connect 4 policy using minimax with alpha-beta pruning.
    """
    board_np = np.array(board, dtype=np.int8)
    depth = 5  # Adjustable depth for computation time
    best_col = -1
    best_score = -np.inf
    max_player = 1
      
    for col in range(COLS):
        if is_valid_move(board_np, col):
            new_board = make_move(board_np, col, 1)
            score = alpha_beta_minimax(new_board, depth - 1, -np.inf, np.inf, False)
            if score > best_score:
                best_score = score
                best_col = col
            # Break ties by preferring center
            elif score == best_score and abs(col - 3) < abs(best_col - 3):
                best_col = col
    return best_col

def is_valid_move(board, col):
    return board[0, col] == 0

def make_move(board, col, player):
    temp = board.copy()
    for row in range(ROWS-1, -1, -1):
        if temp[row, col] == 0:
            temp[row, col] = player
            return temp
    raise ValueError("Invalid move: column full")

def alpha_beta_minimax(board, depth, alpha, beta, maximizing):
    if depth == 0 or is_terminal(board):
        return evaluate(board, maximizing)
    
    if maximizing:
        max_eval = -np.inf
        for col in range(COLS):
            if is_valid_move(board, col):
                new_board = make_move(board, col, 1)
                eval_val = alpha_beta_minimax(new_board, depth - 1, alpha, beta, False)
                max_eval = max(max_eval, eval_val)
                alpha = max(alpha, eval_val)
                if beta <= alpha:
                    break
        return max_eval
    else:
        min_eval = np.inf
        for col in range(COLS):
            if is_valid_move(board, col):
                new_board = make_move(board, col, -1)
                eval_val = alpha_beta_minimax(new_board, depth - 1, alpha, beta, True)
                min_eval = min(min_eval, eval_val)
                beta = min(beta, eval_val)
                if beta <= alpha:
                    break
        return min_eval

def evaluate(board, maximizing):
    """
    Evaluates the board for the maximizing player (us, 1).
    """
    if is_winning(board, 1):
        return 10000
    if is_winning(board, -1):
        return -10000
    if is_draw(board):
        return 0
    
    score = 0
    # Center column preference
    center_col = [int(board[i, 3]) for i in range(ROWS)]
    center_count = center_col.count(1)
    score += center_count * CENTER
    
    # Evaluate all windows
    score += evaluate_directions(board)
    
    return score

def evaluate_directions(board):
    score = 0
    # Horizontal
    for row in range(ROWS):
        for col in range(COLS - 3):
            window = board[row, col:col+4]
            score += evaluate_window(window)
    # Vertical
    for col in range(COLS):
        for row in range(ROWS - 3):
            window = board[row:row+4, col]
            score += evaluate_window(window)
    # Positive diagonal (down-right)
    for row in range(ROWS - 3):
        for col in range(COLS - 3):
            window = [board[row+i, col+i] for i in range(4)]
            score += evaluate_window(np.array(window))
    # Negative diagonal (up-right)
    for row in range(ROWS - 3):
        for col in range(COLS - 3):
            window = [board[row+3-i, col+i] for i in range(4)]
            score += evaluate_window(np.array(window))
    return score

def evaluate_window(window):
    score = 0
    count_1 = np.sum(window == 1)
    count_opp = np.sum(window == -1)
    empty = np.sum(window == 0)
    
    if count_1 == 4:
        score += FOUR
    elif count_opp == 4:
        score -= FOUR
    elif count_1 == 3 and empty == 1:
        score += THREE
    elif count_opp == 3 and empty == 1:
        score -= THREE
    elif count_1 == 2 and empty == 2:
        score += TWO
    elif count_opp == 2 and empty == 2:
        score -= TWO
    return score

def is_terminal(board):
    return is_winning(board, 1) or is_winning(board, -1) or is_draw(board)

def is_winning(board, player):
    # Check for 4 in a row
    # Horizontal
    for row in range(ROWS):
        for col in range(COLS - 3):
            if all(board[row, col + i] == player for i in range(4)):
                return True
    # Vertical
    for col in range(COLS):
        for row in range(ROWS - 3):
            if all(board[row + i, col] == player for i in range(4)):
                return True
    # Diagonals
    for row in range(ROWS - 3):
        for col in range(COLS - 3):
            if all(board[row + i, col + i] == player for i in range(4)):
                return True
            if all(board[row + 3 - i, col + i] == player for i in range(4)):
                return True
    return False

def is_draw(board):
    return np.all(board != 0)

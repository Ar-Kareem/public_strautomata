```python
import copy

def get_valid_moves(board):
    return [c for c in range(7) if board[0][c] == 0]

def drop_disc(board, col, player):
    new_board = copy.deepcopy(board)
    for r in range(5, -1, -1):
        if new_board[r][col] == 0:
            new_board[r][col] = player
            return new_board
    raise ValueError("Column is full")

def check_win(board, player):
    # Horizontal
    for r in range(6):
        for c in range(4):
            if all(board[r][c+i] == player for i in range(4)):
                return True
    # Vertical
    for r in range(3):
        for c in range(7):
            if all(board[r+i][c] == player for i in range(4)):
                return True
    # Diagonal (down-right)
    for r in range(3):
        for c in range(4):
            if all(board[r+i][c+i] == player for i in range(4)):
                return True
    # Diagonal (down-left)
    for r in range(3):
        for c in range(3, 7):
            if all(board[r+i][c-i] == player for i in range(4)):
                return True
    return False

def evaluate(board):
    if check_win(board, 1):
        return 100
    if check_win(board, -1):
        return -100
    score = 0
    center_cols = [2, 3, 4]
    for r in range(6):
        for c in range(7):
            if board[r][c] == 1:
                if c in center_cols:
                    score += 1
                # Bonus for being higher (closer to bottom)
                score += r // 3
            elif board[r][c] == -1:
                if c in center_cols:
                    score -= 1
                score -= r // 3
    return score

def minimax(board, depth, maximizing_player, alpha, beta):
    valid_moves = get_valid_moves(board)
    if depth == 0 or len(valid_moves) == 0:
        return evaluate(board), None
    if maximizing_player:
        max_eval = float('-inf')
        best_col = None
        for col in valid_moves:
            new_board = drop_disc(board, col, 1)
            eval_score, _ = minimax(new_board, depth - 1, False, alpha, beta)
            if eval_score > max_eval:
                max_eval = eval_score
                best_col = col
            alpha = max(alpha, eval_score)
            if beta <= alpha:
                break
        return max_eval, best_col
    else:
        min_eval = float('inf')
        best_col = None
        for col in valid_moves:
            new_board = drop_disc(board, col, -1)
            eval_score, _ = minimax(new_board, depth - 1, True, alpha, beta)
            if eval_score < min_eval:
                min_eval = eval_score
                best_col = col
            beta = min(beta, eval_score)
            if beta <= alpha:
                break
        return min_eval, best_col

def policy(board):
    # Check for immediate win
    for col in get_valid_moves(board):
        new_board = drop_disc(board, col, 1)
        if check_win(new_board, 1):
            return col
    # Check to block opponent's win
    for col in get_valid_moves(board):
        new_board = drop_disc(board, col, -1)
        if check_win(new_board, -1):
            return col
    # Use minimax with alpha-beta pruning
    _, best_col = minimax(board, 5, True, float('-inf'), float('inf'))
    return best_col
```

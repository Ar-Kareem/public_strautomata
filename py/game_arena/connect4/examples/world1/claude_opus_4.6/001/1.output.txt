## Strategy

I'll implement a strong Connect 4 AI using iterative deepening minimax with alpha-beta pruning, combined with a good evaluation function. Key components:

1. **Win/Loss/Draw detection**: Immediately play winning moves and block opponent wins.
2. **Threat detection**: Identify positions where 3-in-a-row with an open spot exist.
3. **Alpha-beta pruning with iterative deepening**: Search as deep as possible within the time limit.
4. **Move ordering**: Try center columns first and prioritize moves that showed promise in shallower searches to maximize pruning efficiency.
5. **Evaluation function**: Score based on number of 2/3/4 in a row patterns, center control, and threat analysis. Also detect "trap" positions (double threats).

import time
import random

def policy(board: list[list[int]]) -> int:
    ROWS, COLS = 6, 7
    b = [row[:] for row in board]
    
    def get_valid_moves():
        return [c for c in range(COLS) if b[0][c] == 0]
    
    def drop_row(col):
        for r in range(ROWS - 1, -1, -1):
            if b[r][col] == 0:
                return r
        return -1
    
    def check_win(player):
        for r in range(ROWS):
            for c in range(COLS - 3):
                if all(b[r][c+i] == player for i in range(4)):
                    return True
        for r in range(ROWS - 3):
            for c in range(COLS):
                if all(b[r+i][c] == player for i in range(4)):
                    return True
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                if all(b[r+i][c+i] == player for i in range(4)):
                    return True
        for r in range(3, ROWS):
            for c in range(COLS - 3):
                if all(b[r-i][c+i] == player for i in range(4)):
                    return True
        return False

    def evaluate_window(window, player):
        opp = -player
        score = 0
        pc = window.count(player)
        oc = window.count(opp)
        ec = window.count(0)
        if pc == 4:
            score += 100000
        elif pc == 3 and ec == 1:
            score += 50
        elif pc == 2 and ec == 2:
            score += 10
        if oc == 3 and ec == 1:
            score -= 80
        elif oc == 2 and ec == 2:
            score -= 8
        return score

    def evaluate(player):
        score = 0
        # Center preference
        for r in range(ROWS):
            if b[r][3] == player:
                score += 6
            elif b[r][3] == -player:
                score -= 6
        # All windows of 4
        for r in range(ROWS):
            for c in range(COLS - 3):
                w = [b[r][c+i] for i in range(4)]
                score += evaluate_window(w, player)
        for r in range(ROWS - 3):
            for c in range(COLS):
                w = [b[r+i][c] for i in range(4)]
                score += evaluate_window(w, player)
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                w = [b[r+i][c+i] for i in range(4)]
                score += evaluate_window(w, player)
        for r in range(3, ROWS):
            for c in range(COLS - 3):
                w = [b[r-i][c+i] for i in range(4)]
                score += evaluate_window(w, player)
        return score

    def minimax(depth, alpha, beta, maximizing, deadline):
        valid = get_valid_moves()
        if not valid:
            return 0, None
        
        if check_win(1):
            return 1000000 + depth, None
        if check_win(-1):
            return -(1000000 + depth), None
        
        if depth == 0 or time.time() > deadline:
            return evaluate(1), None
        
        # Move ordering: center first
        valid.sort(key=lambda c: abs(c - 3))
        
        if maximizing:
            max_eval = -float('inf')
            best_col = valid[0]
            for col in valid:
                r = drop_row(col)
                b[r][col] = 1
                ev, _ = minimax(depth - 1, alpha, beta, False, deadline)
                b[r][col] = 0
                if ev > max_eval:
                    max_eval = ev
                    best_col = col
                alpha = max(alpha, ev)
                if alpha >= beta:
                    break
                if time.time() > deadline:
                    break
            return max_eval, best_col
        else:
            min_eval = float('inf')
            best_col = valid[0]
            for col in valid:
                r = drop_row(col)
                b[r][col] = -1
                ev, _ = minimax(depth - 1, alpha, beta, True, deadline)
                b[r][col] = 0
                if ev < min_eval:
                    min_eval = ev
                    best_col = col
                beta = min(beta, ev)
                if alpha >= beta:
                    break
                if time.time() > deadline:
                    break
            return min_eval, best_col

    valid_moves = get_valid_moves()
    if not valid_moves:
        return 0
    
    # Immediate win check
    for col in valid_moves:
        r = drop_row(col)
        b[r][col] = 1
        if check_win(1):
            b[r][col] = 0
            return col
        b[r][col] = 0
    
    # Block opponent immediate win
    for col in valid_moves:
        r = drop_row(col)
        b[r][col] = -1
        if check_win(-1):
            b[r][col] = 0
            return col
        b[r][col] = 0

    # Avoid moves that give opponent a win
    safe_moves = []
    for col in valid_moves:
        r = drop_row(col)
        b[r][col] = 1
        gives_win = False
        if r > 0:
            b[r-1][col] = -1
            if check_win(-1):
                gives_win = True
            b[r-1][col] = 0
        b[r][col] = 0
        if not gives_win:
            safe_moves.append(col)
    
    if not safe_moves:
        safe_moves = valid_moves

    deadline = time.time() + 0.85
    best_move = safe_moves[0]
    # Prefer center among safe moves
    safe_moves.sort(key=lambda c: abs(c - 3))
    best_move = safe_moves[0]
    
    # Iterative deepening
    for depth in range(1, 30):
        if time.time() > deadline:
            break
        score, move = minimax(depth, -float('inf'), float('inf'), True, deadline)
        if move is not None and move in valid_moves:
            best_move = move
        if abs(score) > 500000:
            break
    
    if best_move in valid_moves and b[0][best_move] == 0:
        return best_move
    return valid_moves[0]

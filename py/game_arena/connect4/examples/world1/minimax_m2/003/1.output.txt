def is_winning(board, player):

                return True

def drop_disc(board, col, player):
    """Simulate dropping a disc in the specified column. Returns the new board or None if column is full."""
    for row in range(5, -1, -1):  # Start from bottom row (5) to top (0)
        if board[row][col] == 0:
            new_board = copy.deepcopy(board)
            new_board[row][col] = player
            return new_board
    return None  # Column is full

def column_full(board, col):
    """Check if a column is full by looking at the top cell."""
    return board[0][col] != 0  # Top cell is filled if column is full

    """Evaluate the board by scoring all possible windows of 4 cells."""
    score = 0
    # Horizontal windows
    for row in range(6):
        for col in range(4):
            window = [board[row][col], board[row][col+1], board[row][col+2], board[row][col+3]]

import random

def policy(board: list[list[int]]) -> int:
    ROWS = 6
    COLS = 7
    WINDOW_LENGTH = 4
    EMPTY = 0
    PLAYER = 1
    OPPONENT = -1

    def get_valid_locations(grid):
        return [c for c in range(COLS) if grid[0][c] == EMPTY]

    def get_next_open_row(grid, col):
        for r in range(ROWS-1, -1, -1):
            if grid[r][col] == EMPTY:
                return r
        return None

    def drop_piece(grid, row, col, piece):
        new_grid = [row[:] for row in grid]
        new_grid[row][col] = piece
        return new_grid

    def score_window(window, piece):
        score = 0
        opp_piece = OPPONENT if piece == PLAYER else PLAYER

        if window.count(piece) == 4:
            score += 10000
        elif window.count(piece) == 3 and window.count(EMPTY) == 1:
            score += 100
        elif window.count(piece) == 2 and window.count(EMPTY) == 2:
            score += 10

        if window.count(opp_piece) == 3 and window.count(EMPTY) == 1:
            score -= 80

        return score

    def evaluate_board(grid, piece):
        score = 0
        # Score center column
        center_array = [grid[r][COLS//2] for r in range(ROWS)]
        center_count = center_array.count(piece)
        score += center_count * 6

        # Score Horizontal
        for r in range(ROWS):
            row_array = grid[r]
            for c in range(COLS - 3):
                window = row_array[c:c+WINDOW_LENGTH]
                score += score_window(window, piece)

        # Score Vertical
        for c in range(COLS):
            col_array = [grid[r][c] for r in range(ROWS)]
            for r in range(ROWS - 3):
                window = col_array[r:r+WINDOW_LENGTH]
                score += score_window(window, piece)

        # Score positive sloped diagonal
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                window = [grid[r+i][c+i] for i in range(WINDOW_LENGTH)]
                score += score_window(window, piece)

        # Score negative sloped diagonal
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                window = [grid[r+3-i][c+i] for i in range(WINDOW_LENGTH)]
                score += score_window(window, piece)

        return score

    def is_terminal_node(grid):

        for piece in [PLAYER, OPPONENT]:
            # Horizontal
            for c in range(COLS-3):
                for r in range(ROWS):
                    if grid[r][c] == piece and grid[r][c+1] == piece and grid[r][c+2] == piece and grid[r][c+3] == piece:
                        return True

            for c in range(COLS):
                for r in range(ROWS-3):
                    if grid[r][c] == piece and grid[r+1][c] == piece and grid[r+2][c] == piece and grid[r+3][c] == piece:

                for r in range(ROWS-3):
                    if grid[r][c] == piece and grid[r+1][c+1] == piece and grid[r+2][c+2] == piece and grid[r+3][c+3] == piece:

                        return True
        return len(get_valid_locations(grid)) == 0

        valid_locations = get_valid_locations(grid)
        is_terminal = is_terminal_node(grid)
        if depth == 0 or is_terminal:
            if is_terminal:
                # Need to check who won
                # Check player win
                for piece in [PLAYER, OPPONENT]:
                    for c in range(COLS-3):
                        for r in range(ROWS):
                            if grid[r][c] == piece and grid[r][c+1] == piece and grid[r][c+2] == piece and grid[r][c+3] == piece:
                                return (None, 100000000000000 if piece == PLAYER else -100000000000000)
                    for c in range(COLS):
                        for r in range(ROWS-3):
                            if grid[r][c] == piece and grid[r+1][c] == piece and grid[r+2][c] == piece and grid[r+3][c] == piece:
                                return (None, 100000000000000 if piece == PLAYER else -100000000000000)
                    for c in range(COLS-3):
                        for r in range(ROWS-3):
                            if grid[r][c] == piece and grid[r+1][c+1] == piece and grid[r+2][c+2] == piece and grid[r+3][c+3] == piece:
                                return (None, 100000000000000 if piece == PLAYER else -100000000000000)

                            if grid[r][c] == piece and grid[r-1][c+1] == piece and grid[r-2][c+2] == piece and grid[r-3][c+3] == piece:
                                return (None, 100000000000000 if piece == PLAYER else -100000000000000)
                return (None, 0)

                return (None, evaluate_board(grid, PLAYER))

        if maximizingPlayer:

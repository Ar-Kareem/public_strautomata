import collections

def policy(board: list[list[int]]) -> int:
    """
    Connect 4 Policy using Minimax with Alpha-Beta Pruning.
    Board: 6 rows (0 top, 5 bottom) x 7 columns.
    Values: 1 (me), -1 (opponent), 0 (empty).
    """
    ROWS = 6
    COLS = 7

    def get_heights(board):
        heights = []
        for c in range(COLS):
            h = 0
            for r in range(ROWS - 1, -1, -1):
                if board[r][c] != 0:
                    h += 1
                else:
                    break
            heights.append(h)
        return heights

    def check_win(board, player):
        # Horizontal
        for r in range(ROWS):
            for c in range(COLS - 3):
                if all(board[r][c+i] == player for i in range(4)): return True
        # Vertical
        for r in range(ROWS - 3):
            for c in range(COLS):
                if all(board[r+i][c] == player for i in range(4)): return True
        # Diagonal /
        for r in range(3, ROWS):
            for c in range(COLS - 3):
                if all(board[r-i][c+i] == player for i in range(4)): return True
        # Diagonal \
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                if all(board[r+i][c+i] == player for i in range(4)): return True
        return False

    def evaluate_window(window, player):
        score = 0
        opp = -player
        if window.count(player) == 4: score += 100
        elif window.count(player) == 3 and window.count(0) == 1: score += 5
        elif window.count(player) == 2 and window.count(0) == 2: score += 2
        
        if window.count(opp) == 3 and window.count(0) == 1: score -= 4
        return score

    def score_position(board, player):
        score = 0
        # Center column preference
        center_array = [board[r][COLS//2] for r in range(ROWS)]
        score += center_array.count(player) * 3

        # Horizontal
        for r in range(ROWS):
            row_array = board[r]
            for c in range(COLS - 3):
                window = row_array[c:c+4]
                score += evaluate_window(window, player)
        # Vertical
        for c in range(COLS):
            col_array = [board[r][c] for r in range(ROWS)]
            for r in range(ROWS - 3):
                window = col_array[r:r+4]
                score += evaluate_window(window, player)
        # Diagonals
        for r in range(ROWS - 3):
            for c in range(COLS - 3):
                window = [board[r+i][c+i] for i in range(4)]
                score += evaluate_window(window, player)
            for c in range(COLS - 3):
                window = [board[r+3-i][c+i] for i in range(4)]
                score += evaluate_window(window, player)
        return score

    def get_valid_locations(board):
        return [c for c in range(COLS) if board[0][c] == 0]

    def drop_piece(board, col, player):
        new_board = [row[:] for row in board]
        for r in range(ROWS-1, -1, -1):
            if new_board[r][col] == 0:
                new_board[r][col] = player
                return new_board
        return new_board

    memo = {}

    def minimax(board, depth, alpha, beta, maximizingPlayer):
        board_tuple = tuple(tuple(row) for row in board)
        if board_tuple in memo and depth <= memo[board_tuple][1]:
            return memo[board_tuple][0]

        valid_locations = get_valid_locations(board)
        is_terminal = len(valid_locations) == 0 or check_win(board, 1) or check_win(board, -1)
        
        if depth == 0 or is_terminal:
            if is_terminal:
                if check_win(board, 1): return (None, 10000000)
                elif check_win(board, -1): return (None, -10000000)
                else: return (None, 0)
            else:
                return (None, score_position(board, 1))

        if maximizingPlayer:
            value = -float('inf')
            column = valid_locations[0]
            # Order moves: middle out
            for col in sorted(valid_locations, key=lambda x: abs(3-x)):
                new_board = drop_piece(board, col, 1)
                new_score = minimax(new_board, depth-1, alpha, beta, False)[1]
                if new_score > value:
                    value = new_score
                    column = col
                alpha = max(alpha, value)
                if alpha >= beta: break
            memo[board_tuple] = (column, depth)
            return column, value
        else:
            value = float('inf')
            column = valid_locations[0]
            for col in sorted(valid_locations, key=lambda x: abs(3-x)):
                new_board = drop_piece(board, col, -1)
                new_score = minimax(new_board, depth-1, alpha, beta, True)[1]
                if new_score < value:
                    value = new_score
                    column = col
                beta = min(beta, value)
                if alpha >= beta: break
            memo[board_tuple] = (column, depth)
            return column, value

    # Instant check for priority moves
    valid_cols = get_valid_locations(board)
    for c in valid_cols:
        if check_win(drop_piece(board, c, 1), 1): return c
    for c in valid_cols:
        if check_win(drop_piece(board, c, -1), -1): return c

    col, score = minimax(board, 5, -float('inf'), float('inf'), True)
    return col

    board_np = np.array(board, dtype=int)

def _would_win(board, col, player):
    """Check if placing a disc in this column for this player would result in an immediate win."""
    # Find the lowest empty row in this column
    for row in range(5, -1, -1):
        if board[row][col] == 0:
            # Temporarily place the disc
            board[row][col] = player
            won = _check_win_at(board, row, col, player)
            # Undo the move
            board[row][col] = 0
            return won
    return False


def _check_win_at(board, row, col, player):
    """Check if placing a disc at (row, col) creates a win for player."""
    directions = [
        (0, 1),   # horizontal
        (1, 0),   # vertical
        (1, 1),   # diagonal down-right
        (1, -1)   # diagonal down-left
    ]

        count = 1  # count the current disc

        r, c = row + dr, col + dc
        while 0 <= r < 6 and 0 <= c < 7 and board[r][c] == player:
            count += 1
            r += dr
            c += dc

        r, c = row - dr, col - dc
        while 0 <= r < 6 and 0 <= c < 7 and board[r][c] == player:
            count += 1
            r -= dr
            c -= dc
        
        if count >= 4:
            return True
    
    return False

    vertical = 0

    horizontal = 0
    for c in range(max(0, col - 3), min(7, col + 4)):

            horizontal += 1
        elif board[row][c] != 0:  # opponent disc

            break  # can't extend through opponent

    r, c = row + 1, col + 1

        r += 1
        c += 1

        diag_dl += 1
        r += 1
        c -= 1
    
    # Weighted score

    # Prefer center columns

    def make_move(board, col, player):

                return row

    def check_four_in_line(board, row, col, player, dr, dc):

        r, c = row + dr, col + dc

            count += 1
            r += dr

        r, c = row - dr, col - dc
        while 0 <= r < 6 and 0 <= c < 7 and board[r][c] == player:
            count += 1
            r -= dr
            c -= dc
        
        return count >= 4

    def is_valid_move(board, col):
        return board[0][col] == 0
    
    def get_valid_moves(board):
        return [col for col in range(7) if is_valid_move(board, col)]

        # Create a copy and drop piece
        new_board = [row[:] for row in board]
        for row in range(5, -1, -1):
            if new_board[row][col] == 0:
                new_board[row][col] = player
                break
        return new_board
    
    def check_win(board, player):
        # Check horizontal
        for row in range(6):
            for col in range(4):
                if all(board[row][col + i] == player for i in range(4)):
                    return True
        
        # Check vertical
        for row in range(3):
            for col in range(7):
                if all(board[row + i][col] == player for i in range(4)):
                    return True

        for row in range(3):

                if all(board[row + i][col + i] == player for i in range(4)):

        for row in range(3):
            for col in range(3, 7):
                if all(board[row + i][col - i] == player for i in range(4)):
                    return True
        
        return False

        count = 0

        for row in range(6):

        for row in range(6 - length + 1):

        # Check diagonals
        for row in range(6 - length + 1):

        opponent_moves = get_valid_moves(new_board)

        # Heuristic evaluation
        score = 0
        
        # Prefer center columns
        score += (3 - abs(3 - col)) * 3
        
        # Count our potential winning sequences
        score += count_sequences(new_board, 1, 3) * 5
        score += count_sequences(new_board, 1, 2) * 2
        
        # Penalize opponent sequences
        score -= count_sequences(new_board, -1, 3) * 5
        score -= count_sequences(new_board, -1, 2) * 2
        
        return score

    valid_moves = get_valid_moves(board)

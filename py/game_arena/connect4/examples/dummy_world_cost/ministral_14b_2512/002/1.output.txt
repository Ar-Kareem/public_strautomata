### **Connect 4 Policy Strategy**

The strategy for this policy is a **hybrid approach** combining:
1. **Immediate Win Detection** – If the current player can win in the next move, they take it.
2. **Block Opponent's Win** – If the opponent has a winning move, the current player blocks it.
3. **Center Control** – Prefer columns in the center (3, 4, 2, 5, 1, 6) for better strategic positioning.
4. **Create Threats** – If no immediate win or block is found, prioritize moves that create **two-in-a-row** (potential future threats).
5. **Randomness for Unpredictability** – If no strong move is found, pick a random legal column to avoid being exploitable by opponents.

This ensures the AI is **aggressive when possible** (forcing wins) but **defensive when necessary** (blocking opponent threats) while maintaining **positional awareness** (favoring the center and creating future threats).

### **Python Implementation**
```python
import numpy as np
from typing import List

def policy(board: List[List[int]]) -> int:
    # Convert board to numpy array for easier manipulation
    np_board = np.array(board, dtype=int)

    # Get legal moves (columns with at least one empty cell)
    legal_moves = [col for col in range(7) if np_board[0, col] == 0]

    # Check for immediate win
    for col in legal_moves:
        new_board = drop_disc(np_board, col, 1)
        if check_win(new_board, 1):
            return col

    # Check for opponent's immediate win and block it
    for col in legal_moves:
        new_board = drop_disc(np_board, col, -1)
        if check_win(new_board, -1):
            return col

    # Prefer center columns (3, 4, 2, 5, 1, 6)
    center_priority = [3, 4, 2, 5, 1, 6]
    for col in center_priority:
        if col in legal_moves:
            return col

    # Create threats (two-in-a-row for current player)
    for col in legal_moves:
        new_board = drop_disc(np_board, col, 1)
        if count_threats(new_board, 1) >= 2:  # At least two threats
            return col

    # If no immediate win/block or strong threat, pick a random legal move
    return np.random.choice(legal_moves)

def drop_disc(board: np.ndarray, col: int, player: int) -> np.ndarray:
    """Drops a disc into the specified column and returns the new board."""
    new_board = board.copy()
    for row in range(6):
        if new_board[row, col] == 0:
            new_board[row, col] = player
            return new_board
    return new_board  # Should not happen if col is legal

def check_win(board: np.ndarray, player: int) -> bool:
    """Checks if the specified player has a winning line (4 in a row)."""
    # Check horizontal
    for row in range(6):
        for col in range(4):
            if all(board[row, col + i] == player for i in range(4)):
                return True

    # Check vertical
    for row in range(3):
        for col in range(7):
            if all(board[row + i, col] == player for i in range(4)):
                return True

    # Check diagonal (top-left to bottom-right)
    for row in range(3):
        for col in range(4):
            if all(board[row + i, col + i] == player for i in range(4)):
                return True

    # Check diagonal (bottom-left to top-right)
    for row in range(3, 6):
        for col in range(4):
            if all(board[row - i, col + i] == player for i in range(4)):
                return True

    return False

def count_threats(board: np.ndarray, player: int) -> int:
    """Counts the number of two-in-a-row (potential threats) for the player."""
    count = 0

    # Check horizontal
    for row in range(6):
        for col in range(5):
            if board[row, col] == player and board[row, col + 1] == player:
                count += 1

    # Check vertical
    for row in range(5):
        for col in range(7):
            if board[row, col] == player and board[row + 1, col] == player:
                count += 1

    # Check diagonal (top-left to bottom-right)
    for row in range(5):
        for col in range(5):
            if board[row, col] == player and board[row + 1, col + 1] == player:
                count += 1

    # Check diagonal (bottom-left to top-right)
    for row in range(5, 6):
        for col in range(2, 7):
            if board[row, col] == player and board[row - 1, col + 1] == player:
                count += 1

    return count
```

### Strategy for the Connect 4 Policy

The strategy for this policy is a **hybrid approach** combining **immediate threat blocking**, **pattern recognition**, and **evaluation-based decision making** with **Monte Carlo Tree Search (MCTS)** for deeper lookahead. Here's the breakdown:

1. **Immediate Threat Blocking**:
   - First, check if the opponent has a winning move (3 in a row with an empty spot). If so, block it immediately.
   - Similarly, check if you can win in the next move (3 in a row with an empty spot). If so, prioritize that move.

2. **Pattern Recognition**:
   - Look for **open threats** (2 in a row with 2 empty spots) and **semi-open threats** (2 in a row with 1 empty spot) for both you and the opponent. Prioritize moves that create or block these patterns.

3. **Evaluation-Based Decision Making**:
   - Use a **static evaluation function** to score the board based on:
     - **Center control**: Dropping in the center (column 3) is highly valued.
     - **Potential lines**: Score potential lines of 3, 2, etc., for both players.
     - **Mobility**: Prefer moves that leave more options for yourself while limiting the opponent.

4. **Monte Carlo Tree Search (MCTS)**:
   - For deeper lookahead, use MCTS to simulate random games from the current board and select the move with the highest win rate. This helps in breaking ties and making more informed decisions in complex positions.

5. **Fallback to Greedy**:
   - If no immediate threats or patterns are found, fall back to a greedy approach (e.g., prefer the center column or the column with the highest potential score).

def policy(board):
    # Check for immediate win or block
    immediate_move = check_immediate_wins(board)
    if immediate_move is not None:
        return immediate_move

def check_immediate_wins(board):
    # Check for opponent's immediate win (block)
    for col in range(7):
        if is_valid_move(board, col):
            new_board = [row[:] for row in board]

def is_valid_move(board, col):
    return board[0][col] == 0

def drop_piece(board, col, player):
    for row in range(5, -1, -1):
        if board[row][col] == 0:
            board[row][col] = player
            break

def is_winning_move(board, player):

                board[row][col+1] == player and

def evaluate_board(board):

    # Evaluate center control

    if is_valid_move(board, center_col):
        score += 10

            drop_piece(new_board, col, 0)
            new_board = [row[:] for row in board]
            drop_piece(new_board, col, -1)
            score -= evaluate_potential(new_board, -1)

    return score

def evaluate_potential(board, player):
    score = 0
    # Check horizontal
    for row in range(6):
        for col in range(7):
            if board[row][col] == player:
                # Check left
                left = col
                while left > 0 and board[row][left-1] == player:
                    left -= 1
                # Check right
                right = col
                while right < 6 and board[row][right+1] == player:
                    right += 1
                length = right - left + 1

                    score += 2

    # Check vertical

                # Check down

                    down += 1
                length = down - up + 1
                if length >= 3:

                elif length == 2:
                    score += 2

        for col in range(7):

def mcts_search(board, iterations=100):
    root = MCTSNode(board, None, 1)
    for _ in range(iterations):
        node = root
        # Selection
        while node.unvisited_children == 0 and node.children:
            node = max(node.children, key=lambda c: c.visit_count)
        # Expansion
        if node.unvisited_children > 0:
            node = node.expand()
        # Simulation
        sim_board = [row[:] for row in node.board]
        sim_player = 1 if node.player == -1 else -1
        while not is_game_over(sim_board):
            col = random.choice([c for c in range(7) if is_valid_move(sim_board, c)])
            drop_piece(sim_board, col, sim_player)
            sim_player *= -1
        # Backpropagation
        winner = -1 if sim_board[5][3] == 1 else 1  # Simplistic winner check
        while node is not None:
            node.visit_count += 1
            node.wins += (1 if winner == node.player else 0)
            node = node.parent
    # Choose the best child
    return max(root.children, key=lambda c: c.visit_count).col

class MCTSNode:
    def __init__(self, board, parent, player):
        self.board = board
        self.parent = parent
        self.player = player
        self.children = []
        self.visit_count = 0
        self.wins = 0
        self.col = None
        self.unvisited_children = 7  # All columns are initially unvisited

    def expand(self):
        # Find all valid moves
        valid_moves = [c for c in range(7) if is_valid_move(self.board, c)]
        if not valid_moves:
            return None
        # Choose a random valid move
        col = random.choice(valid_moves)
        new_board = [row[:] for row in self.board]
        drop_piece(new_board, col, self.player)
        child = MCTSNode(new_board, self, -self.player)
        child.col = col
        self.children.append(child)
        self.unvisited_children -= 1
        return child

def is_game_over(board):
    # Check if the board is full
    for col in range(7):
        if is_valid_move(board, col):
            return False
    # Check for any winning move
    for player in [1, -1]:
        for col in range(7):
            if is_valid_move(board, col):
                new_board = [row[:] for row in board]
                drop_piece(new_board, col, player)
                if is_winning_move(new_board, player):
                    return True
    return True

The strategy for this policy is a heuristic-based evaluation that prioritizes board safety, the removal of checkers from the bar, and aggressive hitting of opponent blots. It selects moves that maximize a positional score, which heavily rewards avoiding hits on the board and leaving only safe points (0 or 1 checker). The policy first generates all legal moves, then computes a score for each potential outcome, and finally selects the move that results in the highest evaluation.

# Helper function to find the source of a move
def find_source(start_state, end_state_tuple, die_val):
    # end_state_tuple is (my_pts, opp_pts, my_bar, opp_bar, my_off)
    # Compare start and end to find what moved
    # Since we know die_val, we can check possible sources
    
    # Check Bar
    if start_state['my_bar'] > end_state_tuple[2]:
        # Bar -> Dest
        return "B"
        
    # Check Off
    if start_state['my_off'] < end_state_tuple[4]:
        # Must have borne off from some point
        # Find which point decreased
        for i in range(24):
            if start_state['my_pts'][i] > end_state_tuple[0][i]:
                return f"A{i}"
                
    # Normal move
    for i in range(24):
        if start_state['my_pts'][i] > end_state_tuple[0][i]:
            # Source decreased
            # Check if dest increased at i - die_val
            dest = i - die_val
            if dest >= 0:
                if start_state['my_pts'][dest] < end_state_tuple[0][dest]:
                    return f"A{i}"
            elif dest < 0:
                # Off
                if start_state['my_off'] < end_state_tuple[4]:
                    return f"A{i}"
    return "P"

def apply_move(state, die_val, source_str):
    # Returns a copy of state after applying move
    # Used to trace sequence for 2-move case
    new_state = {
        'my_pts': list(state['my_pts']),
        'opp_pts': list(state['opp_pts']),
        'my_bar': state['my_bar'],
        'opp_bar': state['opp_bar'],
        'my_off': state['my_off'],
        'opp_off': state['opp_off'],
        'dice': state['dice']
    }
    
    if source_str == "B":
        new_state['my_bar'] -= 1
        dest = 23 - die_val
        if new_state['opp_pts'][dest] == 1:
            new_state['opp_pts'][dest] = 0
            new_state['opp_bar'] += 1
        new_state['my_pts'][dest] += 1
    elif source_str.startswith("A"):
        idx = int(source_str[1:])
        new_state['my_pts'][idx] -= 1
        dest = idx - die_val
        if dest >= 0:
            if new_state['opp_pts'][dest] == 1:
                new_state['opp_pts'][dest] = 0
                new_state['opp_bar'] += 1
            new_state['my_pts'][dest] += 1
        else:
            new_state['my_off'] += 1
    return new_state

The heuristic considers bearing off when possible, moves checkers into the home board with the highest advancement, and avoids leaving blots by preferring moves that reduce exposed single checkers.

def policy(state: dict) -> str:
    my_pts = state['my_pts']
    opp_pts = state['opp_pts']
    my_bar = state['my_bar']
    opp_bar = state['opp_bar']
    my_off = state['my_off']
    opp_off = state['opp_off']
    dice = state['dice']

    def can_bear_off():
        for i in range(6, 24):
            if my_pts[i] > 0:
                return False
        return True

    def get_best_regular_move(die):
        if can_bear_off():
            for i in range(5, -1, -1):
                if my_pts[i] > 0:
                    if i - die < 0:
                        return f"A{i}"
            for i in range(5, -1, -1):
                if my_pts[i] > 0:
                    dest = i - die
                    if 0 <= dest <= 23 and opp_pts[dest] < 2:
                        return f"A{i}"
            return "P"
        else:
            best_advancement = -1
            best_move = "P"
            for i in range(23, -1, -1):
                if my_pts[i] > 0:
                    dest = i - die
                    if 0 <= dest <= 23 and opp_pts[dest] < 2:
                        advancement = 0
                        if i >= 6 and dest < i:
                            advancement = (6 - dest) - (6 - i)
                        elif i < 6 and dest < i:
                            advancement = i - dest
                        if advancement > best_advancement:
                            best_advancement = advancement
                            best_move = f"A{i}"
            return best_move

    if my_bar > 0:
        possible_dice = [d for d in dice if 24 - d >= 0 and opp_pts[24 - d] < 2]
        if not possible_dice:
            return "H:P,P"
        die_used = max(possible_dice)
        remaining_dice = [d for d in dice if d != die_used]
        if len(dice) == 1:
            return "H:B,P"
        else:
            second_move = get_best_regular_move(remaining_dice[0]) if remaining_dice else "P"
            return f"H:B,{second_move}"
    else:
        if not dice:
            return "H:P,P"
        sorted_dice = sorted(dice, reverse=True)
        if len(sorted_dice) == 1:

            if first_move == "P":
                return "H:P,P"

            second_move = get_best_regular_move(sorted_dice[1])

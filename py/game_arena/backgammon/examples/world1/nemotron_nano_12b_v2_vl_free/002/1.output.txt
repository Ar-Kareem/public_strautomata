import random

def policy(state):
    # Prioritize moving checkers from the bar first
    my_bar = state['my_bar']
    if my_bar > 0:
        dice = state['dice']
        if not dice:
            return "H:P,P"
        
        # Use the higher die first if possible
        higher_die = max(dice)
        lower_die = min(dice)
        
        # Check if higher die can be used for bar
        move1 = ""
        move2 = ""
        
        # Function to generate a move from bar to a target point
        def move_bar(die, is_higher):
            if die == 0:
                return ""
            # Assume the move is possible, but prioritize higher die
            return f"H:B,P" if not is_higher else f"L:B,P"
        
        # Try to use both dice
        if higher_die > 0 and lower_die > 0:
            move1 = move_bar(higher_die, True)
            move2 = move_bar(lower_die, False)
            # Combine the moves
            if move1 and move2:
                return f"{'H' if higher_die > lower_die else 'L'}:B,{','.join([move.split(':')[1] for move in [move1, move2] if move])}"
            elif move1:
                return move1 + ",P"
            elif move2:
                return move2 + ",P"
            else:
                return "H:P,P"
        else:
            if higher_die > 0:
                return move_bar(higher_die, True) + ",P"
            elif lower_die > 0:
                return move_bar(lower_die, False) + ",P"
            else:
                return "H:P,P"
    
    # If no checkers on the bar, try to move checkers forward
    my_pts = state['my_pts']
    dice = state['dice']
    if not dice:
        return "H:P,P"
    
    # Example heuristic: move checkers forward (toward home) if possible
    # This is a simplified example and may not be optimal
    moves = []
    for die in dice:
        # Try to move a checker from the current position
        # This is a placeholder logic; real implementation would require more detail
        for i, count in enumerate(my_pts):
            if count > 0 and die <= 23 - i:
                moves.append((i, i + die))
    
    # Generate the move string based on available moves
    if not moves:
        return "H:P,P"
    
    # Prioritize higher die first
    sorted_dice = sorted(dice, reverse=True)
    move1 = ""
    move2 = ""
    
    if len(sorted_dice) >= 1:
        move1 = f"A{sorted_dice[0]}"
    if len(sorted_dice) >= 2:
        move2 = f"A{sum(sorted_dice)}"
    
    return f"H:{move1},{move2}" if move2 else f"H:{move1},P"

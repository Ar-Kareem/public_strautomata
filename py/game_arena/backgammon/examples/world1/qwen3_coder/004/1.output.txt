The strategy for this policy is to implement a priority-based backgammon agent that follows strategic principles:

1. **Bar Priority**: Always handle checkers on the bar first, as they must be re-entered before any other moves.
2. **Safety Over Aggression**: Prefer safe moves over hitting oppurtunities unless it significantly improves position.
3. **Bearing Off Efficiency**: When in the endgame (all checkers in home board), bear off efficiently.
4. **Prime Building**: Try to build blocks (points with 2+ checkers) to control the board.
5. **Flexibility Preservation**: Maintain flexibility by not stacking too many checkers in one spot unnecessarily.
6. **Race Strategy**: When ahead in the race (more checkers borne off), play safely. When behind, take calculated risks.

The implementation handles:
- Parsing the state correctly
- Generating all possible legal move combinations
- Evaluating positions based on a utility function
- Making decisions according to strategic priorities

def policy(state: dict) -> str:
    dice = state['dice']
    if not dice:
        return "H:P,P"
    
    my_pts = state['my_pts']
    opp_pts = state['opp_pts']
    my_bar = state['my_bar']
    opp_bar = state['opp_bar']
    my_off = state['my_off']
    opp_off = state['opp_off']
    
    # Generate all possible moves
    moves = generate_moves(my_pts, opp_pts, my_bar, dice)
    
    if not moves:
        return "H:P,P"
    
    # Evaluate and choose best move
    best_move = choose_best_move(moves, my_pts, opp_pts, my_bar, my_off, opp_off)
    
    return best_move

def generate_moves(my_pts, opp_pts, my_bar, dice):
    moves = []
    
    if len(dice) == 1:
        die = dice[0]
        # Try using this die first
        possible_starts = get_possible_starts(my_pts, opp_pts, my_bar, [die])
        for start in possible_starts:
            new_my_pts, new_opp_pts, new_my_bar = apply_move(my_pts, opp_pts, my_bar, [(start, die)])
            moves.append(([(start, die)], new_my_pts, new_opp_pts, new_my_bar))
                
    elif len(dice) == 2:
        die1, die2 = dice
        # Try both orders
        for first_die, second_die in [(die1, die2), (die2, die1)]:

                temp_my_pts, temp_opp_pts, temp_my_bar = apply_move(my_pts, opp_pts, my_bar, [(start1, first_die)])

def get_possible_starts(my_pts, opp_pts, my_bar, dice):
    possible_starts = []
    
    # If there are checkers on the bar, must move them first
    if my_bar > 0:
        for die in dice:
            target = 24 - die  # Re-enter from bar (opponent's side)
            if target >= 0 and opp_pts[target] < 2:
                possible_starts.append('B')
        return possible_starts
    else:
        # Check for bearing off
        can_bear_off = all(my_pts[i] == 0 for i in range(6, 24))
        
        for die in dice:
            # Normal moves
            for i in range(24):
                if my_pts[i] > 0:
                    target = i - die
                    if target >= 0:
                        if opp_pts[target] < 2:
                            possible_starts.append(f'A{i}')
                    elif can_bear_off and i < 6:  # Bearing off
                        # Can bear off if target < 0 or exact roll
                        if target < 0 or (target == 0 and my_pts[0] > 0):
                            possible_starts.append(f'A{i}')
    
    return list(set(possible_starts))

def apply_move(my_pts, opp_pts, my_bar, moves):
    new_my_pts = my_pts[:]
    new_opp_pts = opp_pts[:]
    new_my_bar = my_bar
    
    for start, die in moves:
        if start == 'B':
            new_my_bar -= 1
            target = 24 - die
            if new_opp_pts[target] == 1:
                # Hit opponent checker
                new_opp_pts[target] = 0
            new_my_pts[target] += 1
        else:
            start_idx = int(start[1:])
            new_my_pts[start_idx] -= 1
            target = start_idx - die
            if target >= 0:
                if new_opp_pts[target] == 1:
                    # Hit opponent checker
                    new_opp_pts[target] = 0
                new_my_pts[target] += 1
            # If target < 0, it's a bear off, so we don't add to any point
    
    return new_my_pts, new_opp_pts, new_my_bar

def choose_best_move(moves, my_pts, opp_pts, my_bar, my_off, opp_off):
    best_score = float('-inf')
    best_move_str = "H:P,P"
    
    for move_sequence, final_my_pts, final_opp_pts, final_my_bar in moves:
        score = evaluate_position(final_my_pts, final_opp_pts, final_my_bar, my_off, opp_off)
        
        if score > best_score:
            best_score = score
            if len(move_sequence) == 0:
                best_move_str = "H:P,P"
            elif len(move_sequence) == 1:
                start, die = move_sequence[0]
                best_move_str = f"H:{start},P" if die == max([d for s, d in move_sequence]) else f"L:{start},P"
            else:
                # Determine order
                first_move, second_move = move_sequence[0], move_sequence[1]
                first_die, second_die = first_move[1], second_move[1]
                
                if first_die > second_die:
                    order = 'H'
                else:
                    order = 'L'
                    
                # Handle case where dice are equal (doubles)
                if first_die == second_die:
                    order = 'H'  # Arbitrary choice
                
                best_move_str = f"{order}:{first_move[0]},{second_move[0]}"
    
    return best_move_str

def evaluate_position(my_pts, opp_pts, my_bar, my_off, opp_off):
    score = 0
    
    # Benefit for bearing off
    score += (15 - sum(my_pts) - my_bar) * 100
    
    # Penalty for checkers on bar
    score -= my_bar * 50
    
    # Benefit for opponent checkers on bar
    opp_bar = 15 - sum(opp_pts) - (15 - my_off - sum(my_pts) - my_bar)
    score += opp_bar * 50
    
    # Benefit for creating blocks (points with 2+ checkers)
    for i in range(24):
        if my_pts[i] >= 2:
            score += 10
        if my_pts[i] >= 3:
            score += 10
        if my_pts[i] >= 4:
            score += 10
            
    # Penalty for blots (single checkers)
    for i in range(24):
        if my_pts[i] == 1:
            score -= 5
    
    # Benefit for controlling key points (5 and 6 pips from home)
    for i in [5, 6, 7, 8]:
        if my_pts[i] >= 2:
            score += 15
    
    # Benefit for advancing checkers (closer to bearing off)
    distance_score = 0
    for i in range(24):
        distance_score += my_pts[i] * (24 - i)
    score -= distance_score / 10
    
    return score

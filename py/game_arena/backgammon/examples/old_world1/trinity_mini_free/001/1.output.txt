def policy(state: dict) -> str:
    my_pts = state['my_pts']
    opp_pts = state['opp_pts']
    my_bar = state['my_bar']
    opp_bar = state['opp_bar']
    dice = state['dice']
    
    # Prioritize hitting opponent's blots
    def hit_move(die):
        for point in range(23, -1, -1):
            if my_pts[point] >= die:
                dest = point - die
                if dest < 0: 
                    continue
                if opp_pts[dest] == 1:
                    return f'{point},{dest}'
        return None
    
    # Check bar moves first
    if my_bar > 0:
        for die in sorted(dice, reverse=True):
            if my_bar >= die:
                # Find valid hit or safe point
                for point in range(23, -1, -1):
                    if my_pts[point] >= die:
                        dest = point - die
                        if dest < 0: 
                            continue
                        if opp_pts[dest] <= 1:
                            return f'H:{point},{dest}'
    # Check point moves
    for die in sorted(dice, reverse=True):
        if my_bar >= die:
            continue
        if die <= 6:
            move = hit_move(die)
            if move:
                return f'H:{move}'
        if die <= 13:
            move = hit_move(die)
            if move:
                return f'L:{move}'
    # Default to safe moves
    for die in sorted(dice, reverse=True):
        if my_bar >= die:
            continue
        for point in range(23, -1, -1):
            if my_pts[point] >= die:
                dest = point - die
                if dest < 0: 
                    continue
                if opp_pts[dest] <= 1:
                    return f'H:{point},{dest}'
    return 'H:P,P'

The strategy for this policy is to prioritize safe and efficient checker movement while minimizing opponent threats. Key principles include:
1. Always move from the bar first if possible
2. Prefer moves that block opponent's home board
3. Maximize bearing off opportunities
4. Avoid leaving checkers on vulnerable points
5. Use higher dice first to maximize mobility

import numpy as np

def policy(state):
    # Extract state variables
    my_pts = np.array(state['my_pts'])
    opp_pts = np.array(state['opp_pts'])
    my_bar = state['my_bar']
    opp_bar = state['opp_bar']
    my_off = state['my_off']
    opp_off = state['opp_off']
    dice = state['dice']
    
    # Helper function to check if a point is safe for landing
    def is_safe(point, opp_pts):
        return opp_pts[point] < 2
    
    # Helper function to check if a point is blocked
    def is_blocked(point, opp_pts):
        return opp_pts[point] >= 2
    
    # Helper function to check if all checkers are in home board
    def all_in_home(my_pts):
        return np.sum(my_pts[:6]) == 15 - my_off
    
    # Helper function to check if bearing off is possible
    def can_bear_off(dice, my_pts):
        if all_in_home(my_pts):
            return True
        return False
    
    # Handle bar moves first
    if my_bar > 0:
        moves = []
        for die in dice:
            if die <= 6 and my_bar > 0:
                # Calculate entry point
                entry_point = 24 - die
                if entry_point >= 0 and entry_point < 24:
                    if opp_pts[entry_point] < 2:
                        moves.append(f"L:B,{entry_point}")
        if moves:
            return moves[0]  # Use first valid bar move
    
    # Handle bearing off if possible
    if can_bear_off(dice, my_pts):
        moves = []
        for die in dice:
            if die <= 6:
                # Find highest point with checkers
                for point in range(23, -1, -1):
                    if my_pts[point] > 0:
                        if point + die >= 24:
                            moves.append(f"L:{point},P")
                        else:
                            moves.append(f"L:{point},{point+die}")
        if moves:
            return moves[0]  # Use first valid bear-off move
    
    # Handle general moves
    moves = []
    for die in sorted(dice, reverse=True):
        for point in range(23, -1, -1):
            if my_pts[point] > 0:
                # Calculate destination
                dest = point + die
                if dest >= 24:
                    # Bearing off
                    moves.append(f"L:{point},P")
                else:
                    # Check if destination is safe
                    if opp_pts[dest] < 2:
                        moves.append(f"L:{point},{dest}")
    
    # If no moves found, return pass
    if not moves:
        return "H:P,P"
    
    # Prefer moves that block opponent's home board
    blocking_moves = [m for m in moves if 'B' in m]
    if blocking_moves:
        return blocking_moves[0]
    
    # Otherwise return first valid move
    return moves[0]

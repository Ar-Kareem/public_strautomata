I'll create a Backgammon policy that makes smart decisions based on board position and dice rolls.

The logic for this policy is:
1. Handle checkers on the bar first (mandatory moves)
2. Evaluate possible moves based on strategic priorities:
   - Safety of checkers (avoiding blots)
   - Building blocks (points with 2+ checkers)
   - Moving checkers toward home board
   - Bearing off when possible
3. Choose moves that maximize a position evaluation score
4. Handle special cases like bearing off and mandatory move rules

import numpy as np

def policy(state: dict) -> str:
    my_pts = state['my_pts']
    opp_pts = state['opp_pts']
    my_bar = state['my_bar']
    opp_bar = state['opp_bar']
    my_off = state['my_off']
    opp_off = state['opp_off']
    dice = state['dice']
    
    # If no dice, pass
    if not dice:
        return "H:P,P"
    
    # Generate all legal moves
    moves = generate_legal_moves(my_pts, opp_pts, my_bar, dice)
    
    # If no legal moves
    if not moves:

            moves_h = generate_legal_moves(my_pts, opp_pts, my_bar, [dice[0]])

            if not moves_h and not moves_l:

                return f"H:{best_move[0]},P"
            else:
                # Choose the better single move
                best_h = select_best_move(moves_h, my_pts, opp_pts, my_bar)
                best_l = select_best_move(moves_l, my_pts, opp_pts, my_bar)
                
                # Evaluate both options
                score_h = evaluate_position_after_move(my_pts, opp_pts, my_bar, [best_h])
                score_l = evaluate_position_after_move(my_pts, opp_pts, my_bar, [best_l])
                
                if score_h >= score_l:
                    return f"H:{best_h[0]},P"
                else:
                    return f"L:{best_l[0]},P"

    # Select the best move
    best_move = select_best_move(moves, my_pts, opp_pts, my_bar)

        # Determine order

def generate_legal_moves(my_pts, opp_pts, my_bar, dice):
    moves = []
    
    # Handle doubles
    if len(dice) == 2 and dice[0] == dice[1]:
        # Use each die separately for 4 moves
        all_dice = [dice[0]] * 4
        moves = generate_moves_recursive(my_pts, opp_pts, my_bar, all_dice, [])
    elif len(dice) == 2:
        # Try both orders
        moves1 = generate_moves_recursive(my_pts, opp_pts, my_bar, [dice[0], dice[1]], [])
        moves2 = generate_moves_recursive(my_pts, opp_pts, my_bar, [dice[1], dice[0]], [])
        moves = moves1 + moves2
    else:
        # Single die
        moves = generate_moves_recursive(my_pts, opp_pts, my_bar, dice, [])
    
    # Remove duplicates
    unique_moves = []
    seen = set()
    for move in moves:
        move_tuple = tuple(move)
        if move_tuple not in seen:
            seen.add(move_tuple)
            unique_moves.append(move)
    
    return unique_moves

def generate_moves_recursive(my_pts, opp_pts, my_bar, remaining_dice, current_move):
    # Base case: no more dice or max moves reached
    if not remaining_dice or len(current_move) >= 2:
        return [current_move[:]]
    
    moves = []
    die = remaining_dice[0]
    next_dice = remaining_dice[1:]
    
    # If we have checkers on the bar, must move them first
    if my_bar > 0:
        dest = 24 - die  # Re-enter from bar
        # Check if destination is open
        if opp_pts[dest] < 2:
            new_my_pts = my_pts[:]
            new_my_pts[dest] += 1
            new_my_bar = my_bar - 1
            
            new_move = current_move + ["B"]
            moves.extend(generate_moves_recursive(new_my_pts, opp_pts, new_my_bar, next_dice, new_move))
    else:
        # Move checkers from points
        # Try all possible source points
        for src in range(24):
            if my_pts[src] > 0:
                dest = src + die
                
                # Check if bearing off
                if dest > 23:
                    # Can only bear off if all checkers are in home board (points 18-23)
                    can_bear_off = all(my_pts[i] == 0 for i in range(18))
                    if not can_bear_off:
                        # Check if we can bear off using the exact die or the highest occupied point
                        highest_point = -1
                        for i in range(23, -1, -1):
                            if my_pts[i] > 0:
                                highest_point = i
                                break
                        
                        if src == highest_point and src <= 23 - die:
                            # We can still move normally
                            pass
                        elif src == highest_point:
                            # This is a legal bearing off
                            pass
                        else:
                            continue  # Can't bear off from this point
                
                # Check if destination is valid
                if dest <= 23 and opp_pts[dest] < 2:
                    # Valid move
                    new_my_pts = my_pts[:]
                    new_my_pts[src] -= 1
                    if dest <= 23:
                        new_my_pts[dest] += 1
                    
                    new_move = current_move + [f"A{src}"]
                    moves.extend(generate_moves_recursive(new_my_pts, opp_pts, my_bar, next_dice, new_move))
                elif dest > 23:
                    # Bearing off
                    new_my_pts = my_pts[:]
                    new_my_pts[src] -= 1
                    
                    new_move = current_move + [f"A{src}"]
                    moves.extend(generate_moves_recursive(new_my_pts, opp_pts, my_bar, next_dice, new_move))
    
    # If no moves possible with this die, pass this die
    if not moves and current_move:
        return [current_move[:]]
    elif not moves:
        return [[]]
    
    return moves

def select_best_move(moves, my_pts, opp_pts, my_bar):
    if not moves:
        return []
    
    best_move = []
    best_score = float('-inf')
    
    for move in moves:
        if move == []:
            continue
            
        score = evaluate_position_after_move(my_pts, opp_pts, my_bar, move)
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move if best_move else []

def evaluate_position_after_move(my_pts, opp_pts, my_bar, move):
    # Make a copy of the position
    new_my_pts = my_pts[:]
    new_my_bar = my_bar
    
    # Apply the moves (simplified - in reality we'd need to know the dice values)
    for move_from in move:
        if move_from == "B" and new_my_bar > 0:
            new_my_bar -= 1
            # We don't know where it goes without the die value, so we'll skip detailed application
        elif move_from.startswith("A"):
            point = int(move_from[1:])
            if new_my_pts[point] > 0:
                new_my_pts[point] -= 1
                # Destination unknown without die, so simplifying
    
    score = 0
    
    # Evaluate safety (number of blots - checkers alone on a point)
    blots = 0
    for i in range(24):
        if new_my_pts[i] == 1:
            blots += 1
    score -= blots * 3  # Penalize blots
    
    # Evaluate builder points (points with 2+ checkers)
    builders = 0
    for i in range(24):
        if new_my_pts[i] >= 2:
            builders += 1
    score += builders * 2  # Reward builders
    
    # Evaluate progress (checkers closer to home)
    progress = 0
    for i in range(24):
        if new_my_pts[i] > 0:
            progress += new_my_pts[i] * (i / 24)  # Higher points are better
    score += progress * 2
    
    # Evaluate checkers borne off
    borne_off = sum(new_my_pts[18:24])
    score += borne_off * 5  # Reward bearing off
    
    # Penalize checkers on bar
    score -= new_my_bar * 4
    
    return score

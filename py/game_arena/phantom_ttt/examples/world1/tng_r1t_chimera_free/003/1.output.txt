

The strategy for this policy is to first attempt to win by completing a line of three, then select the highest-value cell (center, then corners, then edges) if no immediate win is possible. This approach maximizes winning opportunities while efficiently using the available information.

def policy(board: list[list[int]], legal_moves: list[int]) -> int:
    # Convert legal moves to (row, col) tuples
    legal_cells = {(move // 3, move % 3) for move in legal_moves}
    
    # Define all possible winning lines
    lines = [
        # Rows
        [(0, 0), (0, 1), (0, 2)],
        [(1, 0), (1, 1), (1, 2)],
        [(2, 0), (2, 1), (2, 2)],
        # Columns
        [(0, 0), (1, 0), (2, 0)],
        [(0, 1), (1, 1), (2, 1)],
        [(0, 2), (1, 2), (2, 2)],
        # Diagonals
        [(0, 0), (1, 1), (2, 2)],
        [(0, 2), (1, 1), (2, 0)]
    ]
    
    # Check for potential winning moves
    for line in lines:
        my_count = 0
        empty_cells = []
        for (r, c) in line:
            if board[r][c] == 1:
                my_count += 1
            else:
                empty_cells.append((r, c))
        if my_count == 2 and len(empty_cells) == 1:
            win_move = empty_cells[0]
            if win_move in legal_cells:
                return win_move
    
    # Predefined priority order: center, corners, edges
    priority_order = [
        (1, 1),                  # Center
        (0, 0), (0, 2), (2, 0), (2, 2),  # Corners
        (0, 1), (1, 0), (1, 2), (2, 1)   # Edges
    ]
    
    # Select highest priority legal move
    for move in priority_order:
        if move in legal_cells:
            return move
    
    # Fallback to first legal move (should not occur)
    first_move = legal_moves[0]
    return (first_move // 3, first_move % 3)

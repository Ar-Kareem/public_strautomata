Strategy:
- Try to win immediately: if any legal move completes a line where you already have two confirmed marks, play it.
- Otherwise, take the center if available.
- Otherwise pick a corner (prefer corners that create more potential two-in-a-row "threats" for you).
- If no corner is good, pick a side (again preferring moves that create more potential lines for you).
- Avoid playing a cell already confirmed as yours unless there are no other legal choices.
- Break ties with a small randomized tie-breaker so the policy is less predictable.

This heuristic uses the usual strong priorities for 3x3 Tic Tac Toe adapted for Phantom play (where you only see your own confirmed marks). It aims to maximize immediate wins and future offensive potential while avoiding redundant moves.

import random
from typing import List, Tuple

# Lines on the 3x3 board as index triples (0..8)
LINES = [
    (0, 1, 2),
    (3, 4, 5),
    (6, 7, 8),
    (0, 3, 6),
    (1, 4, 7),
    (2, 5, 8),
    (0, 4, 8),
    (2, 4, 6),
]

# Index sets for center/corners/sides
CENTER = 4
CORNERS = {0, 2, 6, 8}
SIDES = {1, 3, 5, 7}


def idx_to_rc(idx: int) -> Tuple[int, int]:
    return divmod(idx, 3)


def score_move(idx: int, board_flat: List[int]) -> float:
    """
    Compute a heuristic score for attempting to mark index `idx`.
    board_flat: length-9 list with 1 for our confirmed marks and 0 for unknown.
    Higher is better. Assumes idx is not currently one of our confirmed marks.
    """
    score = 0.0

    # Prefer center
    if idx == CENTER:
        score += 3.0

    # Prefer corners a bit
    if idx in CORNERS:
        score += 1.5

    # For each line containing idx, evaluate offensive potential
    for a, b, c in LINES:
        if idx not in (a, b, c):
            continue
        # Count our confirmed marks in that line (excluding idx, which we're about to try)
        others = [p for p in (a, b, c) if p != idx]
        our_count = sum(1 for p in others if board_flat[p] == 1)
        unknown_count = sum(1 for p in others if board_flat[p] == 0)

        # If this move would complete our two-in-a-row -> immediate win opportunity
        if our_count == 2:
            score += 100.0  # highest priority: win
        else:
            # Reward moves that increase number of lines where we have 1 (creating future threats)
            # More weight if the line currently has 1 of ours and otherwise unknowns
            if our_count == 1 and unknown_count == 1:
                score += 6.0
            elif our_count == 1 and unknown_count == 0:
                # unlikely but safe to give small bonus
                score += 2.0
            elif our_count == 0 and unknown_count == 2:
                # creates a first mark in a fully unknown line: moderate
                score += 1.0

    return score


def policy(board: List[List[int]], legal_moves: List[int]) -> Tuple[int, int]:
    """
    Choose a move (row, col) for Phantom Tic Tac Toe given:
      - board: 3x3 list of lists with 1 = our confirmed mark, 0 = unknown
      - legal_moves: list of integers 0..8 allowed by the engine
    Returns (row, col).
    """
    # Flatten board to length-9 list for easier indexing
    board_flat = [board[r][c] for r in range(3) for c in range(3)]

    # Ensure legal moves are integers 0..8
    legal = [int(m) for m in legal_moves]

    # Prefer legal moves that are not already our confirmed marks
    non_owned = [m for m in legal if board_flat[m] == 0]
    if non_owned:
        candidate_moves = non_owned
    else:
        # If all legal moves are already our confirmed marks (rare), fall back to legal
        candidate_moves = legal

    # 1) Immediate win check: if any candidate completes a line with two of our marks, play it
    for m in candidate_moves:
        for a, b, c in LINES:
            if m not in (a, b, c):
                continue
            others = [p for p in (a, b, c) if p != m]
            if sum(1 for p in others if board_flat[p] == 1) == 2:
                # Immediate winning attempt
                return idx_to_rc(m)

    # 2) If center is available among candidates, take it
    if CENTER in candidate_moves:
        return idx_to_rc(CENTER)

    # 3) Score each candidate and pick the best (with small random tie-breaker)
    best_score = None
    best_moves = []
    for m in candidate_moves:
        sc = score_move(m, board_flat)
        # random tie-breaker to avoid deterministic behavior: add tiny noise
        sc += random.random() * 1e-6
        if (best_score is None) or (sc > best_score + 1e-9):
            best_score = sc
            best_moves = [m]
        elif abs(sc - best_score) <= 1e-9:
            best_moves.append(m)

    if not best_moves:
        # Fallback: choose a random legal move
        chosen = random.choice(legal)
        return idx_to_rc(chosen)

    # Prefer corners among best_moves, then sides
    corners = [m for m in best_moves if m in CORNERS]
    if corners:
        chosen = random.choice(corners)
        return idx_to_rc(chosen)
    sides = [m for m in best_moves if m in SIDES]
    if sides:
        chosen = random.choice(sides)
        return idx_to_rc(chosen)

    # Otherwise pick random among best moves
    chosen = random.choice(best_moves)
    return idx_to_rc(chosen)

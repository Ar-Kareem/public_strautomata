The strategy for this policy is to prioritize moves that lead to an immediate win first, followed by a heuristic-based order (center, corners, then edges) to maximize the chances of winning in Phantom Tic Tac Toe. This approach focuses on offensive plays while defaulting to a strategic order when no immediate wins are possible.

```python
def priority_order():
    # Highest priority: center, corners, then edges
    return [4, 0, 2, 6, 8, 1, 3, 5, 7]

def check_win(board):
    # Check rows
    for row in board:
        if row[0] + row[1] + row[2] == 3:
            return True
    # Check columns
    for i in range(3):
        if board[0][i] + board[1][i] + board[2][i] == 3:
            return True
    # Check diagonals
    if (board[0][0] + board[1][1] + board[2][2]) == 3:
        return True
    if (board[0][2] + board[1][1] + board[2][0]) == 3:
        return True
    return False

def policy(board, legal_moves):
    winning_moves = []
    for move in legal_moves:
        row = move // 3
        col = move % 3
        board_copy = [row[:] for row in board]
        board_copy[row][col] = 1
        if check_win(board_copy):
            winning_moves.append(move)
    
    if winning_moves:
        # Prioritize among winning moves
        for move in priority_order():
            if move in winning_moves:
                return (move // 3, move % 3)
    
    # No winning moves, follow priority order
    for move in priority_order():
        if move in legal_moves:
            return (move // 3, move % 3)
    # fallback (shouldn't happen as legal_moves is non-empty)
    return (legal_moves[0] // 3, legal_moves[0] % 3)
```



The strategy for this policy is to prioritize winning moves, then select positions that maximize potential winning lines while preferring the center, corners, and edges in that order. It first checks for immediate winning opportunities, then evaluates each possible move based on how many lines (rows, columns, diagonals) it can contribute to, favoring moves that create or extend lines of 1s. If multiple moves have the same score, the center is chosen first, followed by corners and then edges.

def policy(board):
    # Collect all possible moves (cells with 0)
    moves = [(i, j) for i in range(3) for j in range(3) if board[i][j] == 0]
    if not moves:
        return (0, 0)  # Fallback if no moves (shouldn't occur)

    # Check for immediate winning moves
    for (i, j) in moves:
        # Check row
        if sum(board[i]) == 2:
            return (i, j)
        # Check column
        if sum(board[x][j] for x in range(3)) == 2:
            return (i, j)
        # Check main diagonal
        if i == j and sum(board[k][k] for k in range(3)) == 2:
            return (i, j)
        # Check anti-diagonal
        if i + j == 2 and sum(board[k][2 - k] for k in range(3)) == 2:
            return (i, j)

    # Score moves based on potential to create winning lines
    move_scores = []
    for (i, j) in moves:
        score = 0
        # Row potential
        row_ones = sum(board[i]) - board[i][j]
        if row_ones == 1:
            score += 2
        elif row_ones == 0:
            score += 1
        # Column potential
        col_ones = sum(board[x][j] for x in range(3)) - board[i][j]
        if col_ones == 1:
            score += 2
        elif col_ones == 0:
            score += 1
        # Diagonal potentials
        if i == j:
            diag_ones = sum(board[k][k] for k in range(3)) - board[i][j]
            if diag_ones == 1:
                score += 2
            elif diag_ones == 0:
                score += 1
        if i + j == 2:
            diag_ones = sum(board[k][2 - k] for k in range(3)) - board[i][j]
            if diag_ones == 1:
                score += 2
            elif diag_ones == 0:
                score += 1
        move_scores.append(((i, j), score))

    # Select moves with the highest score
    max_score = max(s for (_, s) in move_scores)
    candidates = [move for (move, s) in move_scores if s == max_score]

    # Define priority for tie-breaking
    def priority(move):
        i, j = move
        if (i, j) == (1, 1):
            return 0  # Center first
        elif (i, j) in [(0, 0), (0, 2), (2, 0), (2, 2)]:
            corner_order = [(0, 0), (0, 2), (2, 0), (2, 2)]
            return 1 + corner_order.index(move)  # Corners next
        else:
            edge_order = [(0, 1), (1, 0), (1, 2), (2, 1)]
            return 5 + edge_order.index(move)  # Edges last

    candidates_sorted = sorted(candidates, key=priority)
    return candidates_sorted[0]

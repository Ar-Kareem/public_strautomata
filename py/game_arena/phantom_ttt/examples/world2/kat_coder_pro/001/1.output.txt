import random

def policy(board):
    """
    Phantom Tic Tac Toe policy function.
    
    Args:
        board: 3x3 list of lists with 1 for confirmed marks, 0 for unconfirmed.
        
    Returns:
        tuple (row, col) representing the selected cell.
    """
    # List of all valid moves
    valid_moves = []
    for row in range(3):
        for col in range(3):
            if board[row][col] != 1:  # Avoid confirmed marks
                valid_moves.append((row, col))
    
    if not valid_moves:
        return (0, 0)  # fallback
        
    # Prefer corners (0,0), (0,2), (2,0), (2,2)
    corners = [(r, c) for r, c in valid_moves if r in (0, 2) and c in (0, 2)]
    if corners:
        return random.choice(corners)
    
    # Prefer center (1,1)
    if (1, 1) in valid_moves:
        return (1, 1)
    
    # Otherwise, pick a random valid move
    return random.choice(valid_moves)

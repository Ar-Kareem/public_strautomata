The strategy for this policy is to use a depth-2 minimax search with an evaluation function based on the connectivity difference between the current player and the opponent. The connectivity is measured as the size of the largest connected component of pieces for each player, using 8-directional connectivity. This approach aims to maximize the current player's connectivity while minimizing the opponent's connectivity, leading to moves that help achieve the game objective of connecting all pieces into a single group.

The policy generates all legal moves for the current player, simulates each move, and then evaluates the resulting board state by considering the opponent's best response. The move that maximizes the evaluation function (current player connectivity minus opponent connectivity) is selected. This strategy balances short-term gains with long-term objectives by looking ahead two moves.

import collections

directions = [
    (0, 1),   # right
    (0, -1),  # left
    (1, 0),   # down
    (-1, 0),  # up
    (1, 1),   # down-right
    (1, -1),  # down-left
    (-1, 1),  # up-right
    (-1, -1)  # up-left
]

def generate_moves(board, player):
    moves = []
    for r in range(8):
        for c in range(8):
            if board[r][c] == player:
                for dx, dy in directions:
                    if dx == 0 and dy == 0:
                        continue
                    # Determine line type and line index
                    if dx == 0:
                        line_type = 'row'
                        line_index = r
                    elif dy == 0:
                        line_type = 'column'
                        line_index = c
                    else:
                        if dx * dy > 0:  # same sign
                            line_type = 'diagonal1'
                            line_index = r - c
                        else:
                            line_type = 'diagonal2'
                            line_index = r + c

                    # Count the number of pieces in the line
                    N = 0
                    if line_type == 'row':
                        for j in range(8):
                            if board[line_index][j] != 0:
                                N += 1
                    elif line_type == 'column':
                        for i in range(8):
                            if board[i][line_index] != 0:
                                N += 1
                    elif line_type == 'diagonal1':
                        # diagonal1: i - j = line_index
                        for i in range(8):
                            j = i - line_index
                            if j < 0 or j >= 8:
                                continue
                            if board[i][j] != 0:
                                N += 1
                    elif line_type == 'diagonal2':
                        # diagonal2: i + j = line_index
                        for i in range(8):
                            j = line_index - i
                            if j < 0 or j >= 8:
                                continue
                            if board[i][j] != 0:
                                N += 1

                    if N == 0:
                        continue

                    tr = r + dx * N
                    tc = c + dy * N

                    if tr < 0 or tr >= 8 or tc < 0 or tc >= 8:
                        continue

                    # Check destination is not friendly
                    if board[tr][tc] == player:
                        continue

                    # Check path is clear of enemy pieces
                    path_clear = True
                    for step in range(1, N):
                        path_r = r + dx * step
                        path_c = c + dy * step
                        if board[path_r][path_c] == -player:
                            path_clear = False
                            break
                    if not path_clear:
                        continue

                    moves.append((r, c, tr, tc))
    return moves

def simulate_move(board, move, player):
    from_r, from_c, to_r, to_c = move
    new_board = [row[:] for row in board]
    new_board[from_r][from_c] = 0
    new_board[to_r][to_c] = player
    return new_board

def connectivity(board, player):
    visited = [[False]*8 for _ in range(8)]
    max_size = 0
    for i in range(8):
        for j in range(8):
            if board[i][j] == player and not visited[i][j]:
                size = 0
                queue = collections.deque()
                queue.append((i,j))
                visited[i][j] = True
                while queue:
                    r, c = queue.popleft()
                    size += 1
                    for dr in [-1,0,1]:
                        for dc in [-1,0,1]:
                            if dr == 0 and dc == 0:
                                continue
                            nr, nc = r+dr, c+dc
                            if 0 <= nr < 8 and 0 <= nc < 8 and not visited[nr][nc] and board[nr][nc] == player:
                                visited[nr][nc] = True
                                queue.append((nr,nc))
                if size > max_size:
                    max_size = size
    return max_size

def eval(board):
    return connectivity(board, 1) - connectivity(board, -1)

def minimax(board, depth, is_maximizing):
    if depth == 0:
        return eval(board)
    if is_maximizing:
        moves = generate_moves(board, 1)
        best_value = -float('inf')
        for move in moves:
            new_board = simulate_move(board, move, 1)
            value = minimax(new_board, depth-1, False)
            if value > best_value:
                best_value = value
        return best_value
    else:
        moves = generate_moves(board, -1)
        best_value = float('inf')
        for move in moves:
            new_board = simulate_move(board, move, -1)
            value = minimax(new_board, depth-1, True)
            if value < best_value:
                best_value = value
        return best_value

def minimax_move(board, depth):
    moves = generate_moves(board, 1)
    if not moves:
        return None
    best_move = None
    best_value = -float('inf')
    for move in moves:
        new_board = simulate_move(board, move, 1)
        value = minimax(new_board, depth-1, False)
        if value > best_value:
            best_value = value
            best_move = move
    return best_move

def policy(board):
    best_move = minimax_move(board, 2)
    if best_move is None:

    from_r, from_c, to_r, to_c = best_move

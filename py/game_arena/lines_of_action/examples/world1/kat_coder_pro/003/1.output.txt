def _is_path_clear(board: np.ndarray, from_r: int, from_c: int, to_r: int, to_c: int, dr: int, dc: int) -> bool:
    """Check if the path from (from_r, from_c) to (to_r, to_c) is clear."""

    while r != to_r or c != to_c:
        if board[r, c] == -1:  # Enemy piece blocking
            return False
        r += dr
        c += dc
    
    # Check the destination - can capture enemy or be empty
    if board[to_r, to_c] == 1:  # Friendly piece at destination
        return False
    
    return True

def _make_move(board: np.ndarray, move: str) -> np.ndarray:
    """Apply a move to the board and return the new board state."""
    from_pos, to_pos = move.split(':')
    from_r, from_c = map(int, from_pos.split(','))
    to_r, to_c = map(int, to_pos.split(','))
    
    new_board = board.copy()
    
    # Move the piece
    new_board[to_r, to_c] = new_board[from_r, from_c]
    new_board[from_r, from_c] = 0
    
    # Capture if enemy piece at destination
    if new_board[to_r, to_c] == -1:
        new_board[to_r, to_c] = 1  # Our piece captures
    
    return new_board

def _is_connected(board: np.ndarray, player: int) -> bool:
    """Check if all pieces of the given player are connected."""
    # Find all player's pieces
    player_mask = (board == player)
    piece_positions = np.where(player_mask)
    positions = list(zip(piece_positions[0], piece_positions[1]))
    
    if len(positions) <= 1:
        return True
    
    # BFS to check connectivity
    visited = set()
    stack = [positions[0]]
    visited.add(positions[0])
    
    while stack:
        r, c = stack.pop()
        
        # Check all 8 neighbors
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                
                nr, nc = r + dr, c + dc
                if 0 <= nr < 8 and 0 <= nc < 8:
                    if (nr, nc) not in visited and board[nr, nc] == player:
                        visited.add((nr, nc))
                        stack.append((nr, nc))
    
    return len(visited) == len(positions)

def _compute_connectivity(board: np.ndarray, player: int) -> float:
    """Compute a connectivity score for the given player."""
    player_mask = (board == player)
    piece_positions = np.where(player_mask)
    positions = list(zip(piece_positions[0], piece_positions[1]))
    
    if len(positions) <= 1:
        return len(positions)
    
    # Count connected components using BFS
    visited = set()
    components = 0
    
    for pos in positions:
        if pos not in visited:
            components += 1
            stack = [pos]
            visited.add(pos)
            
            while stack:
                r, c = stack.pop()
                for dr in [-1, 0, 1]:
                    for dc in [-1, 0, 1]:
                        if dr == 0 and dc == 0:
                            continue

                                stack.append((nr, nc))
    
    # Score is number of pieces minus number of components (want fewer components)
    return len(positions) - components

def _rotate_board(board: np.ndarray) -> np.ndarray:
    """Rotate board to swap player perspectives."""
    return -board[::-1, ::-1]

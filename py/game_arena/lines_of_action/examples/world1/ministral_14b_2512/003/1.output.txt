def get_legal_moves(board, from_row, from_col, dr, dc):
    moves = []
    n = board.shape[0]
    count = 0
    # Count pieces in the line in both directions
    # First, count in the positive direction
    r, c = from_row + dr, from_col + dc
    while 0 <= r < n and 0 <= c < n and board[r, c] != 0:
        count += 1
        r += dr
        c += dc
    # Then, count in the negative direction
    r, c = from_row - dr, from_col - dc
    while 0 <= r < n and 0 <= c < n and board[r, c] != 0:
        count += 1
        r -= dr
        c -= dc
    # Now, count includes the piece itself, so subtract 1
    count -= 1

    # Now, find all possible landing positions
    # We need to move exactly 'count' squares in the direction (dr, dc)
    # Also, cannot jump over enemy pieces
    # So, for each step in the direction, check if we can land after 'count' steps
    # without jumping over enemy pieces

    # Check in the positive direction
    r, c = from_row, from_col
    for step in range(1, count + 1):
        r += dr
        c += dc
        if not (0 <= r < n and 0 <= c < n):
            break
        # Check if there's an enemy piece in the path
        # The path is from (from_row, from_col) to (r, c)
        # We need to ensure no enemy pieces are in between
        path_clear = True
        for i in range(1, step):
            check_r, check_c = from_row + i * dr, from_col + i * dc
            if 0 <= check_r < n and 0 <= check_c < n:
                if board[check_r, check_c] == -1:
                    path_clear = False
                    break
        if path_clear:
            # Check if landing on enemy (capture)
            if board[r, c] == -1:
                moves.append(f"{from_row},{from_col}:{r},{c}")
            # Or landing on empty
            elif board[r, c] == 0:
                moves.append(f"{from_row},{from_col}:{r},{c}")

    # Check in the negative direction
    r, c = from_row, from_col
    for step in range(1, count + 1):
        r -= dr
        c -= dc
        if not (0 <= r < n and 0 <= c < n):
            break
        # Check if there's an enemy piece in the path
        path_clear = True
        for i in range(1, step):
            check_r, check_c = from_row - i * dr, from_col - i * dc
            if 0 <= check_r < n and 0 <= check_c < n:
                if board[check_r, check_c] == -1:
                    path_clear = False
                    break
        if path_clear:
            # Check if landing on enemy (capture)
            if board[r, c] == -1:
                moves.append(f"{from_row},{from_col}:{r},{c}")
            # Or landing on empty
            elif board[r, c] == 0:
                moves.append(f"{from_row},{from_col}:{r},{c}")

    return moves

def evaluate_board(board):
    # Count the number of contiguous groups of my pieces
    visited = np.zeros_like(board, dtype=bool)
    groups = 0
    for r in range(8):
        for c in range(8):
            if board[r, c] == 1 and not visited[r, c]:
                groups += 1
                # BFS to mark all connected pieces
                queue = deque([(r, c)])
                visited[r, c] = True
                while queue:
                    x, y = queue.popleft()
                    for dx in [-1, 0, 1]:
                        for dy in [-1, 0, 1]:
                            if dx == 0 and dy == 0:
                                continue
                            nx, ny = x + dx, y + dy
                            if 0 <= nx < 8 and 0 <= ny < 8:
                                if board[nx, ny] == 1 and not visited[nx, ny]:
                                    visited[nx, ny] = True
                                    queue.append((nx, ny))
    # Score is inversely proportional to the number of groups
    score = -groups
    # Bonus for capturing enemy pieces (but this is handled in move generation)
    return score

The strategy for this policy is to evaluate all legal moves by simulating each one and selecting the one that minimizes the number of connected components (using 8-directional connectivity) among the current player's pieces after the move. This greedy approach directly targets the game objective of connecting all pieces into a single group while ensuring legality of moves.

import copy

def get_my_components(board):
    visited = set()
    count = 0
    for r in range(8):
        for c in range(8):
            if board[r][c] == 1 and (r, c) not in visited:
                dfs(board, visited, r, c)
                count += 1
    return count

def dfs(board, visited, r, c):
    stack = [(r, c)]
    visited.add((r, c))
    while stack:
        cr, cc = stack.pop()
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr == 0 and dc == 0:
                    continue
                nr, nc = cr + dr, cc + dc
                if 0 <= nr < 8 and 0 <= nc < 8 and board[nr][nc] == 1 and (nr, nc) not in visited:
                    visited.add((nr, nc))
                    stack.append((nr, nc))

def simulate_move(board, fr, fc, tr, tc):
    new_board = [row[:] for row in board]
    new_board[fr][fc] = 0
    new_board[tr][tc] = 1
    return new_board

def is_legal(board, r, c, dr, dc):
    d = 0
    for k in range(-7, 8):
        nr = r + k * dr
        nc = c + k * dc
        if 0 <= nr < 8 and 0 <= nc < 8 and board[nr][nc] != 0:
            d += 1
    tr = r + d * dr
    tc = c + d * dc
    if not (0 <= tr < 8 and 0 <= tc < 8):
        return None
    if board[tr][tc] == 1:  # Can't land on own piece
        return None
    for k in range(1, d):
        nr = r + k * dr
        nc = c + k * dc
        if board[nr][nc] == -1:  # Can't jump over opponent
            return None
    return (tr, tc)

def policy(board):
    directions = [(dr, dc) for dr in [-1, 0, 1] for dc in [-1, 0, 1] if (dr, dc) != (0, 0)]
    legal_moves = []
    for r in range(8):
        for c in range(8):
            if board[r][c] == 1:
                for dr, dc in directions:
                    target = is_legal(board, r, c, dr, dc)
                    if target:
                        tr, tc = target
                        legal_moves.append((r, c, tr, tc))
    
    if not legal_moves:
        # Fallback, though unlikely to happen
        return "0,0:0,1"  # Illegal, but prevents crash; rules ensure moves exist
    
    best_move = None
    min_components = float('inf')
    for fr, fc, tr, tc in legal_moves:
        new_board = simulate_move(board, fr, fc, tr, tc)
        comps = get_my_components(new_board)
        if comps < min_components:
            min_components = comps
            best_move = (fr, fc, tr, tc)
    
    return f"{best_move[0]},{best_move[1]}:{best_move[2]},{best_move[3]}"

import numpy as np

def get_piece_positions(board):
    return [(row, col) for row in range(8) for col in range(8) if board[row][col] == 1]

def count_connected_pieces(board):
    visited = [[False] * 8 for _ in range(8)]

    def dfs(r, c):
        if r < 0 or r >= 8 or c < 0 or c >= 8 or visited[r][c] or board[r][c] != 1:
            return 0
        visited[r][c] = True
        count = 1
        for dr in [-1, 0, 1]:
            for dc in [-1, 0, 1]:
                if dr != 0 or dc != 0:
                    count += dfs(r + dr, c + dc)
        return count
    
    max_connected = 0
    for r in range(8):
        for c in range(8):
            if board[r][c] == 1 and not visited[r][c]:
                max_connected = max(max_connected, dfs(r, c))
    
    return max_connected

def is_legal_move(board, from_pos, to_pos):
    r1, c1 = from_pos
    r2, c2 = to_pos
    move_distance = abs(r2 - r1) + abs(c2 - c1)
  
    line_count = sum(1 for r in range(8) if board[r][c1] == 1) + \
                 sum(1 for c in range(8) if board[r1][c] == 1) + \
                 sum(1 for d in range(-7, 8) if 0 <= r1 + d < 8 and 0 <= c1 + d < 8 and board[r1 + d][c1 + d] == 1) + \
                 sum(1 for d in range(-7, 8) if 0 <= r1 + d < 8 and 0 <= c1 - d < 8 and board[r1 + d][c1 - d] == 1)

    if move_distance != line_count:
        return False
        
    if board[r2][c2] == -1:
        # Its an opponent's piece; cannot jump over it
        if (abs(r2 - r1) > 1 or abs(c2 - c1) > 1) and (r2 != r1 and c2 != c1):
            return False
        
    return True

def get_legal_moves(board):
    legal_moves = []
    for row, col in get_piece_positions(board):
        for r in range(8):
            for c in range(8):
                if (r, c) != (row, col) and is_legal_move(board, (row, col), (r, c)):
                    legal_moves.append(f"{row},{col}:{r},{c}")
    return legal_moves

def evaluate_board(board):
    score = 0
    score += count_connected_pieces(board) * 10  # Encouraging connected pieces
    return score

def policy(board) -> str:
    legal_moves = get_legal_moves(board)
    if not legal_moves:
        return ''
    
    best_move = None
    best_score = float('-inf')

    for move in legal_moves:
        from_row, from_col, to_row, to_col = map(int, move.replace(':', ',').split(','))
        # Simulate the move
        new_board = np.copy(board)
        new_board[to_row][to_col] = 1
        new_board[from_row][from_col] = 0
        
        # Score the new board
        score = evaluate_board(new_board)

        if score > best_score:
            best_score = score
            best_move = move

    return best_move if best_move else legal_moves[0]  # Fallback move

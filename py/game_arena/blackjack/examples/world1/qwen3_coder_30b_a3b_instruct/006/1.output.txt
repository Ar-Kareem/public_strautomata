        aces = 0

            if card in ["J", "Q", "K"]:
                value += 10
            elif card == "A":
                aces += 1
                value += 11
            else:
                value += int(card)
        
        # Adjust for Aces - if we're over target, change Aces from 11 to 1
        while value > target and aces > 0:

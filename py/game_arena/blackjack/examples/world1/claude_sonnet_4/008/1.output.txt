        """Calculate the best hand value, handling Aces appropriately."""

                total += 11  # Start by counting Ace as 11
            elif card in ["J", "Q", "K"]:
                total += 10
            else:
                total += int(card)

            total -= 10  # Convert an Ace from 11 to 1
            aces -= 1

        remaining = all_cards.copy()

            remaining.remove(card)

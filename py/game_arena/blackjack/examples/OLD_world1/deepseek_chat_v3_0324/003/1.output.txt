```python
import random

def calculate_hand_value(hand, target):
    total = 0
    aces = 0
    
    for card in hand:
        if card == 'A':
            total += 11
            aces += 1
        elif card in ['J', 'Q', 'K']:
            total += 10
        else:
            total += int(card)
    
    # Adjust for aces if needed
    while total > target and aces > 0:
        total -= 10
        aces -= 1
    
    return total

def policy(hand: list[str], target: int) -> str:
    current_value = calculate_hand_value(hand, target)
    
    # If already busted (shouldn't happen as the game would end)
    if current_value > target:
        return "STAY"
    
    # If exactly at target, stay
    if current_value == target:
        return "STAY"
    
    # Calculate probability threshold based on how close we are to target
    distance_to_target = target - current_value
    
    # Base probability to HIT decreases as we get closer to target
    hit_prob = min(1.0, max(0.1, distance_to_target / 15))
    
    # More aggressive when far from target, conservative when close
    if current_value < target - 5:
        # More likely to HIT when we're far below target
        hit_prob = min(1.0, max(0.7, (target - current_value - 5) / 10))
    elif current_value < target - 2:
        hit_prob = 0.5
    elif current_value < target:
        hit_prob = 0.2
    
    # If we have an ace and it's not being used as 1, be more aggressive
    if 'A' in hand and calculate_hand_value(hand, target) + 10 <= target:
        hit_prob = min(1.0, hit_prob + 0.2)
    
    # Random decision based on probability
    if random.random() < hit_prob:
        return "HIT"
    else:
        return "STAY"
```

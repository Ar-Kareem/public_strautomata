import itertools

    Logic:
    1. Determine current hand value and the set of valid remaining cards in our deck.

       Since we don't know the opponent's hand, we calculate the probability

    3. Compare the Expected Win Probability (EWP) of STAY vs. HIT.
       - STAY: We lock in our score. We win if Opponent Busts OR (Opponent <= T AND Opponent < Us).

        if card in ["J", "Q", "K", "10"]: return 10

    def calculate_hand_details(cards: list[str]):

        total = 0
        aces = 0
        for c in cards:
            if c == "A":

                total += 11

                temp -= 10

            is_soft = soft_flag

    def get_remaining_deck(current_hand: list[str]) -> list[str]:

        for card in current_hand:
            deck_counts[card] -= 1
        remaining = []
        for card, count in deck_counts.items():
            remaining.extend([card] * count)
        return remaining

            # But we don't see Opponent draws.
            # So we must stick to the assumption: Opponent plays optimally from start.
            # This means Opponent Distribution is fixed for the game? No, Opponent draws.
            # If we don't see Opponent cards, we can't know their current state.
            # We can only know the *expected* state.
            # But since the game is simultaneous-ish, we assume Opponent plays by a fixed rule 
            # (e.g. Hits if < T-5) starting from scratch.
            # And we assume we play against that fixed rule.
            # So we can calculate the probability of Opponent achieving any score.
            # But wait, if Opponent hits, their state changes.
            # But we don't know when they hit.
            # This is getting complicated.
            
            # Simplification: 
            # Assume Opponent plays exactly like us (Max Expected Value).
            # But since we don't know their hand, we assume they are in the "Expected" state?
            # No.
            
            # Let's go with a simpler, robust policy often used in these contests:
            # Calculate the Expected Value of the next step only, assuming we stop after that step?
            # No, that's greedy and suboptimal.
            
            # Let's implement a look-ahead.
            # What is the value of hitting?
            # Value(Hit) = Sum P(card) * Value(New_State)
            # Value(Stay) = WinProb(Stay)
            # We need Value(New_State).
            # Since we don't know opponent's future moves, we assume Opponent plays a fixed "Conservative" strategy:
            # Stops at T-5.
            # We can simulate our OWN future moves recursively!
            # This is a self-play recursion.
            # V(hand) = max( WinProb_Stay(hand), Sum P(card) * V(hand+card) )
            # We assume Opponent's win distribution is fixed (based on their fixed strategy).
            
            # Let's implement V(hand) with memoization.
            # We need to compute WinProb_Stay(hand) against the fixed Opponent Strategy.
            # We already have a function to compute Opponent Score Distribution.
            
            # Optimization:
            # Opponent strategy is "Hits if < T-6, Stays if >= T-6".
            # Let's implement this.
            
            def get_opp_dist_fixed(t):
                # Returns distribution of Opponent's final score given they play by heuristic
                # We can precompute this or compute on fly.
                # Since T is constant, we can cache this per T.
                if not hasattr(policy, 'opp_fixed_dist'):
                    policy.opp_fixed_dist = {}
                if t in policy.opp_fixed_dist:
                    return policy.opp_fixed_dist[t]
                
                # Recursive calculator for Opponent
                def calc_opp_dist(oh, od, target_t):
                    score, _, _ = calculate_hand_details(list(oh))
                    if score > target_t: return {'bust': 1.0}
                    if not od: return {score: 1.0}
                    
                    # Heuristic decision
                    if score < target_t - 6: # Hit
                        dist = {}
                        for c in set(od):
                            prob = od.count(c) / len(od)
                            new_oh = tuple(sorted(list(oh) + [c]))
                            new_od_list = list(od)
                            new_od_list.remove(c)
                            new_od = tuple(new_od_list)
                            sub = calc_opp_dist(new_oh, new_od, target_t)
                            for k, v in sub.items():
                                dist[k] = dist.get(k, 0) + v * prob
                        return dist
                    else: # Stay
                        return {score: 1.0}

                full_deck = tuple(sorted(list("A2345678910JQK")))
                dist = calc_opp_dist(tuple(), full_deck, t)
                policy.opp_fixed_dist[t] = dist
                return dist

            def get_my_win_prob_stay(current_hand, t):
                # Calculates P(Win) if I stay now
                my_score, _, _ = calculate_hand_details(current_hand)
                if my_score > t: return 0.0 # Should not be called if bust
                
                opp_dist = get_opp_dist_fixed(t)
                p_win = 0
                for score_val, prob in opp_dist.items():
                    if score_val == 'bust':
                        p_win += prob
                    elif my_score > score_val:
                        p_win += prob
                return p_win

            # Now we need Value(Hand) for future decisions.
            # Value(Hand) = max( WinProb(Stay), Avg(Value(Hand+Card)) )
            # We need to memoize Value(Hand).
            
            memo_v = {}
            
            def get_v(current_hand, t):
                # Convert to tuple for memo
                h_tup = tuple(sorted(current_hand))
                if h_tup in memo_v:
                    return memo_v[h_tup]
                
                # 1. Value of Staying
                v_stay = get_my_win_prob_stay(current_hand, t)
                
                # 2. Value of Hitting
                deck = get_remaining_deck(current_hand)
                if not deck:
                    memo_v[h_tup] = v_stay
                    return v_stay
                
                v_hit = 0
                for c in set(deck):
                    prob = deck.count(c) / len(deck)
                    new_hand = current_hand + [c]
                    new_score, _, _ = calculate_hand_details(new_hand)
                    
                    if new_score > t:
                        # We bust. Win Prob is 0 (or P(Opp Bust)? No, we lose immediately).
                        # "If both bust, draw" -> 0 win.
                        # But if we bust and opp doesn't, we lose (-1).
                        # Let's assume we want to maximize Win Prob.
                        # If we bust, we can't win.
                        val_next = 0.0
                    else:
                        val_next = get_v(new_hand, t)
                
                    v_hit += prob * val_next
                
                res = max(v_stay, v_hit)
                memo_v[h_tup] = res
                return res

            # Now, for the current state, what do we do?
            v_stay = get_my_win_prob_stay(hand, target)
            v_hit = 0
            deck = get_remaining_deck(hand)
            if not deck:
                return "STAY"
            
            for c in set(deck):
                prob = deck.count(c) / len(deck)
                new_hand = hand + [c]
                new_score, _, _ = calculate_hand_details(new_hand)
                if new_score > target:
                    val_next = 0.0
                else:
                    val_next = get_v(new_hand, target)
                v_hit += prob * val_next
            
            if v_hit > v_stay:
                return "HIT"
            else:
                return "STAY"

    # If we didn't enter the loop (should not happen), default to STAY
    return "STAY"

# Note: The logic above relies on the opponent playing a fixed heuristic ("Hits if < T-6, Stays if >= T-6").
# This is a reasonable assumption for a competitive environment where opponent strategy is unknown but likely conservative.
# The recursion `get_v` calculates the optimal strategy for ourselves against that opponent.

# Strategy

My strategy combines several key elements for a strong Pentago policy:

1. **Immediate Win Detection**: First, check if any move results in an immediate win for us.

2. **Blocking Opponent Wins**: If we can't win immediately, check if the opponent could win on their next turn and block those threats.

3. **Threat Creation**: Prioritize moves that create multiple threats (potential winning lines with 4 in a row that can become 5).

4. **Positional Evaluation**: Use a heuristic that values:
   - Lines with more of our pieces and no opponent pieces
   - Center positions in quadrants (more flexible for rotations)
   - Potential to form 5-in-a-row after rotations

5. **Minimax with Alpha-Beta Pruning**: For deeper evaluation, use a limited-depth search to evaluate positions, considering both placement and rotation effects.

6. **Line Counting Heuristic**: Score positions based on the number of potential winning lines we control vs opponent controls, weighted by how many pieces are already in each line.

The evaluation function scores each possible line of 5 (horizontal, vertical, diagonal) based on:
- +1000 for 5 in a row (win)
- +100 for 4 in a row with empty 5th spot
- +10 for 3 in a row with 2 empty
- +1 for 2 in a row with 3 empty
- Negative scores for opponent's lines

import numpy as np

def policy(you, opponent) -> str:
    you = np.array(you, dtype=np.int8)
    opponent = np.array(opponent, dtype=np.int8)
    
    def get_empty_cells(y, o):
        empty = []
        for r in range(6):
            for c in range(6):
                if y[r][c] == 0 and o[r][c] == 0:
                    empty.append((r, c))
        return empty
    
    def rotate_quad(board, quad, direction):
        b = board.copy()
        r0 = 0 if quad < 2 else 3
        c0 = 0 if quad % 2 == 0 else 3
        sub = b[r0:r0+3, c0:c0+3].copy()
        if direction == 'L':
            sub = np.rot90(sub, 1)
        else:
            sub = np.rot90(sub, -1)
        b[r0:r0+3, c0:c0+3] = sub
        return b
    
    def check_win(board):
        for r in range(6):
            for c in range(2):
                if np.all(board[r, c:c+5] == 1):
                    return True
        for c in range(6):
            for r in range(2):
                if np.all(board[r:r+5, c] == 1):
                    return True
        for r in range(2):
            for c in range(2):
                if all(board[r+i, c+i] == 1 for i in range(5)):
                    return True
                if all(board[r+i, c+4-i] == 1 for i in range(5)):
                    return True
        return False
    
    def evaluate(y, o):
        score = 0
        lines = []
        for r in range(6):
            for c in range(2):
                lines.append([(r, c+i) for i in range(5)])
        for c in range(6):
            for r in range(2):
                lines.append([(r+i, c) for i in range(5)])
        for r in range(2):
            for c in range(2):
                lines.append([(r+i, c+i) for i in range(5)])
                lines.append([(r+i, c+4-i) for i in range(5)])
        
        for line in lines:
            yc = sum(y[r][c] for r, c in line)
            oc = sum(o[r][c] for r, c in line)
            if oc == 0 and yc > 0:
                score += [0, 1, 10, 100, 1000, 10000][yc]
            if yc == 0 and oc > 0:
                score -= [0, 1, 10, 100, 1000, 10000][oc]
        return score
    
    def apply_move(y, o, r, c, quad, d):
        ny = y.copy()
        no = o.copy()
        ny[r][c] = 1
        ny = rotate_quad(ny, quad, d)
        no = rotate_quad(no, quad, d)
        return ny, no
    
    empty = get_empty_cells(you, opponent)
    rotations = [(q, d) for q in range(4) for d in ['L', 'R']]
    
    best_move = None
    best_score = float('-inf')
    
    for (r, c) in empty:
        for (quad, d) in rotations:
            ny, no = apply_move(you, opponent, r, c, quad, d)
            
            we_win = check_win(ny)
            they_win = check_win(no)
            
            if we_win and not they_win:
                return f"{r+1},{c+1},{quad},{d}"
            
            if we_win and they_win:
                score = 5000
            elif they_win:
                score = -100000
            else:
                score = evaluate(ny, no)
            
            if score > best_score:
                best_score = score
                best_move = (r, c, quad, d)
    
    if best_move is None:
        r, c = empty[0]
        best_move = (r, c, 0, 'L')
    
    r, c, quad, d = best_move
    return f"{r+1},{c+1},{quad},{d}"

def rotate_quadrant(board, quad, dir):
    """
    Rotate the specified quadrant of the 6x6 board by 90 degrees.
    Returns a new board with rotation applied.
    """
    # Create a copy
    new_board = board.copy()
    
    # Define the 3x3 quadrant boundaries
    if quad == 0:
        rows, cols = range(0, 3), range(0, 3)
    elif quad == 1:
        rows, cols = range(0, 3), range(3, 6)
    elif quad == 2:
        rows, cols = range(3, 6), range(0, 3)
    elif quad == 3:
        rows, cols = range(3, 6), range(3, 6)
    else:
        raise ValueError("Invalid quadrant")
    
    # Extract the quadrant
    quad_arr = np.array([[board[r][c] for c in cols] for r in rows])
    
    # Rotate the quadrant: L is anticlockwise, R is clockwise
    if dir == 'L':  # Anticlockwise
        quad_rotated = np.rot90(quad_arr, k=1)  # 90 deg CCW
    elif dir == 'R':  # Clockwise
        quad_rotated = np.rot90(quad_arr, k=3)  # 90 deg CW (same as 270 deg CCW)
    else:
        raise ValueError("Invalid direction")
    
    # Put back the rotated quadrant
    for i, r in enumerate(rows):
        for j, c in enumerate(cols):
            new_board[r][c] = quad_rotated[i][j]
    
    return new_board

def is_win(board, player):
    """
    Check if player (1 for us, -1 for opponent) has 5 or more in a row.
    Returns True if there's a line of 5 or more consecutive pieces.
    """
    size = 6
    # Check horizontal
    for r in range(size):
        for c in range(size - 4):
            if all(board[r][c+i] == player for i in range(5)):
                return True
    
    # Check vertical
    for c in range(size):
        for r in range(size - 4):
            if all(board[r+i][c] == player for i in range(5)):

    for r in range(size - 4):

                return True

            if all(board[r+i][c-i] == player for i in range(5)):

    # Also check for lines of 6

    for c in range(size):

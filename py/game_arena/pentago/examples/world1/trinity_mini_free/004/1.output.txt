

The strategy for thispolicy is to prioritize immediate threats and control of the center quadrants. It uses a heuristic evaluation to select moves that maximize potential future threats while minimizing opponent opportunities. The implementation includes win detection, quadrant rotation, and a fallback for unexplored scenarios.

def check_win(board, player):
    # Check rows for 5 consecutive player marbles
    for r in range(6):
        for c in range(2):
            if all(board[r][c+i] == player for i in range(5)):
                return True
    # Check columns for 5 consecutive player marbles
    for c in range(6):
        for r in range(2):
            if all(board[r+i][c] == player for i in range(5)):
                return True
    # Check diagonals (down-right)
    for r in range(2):
        for c in range(2):
            if all(board[r+i][c+i] == player for i in range(5)):
                return True
    # Check diagonals (down-left)
    for r in range(2):
        for c in range(4, 2, -1):
            if all(board[r+i][c-i] == player for i in range(5)):
                return True
    return False

def rotate_quadrant(board, quad, direction):
    # Define quadrant starting positions
    starts = [(1,1), (1,4), (4,1), (4,4)]
    r0, c0 = starts[quad]
    # Extract 3x3 quadrant
    quadrant = [board[r0+i][c0:c0+3] for i in range(3)]
    # Rotate quadrant
    if direction == 'L':  # Anticlockwise
        rotated = [[quadrant[j][2-i] for j in range(3)] for i in range(3)]
    else:  # Clockwise
        rotated = [[quadrant[2-j][i] for j in range(3)] for i in range(3)]
    # Place rotated quadrant back
    for i in range(3):
        for j in range(3):
            board[r0+i][c0+j] = rotated[i][j]
    return board

def policy(you, opponent):
    # Check for immediate win opportunities

            if you[r][c] == 0 and opponent[r][c] == 0:

                        board = rotate_quadrant(board, quad, direction)

    # Block opponent's immediate win threats
    for r in range(6):

            if you[r][c] == 0 and opponent[r][c] == 0:

    # Heuristic-based move selection
    # Prioritize center quadrants (0-3) over edges (4-5)
    # Prefer rotations that create potential threats
    for r in range(6):
        for c in range(6):
            if you[r][c] == 0 and opponent[r][c] == 0:
                # Prefer center quadrants
                if r < 3 and c < 3: quad = 0
                elif r < 3 and c >= 3: quad = 1
                elif r >= 3 and c < 3: quad = 2
                elif r >= 3 and c >= 3: quad = 3

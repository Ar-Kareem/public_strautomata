The strategy for this policy is:
1. Generate all legal Pentago moves: place on any empty square, then rotate any quadrant left or right.
2. Immediately play a winning move if one exists.
3. Otherwise, avoid moves that allow the opponent an immediate win when possible.
4. Among safe moves, prefer moves that:
   - create strong 5-cell line potential for us,
   - reduce the opponent’s line potential,
   - favor central control.
5. If every move is dangerous, return the best-scoring legal move anyway.
6. The implementation is careful to always return a legal move string.

from typing import List, Tuple

# Pentago policy:
# - always returns a legal move string "row,col,quad,dir"
# - checks immediate wins
# - checks opponent immediate wins after our move
# - uses a fast heuristic evaluation over all legal moves

N = 6

def rotate_subgrid(grid: List[List[int]], quad: int, direction: str) -> List[List[int]]:
    g = [row[:] for row in grid]
    r0 = 0 if quad < 2 else 3
    c0 = 0 if quad % 2 == 0 else 3

    sub = [row[c0:c0+3] for row in g[r0:r0+3]]

    if direction == 'R':
        rot = [[sub[2 - j][i] for j in range(3)] for i in range(3)]
    else:  # 'L'
        rot = [[sub[j][2 - i] for j in range(3)] for i in range(3)]

    for i in range(3):
        for j in range(3):
            g[r0 + i][c0 + j] = rot[i][j]
    return g

def apply_move(you: List[List[int]], opp: List[List[int]], move: Tuple[int, int, int, str]):
    r, c, q, d = move
    ny = [row[:] for row in you]
    no = [row[:] for row in opp]
    ny[r][c] = 1
    ny = rotate_subgrid(ny, q, d)
    no = rotate_subgrid(no, q, d)
    return ny, no

def all_lines():
    lines = []

    # horizontals length 5 and 6 windows
    for r in range(N):
        for c in range(N - 4):
            lines.append([(r, c + k) for k in range(5)])
    # verticals
    for c in range(N):
        for r in range(N - 4):
            lines.append([(r + k, c) for k in range(5)])
    # diag down-right
    for r in range(N - 4):
        for c in range(N - 4):
            lines.append([(r + k, c + k) for k in range(5)])
    # diag down-left
    for r in range(N - 4):
        for c in range(4, N):
            lines.append([(r + k, c - k) for k in range(5)])

    # Also include full length-6 lines to better evaluate potential
    for r in range(N):
        lines.append([(r, c) for c in range(N)])
    for c in range(N):
        lines.append([(r, c) for r in range(N)])
    lines.append([(i, i) for i in range(N)])
    lines.append([(i, N - 1 - i) for i in range(N)])

    # deduplicate
    seen = set()
    uniq = []
    for line in lines:
        t = tuple(line)
        if t not in seen:
            seen.add(t)
            uniq.append(line)
    return uniq

LINES = all_lines()

def has_five(board: List[List[int]]) -> bool:
    # horizontal
    for r in range(N):
        cnt = 0
        for c in range(N):
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
    # vertical
    for c in range(N):
        cnt = 0
        for r in range(N):
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
    # diag down-right from top row and left col
    for sr in range(N):
        r, c = sr, 0
        cnt = 0
        while r < N and c < N:
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
            r += 1
            c += 1
    for sc in range(1, N):
        r, c = 0, sc
        cnt = 0
        while r < N and c < N:
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
            r += 1
            c += 1
    # diag down-left from top row and right col
    for sr in range(N):
        r, c = sr, N - 1
        cnt = 0
        while r < N and c >= 0:
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
            r += 1
            c -= 1
    for sc in range(N - 2, -1, -1):
        r, c = 0, sc
        cnt = 0
        while r < N and c >= 0:
            cnt = cnt + 1 if board[r][c] else 0
            if cnt >= 5:
                return True
            r += 1
            c -= 1
    return False

def legal_moves(you, opp):
    moves = []
    for r in range(N):
        for c in range(N):
            if you[r][c] == 0 and opp[r][c] == 0:
                for q in range(4):
                    moves.append((r, c, q, 'L'))
                    moves.append((r, c, q, 'R'))
    return moves

def immediate_winning_moves(you, opp):
    wins = []
    for mv in legal_moves(you, opp):
        ny, no = apply_move(you, opp, mv)
        yw = has_five(ny)
        ow = has_five(no)
        # If both have five, draw, not a win.
        if yw and not ow:
            wins.append(mv)
    return wins

def heuristic(you, opp):
    # Terminal-ish scoring
    yw = has_five(you)
    ow = has_five(opp)
    if yw and not ow:
        return 10_000_000
    if ow and not yw:
        return -10_000_000
    if yw and ow:
        return 0

    score = 0

    # center preference
    center_weights = [
        [0, 1, 1, 1, 1, 0],
        [1, 2, 2, 2, 2, 1],
        [1, 2, 3, 3, 2, 1],
        [1, 2, 3, 3, 2, 1],
        [1, 2, 2, 2, 2, 1],
        [0, 1, 1, 1, 1, 0],
    ]
    for r in range(N):
        for c in range(N):
            if you[r][c]:
                score += center_weights[r][c] * 3
            elif opp[r][c]:
                score -= center_weights[r][c] * 3

    # line potential
    weights = {
        (1, 0): 3,
        (2, 0): 20,
        (3, 0): 120,
        (4, 0): 1500,
        (5, 0): 500000,
        (6, 0): 700000,
        (0, 1): -3,
        (0, 2): -20,
        (0, 3): -140,
        (0, 4): -1700,
        (0, 5): -500000,
        (0, 6): -700000,
    }

    for line in LINES:
        yc = 0
        oc = 0
        for r, c in line:
            yc += you[r][c]
            oc += opp[r][c]
        if yc and oc:
            continue
        score += weights.get((yc, oc), 0)

        # extra reward for open-ish five windows
        if len(line) == 5:
            if oc == 0:
                score += [0, 2, 12, 80, 1200, 0][yc]
            elif yc == 0:
                score -= [0, 2, 12, 90, 1300, 0][oc]

    return score

def move_to_str(mv):
    r, c, q, d = mv
    return f"{r+1},{c+1},{q},{d}"

def policy(you, opponent) -> str:
    # Convert to plain int lists for robustness
    you = [[int(you[r][c]) for c in range(N)] for r in range(N)]
    opponent = [[int(opponent[r][c]) for c in range(N)] for r in range(N)]

    moves = legal_moves(you, opponent)
    if not moves:
        # Should not happen per prompt, but keep a fallback shape.
        return "1,1,0,L"

    # 1) Immediate winning move
    wins = immediate_winning_moves(you, opponent)
    if wins:
        # Prefer one that also minimizes opponent threats if multiple
        best = None
        best_score = -10**18
        for mv in wins:
            ny, no = apply_move(you, opponent, mv)
            # if opponent also has five it's a draw; immediate_winning_moves excluded that
            s = heuristic(ny, no)
            if s > best_score:
                best_score = s
                best = mv
        return move_to_str(best)

    # 2) Score all moves with safety check
    best_safe = None
    best_safe_score = -10**18
    best_risky = None
    best_risky_score = -10**18

    for mv in moves:
        ny, no = apply_move(you, opponent, mv)

        # If our move creates opponent five too, it's draw/losing in practical terms unless we also win.
        yw = has_five(ny)
        ow = has_five(no)
        if yw and not ow:
            return move_to_str(mv)

        opp_wins = immediate_winning_moves(no, ny)  # opponent to move next
        safe = (len(opp_wins) == 0)

        s = heuristic(ny, no)

        # Penalize giving many immediate wins
        if not safe:
            s -= 300000 * len(opp_wins)

        # Small tactical bonus if we create multiple future threats
        our_next_wins = immediate_winning_moves(ny, no)
        s += 15000 * len(our_next_wins)

        if safe:
            if s > best_safe_score:
                best_safe_score = s
                best_safe = mv
        else:
            if s > best_risky_score:
                best_risky_score = s
                best_risky = mv

    if best_safe is not None:
        return move_to_str(best_safe)
    return move_to_str(best_risky if best_risky is not None else moves[0])

The strategy for this Pentago player is a minimax-like approach with immediate win checks, opponent win prevention, and a heuristic-based move selection. The policy prioritizes immediate wins, blocks the opponent's potential wins, and selects the move that maximizes the number of marbles on the board, assuming this increases the chances of future winning lines.

def policy(you, opponent) -> str:
    def create_combined_board(you, opponent):
        return [[1 if you[r][c] else 2 if opponent[r][c] else 0 for c in range(6)] for r in range(6)]

    def generate_all_legal_moves(board):
        moves = []
        for r in range(6):
            for c in range(6):
                if board[r][c] == 0:
                    for quad in range(4):
                        for direction in ['L', 'R']:
                            moves.append((r + 1, c + 1, quad, direction))
        return moves

    def simulate_move(board, move, player):
        new_board = [row[:] for row in board]
        row, col, quad, direction = move
        r, c = row - 1, col - 1
        new_board[r][c] = player
        start_row = (quad // 2) * 3
        start_col = (quad % 2) * 3
        quadrant = [new_board[start_row + i][start_col : start_col + 3] for i in range(3)]
        if direction == 'R':
            rotated = [list(row) for row in zip(*quadrant[::-1])]
        else:
            rotated = [list(row) for row in zip(*quadrant)][::-1]
        for i in range(3):
            for j in range(3):
                new_board[start_row + i][start_col + j] = rotated[i][j]
        return new_board

    def is_win(board, player):
        for r in range(6):
            for i in range(2):
                if all(board[r][i + j] == player for j in range(5)):
                    return True
        for c in range(6):
            for i in range(2):
                if all(board[i + j][c] == player for j in range(5)):
                    return True
        for r in range(2):
            for c in range(2):
                if all(board[r + i][c + i] == player for i in range(5)):
                    return True
                if all(board[r + i][c + i + 1] == player for i in range(5)):
                    return True
        for r in range(2):
            for c in range(4, 6):
                if all(board[r + i][c - i] == player for i in range(5)):
                    return True
                if c == 5 and all(board[r + i][c - 1 - i] == player for i in range(5)):
                    return True
        return False

    def can_player_win(board, player):
        for r in range(6):
            for c in range(6):
                if board[r][c] == 0:
                    for quad in range(4):
                        for direction in ['L', 'R']:
                            new_board = simulate_move(board, (r + 1, c + 1, quad, direction), player)
                            if is_win(new_board, player):
                                return True
        return False

    def evaluate_move(board):
        my_count = sum(row.count(1) for row in board)
        opp_count = sum(row.count(2) for row in board)
        return my_count - opp_count

    combined_board = create_combined_board(you, opponent)
    all_moves = generate_all_legal_moves(combined_board)

    for move in all_moves:
        new_board = simulate_move(combined_board, move, 1)
        if is_win(new_board, 1):
            return f"{move[0]},{move[1]},{move[2]},{move[3]}"

    safe_moves = []
    for move in all_moves:
        new_board = simulate_move(combined_board, move, 1)
        if not can_player_win(new_board, 2):
            score = evaluate_move(new_board)
            safe_moves.append((move, score))

    if safe_moves:
        best_move = max(safe_moves, key=lambda x: x[1])[0]
        return f"{best_move[0]},{best_move[1]},{best_move[2]},{best_move[3]}"
    else:
        return f"{all_moves[0][0]},{all_moves[0][1]},{all_moves[0][2]},{all_moves[0][3]}"

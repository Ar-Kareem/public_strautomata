def check_win(board):
    for r in range(6):

            if board[r][c] == 1:

                # Check vertical
                if r <= 1 and all(board[r + i][c] == 1 for i in range(5)):
                    return True
                # Check diagonal \
                if r <= 1 and c <= 1 and all(board[r + i][c + i] == 1 for i in range(5)):
                    return True
                # Check diagonal /
                if r <= 1 and c >= 4 and all(board[r + i][c - i] == 1 for i in range(5)):
                    return True
    return False

# Clobber Game Strategy

The strategy for this policy is to implement a Minimax algorithm with alpha-beta pruning to evaluate future game states and choose the optimal move. The algorithm will:

1. Evaluate the current state using a heuristic function that considers:
   - Number of available moves for each player
   - Piece mobility and positioning
   - Control of the center of the board

2. Use Minimax with a depth limit to search future game states:
   - Maximize the score for our moves
   - Minimize the score for opponent moves
   - Use alpha-beta pruning to improve efficiency

3. Select the move that leads to the highest evaluated state

4. If no future states provide a significant advantage, fall back to a greedy approach:
   - Prioritize capturing opponent pieces that have fewer escape options
   - Move towards the center of the board when possible
   - Avoid moves that leave our pieces isolated

import numpy as np
from typing import Tuple, List, Optional

def policy(you: List[int], opponent: List[int]) -> str:
    """
    Clobber game policy using Minimax with alpha-beta pruning.
    
    Args:
        you: 5x6 array representing your pieces
        opponent: 5x6 array representing opponent pieces
    
    Returns:
        String move in format 'row,col,dir'
    """
    # Convert inputs to numpy arrays
    you_board = np.array(you).reshape(5, 6)
    opp_board = np.array(opponent).reshape(5, 6)
    
    # Get all legal moves
    legal_moves = get_legal_moves(you_board, opp_board)
    
    if not legal_moves:
        # No legal moves available - this shouldn't happen in practice
        return "0,0,R"
    
    if len(legal_moves) == 1:
        # Only one legal move
        return legal_moves[0]
    
    # Use minimax to find the best move
    best_move = None
    best_score = float('-inf')
    alpha = float('-inf')
    beta = float('inf')
    
    # Try different search depths based on available time
    for depth in range(1, 4):  # Try depths 1-3
        for move in legal_moves:
            # Apply move to get new state
            new_you, new_opp = apply_move(you_board.copy(), opp_board.copy(), move)
            
            # Evaluate using minimax
            score = minimax(new_you, new_opp, depth - 1, alpha, beta, False)
            
            if score > best_score:
                best_score = score
                best_move = move
                
            alpha = max(alpha, best_score)
            
            # If we have a clearly winning move, take it
            if best_score > 1000:
                return best_move
    
    return best_move if best_move else legal_moves[0]


def get_legal_moves(you: np.ndarray, opponent: np.ndarray) -> List[str]:
    """Get all legal moves for the current player."""
    moves = []
    rows, cols = you.shape
    
    # Define directions: Up, Right, Down, Left
    directions = [(-1, 0, 'U'), (0, 1, 'R'), (1, 0, 'D'), (0, -1, 'L')]
    
    for r in range(rows):
        for c in range(cols):
            if you[r, c] == 1:  # If we have a piece at this position
                for dr, dc, dir_str in directions:
                    nr, nc = r + dr, c + dc
                    # Check if the destination is within bounds and has an opponent piece
                    if 0 <= nr < rows and 0 <= nc < cols and opponent[nr, nc] == 1:
                        moves.append(f"{r},{c},{dir_str}")
    
    return moves


def apply_move(you: np.ndarray, opponent: np.ndarray, move: str) -> Tuple[np.ndarray, np.ndarray]:
    """Apply a move to the board and return the new state."""
    row, col, direction = move.split(',')
    r, c, = int(row), int(col)
    
    # Calculate destination
    if direction == 'U': nr, nc = r - 1, c
    elif direction == 'R': nr, nc = r, c + 1
    elif direction == 'D': nr, nc = r + 1, c
    elif direction == 'L': nr, nc = r, c - 1
    
    # Apply the move
    you[r, c] = 0  # Remove from original position
    you[nr, nc] = 1  # Move to new position
    opponent[nr, nc] = 0  # Capture opponent piece
    
    return you, opponent


def evaluate_board(you: np.ndarray, opponent: np.ndarray) -> float:
    """Evaluate the current board state."""
    score = 0.0
    
    # Count available moves for each player
    you_moves = len(get_legal_moves(you, opponent))
    opp_moves = len(get_legal_moves(opponent, you))
    
    # Base score: difference in available moves
    score += you_moves - opp_moves
    
    # Add positional bonuses
    center_r, center_c = 2, 2.5  # Center of 5x6 board
    
    for r in range(5):
        for c in range(6):
            if you[r, c] == 1:
                # Bonus for being closer to center
                dist_to_center = abs(r - center_r) + abs(c - center_c)
                score += max(0, 2 - dist_to_center)
                
                # Bonus for mobility (potential moves)
                for dr, dc, _ in [(-1,0), (0,1), (1,0), (0,-1)]:
                    nr, nc = r + dr, c + dc
                    if 0 <= nr < 5 and 0 <= nc < 6 and opponent[nr, nc] == 1:
                        score += 0.5
            
            elif opponent[r, c] == 1:
                # Penalty for opponent being close to center
                dist_to_center = abs(r - center_r) + abs(c - center_c)
                score -= max(0, 2 - dist_to_center)
    
    # Large bonus/penalty for winning/losing
    if you_moves == 0:
        score -= 10000  # We lose
    elif opp_moves == 0:
        score += 10000  # We win
    
    return score


def minimax(you: np.ndarray, opponent: np.ndarray, depth: int, alpha: float, beta: float, is_maximizing: bool) -> float:
    """Minimax algorithm with alpha-beta pruning."""
    if depth == 0:
        return evaluate_board(you, opponent)
    
    if is_maximizing:
        max_eval = float('-inf')
        moves = get_legal_moves(you, opponent)
        
        if not moves:
            return -10000 if depth % 2 == 1 else evaluate_board(you, opponent)
        
        for move in moves:
            new_you, new_opp = apply_move(you.copy(), opponent.copy(), move)
            eval_score = minimax(new_you, new_opp, depth - 1, alpha, beta, False)
            max_eval = max(max_eval, eval_score)
            alpha = max(alpha, eval_score)
            
            if beta <= alpha:
                break  # Alpha-beta pruning
        
        return max_eval
    else:
        min_eval = float('inf')
        moves = get_legal_moves(opponent, you)
        
        if not moves:
            return 10000 if depth % 2 == 0 else evaluate_board(you, opponent)
        
        for move in moves:
            new_opp, new_you = apply_move(opponent.copy(), you.copy(), move)
            eval_score = minimax(new_you, new_opp, depth - 1, alpha, beta, True)
            min_eval = min(min_eval, eval_score)
            beta = min(beta, eval_score)
            
            if beta <= alpha:
                break  # Alpha-beta pruning
        
        return min_eval

The strategy is to implement a greedy policy that selects the move causing the new position to have the maximum number of adjacent opponent pieces, enabling more capture options in the next turn. This approach ensures high-scoring moves that maintain aggressive play by maximizing immediate threat options.

def policy(you: list[int], opponent: list[int]) -> str:
    # Define directions in order of priority for tie-breaking
    directions = [
        ('R', (0, 1)),  # Right
        ('D', (1, 0)),  # Down
        ('U', (-1, 0)), # Up
        ('L', (0, -1))  # Left
    ]
    
    legal_moves = []
    for row in range(5):
        for col in range(6):
            if you[row][col] == 1:
                for dir_name, (dr, dc) in directions:
                    new_row = row + dr
                    new_col = col + dc
                    # Check if the move is within bounds and captures an opponent
                    if (0 <= new_row < 5 and 0 <= new_col < 6 and 
                        opponent[new_row][new_col] == 1):
                        legal_moves.append((row, col, dir_name))
    
    # Select move with the highest number of adjacent opponent pieces in the new position
    best_score = -1
    best_move = legal_moves[0]  # At least one move exists per problem constraints
    for move in legal_moves:
        _, _, _, tr, tc = move  # Extract target cell coordinates
        score = 0
        # Check all four adjacent cells around the target position
        for check_dir in directions:
            check_dr, check_dc = check_dir[1]
            x, y = tr + check_dr, tc + check_dc
            if 0 <= x < 5 and 0 <= y < 6 and opponent[x][y] == 1:
                score += 1
        # Update best_move if current move has higher score or is first in case of tie
        if score > best_score or (score == best_score and move < best_move):
            best_score = score
            best_move = move
    
    # Extract and format the best move
    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

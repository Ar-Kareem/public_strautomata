def policy(you: list[int], opponent: list[int]) -> str:
    # Convert the linear list representation into a 2D grid
    board = [you[i*6:(i+1)*6] for i in range(5)]
    opp_board = [opponent[i*6:(i+1)*6] for i in range(5)]
    
    directions = {
        'U': (-1, 0),
        'R': (0, 1),
        'D': (1, 0),
        'L': (0, -1)
    }
    
    moves = []
    
    # Gather all possible moves
    for row in range(5):
        for col in range(6):
            if board[row][col] == 1:  # If we have a piece here

                    nrow, ncol = row + dr, col + dc
                    # Check the bounds and if the opponent has a piece
                    if 0 <= nrow < 5 and 0 <= ncol < 6 and opp_board[nrow][ncol] == 1:
                        moves.append((row, col, dir))
    
    if moves:
        # From possible moves, choose one. This could be enhanced with heuristics.
        # For now, we just take the first available move (could be randomized for variation).
        chosen_move = moves[0]

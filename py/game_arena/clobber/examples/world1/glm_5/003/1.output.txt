The strategy for this policy is to use an **Iterative Deepening Negamax search with Alpha-Beta Pruning**. The core of the strategy relies on a **Mobility-Based Evaluation Function**, which estimates the value of a board state by calculating the difference between the number of moves available to the current player and the number of moves available to the opponent. This captures the essence of Clobber: restricting the opponent's options while preserving one's own.

To maximize performance within the 1-second time limit:
1.  **Bitboard Representation**: The 5x6 board is represented using 30-bit integers for the player and the opponent. This allows for extremely fast move generation and board updates using bitwise operations.
2.  **Efficient Move Generation**: Legal moves are generated by identifying adjacent opponent pieces using bitwise shifts, with special handling for column boundaries to prevent wrapping.
3.  **Move Ordering**: Moves are sorted heuristically before exploration. Moves that leave the opponent with the fewest available replies (minimizing opponent mobility) are explored first. This significantly increases the likelihood of alpha-beta cut-offs.
4.  **Iterative Deepening**: The search runs at increasing depths (starting from 1) until the time limit is approached. This ensures the AI returns the best move found so far, even if the search is cut short.

import time

def policy(you, opponent):
    """
    Calculates the next move for Clobber using Iterative Deepening Negamax
    with a Mobility-based heuristic and Bitboard representation.
    """
    start_time = time.time()
    TIME_LIMIT = 0.95
    
    # --- 1. State Conversion to Bitboards ---
    # Board is 5 rows x 6 cols (30 cells).
    # We map (r, c) to bit index k = r * 6 + c.
    # Bit 0 is (0,0), Bit 29 is (4,5).
    
    my_bits = 0
    opp_bits = 0
    
    # Detect input shape (flat vs 2D) and convert
    if you and isinstance(you[0], int):
        # Flat list input
        for i in range(30):
            if you[i] == 1:
                my_bits |= (1 << i)
            elif opponent[i] == 1:
                opp_bits |= (1 << i)
    else:
        # 2D list input
        for r in range(5):
            for c in range(6):
                idx = r * 6 + c
                if you[r][c] == 1:
                    my_bits |= (1 << idx)
                elif opponent[r][c] == 1:
                    opp_bits |= (1 << idx)

    # --- 2. Precomputed Masks ---
    # Mask for Column 0 (indices 0, 6, 12, 18, 24)
    MASK_COL_0 = sum(1 << (r * 6) for r in range(5))
    # Mask for Column 5 (indices 5, 11, 17, 23, 29)
    MASK_COL_5 = sum(1 << (r * 6 + 5) for r in range(5))
    
    # Valid targets for capture from the perspective of the mover
    # Moves Right: Opponent must NOT be in Col 0 (wraps from Col 5)
    # Moves Left: Opponent must NOT be in Col 5 (wraps from Col 0)
    NOT_COL_0 = (~MASK_COL_0) & ((1 << 30) - 1)
    NOT_COL_5 = (~MASK_COL_5) & ((1 << 30) - 1)

    # --- 3. Helper Functions ---
    
    def count_moves(me, other):
        """
        Efficiently calculates the number of legal moves for 'me' against 'other'.
        Uses bitwise operations to find adjacent opponent pieces.
        """
        # Up: My piece at k, Opponent at k-6 (valid if opp exists at k-6)
        # logic: shift opp UP (<<6), check intersection with me.
        # bits of me that can capture UP: me & (opp << 6)
        # Valid because my piece at row > 0 means opp at row-1 is valid.
        count = (me & (other << 6)).bit_count()
        
        # Down: My piece at k, Opponent at k+6
        count += (me & (other >> 6)).bit_count()
        
        # Right: My piece at k, Opponent at k+1. 
        # Opponent must NOT be in Col 0 (due to wraparound).
        count += (me & ((other & NOT_COL_0) >> 1)).bit_count()
        
        # Left: My piece at k, Opponent at k-1.
        # Opponent must NOT be in Col 5.
        count += (me & ((other & NOT_COL_5) << 1)).bit_count()
        
        return count

    def generate_moves(me, other):
        """
        Generates a list of legal moves (from_idx, to_idx).
        """
        moves = []
        
        # Up: Me at k -> to k-6. Opponent at k-6.
        sources = me & (other << 6)
        m = sources
        while m:
            lsb = m & -m
            idx = (lsb.bit_length() - 1)
            moves.append((idx, idx - 6))
            m -= lsb
            
        # Down
        sources = me & (other >> 6)
        m = sources
        while m:
            lsb = m & -m
            idx = (lsb.bit_length() - 1)
            moves.append((idx, idx + 6))
            m -= lsb
            
        # Right
        sources = me & ((other & NOT_COL_0) >> 1)
        m = sources
        while m:
            lsb = m & -m
            idx = (lsb.bit_length() - 1)
            moves.append((idx, idx + 1))
            m -= lsb
            
        # Left
        sources = me & ((other & NOT_COL_5) << 1)
        m = sources
        while m:
            lsb = m & -m
            idx = (lsb.bit_length() - 1)
            moves.append((idx, idx - 1))
            m -= lsb
            
        return moves

    # --- 4. Negamax Search ---
    
    def negamax(me, other, depth, alpha, beta):
        """
        Returns (score, best_move_tuple).
        Score is from perspective of current player 'me'.
        """
        # Terminal condition: No moves = Loss
        # We can check moves count quickly.
        # If depth is 0, we evaluate.
        
        moves = generate_moves(me, other)
        
        if not moves:
            return -10000 - depth, None # Loss penalty
            
        if depth == 0:
            # Evaluation: Mobility Difference
            my_mob = len(moves) # we already generated, might as well use it
            # Note: calculating opponent mobility is still needed
            opp_mob = count_moves(other, me)
            return my_mob - opp_mob, None

        # Move Ordering: Heuristic to try moves that limit opponent mobility first
        # This improves alpha-beta pruning efficiency.
        scored_moves = []
        for f, t in moves:
            # Apply move to test resulting mobility
            new_me = me ^ ((1 << f) | (1 << t))
            new_opp = other ^ (1 << t)
            # Opponent mobility after this move
            opp_mob_after = count_moves(new_opp, new_me)
            # We want to minimize opponent mobility, so lower score is better for sorting
            scored_moves.append((opp_mob_after, f, t))
        
        scored_moves.sort(key=lambda x: x[0]) # Sort by opponent mobility ASC
        
        best_move = scored_moves[0][1:] # Default to first best looking move
        max_eval = -float('inf')
        
        for _, f, t in scored_moves:
            # Apply Move
            # My piece moves from f to t. Opponent piece at t removed.
            new_me = me ^ ((1 << f) | (1 << t))
            new_opp = other ^ (1 << t)
            
            # Recursive call with swapped roles
            eval_val, _ = negamax(new_opp, new_me, depth - 1, -beta, -alpha)
            eval_val = -eval_val
            
            if eval_val > max_eval:
                max_eval = eval_val
                best_move = (f, t)
            
            alpha = max(alpha, eval_val)
            if alpha >= beta:
                break
                
        return max_eval, best_move

    # --- 5. Iterative Deepening Loop ---
    
    best_move_found = None
    
    # At least depth 1
    current_depth = 1
    while True:
        try:
            score, move = negamax(my_bits, opp_bits, current_depth, -float('inf'), float('inf'))
            if move:
                best_move_found = move
            
            # If we found a winning line, stop
            if score > 9000:
                break
                
            # Check time
            if time.time() - start_time > TIME_LIMIT:
                break
            
            current_depth += 1
            # Safety break for extremely deep searches (unlikely in 1s)
            if current_depth > 20:
                break
        except Exception:
            # Fallback if search fails
            break

    # Fallback if no move found (should not happen if logic is correct)
    if best_move_found is None:
        moves = generate_moves(my_bits, opp_bits)
        if moves:
            best_move_found = moves[0]
        else:
            return "0,0,U" # Should not be reached if called legally

    # --- 6. Format Output ---
    f, t = best_move_found
    r, c = f // 6, f % 6
    diff = t - f
    
    if diff == -6: d_char = 'U'
    elif diff == 6: d_char = 'D'
    elif diff == -1: d_char = 'L'
    elif diff == 1: d_char = 'R'
    else: d_char = 'U' # Fallback

    return f"{r},{c},{d_char}"

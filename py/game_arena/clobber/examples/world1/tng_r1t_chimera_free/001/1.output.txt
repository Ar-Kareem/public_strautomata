

The strategy is to select a legal move that captures an opponent's piece which has the fewest adjacent opponent pieces. This aims to limit the opponent's future moves by removing less connected pieces first. We iterate through each cell, check all possible moves, evaluate them based on the opponent's piece connectivity, and choose the first optimal move in row-major order.<code>
def policy(you: list[int], opponent: list[int]) -> str:
    legal_moves = []
    dirs = [('U', -1, 0), ('R', 0, 1), ('D', 1, 0), ('L', 0, -1)]
    for row in range(5):
        for col in range(6):
            if you[row][col]:
                for d, dr, dc in dirs:
                    new_row = row + dr
                    new_col = col + dc
                    if 0 <= new_row < 5 and 0 <= new_col < 6 and opponent[new_row][new_col]:
                        legal_moves.append((row, col, d, new_row, new_col))
    
    # Precompute scores for each move based on opponent's piece connectivity
    def get_score(dest_r, dest_c):
        count = 0
        for dr, dc in [(-1, 0), (1, 0), (0, -1), (0, 1)]:
            nr, nc = dest_r + dr, dest_c + dc
            if 0 <= nr < 5 and 0 <= nc < 6 and opponent[nr][nc]:
                count += 1
        return count
    
    scored_moves = []
    for move in legal_moves:
        score = get_score(move[3], move[4])
        scored_moves.append((score, move))
    
    # Select the move with the lowest score, earliest in the list in case of ties
    min_score = min(scored_moves, key=lambda x: x[0])[0]
    candidates = [m for score, m in scored_moves if score == min_score]
    best_move = candidates[0]
    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

The strategy is to minimize the opponent's next move options by choosing the move that leaves them with the fewest possible legal moves. For each possible move, we simulate the board state after the move and count how many moves the opponent can make in response. We select the move that results in the smallest number of opponent moves, prioritizing moves that lead to immediate wins (zero moves for opponent). Code utilizes nested helper functions for clarity and efficiency.

def policy(you: list[int], opponent: list[int]) -> str:
    def direction_to_offset(dir: str) -> tuple[int, int]:
        if dir == 'U':
            return (-1, 0)
        elif dir == 'D':
            return (1, 0)
        elif dir == 'L':
            return (0, -1)
        elif dir == 'R':
            return (0, 1)
        else:
            raise ValueError(f"Invalid direction: {dir}")

    def get_legal_moves(player: list[int], other: list[int]) -> list[tuple[int, int, str]]:
        moves = []
        rows = len(player)
        cols = len(player[0])
        for r in range(rows):
            for c in range(cols):
                if player[r][c] == 1:
                    if r > 0 and other[r-1][c] == 1:
                        moves.append((r, c, 'U'))
                    if r < rows-1 and other[r+1][c] == 1:
                        moves.append((r, c, 'D'))
                    if c > 0 and other[r][c-1] == 1:
                        moves.append((r, c, 'L'))
                    if c < cols-1 and other[r][c+1] == 1:
                        moves.append((r, c, 'R'))
        return moves

    # Generate current legal moves
    current_moves = get_legal_moves(you, opponent)
    best_move = None
    min_opponent_moves = float('inf')

    for move in current_moves:
        r, c, dir = move
        dr, dc = direction_to_offset(dir)
        new_r = r + dr
        new_c = c + dc

        # Create board copies
        new_you = [row.copy() for row in you]
        new_opponent = [row.copy() for row in opponent]
        
        # Execute move
        new_you[r][c] = 0
        new_you[new_r][new_c] = 1
        new_opponent[new_r][new_c] = 0
        
        # Calculate opponent's response moves
        opp_moves = get_legal_moves(new_opponent, new_you)
        move_count = len(opp_moves)

        if move_count < min_opponent_moves:
            min_opponent_moves = move_count
            best_move = move
            # Early exit if found perfect move
            if move_count == 0:
                break

    # Format result
    return f"{best_move[0]},{best_move[1]},{best_move[2]}"

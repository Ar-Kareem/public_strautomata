The strategy for this Clobber AI is to use a Greedy Minimax approach with a simple heuristic. Since the game of Clobber is finite and the board is small (5x6), the number of moves decreases as the game progresses. The evaluation function focuses on the "mobility" difference (the number of available moves for me versus the number of available moves for the opponent). By maximizing my future options while minimizing the opponent's, the policy forces the opponent into a state with no legal moves. To ensure efficiency within once second, it performs a 1-ply search to prioritize moves that leave the opponent with the fewest responses.

import numpy as np

def policy(you: list[list[int]], opponent: list[list[int]]) -> str:
    """
    Implements a move-selection policy for Clobber on a 5x6 board.
    Returns move as 'row,col,dir'.
    """
    rows = 5
    cols = 6
    directions = [(-1, 0, 'U'), (1, 0, 'D'), (0, -1, 'L'), (0, 1, 'R')]
    
    my_pos = np.array(you)
    opp_pos = np.array(opponent)

    def get_legal_moves(player_board, enemy_board):
        moves = []
        for r in range(rows):
            for c in range(cols):
                if player_board[r, c] == 1:
                    for dr, dc, d_str in directions:
                        nr, nc = r + dr, c + dc
                        if 0 <= nr < rows and 0 <= nc < cols:
                            if enemy_board[nr, nc] == 1:
                                moves.append((r, c, d_str, nr, nc))
        return moves

    legal_moves = get_legal_moves(my_pos, opp_pos)
    
    # If for some reason we are called with no moves, though the prompt says we won't be:
    if not legal_moves:
        return "0,0,U"

    best_move = None
    min_opp_responses = float('inf')

    for r, c, d_str, nr, nc in legal_moves:
        # Simulate move
        # My piece moves from (r, c) to (nr, nc)
        new_my_pos = my_pos.copy()
        new_opp_pos = opp_pos.copy()
        
        new_my_pos[r, c] = 0
        new_my_pos[nr, nc] = 1
        new_opp_pos[nr, nc] = 0
        
        # Count opponent's legal moves in the next state
        # In the next state, the 'opponent' becomes the 'player'
        opp_moves = get_legal_moves(new_opp_pos, new_my_pos)
        num_opp_moves = len(opp_moves)
        
        # Simple heuristic: Leave opponent with minimum possible moves
        if num_opp_moves < min_opp_responses:
            min_opp_responses = num_opp_moves
            best_move = f"{r},{c},{d_str}"
            
        # Optimization: If we find a move that leaves opponent with 0 responses, take it immediately
        if num_opp_moves == 0:
            return f"{r},{c},{d_str}"

    if best_move:
        return best_move
    
    # Fallback to the first legal move
    r_start, c_start, d_char, _, _ = legal_moves[0]
    return f"{r_start},{c_start},{d_char}"

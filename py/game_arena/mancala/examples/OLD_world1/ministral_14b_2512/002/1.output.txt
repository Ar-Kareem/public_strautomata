import numpy as np
from collections import deque

def policy(you, opponent):
    # Simple heuristic-based policy with some minimax search
    # First try to capture opponent's seeds if possible
    for i in range(6):
        if you[i] > 0:
            # Simulate the move
            temp_you = you.copy()
            temp_opponent = opponent.copy()
            seeds = temp_you[i]
            temp_you[i] = 0

            # Distribute seeds
            pos = i + 1
            while seeds > 0:
                if pos == 6:  # Your store
                    temp_you[6] += 1
                elif pos < 6:  # Your houses

                        # Capture opportunity
                        temp_you[6] += temp_you[pos] + temp_opponent[5 - pos]
                        temp_you[pos] = 0
                        temp_opponent[5 - pos] = 0
                else:  # Opponent's houses
                    temp_opponent[pos - 7] += 1
                seeds -= 1
                pos += 1
                if pos > 12:  # Wrap around
                    pos = 0

            # Check if capture happened
            if temp_you[6] > you[6] and temp_opponent[5 - i] == 0 and temp_you[i] == 1:
                return i

    # If no immediate capture, use a simple heuristic
    # Prefer moves that leave opponent with empty houses
    best_score = -float('inf')
    best_move = -1

    for i in range(6):
        if you[i] == 0:
            continue

        # Simulate the move
        temp_you = you.copy()
        temp_opponent = opponent.copy()
        seeds = temp_you[i]
        temp_you[i] = 0

        # Distribute seeds
        pos = i + 1
        while seeds > 0:

                temp_you[6] += 1

                temp_you[pos] += 1

                    # Capture opportunity
                    temp_you[6] += temp_you[pos] + temp_opponent[5 - pos]
                    temp_you[pos] = 0

            else:  # Opponent's houses
                temp_opponent[pos - 7] += 1
            seeds -= 1
            pos += 1
            if pos > 12:  # Wrap around
                pos = 0

        # Evaluate the position
        score = evaluate_position(temp_you, temp_opponent)

        if score > best_score:
            best_score = score
            best_move = i

    return best_move

def evaluate_position(you, opponent):
    # Simple evaluation function
    score = 0

    # Your store advantage
    score += you[6] * 0.5

    # Opponent's store disadvantage
    score -= opponent[6] * 0.5

    # Prefer moves that leave opponent with empty houses
    for i in range(6):
        if opponent[i] == 0:
            score += 1
        if you[i] == 0:
            score -= 1

    # Prefer moves that create potential captures
    for i in range(6):
        if you[i] == 1 and opponent[5 - i] > 0:
            score += 2
        if opponent[i] == 1 and you[5 - i] > 0:
            score -= 2

    # Prefer moves that distribute seeds evenly
    you_variance = np.var(you[:6])
    opponent_variance = np.var(opponent[:6])
    score += (1 - you_variance) * 0.1
    score -= (1 - opponent_variance) * 0.1

    return score

This implementation:

1. First checks for immediate capture opportunities (moves that would result in capturing opponent's seeds)
2. If no immediate captures are found, uses a heuristic evaluation function to choose the best move
3. The evaluation function considers:
   - Store differences
   - Empty opponent houses (good)
   - Potential capture opportunities
   - Seed distribution variance (more even distribution is better)

The policy is designed to be computationally efficient while still making reasonably good moves. It prioritizes:
- Capturing opponent's seeds when possible
- Creating situations where you can capture in future moves
- Balancing your seed distribution
- Minimizing opponent's store advantage

The evaluation function uses simple heuristics that work well for Kalah, though more sophisticated approaches could be implemented if needed.

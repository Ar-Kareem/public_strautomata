def policy(you: list[int], opponent: list[int]) -> int:
    import copy

    def get_distribution_order(start_house: int):
        current = start_house + 1
        while True:
            if current < 6:
                yield ('you', current)
                current += 1
            elif current == 6:
                yield ('you_store',)
                current = 0
                while current < 6:
                    yield ('opponent', current)
                    current += 1
                current = 0

    def simulate_move(you_state: list[int], opp_state: list[int], move: int):
        new_you = copy.deepcopy(you_state)
        new_opp = copy.deepcopy(opp_state)
        seeds = new_you[move]
        new_you[move] = 0

        gen = get_distribution_order(move)
        last_pos = None
        for _ in range(seeds):
            pos = next(gen)
            if pos[0] == 'you':
                new_you[pos[1]] += 1
                last_pos = pos
            elif pos[0] == 'you_store':
                new_you[6] += 1
                last_pos = pos
            elif pos[0] == 'opponent':
                new_opp[pos[1]] += 1
                last_pos = pos

        extra = last_pos and last_pos[0] == 'you_store'
        captured = 0
        if last_pos and last_pos[0] == 'you' and len(last_pos) == 2:
            house = last_pos[1]
            if new_you[house] == 1:
                opp_house = 5 - house
                if new_opp[opp_house] > 0:
                    captured = new_you[house] + new_opp[opp_house]
                    new_you[6] += captured
                    new_you[house] = 0
                    new_opp[opp_house] = 0

        return new_you, new_opp, extra, captured

    best_move = -1
    best_score = -float('inf')
    original_store = you[6]

    for i in range(6):
        if you[i] == 0:
            continue

        new_you, new_opp, extra, captured = simulate_move(you, opponent, i)
        store_diff = new_you[6] - original_store
        score = extra * 10000 + captured * 100 + store_diff

        if score > best_score or (score == best_score and i < best_move):
            best_score = score
            best_move = i

    return best_move

import random

def simulate_move(you: list[int], opponent: list[int], i: int) -> tuple[list[int], list[int]]:
    # Create copies of the boards
    y = you[:]
    o = opponent[:]
    
    # Number of seeds to distribute
    seeds = y[i]
    y[i] = 0
    
    # The sowing starts from position i+1, and cycles through the board
    # Board layout in order: y[0] to y[5], y[6] (store), o[0] to o[5], then back to y[0]
    # Total 12 positions: 0-5: y[], 6: y[6], 7-11: o[0-5]
    # Skip opponent's store (would be 12)
    
    # Calculate the last position after distributing
    last_p = (i + seeds) % 12
    
    # Distribute the seeds
    for k in range(seeds):
        p = (i + 1 + k) % 12
        if p < 6:
            y[p] += 1
        elif p == 6:
            y[6] += 1
        else:  # 7-11
            o[p - 7] += 1
    
    # Determine last position for special rules
    if last_p < 6:
        last_player, last_pos = 0, last_p
    elif last_p == 6:
        last_player, last_pos = 0, 6
    else:
        last_player, last_pos = 1, last_p - 7
    
    # Capture if last seed lands in empty house on your side and opposite has seeds
    if last_player == 0 and last_pos < 6 and y[last_pos] == 1 and o[5 - last_pos] > 0:
        captured_yours = 1
        captured_opp = o[5 - last_pos]
        y[last_pos] -= captured_yours
        o[5 - last_pos] = 0
        y[6] += captured_yours + captured_opp
    
    # Return the new board states (we don't handle extra turns in simulation for simplicity)
    return y, o

def policy(you: list[int], opponent: list[int]) -> int:
    # Get possible moves
    possible_moves = [i for i in range(6) if you[i] > 0]
    
    # Simulate each possible move and compute score
    scores = []
    for i in possible_moves:
        new_y, new_o = simulate_move(you, opponent, i)
        score = new_y[6] - new_o[6]
        scores.append((score, i))
    
    # Find the best score
    if not scores:
        # Should not happen, but just in case
        return 0
    best_score = max(x[0] for x in scores)
    
    # Get all moves with the best score
    candidates = [x[1] for x in scores if x[0] == best_score]
    
    # Randomly choose one if ties
    return random.choice(candidates)

import random

def policy(you: list[int], opponent: list[int]) -> int:
    # Simulate each possible move and score them
    scores = []
    legal_moves = [i for i in range(6) if you[i] > 0]
    for i in legal_moves:
        new_you, new_opp, extra, captured = simulate_move(you, opponent, i)
        # Heuristic score: store + bonus for extra turn + captured seeds
        score = new_you[6] + 10 * extra + captured
        scores.append((score, i))
    # Choose the move with the highest score; if tie, pick randomly among max
    max_score = max(scores, key=lambda x: x[0])[0]
    best_moves = [i for s, i in scores if s == max_score]
    return random.choice(best_moves)

def simulate_move(you: list[int], opponent: list[int], i: int):
    # Make copies
    y = you[:]
    o = opponent[:]
    seeds = y[i]
    y[i] = 0
    extra_move = False
    captured = 0
    index = i + 1
    for _ in range(seeds):
        if index == 6:  # my store
            y[6] += 1
        elif index <= 5:
            y[index] += 1
        elif index >= 7:
            o[index - 7] += 1
        index = (index + 1) % 13
    # Determine last seed position
    if seeds > 0:
        last_pos = (i + seeds) % 13
        if last_pos == 6:
            extra_move = True
        elif last_pos <= 5 and y[last_pos] == 1 and o[5 - last_pos] > 0:
            captured = 1 + o[5 - last_pos]
            y[6] += captured
            y[last_pos] = 0
            o[5 - last_pos] = 0
    return y, o, extra_move, captured

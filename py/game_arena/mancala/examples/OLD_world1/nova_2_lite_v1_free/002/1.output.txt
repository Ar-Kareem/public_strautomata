def policy(you: list[int], opponent: list[int]) -> int:
    """
    Selects the best move for Kalah based on maximizing immediate store gain, captures, and extra moves.
    """
    best_score = -float('inf')
    best_move = None
    
    for i in range(6):
        if you[i] == 0:
            continue
        
        seeds = you[i]
        # Simulate distribution to determine where last seed lands and calculate store gain
        temp_you = you.copy()
        temp_opponent = opponent.copy()
        
        temp_you[i] = 0
        current = i + 1
        count = seeds
        last_house = -1
        extra_move = False
        capture = False
        
        while count > 0:
            if current == 6:  # Our store
                temp_you[6] += 1
                count -= 1
                last_house = 6
                extra_move = True
            elif current <= 5:  # Our houses
                temp_you[current] += 1
                count -= 1
                last_house = current
            elif current >= 7:
                # Map to opponent's side
                opp_index = current - 7
                if opp_index == 6:
                    current = 0  # Skip opponent's store, wrap to opponent's house 0
                    continue
                temp_opponent[opp_index] += 1
                count -= 1
                last_house = -opp_index  # Mark as opponent's house
            current += 1
        
        # Check for capture
        if 0 <= last_house <= 5 and temp_you[last_house] == 1:
            opp_index = 5 - last_house
            if temp_opponent[opp_index] > 0:
                # Capture
                temp_you[6] += temp_you[last_house] + temp_opponent[opp_index]
                temp_you[last_house] = 0
                temp_opponent[opp_index] = 0
                capture = True
        
        # Calculate score: store gain, extra move, capture, and remaining seeds in our houses
        store_gain = temp_you[6] - you[6]
        score = store_gain * 10
        if extra_move:
            score += 5
        if capture:
            score += 8
        # Penalize remaining seeds in our houses
        remaining = sum(temp_you[:6])
        score -= remaining
        
        if score > best_score:
            best_score = score
            best_move = i
    
    return best_move

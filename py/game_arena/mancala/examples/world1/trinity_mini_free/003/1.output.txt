def policy(you: list[int], opponent: list[int]) -> int:
    best_move = None
    best_score = -float('inf')
    original_you = you.copy()
    
    for i in range(6):
        if you[i] == 0:
            continue
            
        # Simulate move i
        you_new = you.copy()
        opponent_new = opponent.copy()
        you_new[i] = 0  # Remove all seeds from house i
        
        # Build distribution order
        order = []
        # Step 1: Your houses from i+1 to 5
        for j in range(i + 1, 6):
            order.append(j)
        # Step 2: Your store (index 6)
        order.append(6)
        # Step 3: Opponent's houses from 0 to 5
        for j in range(6):
            order.append(j)
        # Step 4: Your houses from 0 to i-1
        for j in range(i):
            order.append(j)
        
        # Distribute seeds
        for _ in range(original_you[i]):
            idx = order.pop(0)
            you_new[idx] += 1
        
        # Check for capture
        last_index = idx  # Last house index where a seed was placed
        if last_index < 6:  # If it's a house (0-5)
            if original_you[last_index] == 0:  # Was empty before move
                opposite_house = 5 - last_index
                if opponent_new[opposite_house] > 0:
                    you_new[6] += opponent_new[opposite_house]
                    opponent_new[opposite_house] = 0
        
        # Evaluate new state
        your_total = sum(you_new[0:6]) + you_new[6]
        opponent_total = sum(opponent_new[0:6]) + opponent_new[6]
        score = your_total - opponent_total
        
        # Update best move
        if score > best_score:
            best_score = score

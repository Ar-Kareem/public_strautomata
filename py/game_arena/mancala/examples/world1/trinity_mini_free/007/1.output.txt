

The strategy for this policy isto prioritize moves that maximize immediate seed gain while considering future opportunities. It evaluates each possible move by simulating the sowing process and calculating a heuristic score based on seed differences, capture potential, and extra moves. The move with the highest score is selected.

def simulate_move(you, opponent, move):
    new_you = you.copy()
    new_opponent = opponent.copy()
    seeds = new_you[move]
    new_you[move] = 0
    order = list(range(move + 1, 6)) + [6] + list(range(0, 6)) + list(range(0, 6))
    last_pos = None
    for _ in range(seeds):
        pos = order.pop(0)
        if pos == 6:
            continue
        if pos < 6:
            new_you[pos] += 1
        else:
            new_opponent[pos] += 1
        last_pos = pos
    if last_pos < 6 and last_pos >= 0:
        if new_opponent[5 - move] > 0:
            new_you[6] += new_opponent[5 - move]
            new_opponent[5 - move] = 0
            extra_moves = 1
        else:
            extra_moves = 0
    else:
        extra_moves = 0
        if last_pos == 6:
            extra_moves = 1
    return new_you, new_opponent, extra_moves

def policy(you, opponent):
    best_move = None
    best_eval = -float('inf')
    for move in range(6):
        if you[move] == 0:
            continue
        new_you, new_opponent, extra_moves = simulate_move(you, opponent, move)
        eval_val = (new_you[6] - new_opponent[6]) + (sum(new_you[0:6]) - sum(new_opponent[0:6])) + (extra_moves * 10)
        if eval_val > best_eval:
            best_eval = eval_val
            best_move = move
    return best_move



The strategy for this policy is to prioritize moves that grant extra turns, followed by those enabling captures for immediate gains, while also considering future move potential through seed retention bonuses. It simulates each possible move to evaluate outcomes based on these criteria, ensuring optimal decision-making.

def policy(you: list[int], opponent: list[int]) -> int:
    best_score = -float('inf')
    best_move = 0  # default to first legal move

    for move_i in range(6):
        if you[move_i] == 0:
            continue  # skip empty house

        # Simulate the move
        new_you = you.copy()
        new_opponent = opponent.copy()
        n_seeds = new_you[move_i]
        new_you[move_i] = 0

        current_side = 'you'
        pos = move_i + 1
        last_type = None  # 'you_store', 'you_house', 'opponent_house'
        last_house = None  # if type is house, this is the index

        for _ in range(n_seeds):
            if current_side == 'you':
                if pos < 6:
                    new_you[pos] += 1
                    last_type = 'you_house'
                    last_house = pos
                    pos += 1
                elif pos == 6:
                    new_you[6] += 1
                    last_type = 'you_store'
                    current_side = 'opponent'
                    pos = 0
                else:
                    current_side = 'opponent'
                    pos = 0
                    new_opponent[pos] += 1
                    last_type = 'opponent_house'
                    last_house = pos
                    pos += 1
            else:
                if pos < 6:
                    new_opponent[pos] += 1
                    last_type = 'opponent_house'
                    last_house = pos
                    pos += 1
                else:
                    current_side = 'you'
                    pos = 0
                    new_you[pos] += 1
                    last_type = 'you_house'
                    last_house = pos
                    pos += 1

        # Check for extra turn
        extra_turn = (last_type == 'you_store')

        # Check for capture
        captured = 0
        if last_type == 'you_house':
            house_i = last_house
            if new_you[house_i] == 1 and new_opponent[5 - house_i] > 0:
                captured = 1 + new_opponent[5 - house_i]
                new_you[6] += captured
                new_you[house_i] = 0
                new_opponent[5 - house_i] = 0

        # Calculate immediate gain
        immediate_gain = new_you[6] - you[6]

        # Compute score
        score = 0
        if extra_turn:
            score += 1000 + immediate_gain
        elif captured > 0:
            score += 500 + captured + immediate_gain
        else:
            score += immediate_gain

        # Heuristic for remaining seeds
        remaining_seeds = sum(new_you[:6])
        score += remaining_seeds * 0.01

        # Update best move
        if score > best_score or (score == best_score and move_i < best_move):
            best_score = score
            best_move = move_i

    return best_move

The strategy for this policy is to use a single-ply lookahead with a heuristic evaluation function. This balances speed and effectiveness for the 6-house Kalah variant.

1.  **Move Generation**: The policy iterates through all legal moves (houses 0-5 with seeds).
2.  **Simulation**: For each move, it performs a "dry run" of the game mechanics (sowing, extra turn logic, and capture logic) to determine the resulting board state and score difference.
3.  **Heuristic Evaluation**: The resulting state is evaluated using a weighted score:
    *   **Immediate Win**: Prioritized highest.
    *   **Seed Gain**: The difference in total seeds between the player and opponent.
    *   **Positional Bonus**: Seeds landing in "safe" houses (low indices) or houses adjacent to the store are favored to enable captures or extra turns.
    *   **Next Turn Control**: If the move grants an extra turn, the evaluation looks one step ahead to the opponent's best response, penalizing moves that leave the opponent with strong options or allow them to close the score gap significantly.
4.  **Selection**: The move with the highest heuristic score is chosen.

def policy(you: list[int], opponent: list[int]) -> int:
    # Deep copy helper to avoid mutating original state during lookahead
    def copy_state(y, o):
        return y[:], o[:]

    def is_game_over(y, o):
        return sum(y[:6]) == 0 or sum(o[:6]) == 0

    def get_final_score(y, o):
        # If game is over, remaining seeds go to owner
        y_score = y[6] + sum(y[:6])
        o_score = o[6] + sum(o[:6])
        return y_score, o_score

    def evaluate_state(y, o, depth=0, is_maximizing=True):
        # Basic terminal check
        if is_game_over(y, o):
            y_score, o_score = get_final_score(y, o)
            if y_score > o_score:
                return 100000 # Win
            elif y_score < o_score:
                return -100000 # Loss
            else:
                return 0 # Draw

        # Heuristic Score Calculation
        # 1. Seed Difference (Primary driver)
        my_seeds = sum(y[:6]) + y[6]
        opp_seeds = sum(o[:6]) + o[6]
        score = (my_seeds - opp_seeds) * 10

        # 2. Positional Bonus (Safe seeds and seeds near store)
        # Seeds in later houses (3,4,5) are closer to store for quick points
        # Seeds in earlier houses (0,1) are safer from opponent capture
        # We weight 0 and 1 highly, and 3,4,5 moderately
        pos_bonus = 0
        for i in range(6):
            if y[i] > 0:
                if i <= 1: pos_bonus += 2
                elif i == 2: pos_bonus += 1
                elif i >= 3: pos_bonus += 2
        score += pos_bonus

        # 3. Threat/Risk (Opponent's potential captures)
        # Penalize if we have empty houses where opponent has seeds across from them
        # (Opponent index 5-i maps to our index i)
        # If our house i is empty, and opponent house (5-i) > 0, we are vulnerable if opponent lands there
        risk = 0
        for i in range(6):
            if y[i] == 0 and o[5-i] > 0:
                risk += (o[5-i] * 2)
        score -= risk

        # 4. One-ply lookahead for root node (to prevent giving opponent immediate advantage)
        # Only do this if we are at the root (depth 0) and not in a recursive call
        if depth == 0:
            # This branch is handled by the specific move evaluation logic below
            pass

        return score

    # Main Move Selection Logic
    best_move = -1
    best_score = -float('inf')

    # Get legal moves
    legal_moves = [i for i in range(6) if you[i] > 0]

    for move in legal_moves:
        # Simulate the move
        y_temp, o_temp = copy_state(you, opponent)
        
        # 1. Sowing
        seeds = y_temp[move]
        y_temp[move] = 0
        idx = move
        last_land = -1
        
        while seeds > 0:

            # Wrap around logic
            if idx == 6: # Our store
                if seeds == 1: # Special case: if this is the last seed, it lands in store
                    # But we increment store, so logic is standard
                    pass
                y_temp[6] += 1
                seeds -= 1
                if seeds == 0:
                    last_land = 6
                    break

                idx = 0
            
            # Standard board sowing (0-5 you, 6 you store, 7-12 opp houses, 13 opp store skipped)
            # Let's map indices to array access
            if idx < 6: # You houses
                y_temp[idx] += 1
            elif idx < 13: # Opponent houses (indices 7-12 map to opp 0-5)
                o_temp[idx-7] += 1
            
            seeds -= 1
            last_land = idx

        # 2. Extra Turn Check
        is_extra_turn = (last_land == 6)
        
        # 3. Capture Check
        capture_happened = False
        if not is_extra_turn and 0 <= last_land <= 5: # Landed in our house
            if y_temp[last_land] == 1: # Was empty (now has 1)

                if o_temp[opp_idx] > 0:

            elif y_final < o_final:

                                            o_opp[13] += o_opp[house_idx] + y_opp[our_vuln_idx] # Add to opp store
                                            o_opp[house_idx] = 0
                                            y_opp[our_vuln_idx] = 0

        if current_score > best_score:

            best_move = move

    # Fallback if something goes wrong (shouldn't happen given legal_moves)
    if best_move == -1 and legal_moves:
        return legal_moves[0]
        
    return best_move

The strategy for this policy is to prioritize moves that lead to extra turns, capture opportunities, and prevent the opponent from gaining an advantageous position. We will start by implementing a basic policy that tries to maximize score. We will add improvements iteratively. First, always play a move which results in another turn, if possible. Second, capture if possible. Third, try to sow into empty pits on our side if immediately possible and able to capture. Fourth, pick the move that maximizes the number of stones in our store after one move. Fifth, avoid giving opponent good sowing positions. Sixth, explore possible scenarios a few turns in advance.

```python
def policy(you: list[int], opponent: list[int]) -> int:
    """
    Chooses the next move for Kalah based on the current board state.

    Args:
        you: A list of 7 integers representing your side of the board.
             you = [h0, h1, h2, h3, h4, h5, store]
        opponent: A list of 7 integers representing the opponent's side of the board.
                  opponent = [h0, h1, h2, h3, h4, h5, store]

    Returns:
        An integer representing the chosen house to move (0-5).
    """

    def simulate_move(house, you, opponent):
        """Simulates a move and returns the resulting board state and whether another turn is granted."""
        seeds = you[house]
        you = you[:]  # Create a copy
        opponent = opponent[:] # Create a copy
        you[house] = 0
        current_house = house + 1
        own_side = True

        while seeds > 0:
            if own_side:
                if current_house == 6:
                    you[6] += 1
                    seeds -= 1
                    if seeds == 0:
                        return you, opponent, True  # Another turn
                else:
                    you[current_house] += 1
                    seeds -= 1
            else:
                if current_house == 6:
                    current_house = 0 # skip opponent's store
                else:
                    opponent[current_house] += 1
                    seeds -= 1
            
            current_house += 1
            if own_side:
                if current_house > 6:
                    own_side = False
                    current_house = 0
            else:
                if current_house > 6:
                    own_side = True
                    current_house = 0

        # Check for capture
        if own_side and current_house > 0 and current_house <= 6:
            last_house = current_house - 1
            if you[last_house] == 1 and opponent[5 - last_house] > 0:
                you[6] += 1 + opponent[5 - last_house]
                you[last_house] = 0
                opponent[5 - last_house] = 0

        return you, opponent, False


    def get_legal_moves(board):
        """Returns a list of legal moves (house indices) for the given board."""
        legal_moves = []
        for i in range(6):
            if board[i] > 0:
                legal_moves.append(i)
        return legal_moves


    legal_moves = get_legal_moves(you)

    # 1. Check for moves that give another turn
    for move in legal_moves:
        temp_you, temp_opponent, another_turn = simulate_move(move, you, opponent)
        if another_turn:
            return move

    # 2. Check for captures
    for move in legal_moves:
      temp_you, temp_opponent, _ = simulate_move(move, you, opponent)
      if temp_you[6] > you[6]: # capture nets some points
        return move

    # 3. Prioritize sowing into empty pits to set up captures in the future

    # 4. Maximize score after one move
    best_move = -1
    best_score = -1
    for move in legal_moves:
        temp_you, temp_opponent, _ = simulate_move(move, you, opponent)
        if temp_you[6] > best_score:
            best_score = temp_you[6]
            best_move = move

    if best_move != -1:
        return best_move


    # 5. If all else fails, just return the first legal move
    return legal_moves[0]
```

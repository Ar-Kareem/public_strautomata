def policy(you: list[int], opponent: list[int]) -> int:
    # Simulate a move and return new state + extra_turn flag
    def simulate_move(you, opponent, move):
        you = you[:]
        opponent = opponent[:]
        
        seeds = you[move]
        you[move] = 0
        
        pos = move
        player_side = True  # True = you, False = opponent
        
        while seeds > 0:
            pos += 1
            
            if player_side:
                if pos <= 6:
                    you[pos] += 1
                    seeds -= 1
                    if seeds == 0:
                        # Last seed
                        if pos == 6:
                            # Extra turn
                            return you, opponent, True
                        elif pos < 6 and you[pos] == 1:
                            # Capture check
                            opp_pos = 5 - pos
                            if opponent[opp_pos] > 0:
                                you[6] += you[pos] + opponent[opp_pos]
                                you[pos] = 0
                                opponent[opp_pos] = 0
                else:
                    # Switch to opponent side
                    player_side = False
                    pos = -1
            else:
                if pos <= 5:
                    opponent[pos] += 1
                    seeds -= 1
                    if seeds == 0:
                        # Last seed on opponent side, no special effects
                        pass
                else:
                    # Skip opponent store, go back to our side
                    player_side = True
                    pos = -1
        
        return you, opponent, False
    
    def is_game_over(you, opponent):
        return sum(you[:6]) == 0 or sum(opponent[:6]) == 0
    
    def get_legal_moves(you):
        return [i for i in range(6) if you[i] > 0]

        # Simple evaluation: difference in stores
        return you[6] - opponent[6]

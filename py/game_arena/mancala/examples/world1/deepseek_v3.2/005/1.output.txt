def policy(you: list[int], opponent: list[int]) -> int:
    # Helper function to simulate a move
    def simulate_move(you_state, opp_state, house):
        """Return (new_you, new_opponent, extra_turn)"""
        you_copy = you_state[:]
        opp_copy = opp_state[:]
        seeds = you_copy[house]
        you_copy[house] = 0
        pos = house + 1
        player = 0  # 0: you, 1: opponent
        last_house = None
        last_player = None
        
        while seeds > 0:
            if player == 0:
                if pos <= 5:
                    you_copy[pos] += 1
                    last_house = pos
                    last_player = 0
                    seeds -= 1
                    pos += 1
                elif pos == 6:
                    you_copy[6] += 1
                    last_house = 6
                    last_player = 0
                    seeds -= 1
                    pos = 0
                    player = 1
                else:
                    pos = 0
                    player = 1
            else:  # player == 1 (opponent's side)
                if pos <= 5:
                    opp_copy[pos] += 1
                    last_house = pos
                    last_player = 1
                    seeds -= 1
                    pos += 1

                    # skip opponent's store
                    pos = 0
                    player = 0
                else:
                    pos = 0
                    player = 0

                    you_copy[6] += you_copy[last_house] + opp_copy[opp_house]
                    you_copy[last_house] = 0

        """Check if game is over (one side has no seeds in houses)."""

            # Player has no seeds, opponent collects all their seeds
            o[6] += sum(o[:6])
            for i in range(6):
                o[i] = 0
        elif all(s == 0 for s in o[:6]):
            # Opponent has no seeds, player collects
            p[6] += sum(p[:6])
            for i in range(6):
                p[i] = 0
        return p, o

def simulate_move(you, opponent, house):
    # Copy the states
    new_you = you.copy()
    new_opponent = opponent.copy()
    
    # Pick up seeds
    seeds = new_you[house]
    new_you[house] = 0
    
    # Distribute seeds
    pos = house
    player = 0  # 0 = you, 1 = opponent
    
    while seeds > 0:
        pos += 1
        
        if player == 0:  # Your turn
            if pos <= 5:  # Your houses
                new_you[pos] += 1
            elif pos == 6:  # Your store
                new_you[6] += 1
            else:  # pos >= 7, switch to opponent
                pos = 0
                player = 1
                new_opponent[0] += 1
        else:  # Opponent side
            if pos <= 5:  # Opponent houses
                new_opponent[pos] += 1
            elif pos == 6:  # Skip opponent store
                pos = -1  # Will become 0 on next iteration
                player = 0
                continue
        
        seeds -= 1
    
    # Check for captures and extra moves
    extra_move = False
    
    if player == 0 and pos == 6:  # Landed in your store
        extra_move = True
    elif player == 0 and 0 <= pos <= 5:  # Landed in your house

                new_you[pos] = 0
                new_opponent[opposite] = 0

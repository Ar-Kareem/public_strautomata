def policy(you: list[int], opponent: list[int]) -> int:

        you_new = you_state[:]
        opp_new = opp_state[:]
        
        if you_new[move] == 0:
            return you_new, opp_new, False, False
        
        seeds = you_new[move]
        you_new[move] = 0
        pos = move

        for _ in range(seeds):
            pos += 1

                you_new[pos] += 1

                    pos = 0
                    you_new[pos] += 1
                else:
                    opp_new[pos - 7] += 1
            else:  # Wrap around
                pos = pos % 14

                    you_new[6] += 1
                elif pos < 6:  # Our houses
                    you_new[pos] += 1
                else:  # Opponent houses (pos 7-12 maps to 0-5)

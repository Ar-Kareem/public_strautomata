def policy(you: list[int], opponent: list[int]) -> int:
    """
    Select the best move for Kalah based on immediate gains and strategic positioning.
    
    Args:
        you: Your side [h0,h1,h2,h3,h4,h5,store]

                    new_you[6] += 1
                    current_pos += 1
                    if current_pos == 7:
                        side = 'opponent'
                        current_pos = 0
                elif current_pos < 6:  # Our houses
                    new_you[current_pos] += 1
                    current_pos += 1
                else:  # Wrap around to opponent side
                    side = 'opponent'
                    current_pos = 0
                    new_opponent[current_pos] += 1
                    current_pos += 1
            else:  # opponent side
                if current_pos < 6:  # opponent houses
                    new_opponent[current_pos] += 1
                    current_pos += 1
                else:  # wrap around to our side
                    side = 'you'
                    current_pos = 0
                    new_you[current_pos] += 1
                    current_pos += 1

        extra_turn = (side == 'you' and current_pos - 1 == 6)

        capture = False
        if side == 'you' and 0 <= current_pos - 1 <= 5:

                opposite_pos = 5 - landing_pos

                    captured_seeds = 1 + new_opponent[opposite_pos]

        return new_you, new_opponent, extra_turn, capture
    
    # Evaluate all legal moves
    best_move = None

    for i in range(6):

            new_you, new_opponent, extra_turn, capture = simulate_move(you, opponent, i)

            score = 0

            seed_gain = new_you[6] - you[6]

            if extra_turn:

            # Capture is good
            if capture:

            # Consider how many seeds we're distributing (future potential)

                our_houses_total = sum(new_you[0:6])

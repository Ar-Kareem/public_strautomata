def simulate_move(move: int, seeds: int) -> int:
    """Simulate a move and return the final position (0-12) where 6 is your store."""
    # Start position after taking seeds from house 'move'
    # We'll number positions as:
    # 0-5: your houses (0-5)  
    # 6: your store
    # 7-12: opponent houses (0-5)
    # 13-18: your houses again (0-5) 
    # etc.
    
    start_pos = move + 1  # First position to place a seed
    final_pos = start_pos + seeds - 1  # Last seed placed here (0-based from your house 0)
    
    # But we need to skip opponent's store (position 19 in our mapping would be opponent store)
    # Opponent store would be at positions 6 + 1 + 6 + 6*n + 6 = 19 + 13*n ... 
    # Actually let's think differently.
    
    # Positions in cycle:
    # 0-5: your houses  
    # 6: your store
    # 7-12: opponent houses
    # 13-18: your houses again
    # 19: your store again (but we never place here as it's opponent's store in actuality)
    # 20-25: opponent houses again
    # Total cycle length = 13 (6+1+6)
    
    # The opponent's store is not in the cycle, it's skipped.
    # So positions map to:
    # 0-5 -> you[0-5] 
    # 6 -> you[6] 
    # 7-12 -> opponent[0-5] 
    # 13-18 -> you[0-5] 
    # 19 -> skipped (opponent store) 
    # 20-25 -> opponent[0-5] 
    # 26 -> skipped again
    # etc.
    
    # So we reduce modulo 13, and if result >= 7 and (result-7) % 13 >= 6, we adjust
    
    # Actually simpler:
    # Every 13 positions, we skip 1 (opponent's store)
    # So for position p, the actual position is:
    # - if p < 7: as is 
    # - if p >= 7: we subtract floor((p-7)/6) because we skip 1 every 6 opponent positions
    
    # No, even simpler. We just note that in every block of 13:
    # positions 0-5: your houses (6 positions)  
    # position 6: your store (1 position)
    # positions 7-12: opponent houses (6 positions, we skip 1)
    
    # No wait that's wrong. We don't place in opponent store.
    
    # Let's re-read:
    # Place seeds "in order" 
    # 1. you[i+1] through you[5] 
    # 2. you[6]
    # 3. opponent[0] through opponent[5] 
    # 4. you[0] etc.
    
    # So the sequence is:
    # you[i+1], you[i+2], ..., you[5], you[6], opponent[0], opponent[1], ..., opponent[5], you[0], you[1], ...
    
    # Map this to positions: 0,1,2,... 
    # Positions 0 to (5-i): you[i+1] to you[5]
    # Position (5-i)+1: you[6]  
    # Positions (5-i)+2 to (5-i)+1+6: opponent[0] to opponent[5]
    # Positions (5-i)+1+6+1 onwards: you[0] onwards
    # So position p maps to:
    # p <= 5-i: you[i+p+1] 
    # p == 5-i+1: you[6]
    # p > 5-i+1: see below

    # It's placed in position (seeds - 1) starting from position 0 being you[i+1]
    
    # Offset from you[0] is (i+1) + (seeds-1) = i + seeds 
    # But we skip opponent store
    
    total_steps = seeds 
    current_pos = move  # We start "from" this house (but don't place here)
    
    for step in range(total_steps):
        current_pos += 1
        # Normalize current_pos with wrapping and skipping opponent store  
        # Map 0-5: you houses
        # Map 6: your store 
        # Map 7-12: opponent houses
        # Map 13+: wrap around
        
        # To skip opponent store (which logically would occupy a position),
        # we note that after moving through 13 positions (6 yours + 1 store + 6 opponent),
        # the pattern repeats but we need to skip the opponent's store slot
        
        # A clearer approach: model coordinates as a list and advance
    
    # Let's hand-code the pattern:
    if seeds == 0:
        return move  # No seeds moved
    
    # Need to place 'seeds' seeds starting from house after 'move' 
    positions = []  # What actual positions get seeds
    
    start_house = move + 1
    seeds_to_place = seeds
    pos = start_house
    
    while seeds_to_place > 0:
        # pos tells us where we are:
        # 0-5: your houses
        # 6: your store  
        # 7-12: opponent houses
        # >=13: normalize
        
        normalized_pos = pos % 13
        if normalized_pos == 12:  # This would be opponent's store
            pos += 1
            continue  # Skip this position
            
        positions.append(normalized_pos)
        seeds_to_place -= 1
        pos += 1
    
    return positions[-1] if positions else move


def is_capture_move(you: list[int], opponent: list[int], move: int) -> tuple[bool, int]:
    """Check if a move results in a capture and return (is_capture, captured_seeds)."""
    seeds = you[move]
    if seeds == 0:
        return (False, 0)
        
    final_position = simulate_move(move, seeds)
    
    # Check if it's a capture:
    # 1. Final position is your house (0-5) 
    # 2. That house was empty before placing the last seed
    # 3. Opposite house has seeds
    
    if not (0 <= final_position <= 5):
        return (False, 0)
        
    # Need to determine if final house was empty before placing the last seed
    # This is tricky without full simulation, 
    # but a good approximation is that if you[final_position] was 0, it was empty
    
    # Actually, to be precise, it was empty at the moment we placed the last seed
    # If it's the final house we place a seed in, and if no previous seeds 
    # were placed there during this move, and it was 0 originally, then yes.
    
    # But we removed all seeds from move index first, so if move == final_position,
    # then that house became 0 before we started planting.
    
    # A simple approximation: if you[final_position] is currently 0 (and final_position != move),
    # or final_position == move (which means we emptied it):
    if final_position == move:
        was_empty = True  # We removed all seeds from it
    else:
        was_empty = (you[final_position] == 0)  # Was empty before our move
        
    if not was_empty:
        return (False, 0)
        
    # Check opposite house
    opposite_house = 5 - final_position
    if opponent[opposite_house] == 0:
        return (False, 0)
        
    # It's a capture
    captured = 1 + opponent[opposite_house]  # The seed we landed on + seeds in opposite house
    return (True, captured)

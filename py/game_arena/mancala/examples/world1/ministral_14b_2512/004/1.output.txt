   - Your remaining houses (from `i+1` to `5`),
   - Your store (index 6),
   - Opponent's houses (indices 0 to 5),
   - Back to your houses (indices 0 to 5), and so on.
   - The opponent's store (index 6) is skipped.
3. **Special Effects**:
   - **Extra Move**: If the last seed lands in your store (index 6), you get another move.
   - **Capture**: If the last seed lands in one of your empty houses (indices 0-5) and the opposite house on the opponent's side has seeds, you capture all seeds from both houses (the empty one and the opposite one) and move them to your store.
4. **Game End**: The game ends when one player has no seeds in all their houses (indices 0-5). The remaining seeds in the opponent's houses are moved to their store, and the player with the most seeds in their store wins.

```python
def simulate_move(you, opponent, i):
    seeds = you[i]
    you[i] = 0
    index = i
    last_seed_pos = None
    # Distribute seeds
    while seeds > 0:
        if index == 6:  # your store
            index = 0
        elif index == 7:  # opponent's store (skip)
            index = 0
        else:
            you[index] += 1
            seeds -= 1
            if seeds == 0:
                last_seed_pos = index
            index += 1
            if index > 7:  # wrap around
                index = 0
    # Check for capture
    capture = False
    if last_seed_pos != 6 and last_seed_pos != 7 and you[last_seed_pos] == 1:
        opposite_pos = 5 - last_seed_pos
        if opponent[opposite_pos] > 0:
            you[6] += you[last_seed_pos] + opponent[opposite_pos]
            you[last_seed_pos] = 0
            opponent[opposite_pos] = 0
            capture = True
    # Check for extra move
    extra_move = (last_seed_pos == 6)
    return you, opponent, capture, extra_move
```

```python
def policy(you, opponent):
    legal_moves = [i for i in range(6) if you[i] > 0]
    best_score = -float('inf')
    best_move = legal_moves[0]

    for i in legal_moves:
        # Simulate the move
        new_you = you.copy()
        new_opponent = opponent.copy()
        capture, extra_move = simulate_move(new_you, new_opponent, i)

        # Calculate score
        score = 0
        if capture:
            score += 100
        if extra_move:
            score += 50
        # Prefer moves that leave opponent with fewer seeds
        opponent_seed_count = sum(new_opponent[:6])
        score -= opponent_seed_count
        # Prefer moves that leave you with more seeds
        your_seed_count = sum(new_you[:6])
        score += your_seed_count

        if score > best_score:
            best_score = score
            best_move = i

    return best_move
```

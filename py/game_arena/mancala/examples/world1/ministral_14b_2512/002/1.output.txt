def policy(you, opponent):
    best_move = -1
    best_score = -float('inf')

    for i in range(6):
        if you[i] == 0:
            continue  # Skip empty houses

        # Simulate the move
        new_you = you.copy()
        new_opponent = opponent.copy()
        seeds = new_you[i]
        new_you[i] = 0

        # Distribute seeds
        pos = i
        for seed in range(seeds):
            pos = (pos + 1) % 14  # 14 positions: 6 you + 6 opponent + 2 stores (but skip opponent's store)
            # Map pos to actual indices
            if pos < 6:
                new_you[pos] += 1
            elif pos < 12:
                new_opponent[pos - 6] += 1
            # pos == 12 or 13: skip opponent's store (pos 12 is you[6], pos 13 is opponent[6], but we skip opponent[6])

        # Check for capture or extra move
        last_pos = (i + seeds) % 14
        if last_pos < 6:
            # Last seed in my house
            if new_you[last_pos] == 1 and new_opponent[5 - last_pos] > 0:
                # Capture
                captured = new_opponent[5 - last_pos] + 1
                new_you[6] += captured
                new_you[last_pos] = 0
                new_opponent[5 - last_pos] = 0
                score = captured * 100  # High score for capture
            else:
                score = 0
        elif last_pos == 6:
            # Last seed in my store, extra move
            score = 10  # Moderate score for extra move
        else:
            score = 0

        # Additional scoring based on board state
        # For example, prefer moves that leave opponent with fewer seeds
        opponent_total = sum(new_opponent[:6])
        score += -opponent_total  # Higher score if opponent has fewer seeds

        # Update best move
        if score > best_score:
            best_score = score
            best_move = i

    return best_move

        pos = i
        for seed in range(seeds):
            pos = (pos + 1) % 14
            if pos < 6:
                new_you[pos] += 1
            elif pos < 12:
                new_opponent[pos - 6] += 1

        # Check for capture or extra move
        last_pos = (i + seeds) % 14
        if last_pos < 6:
            if new_you[last_pos] == 1 and new_opponent[5 - last_pos] > 0:
                # Capture
                captured = new_opponent[5 - last_pos] + 1
                new_you[6] += captured
                new_you[last_pos] = 0
                new_opponent[5 - last_pos] = 0
                score = captured * 100
            else:
                score = 0
        elif last_pos == 6:
            # Extra move
            score = 10
        else:
            score = 0

        # Additional scoring: prefer moves that leave opponent with fewer seeds
        opponent_total = sum(new_opponent[:6])
        score += -opponent_total

        # Prefer moves that leave my houses more balanced
        you_variance = sum((x - (sum(new_you[:6]) / 6)) ** 2 for x in new_you[:6])
        score += -you_variance / 100  # Lower variance is better

        # Update best move
        if score > best_score:
            best_score = score
            best_move = i

    return best_move

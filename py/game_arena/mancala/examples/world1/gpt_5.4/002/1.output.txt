The strategy for this policy is a fast alpha-beta search with strong Kalah-specific move ordering and a robust fallback. It models the full game rules including extra turns, captures, and game-end sweeping. The policy prioritizes immediate tactical gains, especially extra moves and captures, while searching several plies ahead within a tight time budget. The evaluation function emphasizes store advantage, near-term material, move availability, extra-turn potential, and capture threats. If time gets tight for any reason, it still always returns a legal move.

import time
from typing import List, Tuple

INF = 10**18

# Tuned for a 1s arena budget with Python.
TIME_LIMIT = 0.92


def legal_moves(you: List[int]) -> List[int]:
    return [i for i in range(6) if you[i] > 0]


def game_over_side(side: List[int]) -> bool:
    return side[0] == 0 and side[1] == 0 and side[2] == 0 and side[3] == 0 and side[4] == 0 and side[5] == 0


def finalize_if_terminal(you: List[int], opp: List[int]) -> Tuple[List[int], List[int], bool]:
    y_empty = game_over_side(you)
    o_empty = game_over_side(opp)
    if not y_empty and not o_empty:
        return you, opp, False

    you = you[:]
    opp = opp[:]
    if y_empty:
        rem = sum(opp[:6])
        opp[6] += rem
        for i in range(6):
            opp[i] = 0
    if o_empty:
        rem = sum(you[:6])
        you[6] += rem
        for i in range(6):
            you[i] = 0
    return you, opp, True


def apply_move(you: List[int], opp: List[int], move: int) -> Tuple[List[int], List[int], bool, bool]:
    """
    Returns:
      new_you, new_opp, extra_turn, terminal
    Perspective remains from the same player ("you") if extra_turn is True.
    If extra_turn is False, boards are swapped so caller's next player-to-move is first.
    """
    you = you[:]
    opp = opp[:]
    seeds = you[move]
    you[move] = 0

    side = 0  # 0 = you side incl. store, 1 = opp side houses only
    idx = move + 1
    last_side = None
    last_idx = None

    while seeds > 0:
        if side == 0:
            while idx <= 6 and seeds > 0:
                you[idx] += 1
                seeds -= 1
                last_side = 0
                last_idx = idx
                idx += 1
            if seeds == 0:
                break
            side = 1
            idx = 0
        else:
            while idx <= 5 and seeds > 0:
                opp[idx] += 1
                seeds -= 1
                last_side = 1
                last_idx = idx
                idx += 1
            if seeds == 0:
                break
            side = 0
            idx = 0

    extra_turn = (last_side == 0 and last_idx == 6)

    # Capture
    if last_side == 0 and 0 <= last_idx <= 5 and you[last_idx] == 1 and opp[5 - last_idx] > 0:
        you[6] += you[last_idx] + opp[5 - last_idx]
        you[last_idx] = 0
        opp[5 - last_idx] = 0

    you, opp, terminal = finalize_if_terminal(you, opp)

    if terminal:
        return you, opp, False, True

    if extra_turn:
        return you, opp, True, False
    else:
        return opp, you, False, False


def count_extra_turn_moves(you: List[int]) -> int:
    c = 0
    for i in range(6):
        s = you[i]
        if s <= 0:
            continue
        dist = 6 - i
        if s % 13 == dist:
            c += 1
    return c


def immediate_capture_gain(you: List[int], opp: List[int], move: int) -> int:
    """
    Estimate capture gained directly by this move, 0 if none.
    """
    seeds = you[move]
    if seeds <= 0:
        return 0

    # Simulate landing location cheaply.
    side = 0
    idx = move + 1
    y = you[:]
    o = opp[:]
    y[move] = 0
    last_side = None
    last_idx = None

    while seeds > 0:
        if side == 0:
            while idx <= 6 and seeds > 0:
                y[idx] += 1
                seeds -= 1
                last_side = 0
                last_idx = idx
                idx += 1
            if seeds == 0:
                break
            side = 1
            idx = 0
        else:
            while idx <= 5 and seeds > 0:
                o[idx] += 1
                seeds -= 1
                last_side = 1
                last_idx = idx
                idx += 1
            if seeds == 0:
                break
            side = 0
            idx = 0

    if last_side == 0 and 0 <= last_idx <= 5 and y[last_idx] == 1 and o[5 - last_idx] > 0:
        return 1 + o[5 - last_idx]
    return 0


def evaluate(you: List[int], opp: List[int]) -> int:
    # Terminal-aware large reward.
    y, o, terminal = finalize_if_terminal(you, opp)
    if terminal:
        return (y[6] - o[6]) * 100000

    store_diff = you[6] - opp[6]
    pit_diff = sum(you[:6]) - sum(opp[:6])

    my_moves = sum(1 for x in you[:6] if x > 0)
    opp_moves = sum(1 for x in opp[:6] if x > 0)

    my_extra = count_extra_turn_moves(you)
    opp_extra = count_extra_turn_moves(opp)

    # Potential captures next move.
    my_cap = 0
    opp_cap = 0
    for i in range(6):
        if you[i] > 0:
            g = immediate_capture_gain(you, opp, i)
            if g > my_cap:
                my_cap = g
        if opp[i] > 0:
            g = immediate_capture_gain(opp, you, i)
            if g > opp_cap:
                opp_cap = g

    # Slight preference for seeds on our side and mobility, but store dominates.
    score = (
        1200 * store_diff
        + 35 * pit_diff
        + 22 * (my_moves - opp_moves)
        + 85 * (my_extra - opp_extra)
        + 55 * (my_cap - opp_cap)
    )

    # Endgame sharpening.
    total_board = sum(you[:6]) + sum(opp[:6])
    if total_board <= 18:
        score += 90 * store_diff + 20 * pit_diff

    return score


def move_priority(you: List[int], opp: List[int], move: int) -> int:
    """
    Heuristic move ordering: extra turn first, then captures, then store gain / tactical shape.
    """
    seeds = you[move]
    extra = 1 if seeds % 13 == (6 - move) else 0
    cap = immediate_capture_gain(you, opp, move)

    ny, no, ex, term = apply_move(you, opp, move)
    if term:
        terminal_gain = ny[6] - no[6]
        return 10_000_000 + terminal_gain * 1000
    val = 0
    val += extra * 1_000_000
    val += cap * 50_000
    # Since apply_move swaps on non-extra, compare from original mover's perspective.
    if ex:
        val += (ny[6] - you[6]) * 2000
        val += evaluate(ny, no)
    else:
        val += ((no[6] - opp[6]) - (ny[6] - you[6])) * 2000  # small tactical term
        val += -evaluate(ny, no)
    # Slight preference for rightmost houses in many openings/endgames.
    val += move
    return val


class Searcher:
    def __init__(self, deadline: float):
        self.deadline = deadline
        self.tt = {}

    def out_of_time(self) -> bool:
        return time.perf_counter() >= self.deadline

    def key(self, you: List[int], opp: List[int], depth: int) -> Tuple:
        return (tuple(you), tuple(opp), depth)

    def negamax(self, you: List[int], opp: List[int], depth: int, alpha: int, beta: int) -> int:
        if self.out_of_time():
            raise TimeoutError

        y, o, terminal = finalize_if_terminal(you, opp)
        if terminal:
            return (y[6] - o[6]) * 100000
        if depth <= 0:
            return evaluate(you, opp)

        k = self.key(you, opp, depth)
        tt_entry = self.tt.get(k)
        if tt_entry is not None:
            return tt_entry

        moves = legal_moves(you)
        if not moves:
            yy, oo, _ = finalize_if_terminal(you, opp)
            return (yy[6] - oo[6]) * 100000

        moves.sort(key=lambda m: move_priority(you, opp, m), reverse=True)

        best = -INF
        for m in moves:
            ny, no, extra, term = apply_move(you, opp, m)
            if term:
                val = (ny[6] - no[6]) * 100000
            elif extra:
                val = self.negamax(ny, no, depth - 1, alpha, beta)
            else:
                val = -self.negamax(ny, no, depth - 1, -beta, -alpha)

            if val > best:
                best = val
            if best > alpha:
                alpha = best
            if alpha >= beta:
                break

        self.tt[k] = best
        return best


def choose_fallback_move(you: List[int], opp: List[int]) -> int:
    moves = legal_moves(you)
    best_move = moves[0]
    best_score = -INF
    for m in moves:
        ny, no, extra, term = apply_move(you, opp, m)
        if term:
            score = (ny[6] - no[6]) * 100000
        elif extra:
            score = evaluate(ny, no) + 5000
        else:
            score = -evaluate(ny, no)
        score += move_priority(you, opp, m)
        if score > best_score:
            best_score = score
            best_move = m
    return best_move


def policy(you: List[int], opponent: List[int]) -> int:
    moves = legal_moves(you)
    if not moves:
        return 0  # should never happen by problem statement
    if len(moves) == 1:
        return moves[0]

    start = time.perf_counter()
    deadline = start + TIME_LIMIT

    # Strong legal fallback immediately.
    best_move = choose_fallback_move(you, opponent)

    searcher = Searcher(deadline)

    # Iterative deepening.
    depth = 1
    try:
        while True:
            if time.perf_counter() >= deadline:
                break

            ordered = sorted(moves, key=lambda m: move_priority(you, opponent, m), reverse=True)
            current_best_move = ordered[0]
            current_best_score = -INF

            alpha = -INF
            beta = INF

            for m in ordered:
                if time.perf_counter() >= deadline:
                    raise TimeoutError
                ny, no, extra, term = apply_move(you, opponent, m)
                if term:
                    score = (ny[6] - no[6]) * 100000
                elif extra:
                    score = searcher.negamax(ny, no, depth - 1, alpha, beta)
                else:
                    score = -searcher.negamax(ny, no, depth - 1, -beta, -alpha)

                if score > current_best_score:
                    current_best_score = score
                    current_best_move = m
                if score > alpha:
                    alpha = score

            best_move = current_best_move
            depth += 1

            # Avoid too-deep searches in huge branching positions.
            if depth > 14:
                break

    except TimeoutError:
        pass
    except Exception:
        # Must always return a legal move.
        pass

    if you[best_move] > 0:
        return best_move
    for m in moves:
        if you[m] > 0:
            return m
    return moves[0]

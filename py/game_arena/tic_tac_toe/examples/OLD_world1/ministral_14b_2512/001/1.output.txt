Here's a Python implementation for a Tic Tac Toe policy function that uses a combination of heuristic evaluation and minimax search with alpha-beta pruning for optimal play:

```python
import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Returns the best move for the current player (1) on the given board.
    Uses minimax with alpha-beta pruning for optimal play.
    """
    def evaluate(board: List[List[int]]) -> int:
        """Evaluate the board state for the current player (1)."""
        # Check rows
        for row in board:
            if row[0] == row[1] == row[2] != 0:
                return 1 if row[0] == 1 else -1

        # Check columns
        for col in range(3):
            if board[0][col] == board[1][col] == board[2][col] != 0:
                return 1 if board[0][col] == 1 else -1

        # Check diagonals
        if board[0][0] == board[1][1] == board[2][2] != 0:
            return 1 if board[0][0] == 1 else -1
        if board[0][2] == board[1][1] == board[2][0] != 0:
            return 1 if board[0][2] == 1 else -1

        return 0  # No winner yet

    def is_terminal(board: List[List[int]]) -> bool:
        """Check if the game is over."""
        return evaluate(board) != 0 or all(cell != 0 for row in board for cell in row)

    def minimax(board: List[List[int]], depth: int, alpha: int, beta: int, maximizing_player: bool) -> Tuple[int, int]:
        """Minimax algorithm with alpha-beta pruning."""
        if is_terminal(board):
            score = evaluate(board)
            return (0, 0) if score == 0 else (None, None)  # Return dummy move for terminal states

        if maximizing_player:
            best_move = None
            best_score = -float('inf')
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = 1
                        move = minimax(board, depth + 1, alpha, beta, False)
                        board[i][j] = 0
                        if move[0] is not None:  # If the opponent didn't win
                            score = -move[1] if move[1] != -float('inf') else -float('inf')
                            if score > best_score:
                                best_score = score
                                best_move = (i, j)
                            alpha = max(alpha, best_score)
                            if beta <= alpha:
                                break
                if beta <= alpha:
                    break
            return (best_move[0], best_move[1], best_score) if best_move else (None, None, -float('inf'))
        else:
            best_move = None
            best_score = float('inf')
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = -1
                        move = minimax(board, depth + 1, alpha, beta, True)
                        board[i][j] = 0
                        if move[0] is not None:  # If we didn't win
                            score = -move[1] if move[1] != float('inf') else float('inf')
                            if score < best_score:
                                best_score = score
                                best_move = (i, j)
                            beta = min(beta, best_score)
                            if beta <= alpha:
                                break
                if beta <= alpha:
                    break
            return (best_move[0], best_move[1], best_score) if best_move else (None, None, float('inf'))

    # Find the best move
    best_move = None
    best_score = -float('inf')
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                move = minimax(board, 0, -float('inf'), float('inf'), False)
                board[i][j] = 0
                if move[0] is not None and move[2] > best_score:
                    best_score = move[2]
                    best_move = (i, j)

    return best_move if best_move else (0, 0)  # Fallback to (0,0) if no move found (shouldn't happen)
```

### Explanation:
1. **Evaluation Function**: Checks for winning conditions (rows, columns, diagonals) and returns a score (1 for player win, -1 for opponent win, 0 for draw).
2. **Terminal Check**: Determines if the game is over (either a win or a full board).
3. **Minimax with Alpha-Beta Pruning**: Recursively evaluates all possible moves, pruning branches that cannot improve the current best score. This ensures optimal play by always choosing the move that maximizes the player's chance of winning.
4. **Move Selection**: The outer loop tries all possible moves, uses minimax to evaluate them, and selects the best one.

### Notes:
- The function assumes the current player is `1` (as per the problem statement).
- The minimax algorithm is guaranteed to find the optimal move in Tic Tac Toe (since the game is small and finite).
- The implementation is efficient enough to run within the 1-second constraint on modern hardware. For even faster performance, you could memoize the board states, but this is not strictly necessary for Tic Tac Toe.

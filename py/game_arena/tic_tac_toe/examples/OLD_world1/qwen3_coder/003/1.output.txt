def policy(board: list[list[int]]) -> tuple[int, int]:
    import random
    
    # Convert board to player perspective (1 is us, -1 is opponent, 0 is empty)
    # Board is already in the right format
    
    def check_win(board, player):
        # Check rows
        for i in range(3):
            if all(board[i][j] == player for j in range(3)):
                return True
        # Check columns
        for j in range(3):
            if all(board[i][j] == player for i in range(3)):
                return True
        # Check diagonals
        if all(board[i][i] == player for i in range(3)):
            return True
        if all(board[i][2-i] == player for i in range(3)):
            return True
        return False
    
    def is_board_full(board):
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    return False
        return True
    
    # Check if we can win immediately
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1  # Try placing our mark
                if check_win(board, 1):
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # Check if we need to block opponent from winning
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = -1  # Try placing opponent's mark
                if check_win(board, -1):
                    board[i][j] = 0  # Reset
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # If no immediate win or block, try to take center
    if board[1][1] == 0:
        return (1, 1)
    
    # Try to take corners
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    empty_corners = [(i, j) for i, j in corners if board[i][j] == 0]
    if empty_corners:
        return random.choice(empty_corners)
    
    # Take any remaining side
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    empty_sides = [(i, j) for i, j in sides if board[i][j] == 0]
    if empty_sides:
        return random.choice(empty_sides)
    
    # Should never reach here if board has empty spaces
    # but just in case, find any empty spot
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                return (i, j)

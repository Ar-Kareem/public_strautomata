import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Convert board to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Check for winning move
    for i in range(3):
        for j in range(3):
            if board_np[i, j] == 0:
                # Try placing our mark here
                board_np[i, j] = 1
                if check_winner(board_np) == 1:
                    return (i, j)
                # Reset the board
                board_np[i, j] = 0
    
    # Check for blocking opponent's winning move
    for i in range(3):
        for j in range(3):
            if board_np[i, j] == 0:
                # Try placing opponent's mark here
                board_np[i, j] = -1
                if check_winner(board_np) == -1:
                    return (i, j)
                # Reset the board
                board_np[i, j] = 0
    
    # If no immediate win/block, use minimax with depth limit
    _, move = minimax(board_np, 4, True, float('-inf'), float('inf'))
    return move

def check_winner(board: np.ndarray) -> int:
    # Check rows
    for row in board:
        if abs(sum(row)) == 3:
            return row[0]
    
    # Check columns
    for col in range(3):
        if abs(sum(board[:, col])) == 3:
            return board[0, col]
    
    # Check diagonals
    if abs(sum(board.diagonal())) == 3:
        return board[0, 0]
    
    if abs(sum(np.fliplr(board).diagonal())) == 3:
        return board[0, 2]
    
    return 0

def is_board_full(board: np.ndarray) -> bool:
    return not np.any(board == 0)

def minimax(board: np.ndarray, depth: int, is_maximizing: bool, alpha: float, beta: float) -> Tuple[int, Tuple[int, int]]:
    winner = check_winner(board)
    
    # Terminal conditions
    if winner == 1:  # We win
        return 10 + depth, (-1, -1)
    if winner == -1:  # Opponent wins
        return -10 - depth, (-1, -1)
    if is_board_full(board) or depth == 0:  # Draw or depth limit
        return 0, (-1, -1)
    
    best_move = (-1, -1)
    
    if is_maximizing:  # Our turn (maximize)
        max_eval = float('-inf')
        for i in range(3):
            for j in range(3):
                if board[i, j] == 0:
                    board[i, j] = 1
                    eval_score, _ = minimax(board, depth - 1, False, alpha, beta)
                    board[i, j] = 0
                    
                    if eval_score > max_eval:
                        max_eval = eval_score
                        best_move = (i, j)
                    
                    alpha = max(alpha, eval_score)
                    if beta <= alpha:
                        break
            if beta <= alpha:
                break
        return max_eval, best_move
    else:  # Opponent's turn (minimize)
        min_eval = float('inf')
        for i in range(3):
            for j in range(3):
                if board[i, j] == 0:
                    board[i, j] = -1
                    eval_score, _ = minimax(board, depth - 1, True, alpha, beta)
                    board[i, j] = 0
                    
                    if eval_score < min_eval:
                        min_eval = eval_score
                        best_move = (i, j)
                    
                    beta = min(beta, eval_score)
                    if beta <= alpha:
                        break
            if beta <= alpha:
                break
        return min_eval, best_move

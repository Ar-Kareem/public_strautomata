Here's a Python implementation of a Tic Tac Toe AI that uses the Minimax algorithm to determine the best move. The AI will always play optimally, either winning the game, forcing a draw, or blocking the opponent from winning.

```python
import numpy as np

def policy(board: list[list[int]]) -> int:
    # Convert the board to a 2D numpy array for easier manipulation
    board_np = np.array(board)
    
    # Available moves (1-9) corresponding to empty cells (0)
    available_moves = [i+1 for i in range(9) if board_np.flatten()[i] == 0]
    
    # If the board is empty (first move), choose the center for optimal play
    if len(available_moves) == 9:
        return 5
    
    # Use Minimax to find the best move
    best_score = -np.inf
    best_move = available_moves[0]  # default to first available move
    
    for move in available_moves:
        # Convert move (1-9) to row and column indices (0-2)
        row = (move - 1) // 3
        col = (move - 1) % 3
        
        # Make the move
        board_np[row, col] = 1
        
        # Calculate the score for this move
        score = minimax(board_np, 0, False)
        
        # Undo the move
        board_np[row, col] = 0
        
        # Update the best move
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

def minimax(board: np.ndarray, depth: int, is_maximizing: bool) -> int:

    result = check_winner(board)
    if result is not None:
        return result
    
    if is_maximizing:
        best_score = -np.inf
        for i in range(3):
            for j in range(3):
                if board[i, j] == 0:
                    board[i, j] = 1
                    score = minimax(board, depth + 1, False)
                    board[i, j] = 0
                    best_score = max(score, best_score)
        return best_score
    else:
        best_score = np.inf
        for i in range(3):
            for j in range(3):
                if board[i, j] == 0:
                    board[i, j] = -1
                    score = minimax(board, depth + 1, True)
                    board[i, j] = 0
                    best_score = min(score, best_score)
        return best_score

def check_winner(board: np.ndarray) -> int | None:
    # Check rows, columns, and diagonals for a win
    for i in range(3):
        if board[i, 0] == board[i, 1] == board[i, 2] != 0:
            return board[i, 0]  # 1 for AI win, -1 for opponent win
        if board[0, i] == board[1, i] == board[2, i] != 0:
            return board[0, i]
    
    # Check diagonals
    if board[0, 0] == board[1, 1] == board[2, 2] != 0:
        return board[0, 0]
    if board[0, 2] == board[1, 1] == board[2, 0] != 0:
        return board[0, 2]
    
    # Check for a draw (no empty cells left)
    if np.all(board != 0):
        return 0
    
    # Game not over yet
    return None
```

### Explanation:
1. **policy(board)**: This is the main function that decides the next move. It converts the board to a numpy array, checks available moves, and uses the Minimax algorithm to determine the best move.
2. **Minimax Algorithm**: This recursive algorithm simulates all possible moves and assumes the opponent plays optimally. It returns a score for each possible move:
   - `1` if the AI can win,
   - `-1` if the opponent can win,
   - `0` for a draw.
3. **check_winner(board)**: Utility function to check if the current board state is a win, loss, or draw.

The AI will always play optimally, making the best possible move to win or force a draw. If you want to test it, you can call `policy(board)` with the current state of the board.

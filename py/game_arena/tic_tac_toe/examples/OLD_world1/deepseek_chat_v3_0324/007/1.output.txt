Here's a Python implementation of a Tic Tac Toe AI that uses the minimax algorithm with alpha-beta pruning to determine the optimal move. The AI will always play optimally, either winning or forcing a draw:

```python
import numpy as np

def check_winner(board):
    """Check if there's a winner and return the winner (1 or -1) if found, else 0."""
    # Check rows and columns
    for i in range(3):
        if board[i][0] == board[i][1] == board[i][2] != 0:
            return board[i][0]
        if board[0][i] == board[1][i] == board[2][i] != 0:
            return board[0][i]
    
    # Check diagonals
    if board[0][0] == board[1][1] == board[2][2] != 0:
        return board[0][0]
    if board[0][2] == board[1][1] == board[2][0] != 0:
        return board[0][2]
    
    return 0

def is_board_full(board):
    """Check if the board is full."""
    return all(all(cell != 0 for cell in row) for row in board)

def minimax(board, depth, is_maximizing, alpha, beta):
    """Minimax algorithm with alpha-beta pruning."""
    winner = check_winner(board)
    if winner == 1:
        return 10 - depth
    if winner == -1:
        return depth - 10
    if is_board_full(board):
        return 0
    
    if is_maximizing:
        best_score = -float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = 1
                    score = minimax(board, depth + 1, False, alpha, beta)
                    board[i][j] = 0
                    best_score = max(score, best_score)
                    alpha = max(alpha, best_score)
                    if beta <= alpha:
                        break
        return best_score
    else:
        best_score = float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = -1
                    score = minimax(board, depth + 1, True, alpha, beta)
                    board[i][j] = 0
                    best_score = min(score, best_score)
                    beta = min(beta, best_score)
                    if beta <= alpha:
                        break
        return best_score

def policy(board):
    """Determine the best move for the current board state."""
    best_score = -float('inf')
    best_move = (-1, -1)
    
    # If board is empty, return center (best first move)
    if all(all(cell == 0 for cell in row) for row in board):
        return (1, 1)
    
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                score = minimax(board, 0, False, -float('inf'), float('inf'))
                board[i][j] = 0
                if score > best_score:
                    best_score = score
                    best_move = (i, j)
    
    return best_move
```

This implementation:

1. Uses the minimax algorithm with alpha-beta pruning to evaluate all possible moves
2. Always makes the optimal move (will never lose if it can force a draw)
3. Returns immediately with the center position if the board is empty (statistically best first move)
4. Evaluates each possible move by simulating future game states
5. Uses a scoring system where:
   - A win for the AI (1) scores +10
   - A win for the opponent (-1) scores -10
   - A draw scores 0
   - The score is adjusted by depth to prefer quicker wins/longer losses

The algorithm is efficient enough to run within the 1-second constraint for all possible Tic Tac Toe positions.

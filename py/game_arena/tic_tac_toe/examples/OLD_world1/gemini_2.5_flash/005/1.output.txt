import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the next move for a Tic Tac Toe game using a simple heuristic policy.

    Args:
        board: A 3x3 list of lists representing the Tic Tac Toe board.
               0: empty, 1: AI (you), -1: opponent.

    Returns:
        A tuple of two integers (row, col) indicating the chosen move.
    """

    player = 1
    opponent = -1

    # Check for a winning move
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                # Try placing a piece and see if it wins
                board[r][c] = player
                if check_win(board, player):
                    board[r][c] = 0  # Undo the move
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # Check for a blocking move (opponent win)
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                # Try placing opponent's piece and see if it wins
                board[r][c] = opponent
                if check_win(board, opponent):
                    board[r][c] = 0  # Undo the move
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # Take the center if available
    if board[1][1] == 0:
        return (1, 1)

    # Take any available corner
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    random.shuffle(corners) # Randomize corner choice
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)

    # Take any available side
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    random.shuffle(sides) # Randomize side choice
    for r, c in sides:
        if board[r][c] == 0:
            return (r, c)

    # If no strategic move, return the first available empty cell (should not happen in a valid game)
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                return (r, c)

    # Should never reach here in a valid Tic Tac Toe game
    return (-1, -1)


def check_win(board: list[list[int]], player_id: int) -> bool:
    """
    Checks if the given player has won the game.
    """
    # Check rows
    for r in range(3):
        if all(board[r][c] == player_id for c in range(3)):
            return True
    # Check columns
    for c in range(3):
        if all(board[r][c] == player_id for r in range(3)):
            return True
    # Check diagonals
    if all(board[i][i] == player_id for i in range(3)):
        return True
    if all(board[i][2 - i] == player_id for i in range(3)):
        return True
    return False

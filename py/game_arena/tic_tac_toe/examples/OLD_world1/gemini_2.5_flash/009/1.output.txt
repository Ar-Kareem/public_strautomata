import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the next move for a Tic Tac Toe game using a minimax-like approach
    with a focus on winning, blocking, and then strategic positioning.
    """

    player = 1
    opponent = -1

    # Define helper function to check for a winner
    def check_win(current_board, p):
        # Check rows
        for row in current_board:
            if all(cell == p for cell in row):
                return True
        # Check columns
        for col in range(3):
            if all(current_board[row][col] == p for row in range(3)):
                return True
        # Check diagonals
        if all(current_board[i][i] == p for i in range(3)) or \
           all(current_board[i][2 - i] == p for i in range(3)):
            return True
        return False

    # Get all available moves
    available_moves = []
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                available_moves.append((r, c))

    # 1. Win if possible
    for r, c in available_moves:
        temp_board = [row[:] for row in board]  # Create a copy
        temp_board[r][c] = player
        if check_win(temp_board, player):
            return r, c

    # 2. Block opponent's win if possible
    for r, c in available_moves:
        temp_board = [row[:] for row in board]
        temp_board[r][c] = opponent
        if check_win(temp_board, opponent):
            return r, c

    # 3. Take the center if available
    if (1, 1) in available_moves:
        return 1, 1

    # 4. Take any available corner
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    random.shuffle(corners) # Randomize corner choice
    for r, c in corners:
        if (r, c) in available_moves:
            return r, c

    # 5. Take any available side (edge)
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    random.shuffle(sides) # Randomize side choice
    for r, c in sides:
        if (r, c) in available_moves:
            return r, c

    # If somehow no winning, blocking, or strategic moves are found,
    # just pick a random available move (shouldn't happen with full logic)
    if available_moves:
        return random.choice(available_moves)

    # Should not reach here in a playable game if there are available moves
    # But as a fallback, return (0,0) if board is full, for instance.
    return 0, 0

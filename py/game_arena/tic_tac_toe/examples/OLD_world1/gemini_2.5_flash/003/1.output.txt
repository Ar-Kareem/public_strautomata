import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the best next move for a Tic-Tac-Toe game.

    Args:
        board: A 3x3 list of lists representing the Tic-Tac-Toe board.
               0: empty, 1: AI (you), -1: opponent.

    Returns:
        A tuple (row, col) indicating the chosen move.
    """
    player = 1  # AI is player 1

    # 1. Check for a winning move
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                board[r][c] = player
                if check_win(board, player):
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # 2. Block opponent's winning move
    opponent = -1
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                board[r][c] = opponent
                if check_win(board, opponent):
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # 3. Take the center if available
    if board[1][1] == 0:
        return (1, 1)

    # 4. Take a corner if available
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    random.shuffle(corners)  # Randomize corner choice
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)

    # 5. Take any empty side square
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    random.shuffle(sides)  # Randomize side choice
    for r, c in sides:
        if board[r][c] == 0:
            return (r, c)

    # If no moves found (should not happen in a valid game)
    # This part should theoretically be unreachable if the board can still have moves.
    # As a fallback, find any empty cell.
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                return (r, c)
    
    # If no empty cells and no winning moves (e.g., game already over)
    # This case should also not be reached in a typical game cycle where `policy` is called
    # when a move is expected.
    return (-1, -1) # Indicate an error or no valid moves


def check_win(board: list[list[int]], player: int) -> bool:
    """
    Checks if the given player has won the game.
    """
    # Check rows
    for r in range(3):
        if all(board[r][c] == player for c in range(3)):
            return True
    # Check columns
    for c in range(3):
        if all(board[r][c] == player for r in range(3)):
            return True
    # Check diagonals
    if all(board[i][i] == player for i in range(3)):
        return True
    if all(board[i][2 - i] == player for i in range(3)):
        return True
    return False

if __name__ == '__main__':
    # Initial empty board
    empty_board = [
        [0, 0, 0],

    # Board with opponent about to win
    win_block_board = [

        [0, 0, 0]
    ]
    print(f"Block win move: {policy(win_block_board)}") # Should pick (2,1)

    # Board where AI can win
    ai_win_board = [
        [1, -1, 0],
        [1, 0, -1],
        [0, 0, 0]
    ]
    print(f"AI win move: {policy(ai_win_board)}") # Should pick (2,0)

    # Full board (no moves, should return (-1,-1) or any empty based on last fallback)
    full_board = [
        [1, -1, 1],
        [-1, 1, -1],
        [1, -1, 0] # One empty slot, should pick (2,2)
    ]
    print(f"Full board move: {policy(full_board)}")

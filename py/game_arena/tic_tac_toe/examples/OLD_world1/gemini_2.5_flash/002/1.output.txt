import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the next move for a Tic Tac Toe game.

    Args:
        board: A 3x3 list of lists representing the Tic Tac Toe board.
               0: empty, 1: player (you), -1: opponent.

    Returns:
        A tuple of two integers (row, col) indicating the chosen move.
    """

    player = 1
    opponent = -1

    # Check for immediate win opportunities
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                # If placing a piece here wins the game, do it
                board[i][j] = player
                if check_win(board, player):
                    return (i, j)
                board[i][j] = 0  # Undo the move

    # Block opponent's win opportunities
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                # If placing an opponent's piece here wins them the game, block it
                board[i][j] = opponent
                if check_win(board, opponent):
                    return (i, j)
                board[i][j] = 0  # Undo the move

    # Prioritize center
    if board[1][1] == 0:
        return (1, 1)

    # Prioritize corners
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    random.shuffle(corners) # Randomize corner choice for less predictability
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)

    # Take any available side
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    random.shuffle(sides) # Randomize side choice
    for r, c in sides:
        if board[r][c] == 0:
            return (r, c)

    # This should theoretically not be reached in a valid game unless the board is full
    # In case of full board, return a random empty spot (should not happen)
    empty_cells = [(r, c) for r in range(3) for c in range(3) if board[r][c] == 0]
    if empty_cells:
        return random.choice(empty_cells)
    
    # If no empty cells and no win/block, this implies draw or full board
    # Returning a default or signalling a draw could be implemented, but given
    # the game context, we expect an empty cell to always be available until end.
    raise Exception("No valid moves left, board is full or error.")


def check_win(board: list[list[int]], player: int) -> bool:
    """
    Checks if the given player has won the game.
    """
    # Check rows
    for row in board:
        if all(cell == player for cell in row):
            return True
    
    # Check columns
    for col_idx in range(3):
        if all(board[row_idx][col_idx] == player for row_idx in range(3)):
            return True
            
    # Check diagonals
    if all(board[i][i] == player for i in range(3)):
        return True
    if all(board[i][2 - i] == player for i in range(3)):
        return True
            
    return False

"""
Tic‑Tack‑Toe AI – returns the best move (row, col) for player 1.
Uses a simple minimax with memoization (no recursion depth issues for 3×3).
If multiple moves are equally good, one is chosen randomly.
"""

import random
from typing import List, Tuple, Dict


def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Returns the optimal move for player 1 (you) on a 3×3 Tic‑Tack‑Toe board.

    Parameters
    ----------
    board : list[list[int]]
        3×3 board where
           0 = empty cell,
           1 = your mark,
          -1 = opponent's mark.

    Returns
    -------
    tuple[int, int]
        (row, column) coordinates (each from 0 to 2) of the chosen move.
    """
    # collect all empty cells
    empty = [(r, c) for r in range(3) for c in range(3) if board[r][c] == 0]

    if not empty:
        # board is full – should not happen in a normal game, but return (0,0) as placeholder
        return (0, 0)

    # 1. Check for an immediate win for us (if we can place a 1 and win)
    for r, c in empty:
        b_copy = [row[:] for row in board]   # shallow copy of rows, then deep copy rows
        b_copy[r][c] = 1
        if is_winner(b_copy, 1):
            return (r, c)

    # 2. Check if opponent can win on their next move; block it
    for r, c in empty:
        b_copy = [row[:] for row in board]
        b_copy[r][c] = -1
        if is_winner(b_copy, -1):
            return (r, c)

    # 3. Use minimax to find the move with the highest minimax score
    best_score = -float('inf')
    best_moves = []
    for move in empty:
        r, c = move
        b_copy = [row[:] for row in board]
        b_copy[r][c] = 1
        score = minimax(b_copy, False)  # maximizing turn is false because opponent moves next
        if score > best_score:
            best_score = score
            best_moves = [move]
        elif score == best_score:
            best_moves.append(move)

    # pick randomly among equally optimal moves
    return random.choice(best_moves)


############################################################
# Helper functions: win detection and minimax search       #
############################################################


def is_winner(board: List[List[int]], player: int) -> bool:
    """Check rows, columns, and diagonals for a winning line for the given player."""
    # rows
    for row in board:
        if all(cell == player for cell in row):
            return True
    # columns
    for col in range(3):
        if all(board[row][col] == player for row in range(3)):
            return True
    # diagonals
    if all(board[i][i] == player for i in range(3)):
        return True
    if all(board[i][2 - i] == player for i in range(3)):
        return True
    return False


def is_terminal(board: List[List[int]]) -> bool:
    """True if the board is full or someone has won."""
    if is_winner(board, 1):
        return True
    if is_winner(board, -1):
        return True
    # check for any empty cell
    for row in board:
        if 0 in row:
            return False
    return True  # board is full


def minimax(board: List[List[int]], is_maximizing: bool) -> int:
    """
    Minimax with no recursion depth limit needed for 3×3.
    Returns the optimal utility value for the current player.
    """
    if is_terminal(board):
        if is_winner(board, 1):
            return 1
        elif is_winner(board, -1):
            return -1
        else:  # draw
            return 0

    if is_maximizing:
        best_val = -float('inf')
        for r, c in [(r, c) for r in range(3) for c in range(3) if board[r][c] == 0]:
            board[r][c] = 1
            val = minimax(board, False)
            board[r][c] = 0
            best_val = max(best_val, val)
        return best_val
    else:
        best_val = float('inf')
        for r, c in [(r, c) for r in range(3) for c in range(3) if board[r][c] == 0]:
            board[r][c] = -1
            val = minimax(board, True)
            board[r][c] = 0
            best_val = min(best_val, val)
        return best_val

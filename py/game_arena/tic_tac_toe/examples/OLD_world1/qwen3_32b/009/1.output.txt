def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(bd):
        for player in [1, -1]:
            # Check rows
            for row in bd:
                if all(cell == player for cell in row):
                    return player
            # Check columns
            for col in range(3):
                if all(bd[row][col] == player for row in range(3)):
                    return player
            # Check diagonals
            if all(bd[i][i] == player for i in range(3)):
                return player
            if all(bd[i][2 - i] == player for i in range(3)):
                return player
        # Check for draw
        if all(cell != 0 for row in bd for cell in row):
            return 0
        return None

    def minimax(current_board, is_maximizing):
        winner = check_winner(current_board)
        if winner is not None:
            if winner == 1:
                return (None, 1)
            elif winner == -1:
                return (None, -1)
            else:
                return (None, 0)
        
        best_score = -float('inf') if is_maximizing else float('inf')
        best_move = None
        player = 1 if is_maximizing else -1

        for i in range(3):
            for j in range(3):
                if current_board[i][j] == 0:
                    new_board = [row.copy() for row in current_board]
                    new_board[i][j] = player
                    move, score = minimax(new_board, not is_maximizing)
                    if is_maximizing:
                        if score > best_score:
                            best_score = score
                            best_move = (i, j)
                    else:
                        if score < best_score:
                            best_score = score
                            best_move = (i, j)
        return (best_move, best_score)
    
    best_move, _ = minimax(board, True)
    return best_move

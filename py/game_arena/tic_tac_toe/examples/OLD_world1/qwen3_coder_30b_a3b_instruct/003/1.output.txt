import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Determines the next move for a 3x3 Tic Tac Toe game using a minimax algorithm
    with alpha-beta pruning for optimal play.
    
    Args:
        board: 3x3 list of lists representing the game state
               0 = empty, 1 = AI player, -1 = opponent
    
    Returns:
        Tuple of (row, column) indicating the next move (0-2, 0-2)
    """
    
    # Convert to numpy array for easier manipulation
    board = np.array(board)
    
    def get_empty_cells(board):
        """Returns list of empty positions"""
        empty = []
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    empty.append((i, j))
        return empty
    
    def check_winner(board):
        """Returns 1 if AI wins, -1 if opponent wins, 0 if no winner"""
        # Check rows
        for i in range(3):
            if board[i][0] == board[i][1] == board[i][2] != 0:
                return board[i][0]
        
        # Check columns
        for j in range(3):
            if board[0][j] == board[1][j] == board[2][j] != 0:
                return board[0][j]
        
        # Check diagonals
        if board[0][0] == board[1][1] == board[2][2] != 0:
            return board[0][0]
        if board[0][2] == board[1][1] == board[2][0] != 0:
            return board[0][2]
        
        return 0  # No winner
    
    def minimax(board, depth, is_maximizing, alpha, beta):
        """Minimax algorithm with alpha-beta pruning"""
        
        winner = check_winner(board)
        
        # Terminal conditions
        if winner == 1:  # AI wins
            return 10 - depth
        elif winner == -1:  # Opponent wins
            return depth - 10
        elif len(get_empty_cells(board)) == 0:  # Tie
            return 0
        
        if is_maximizing:  # AI's turn (maximize)
            max_eval = -float('inf')
            for i, j in get_empty_cells(board):
                board[i][j] = 1
                eval_score = minimax(board, depth + 1, False, alpha, beta)
                board[i][j] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break  # Beta cutoff
            return max_eval
        else:  # Opponent's turn (minimize)
            min_eval = float('inf')
            for i, j in get_empty_cells(board):
                board[i][j] = -1
                eval_score = minimax(board, depth + 1, True, alpha, beta)
                board[i][j] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break  # Alpha cutoff
            return min_eval
    
    # Find the best move
    best_score = -float('inf')
    best_move = None
    
    # Try all empty positions
    for i, j in get_empty_cells(board):
        board[i][j] = 1  # Make the move
        score = minimax(board, 0, False, -float('inf'), float('inf'))
        board[i][j] = 0  # Undo the move
        
        if score > best_score:
            best_score = score
            best_move = (i, j)
    
    return best_move

This implementation uses a minimax algorithm with alpha-beta pruning to determine the optimal move for an AI player in Tic Tac Toe:

1. It recursively evaluates all possible future game states
2. For each move, it determines if the game would result in a win for the AI (10 - depth), loss for the AI (depth - 10), or a tie (0)
3. It uses alpha-beta pruning to significantly reduce the number of nodes evaluated
4. The AI always tries to maximize its chances of winning while minimizing the opponent's chances
5. When multiple moves result in the same score, it will select one (typically the first one encountered)

The time complexity is significantly reduced due to alpha-beta pruning and works within the 1-second execution time constraint for a 3x3 board.

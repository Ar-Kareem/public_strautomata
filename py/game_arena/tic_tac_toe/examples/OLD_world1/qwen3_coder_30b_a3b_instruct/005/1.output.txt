def policy(board: list[list[int]]) -> tuple[int, int]:
    import numpy as np
    
    # Convert board to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Check if we can win in one move
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:  # Empty spot
                # Try placing our mark there
                board_np[i, j] = 1
                # Check if this wins
                if check_win(board_np, 1):
                    board_np[i, j] = 0  # Reset
                    return (i, j)
                board_np[i, j] = 0  # Reset
    
    # Check if opponent can win in one move and block them
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:  # Empty spot
                # Try placing opponent's mark there
                board_np[i, j] = -1
                # Check if this wins for opponent
                if check_win(board_np, -1):
                    board_np[i, j] = 0  # Reset
                    return (i, j)
                board_np[i, j] = 0  # Reset
    
    # If no immediate threat or win, try to take center
    if board[1][1] == 0:
        return (1, 1)
    
    # Try to take a corner
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)
    
    # Otherwise, take any remaining empty spot
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                return (i, j)

def check_win(board, player):
    # Check rows
    for i in range(3):
        if all(board[i, j] == player for j in range(3)):
            return True
    
    # Check columns
    for j in range(3):
        if all(board[i, j] == player for i in range(3)):
            return True
    
    # Check diagonals
    if all(board[i, i] == player for i in range(3)):
        return True
    if all(board[i, 2-i] == player for i in range(3)):
        return True
    
    return False

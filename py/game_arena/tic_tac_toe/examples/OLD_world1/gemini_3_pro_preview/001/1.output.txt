Here is the complete Python code for the Tic Tac Toe policy. It uses a Minimax algorithm to ensure optimal play.

```python
def policy(board: list[list[int]]) -> int:
    """
    Determines the optimal move for the current player (1) against an opponent (-1).
    
    Args:
        board: A 3x3 list of lists where:
               0 indicates an empty cell,
               1 indicates the AI player,
               -1 indicates the opponent.
               
    Returns:
        int: A number between 1 and 9 indicating the chosen move.
             1 2 3
             4 5 6
             7 8 9
    """
    
    # Constants
    PLAYER = 1
    OPPONENT = -1
    EMPTY = 0
    
    # Identify all valid moves as (row, col) tuples
    valid_moves = []
    for r in range(3):
        for c in range(3):
            if board[r][c] == EMPTY:
                valid_moves.append((r, c))
    
    # If the board is full, valid_moves is empty
    if not valid_moves:
        return 0

    # Helper function to check for a win
    def check_winner(b):
        # Check Rows
        for row in b:
            s = sum(row)
            if s == 3: return PLAYER
            if s == -3: return OPPONENT
            
        # Check Columns
        for c in range(3):
            s = b[0][c] + b[1][c] + b[2][c]
            if s == 3: return PLAYER
            if s == -3: return OPPONENT
            
        # Check Diagonals
        d1 = b[0][0] + b[1][1] + b[2][2]
        if d1 == 3: return PLAYER
        if d1 == -3: return OPPONENT
        
        d2 = b[0][2] + b[1][1] + b[2][0]
        if d2 == 3: return PLAYER
        if d2 == -3: return OPPONENT
        
        return None

    # --- PRIORITY 1: WIN IMMEDIATELY ---
    for r, c in valid_moves:
        board[r][c] = PLAYER
        if check_winner(board) == PLAYER:
            return r * 3 + c + 1
        board[r][c] = EMPTY # Undo move

    # --- PRIORITY 2: BLOCK OPPONENT WIN ---
    for r, c in valid_moves:
        board[r][c] = OPPONENT
        if check_winner(board) == OPPONENT:
            return r * 3 + c + 1
        board[r][c] = EMPTY # Undo move

    # --- PRIORITY 3: TAKE CENTER (Heuristic Optimization) ---
    # Taking the center is statistically the best opening/response in neutral states.
    if board[1][1] == EMPTY:
        return 5

    # --- PRIORITY 4: MINIMAX ALGORITHM ---
    # If no immediate win/loss, search game tree for optimal move.
    
    memo = {}

    def minimax(b, depth, is_maximizing):
        # Create a hashable representation of the board state
        state = tuple(tuple(row) for row in b)
        if (state, is_maximizing) in memo:
            return memo[(state, is_maximizing)]

        winner = check_winner(b)
        if winner == PLAYER:
            return 10 - depth  # Prefer winning sooner
        if winner == OPPONENT:
            return depth - 10  # Prefer losing later (if inevitable)
        
        # Check for Draw (no empty cells left)
        # Note: We check available cells by scanning b
        has_moves = False
        for r in range(3):
            for c in range(3):
                if b[r][c] == EMPTY:
                    has_moves = True
                    break
            if has_moves: break
        
        if not has_moves:
            return 0

        if is_maximizing:
            best_score = -1000
            for r in range(3):
                for c in range(3):
                    if b[r][c] == EMPTY:
                        b[r][c] = PLAYER
                        score = minimax(b, depth + 1, False)
                        b[r][c] = EMPTY
                        best_score = max(best_score, score)
            memo[(state, is_maximizing)] = best_score
            return best_score
        else:
            best_score = 1000
            for r in range(3):
                for c in range(3):
                    if b[r][c] == EMPTY:
                        b[r][c] = OPPONENT
                        score = minimax(b, depth + 1, True)
                        b[r][c] = EMPTY
                        best_score = min(best_score, score)
            memo[(state, is_maximizing)] = best_score
            return best_score

    # Evaluate remaining moves
    best_move = valid_moves[0]
    best_val = -1000

    for r, c in valid_moves:
        board[r][c] = PLAYER
        # After we move, it's opponent's turn (minimizing step)
        move_val = minimax(board, 0, False)
        board[r][c] = EMPTY

        if move_val > best_val:
            best_val = move_val
            best_move = (r, c)

    return best_move[0] * 3 + best_move[1] + 1
```

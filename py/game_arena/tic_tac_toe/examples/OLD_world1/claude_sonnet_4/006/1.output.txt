Looking at this Tic Tac Toe problem, I need to implement a strategy that can play optimally. I'll use the minimax algorithm with alpha-beta pruning for efficiency.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(board):
        # Check rows
        for row in board:
            if row[0] == row[1] == row[2] != 0:
                return row[0]
        
        # Check columns
        for col in range(3):
            if board[0][col] == board[1][col] == board[2][col] != 0:
                return board[0][col]
        
        # Check diagonals
        if board[0][0] == board[1][1] == board[2][2] != 0:
            return board[0][0]
        if board[0][2] == board[1][1] == board[2][0] != 0:
            return board[0][2]
        
        return 0
    
    def is_terminal(board):
        if check_winner(board) != 0:
            return True
        # Check if board is full
        for row in board:
            if 0 in row:
                return False
        return True
    
    def get_empty_cells(board):
        empty = []
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    empty.append((i, j))
        return empty
    
    def minimax(board, depth, is_maximizing, alpha, beta):
        winner = check_winner(board)
        
        if winner == 1:  # AI wins
            return 10 - depth
        elif winner == -1:  # Opponent wins
            return depth - 10
        elif is_terminal(board):  # Draw
            return 0
        
        if is_maximizing:
            max_eval = float('-inf')
            for row, col in get_empty_cells(board):
                board[row][col] = 1
                eval_score = minimax(board, depth + 1, False, alpha, beta)
                board[row][col] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for row, col in get_empty_cells(board):
                board[row][col] = -1
                eval_score = minimax(board, depth + 1, True, alpha, beta)
                board[row][col] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    # Find the best move
    best_score = float('-inf')
    best_move = None
    
    for row, col in get_empty_cells(board):
        board[row][col] = 1
        score = minimax(board, 0, False, float('-inf'), float('inf'))
        board[row][col] = 0
        
        if score > best_score:
            best_score = score
            best_move = (row, col)
    
    return best_move
```

This implementation:

1. **Uses minimax with alpha-beta pruning** for optimal play while maintaining efficiency
2. **Evaluates positions** by checking for wins, losses, and draws
3. **Scores positions** with depth consideration (faster wins/slower losses preferred)
4. **Handles the endgame perfectly** by exploring all possible outcomes
5. **Runs efficiently** within the 1-second time limit due to alpha-beta pruning

The algorithm will always make the optimal move, ensuring it never loses when a draw or win is possible.

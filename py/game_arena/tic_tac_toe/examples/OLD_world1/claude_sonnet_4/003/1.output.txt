Looking at this Tic Tac Toe problem, I need to implement a strategy that can play optimally. I'll use the minimax algorithm with alpha-beta pruning for efficiency.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(board):
        # Check rows
        for row in board:
            if row[0] == row[1] == row[2] != 0:
                return row[0]
        
        # Check columns
        for col in range(3):
            if board[0][col] == board[1][col] == board[2][col] != 0:
                return board[0][col]
        
        # Check diagonals
        if board[0][0] == board[1][1] == board[2][2] != 0:
            return board[0][0]
        if board[0][2] == board[1][1] == board[2][0] != 0:
            return board[0][2]
        
        return 0
    
    def is_board_full(board):
        return all(board[i][j] != 0 for i in range(3) for j in range(3))
    
    def minimax(board, depth, is_maximizing, alpha, beta):
        winner = check_winner(board)
        
        # Terminal conditions
        if winner == 1:  # AI wins
            return 10 - depth
        elif winner == -1:  # Opponent wins
            return depth - 10
        elif is_board_full(board):  # Draw
            return 0
        
        if is_maximizing:
            max_eval = float('-inf')
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = 1
                        eval_score = minimax(board, depth + 1, False, alpha, beta)
                        board[i][j] = 0
                        max_eval = max(max_eval, eval_score)
                        alpha = max(alpha, eval_score)
                        if beta <= alpha:
                            break
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = -1
                        eval_score = minimax(board, depth + 1, True, alpha, beta)
                        board[i][j] = 0
                        min_eval = min(min_eval, eval_score)
                        beta = min(beta, eval_score)
                        if beta <= alpha:
                            break
                if beta <= alpha:
                    break
            return min_eval
    
    # Find the best move
    best_score = float('-inf')
    best_move = None
    
    # Try center first if available (good heuristic)
    if board[1][1] == 0:
        board[1][1] = 1
        score = minimax(board, 0, False, float('-inf'), float('inf'))
        board[1][1] = 0
        if score > best_score:
            best_score = score
            best_move = (1, 1)
    
    # Try corners
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    for i, j in corners:
        if board[i][j] == 0:
            board[i][j] = 1
            score = minimax(board, 0, False, float('-inf'), float('inf'))
            board[i][j] = 0
            if score > best_score:
                best_score = score
                best_move = (i, j)
    
    # Try edges
    edges = [(0, 1), (1, 0), (1, 2), (2, 1)]
    for i, j in edges:
        if board[i][j] == 0:
            board[i][j] = 1
            score = minimax(board, 0, False, float('-inf'), float('inf'))
            board[i][j] = 0
            if score > best_score:
                best_score = score
                best_move = (i, j)
    
    return best_move
```

This implementation:

1. Uses minimax algorithm with alpha-beta pruning for optimal play
2. Evaluates positions by checking for wins, losses, and draws
3. Prioritizes moves by trying center first, then corners, then edges (good Tic Tac Toe heuristics)
4. Returns the optimal move as a tuple (row, col)
5. Should run well within the 1-second time limit due to alpha-beta pruning and the small 3x3 board size

The algorithm will play perfectly and either win or draw against any opponent.

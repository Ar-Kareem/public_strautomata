Looking at this Tic Tac Toe problem, I need to implement a strategy that plays optimally. I'll use the minimax algorithm with alpha-beta pruning for efficiency.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(b):
        # Check rows, columns, and diagonals
        for i in range(3):
            if b[i][0] == b[i][1] == b[i][2] != 0:
                return b[i][0]
            if b[0][i] == b[1][i] == b[2][i] != 0:
                return b[0][i]
        
        if b[0][0] == b[1][1] == b[2][2] != 0:
            return b[0][0]
        if b[0][2] == b[1][1] == b[2][0] != 0:
            return b[0][2]
        
        return 0
    
    def is_full(b):
        return all(b[i][j] != 0 for i in range(3) for j in range(3))
    
    def minimax(b, depth, is_maximizing, alpha, beta):
        winner = check_winner(b)
        
        if winner == 1:  # AI wins
            return 10 - depth
        elif winner == -1:  # Opponent wins
            return depth - 10
        elif is_full(b):  # Tie
            return 0
        
        if is_maximizing:
            max_eval = float('-inf')
            for i in range(3):
                for j in range(3):
                    if b[i][j] == 0:
                        b[i][j] = 1
                        eval_score = minimax(b, depth + 1, False, alpha, beta)
                        b[i][j] = 0
                        max_eval = max(max_eval, eval_score)
                        alpha = max(alpha, eval_score)
                        if beta <= alpha:
                            break
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for i in range(3):
                for j in range(3):
                    if b[i][j] == 0:
                        b[i][j] = -1
                        eval_score = minimax(b, depth + 1, True, alpha, beta)
                        b[i][j] = 0
                        min_eval = min(min_eval, eval_score)
                        beta = min(beta, eval_score)
                        if beta <= alpha:
                            break
                if beta <= alpha:
                    break
            return min_eval
    
    # Find the best move
    best_score = float('-inf')
    best_move = (0, 0)
    
    # Try center first (good heuristic)
    if board[1][1] == 0:
        board[1][1] = 1
        score = minimax(board, 0, False, float('-inf'), float('inf'))
        board[1][1] = 0
        if score > best_score:
            best_score = score
            best_move = (1, 1)
    
    # Try corners next
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    for i, j in corners:
        if board[i][j] == 0:
            board[i][j] = 1
            score = minimax(board, 0, False, float('-inf'), float('inf'))
            board[i][j] = 0
            if score > best_score:
                best_score = score
                best_move = (i, j)
    
    # Try edges last
    edges = [(0, 1), (1, 0), (1, 2), (2, 1)]
    for i, j in edges:
        if board[i][j] == 0:
            board[i][j] = 1
            score = minimax(board, 0, False, float('-inf'), float('inf'))
            board[i][j] = 0
            if score > best_score:
                best_score = score
                best_move = (i, j)
    
    return best_move
```

This implementation:

1. **Uses minimax with alpha-beta pruning** for optimal play while maintaining efficiency
2. **Prioritizes moves strategically** - center first, then corners, then edges - which helps with pruning
3. **Handles all game states** - winning, losing, and draw conditions
4. **Uses depth in scoring** to prefer quicker wins and slower losses
5. **Stays within time constraints** due to alpha-beta pruning reducing the search space

The algorithm will never lose against optimal play and will win against suboptimal opponents.

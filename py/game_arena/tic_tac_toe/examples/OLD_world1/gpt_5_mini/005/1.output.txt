from typing import List, Tuple, Optional

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Return the best move (row, col) for player 1 on a 3x3 Tic Tac Toe board.
    board is a list of 3 lists, each with 3 ints: 0 empty, 1 (us), -1 (opponent).
    Uses minimax with alpha-beta pruning to choose an optimal move.
    """

    def check_winner(b: List[List[int]]) -> Optional[int]:
        # Returns 1 if player 1 wins, -1 if opponent wins, None if no winner yet
        lines = []
        # rows and columns
        for i in range(3):
            lines.append(b[i][0] + b[i][1] + b[i][2])  # row i
            lines.append(b[0][i] + b[1][i] + b[2][i])  # col i
        # diagonals
        lines.append(b[0][0] + b[1][1] + b[2][2])
        lines.append(b[0][2] + b[1][1] + b[2][0])
        for s in lines:
            if s == 3:
                return 1
            if s == -3:
                return -1
        return None

    # Count pieces to determine whose turn it is; typically it's our turn when counts equal
    count_1 = sum(cell == 1 for row in board for cell in row)
    count_m1 = sum(cell == -1 for row in board for cell in row)
    # Next to move (based on board): 1 if equal counts, otherwise -1
    # We will assume board passed is a position where it's our turn (1) if counts equal.
    next_to_move = 1 if count_1 == count_m1 else -1

    # Minimax with alpha-beta. Score from perspective of player 1.
    def minimax(b: List[List[int]], turn: int, depth: int, alpha: int, beta: int) -> Tuple[int, Optional[Tuple[int, int]]]:
        w = check_winner(b)
        if w is not None:
            # return signed score; prefer faster wins and slower losses
            if w == 1:
                return (10 - depth, None)
            else:
                return (-10 + depth, None)
        # check for draw
        empties = [(r, c) for r in range(3) for c in range(3) if b[r][c] == 0]
        if not empties:
            return (0, None)

        best_move: Optional[Tuple[int, int]] = None
        if turn == 1:
            max_eval = -999
            for (r, c) in empties:
                b[r][c] = 1
                score, _ = minimax(b, -1, depth + 1, alpha, beta)
                b[r][c] = 0
                if score > max_eval:
                    max_eval = score
                    best_move = (r, c)
                alpha = max(alpha, score)
                if beta <= alpha:
                    break
            return (max_eval, best_move)
        else:
            min_eval = 999
            for (r, c) in empties:
                b[r][c] = -1
                score, _ = minimax(b, 1, depth + 1, alpha, beta)
                b[r][c] = 0
                if score < min_eval:
                    min_eval = score
                    best_move = (r, c)
                beta = min(beta, score)
                if beta <= alpha:
                    break
            return (min_eval, best_move)

    # If it's not our turn (next_to_move != 1), we still choose a legal move for 1.
    # But minimax expects the actual next player; to produce our move we simulate from current node
    # and choose the move for player 1 when it becomes 1's turn. The simplest robust approach:
    # If it's our turn, run minimax root choosing among moves for 1.
    # If it's opponent's turn, simulate opponent optimal move(s) and then choose our best response.
    if next_to_move == 1:
        _, move = minimax(board, 1, 0, -999, 999)
        if move is None:
            # fallback: pick first empty
            for r in range(3):
                for c in range(3):
                    if board[r][c] == 0:
                        return (r, c)
        return move  # type: ignore
    else:
        # It's opponent to move; simulate opponent optimal moves and pick our best reply among them.
        # Find opponent moves that minimize our outcome, then choose our best response to those moves
        empties = [(r, c) for r in range(3) for c in range(3) if board[r][c] == 0]
        best_for_us: Optional[Tuple[int, int]] = None
        best_score_for_us = -999
        # For each possible opponent move, apply it, then choose our response by minimax assuming we move next.
        for (orow, ocol) in empties:
            board[orow][ocol] = -1
            score_after_opp_move, _ = minimax(board, 1, 1, -999, 999)
            # score_after_opp_move is the best score for us after opponent played (assuming optimal play thereafter).
            # We want to pick the move (for us) that maximizes our score, but since opponent chose their move,
            # we instead pick the initial move that would follow from the resulting position. To be safe,
            # pick the best immediate reply from the board position.
            # Find our best reply move explicitly
            _, our_reply = minimax(board, 1, 1, -999, 999)
            board[orow][ocol] = 0
            if our_reply is None:
                continue
            # Evaluate score for that reply by simulating it
            r, c = our_reply
            board[r][c] = 1
            final_score, _ = minimax(board, -1, 2, -999, 999)
            board[r][c] = 0
            if final_score > best_score_for_us:
                best_score_for_us = final_score
                best_for_us = our_reply
        if best_for_us is not None:
            return best_for_us
        # fallback
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    return (r, c)
        # Should never reach here
        return (0, 0)

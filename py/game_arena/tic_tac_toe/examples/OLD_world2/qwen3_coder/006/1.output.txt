import random
from typing import List, Tuple, Optional

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Check for winning move
    move = find_winning_move(board, 1)
    if move:
        return move
    
    # Block opponent's winning move
    move = find_winning_move(board, -1)
    if move:
        return move
    
    # Try to take center positions
    center_moves = [(1,1), (1,2), (2,1), (2,2)]
    available_centers = [move for move in center_moves if board[move[0]][move[1]] == 0]
    if available_centers:
        return random.choice(available_centers)
    
    # Try to take corners
    corner_moves = [(0,0), (0,3), (3,0), (3,3)]
    available_corners = [move for move in corner_moves if board[move[0]][move[1]] == 0]
    if available_corners:
        return random.choice(available_corners)
    
    # Take any available spot
    empty_spots = [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]
    if empty_spots:
        return random.choice(empty_spots)
    
    # Should never happen in a valid game
    raise Exception("No valid moves available")

def find_winning_move(board: List[List[int]], player: int) -> Optional[Tuple[int, int]]:
    # Check rows
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                board[i][j] = player
                if check_win(board, player):
                    board[i][j] = 0  # Reset
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # Check columns
    for j in range(4):
        for i in range(4):
            if board[i][j] == 0:
                board[i][j] = player
                if check_win(board, player):
                    board[i][j] = 0  # Reset
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # Check diagonals
    for i in range(4):
        if board[i][i] == 0:
            board[i][i] = player
            if check_win(board, player):
                board[i][i] = 0  # Reset
                return (i, i)
            board[i][i] = 0  # Reset
    
    for i in range(4):
        if board[i][3-i] == 0:
            board[i][3-i] = player
            if check_win(board, player):
                board[i][3-i] = 0  # Reset
                return (i, 3-i)
            board[i][3-i] = 0  # Reset
    
    return None

def check_win(board: List[List[int]], player: int) -> bool:
    # Check rows
    for i in range(4):
        if all(board[i][j] == player for j in range(4)):
            return True
    
    # Check columns
    for j in range(4):
        if all(board[i][j] == player for i in range(4)):
            return True
    
    # Check diagonals
    if all(board[i][i] == player for i in range(4)):
        return True
    if all(board[i][3-i] == player for i in range(4)):
        return True
    
    return False

import random
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Convert to convenient format
    me = 1
    opponent = -1
    
    # Check for winning moves
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                # Try placing our piece here
                board[i][j] = me
                if check_win(board, me):
                    board[i][j] = 0  # Reset
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # Block opponent's winning moves
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                # Try placing opponent's piece here
                board[i][j] = opponent
                if check_win(board, opponent):
                    board[i][j] = 0  # Reset
                    return (i, j)
                board[i][j] = 0  # Reset
    
    # Try to take center positions
    center_positions = [(1,1), (1,2), (2,1), (2,2)]
    empty_centers = [(i, j) for i, j in center_positions if board[i][j] == 0]
    if empty_centers:
        return random.choice(empty_centers)
    
    # Try to take corner positions
    corner_positions = [(0,0), (0,3), (3,0), (3,3)]
    empty_corners = [(i, j) for i, j in corner_positions if board[i][j] == 0]
    if empty_corners:
        return random.choice(empty_corners)
    
    # Take any available position
    empty_positions = [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]
    if empty_positions:
        return random.choice(empty_positions)
    
    # Should never reach here if board has empty spaces
    return (0, 0)

def check_win(board: List[List[int]], player: int) -> bool:
    # Check rows
    for i in range(4):
        if all(board[i][j] == player for j in range(4)):
            return True
    
    # Check columns
    for j in range(4):
        if all(board[i][j] == player for i in range(4)):
            return True
    
    # Check diagonals
    if all(board[i][i] == player for i in range(4)):
        return True
    if all(board[i][3-i] == player for i in range(4)):
        return True
    
    return False

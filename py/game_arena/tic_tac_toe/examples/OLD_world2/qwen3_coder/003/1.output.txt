import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Convert to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Check for winning move
    for i in range(4):
        for j in range(4):
            if board_np[i, j] == 0:
                # Try placing our piece
                board_np[i, j] = 1
                if check_winner(board_np, 1):
                    return (i, j)
                # Reset
                board_np[i, j] = 0
    
    # Check for blocking opponent's winning move
    for i in range(4):
        for j in range(4):
            if board_np[i, j] == 0:
                # Try placing opponent's piece
                board_np[i, j] = -1
                if check_winner(board_np, -1):
                    # Block this move
                    return (i, j)
                # Reset
                board_np[i, j] = 0
    
    # Try to take center positions
    center_positions = [(1, 1), (1, 2), (2, 1), (2, 2)]
    for pos in center_positions:
        if board_np[pos[0], pos[1]] == 0:
            return pos
    
    # Try to take corners
    corners = [(0, 0), (0, 3), (3, 0), (3, 3)]
    for corner in corners:
        if board_np[corner[0], corner[1]] == 0:
            return corner
    
    # Take any available position
    for i in range(4):
        for j in range(4):
            if board_np[i, j] == 0:
                return (i, j)
    
    # This should never happen in a valid game
    return (0, 0)

def check_winner(board: np.ndarray, player: int) -> bool:
    # Check rows
    for i in range(4):
        if np.all(board[i, :] == player):
            return True
    
    # Check columns
    for j in range(4):
        if np.all(board[:, j] == player):
            return True
    
    # Check diagonals
    if np.all(np.diag(board) == player):
        return True
    if np.all(np.diag(np.fliplr(board)) == player):
        return True
    
    return False

import random
from typing import List, Tuple, Optional

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Constants
    PLAYER = 1
    OPPONENT = -1
    EMPTY = 0
    SIZE = 4
    
    # Check for winning move
    move = find_winning_move(board, PLAYER)
    if move:
        return move
    
    # Block opponent's winning move
    move = find_winning_move(board, OPPONENT)
    if move:
        return move
    
    # Try to take center positions
    center_moves = [(1,1), (1,2), (2,1), (2,2)]
    empty_center_moves = [move for move in center_moves if board[move[0]][move[1]] == EMPTY]
    if empty_center_moves:
        return random.choice(empty_center_moves)
    
    # Try to take corner positions
    corner_moves = [(0,0), (0,3), (3,0), (3,3)]
    empty_corner_moves = [move for move in corner_moves if board[move[0]][move[1]] == EMPTY]
    if empty_corner_moves:
        return random.choice(empty_corner_moves)
    
    # Make any valid move
    empty_positions = []
    for i in range(SIZE):
        for j in range(SIZE):
            if board[i][j] == EMPTY:
                empty_positions.append((i, j))
    
    if empty_positions:
        return random.choice(empty_positions)
    
    # Should not happen in a valid game
    raise Exception("No valid moves available")

def find_winning_move(board: List[List[int]], player: int) -> Optional[Tuple[int, int]]:
    SIZE = 4
    EMPTY = 0
    
    # Check rows
    for i in range(SIZE):
        row = board[i]
        if row.count(player) == 3 and row.count(EMPTY) == 1:
            j = row.index(EMPTY)
            return (i, j)
    
    # Check columns
    for j in range(SIZE):
        col = [board[i][j] for i in range(SIZE)]
        if col.count(player) == 3 and col.count(EMPTY) == 1:
            i = col.index(EMPTY)
            return (i, j)
    
    # Check main diagonal
    diag1 = [board[i][i] for i in range(SIZE)]
    if diag1.count(player) == 3 and diag1.count(EMPTY) == 1:
        i = diag1.index(EMPTY)
        return (i, i)
    
    # Check anti-diagonal
    diag2 = [board[i][SIZE-1-i] for i in range(SIZE)]
    if diag2.count(player) == 3 and diag2.count(EMPTY) == 1:
        i = diag2.index(EMPTY)
        return (i, SIZE-1-i)
    
    return None

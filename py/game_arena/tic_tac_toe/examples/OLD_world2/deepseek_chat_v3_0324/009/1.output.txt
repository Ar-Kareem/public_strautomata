I'll create a Python implementation for a 4x4 Tic Tac Toe AI. The AI will use a minimax algorithm with alpha-beta pruning to determine the best move. For efficiency in the 1-second time constraint, I'll implement a depth-limited search.

```python
import numpy as np
from typing import Optional

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert board to numpy array for easier manipulation
    board_array = np.array(board)
    player = 1  # Our AI is always player 1
    
    # Check if we can win in one move
    winning_move = find_winning_move(board_array, player)
    if winning_move is not None:
        return winning_move
    
    # Check if we need to block opponent's winning move
    winning_move = find_winning_move(board_array, -player)
    if winning_move is not None:
        return winning_move
    
    # Check for center control if empty
    center = [(1, 1), (1, 2), (2, 1), (2, 2)]
    for row, col in center:
        if board_array[row][col] == 0:
            return row, col
    
    # If none of the above, use minimax with alpha-beta pruning
    best_score = -float('inf')
    best_move = None
    
    for row in range(4):
        for col in range(4):
            if board_array[row][col] == 0:
                board_array[row][col] = player
                score = minimax(board_array, 3, False, -float('inf'), float('inf'), player)
                board_array[row][col] = 0
                
                if score > best_score:
                    best_score = score
                    best_move = (row, col)
    
    return best_move if best_move is not None else first_available_move(board)

def find_winning_move(board: np.ndarray, player: int) -> Optional[tuple[int, int]]:
    for row in range(4):
        for col in range(4):
            if board[row][col] == 0:
                board[row][col] = player
                if check_win(board, player):
                    board[row][col] = 0
                    return row, col
                board[row][col] = 0
    return None

def check_win(board: np.ndarray, player: int) -> bool:
    # Check rows
    for row in range(4):
        if np.all(board[row, :] == player):
            return True
    
    # Check columns
    for col in range(4):
        if np.all(board[:, col] == player):
            return True
    
    # Check diagonals
    if np.all(np.diag(board) == player):
        return True
    if np.all(np.diag(np.fliplr(board)) == player):
        return True
    
    return False

def minimax(board: np.ndarray, depth: int, is_maximizing: bool, alpha: float, beta: float, player: int) -> int:
    opponent = -player
    
    if check_win(board, player):
        return 10
    if check_win(board, opponent):
        return -10
    if is_board_full(board) or depth == 0:
        return 0
    
    if is_maximizing:
        best_score = -float('inf')
        for row in range(4):
            for col in range(4):
                if board[row][col] == 0:
                    board[row][col] = player
                    score = minimax(board, depth - 1, False, alpha, beta, player)
                    board[row][col] = 0
                    best_score = max(score, best_score)
                    alpha = max(alpha, best_score)
                    if beta <= alpha:
                        break
        return best_score
    else:
        best_score = float('inf')
        for row in range(4):
            for col in range(4):
                if board[row][col] == 0:
                    board[row][col] = opponent
                    score = minimax(board, depth - 1, True, alpha, beta, player)
                    board[row][col] = 0
                    best_score = min(score, best_score)
                    beta = min(beta, best_score)
                    if beta <= alpha:
                        break
        return best_score

def is_board_full(board: np.ndarray) -> bool:
    return not np.any(board == 0)

def first_available_move(board: list[list[int]]) -> tuple[int, int]:
    for row in range(4):
        for col in range(4):
            if board[row][col] == 0:
                return row, col
    return 0, 0  # fallback, should never be reached in a non-full board
```

This implementation:

1. First checks for immediate winning moves for itself (to win immediately if possible)
2. Then checks if the opponent has an immediate winning move (to block them)
3. Prioritizes controlling the center squares (which are strategically valuable)
4. Falls back to a depth-limited minimax search with alpha-beta pruning to look ahead a few moves
5. Has a fallback to pick the first available move if needed

The minimax search is limited to depth 3 for performance reasons, ensuring it stays within the 1-second time constraint. The evaluation function is simple: +10 for wins, -10 for losses, and 0 for draws or non-terminal states.

This strategy is effective for 4x4 Tic Tac Toe while maintaining reasonable performance within the time constraints.

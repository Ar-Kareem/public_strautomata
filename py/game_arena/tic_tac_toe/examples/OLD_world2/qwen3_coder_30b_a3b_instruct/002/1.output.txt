import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Convert to numpy array for easier manipulation
    grid = np.array(board)
    
    # Check for winning moves
    for i in range(4):
        for j in range(4):
            if grid[i, j] == 0:  # Empty cell
                # Temporarily place our mark
                grid[i, j] = 1
                if check_winner(grid) == 1:
                    grid[i, j] = 0  # Restore
                    return (i, j)
                grid[i, j] = 0  # Restore
    
    # Check for blocking opponent winning moves
    for i in range(4):
        for j in range(4):
            if grid[i, j] == 0:  # Empty cell
                # Temporarily place opponent's mark
                grid[i, j] = -1
                if check_winner(grid) == -1:
                    grid[i, j] = 0  # Restore
                    return (i, j)
                grid[i, j] = 0  # Restore
    
    # Prefer center positions when possible
    center_positions = [(1,1), (1,2), (2,1), (2,2)]
    available_centers = [(i,j) for i,j in center_positions if grid[i,j] == 0]
    
    if available_centers:
        # Prioritize center positions
        for i, j in available_centers:
            # Check if this is a good center move
            if is_good_center_move(grid, i, j):
                return (i, j)
        
        # If no particularly good center move, return first available
        return available_centers[0]
    
    # Prefer corners
    corner_positions = [(0,0), (0,3), (3,0), (3,3)]
    available_corners = [(i,j) for i,j in corner_positions if grid[i,j] == 0]
    
    if available_corners:
        return available_corners[0]
    
    # Fallback: pick any remaining empty spot
    for i in range(4):
        for j in range(4):
            if grid[i, j] == 0:
                return (i, j)
    
    # Should not reach here in a valid game state
    return (0, 0)

def check_winner(grid: np.ndarray) -> int:
    # Check rows
    for row in grid:
        if np.all(row == 1):
            return 1
        if np.all(row == -1):
            return -1
    
    # Check columns
    for col in grid.T:
        if np.all(col == 1):
            return 1
        if np.all(col == -1):
            return -1
    
    # Check diagonals
    if grid[0,0] == grid[1,1] == grid[2,2] == grid[3,3] != 0:
        return grid[0,0]
    if grid[0,3] == grid[1,2] == grid[2,1] == grid[3,0] != 0:
        return grid[0,3]
    
    return 0

def is_good_center_move(grid: np.ndarray, row: int, col: int) -> bool:
    # A good center move is one that creates multiple threat opportunities or 
    # is part of potential winning lines
    grid[row, col] = 1  # Temporarily mark
    
    # Count how many potential winning lines this move creates for us
    count = 0
    
    # Check row
    row_array = grid[row, :]
    if np.sum(row_array == 1) == 3 and np.sum(row_array == 0) == 1:
        count += 1
    
    # Check column
    col_array = grid[:, col]
    if np.sum(col_array == 1) == 3 and np.sum(col_array == 0) == 1:
        count += 1
    
    # Check diagonals if applicable
    if row == col:  # Main diagonal
        diag = np.array([grid[i,i] for i in range(4)])
        if np.sum(diag == 1) == 3 and np.sum(diag == 0) == 1:
            count += 1
    
    if row + col == 3:  # Anti-diagonal
        diag = np.array([grid[i,3-i] for i in range(4)])
        if np.sum(diag == 1) == 3 and np.sum(diag == 0) == 1:
            count += 1
    
    grid[row, col] = 0  # Restore
    return count > 0

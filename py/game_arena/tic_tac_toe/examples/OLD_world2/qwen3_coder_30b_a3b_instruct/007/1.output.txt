import numpy as np
from typing import Tuple

def policy(board: list[list[int]]) -> Tuple[int, int]:
    # Convert board to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Get all empty positions
    empty_positions = np.where(board_np == 0)
    empty_positions = list(zip(empty_positions[0], empty_positions[1]))
    
    if not empty_positions:
        return (0, 0)  # Should not happen in valid game
    
    # Check for winning move
    for row, col in empty_positions:
        board_np[row, col] = 1  # Try our move
        if check_winner(board_np) == 1:
            board_np[row, col] = 0  # Restore
            return (row, col)
        board_np[row, col] = 0  # Restore
    
    # Check for blocking move
    for row, col in empty_positions:
        board_np[row, col] = -1  # Try opponent's move
        if check_winner(board_np) == -1:
            board_np[row, col] = 0  # Restore
            return (row, col)
        board_np[row, col] = 0  # Restore
    
    # Prefer center positions
    center_positions = [(1, 1), (1, 2), (2, 1), (2, 2)]
    available_center = [pos for pos in center_positions if pos in empty_positions]
    
    if available_center:
        return available_center[0]
    
    # Prefer corners
    corner_positions = [(0, 0), (0, 3), (3, 0), (3, 3)]
    available_corners = [pos for pos in corner_positions if pos in empty_positions]
    
    if available_corners:
        return available_corners[0]
    
    # Otherwise, choose a random empty position
    return empty_positions[0]

def check_winner(board: np.ndarray) -> int:
    # Check rows
    for row in board:
        if np.all(row == 1):
            return 1
        if np.all(row == -1):
            return -1
    
    # Check columns
    for col in board.T:
        if np.all(col == 1):
            return 1
        if np.all(col == -1):
            return -1
    
    # Check diagonals
    if board[0, 0] == board[1, 1] == board[2, 2] == board[3, 3] != 0:
        return board[0, 0]
    if board[0, 3] == board[1, 2] == board[2, 1] == board[3, 0] != 0:
        return board[0, 3]
    
    return 0  # No winner

import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    # Convert to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Check if we can win in one move
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:  # Empty cell
                # Temporarily place our mark
                board_np[i, j] = 1
                if check_winner(board_np) == 1:
                    board_np[i, j] = 0
                    return (i, j)
                board_np[i, j] = 0
    
    # Check if opponent can win in one move and block
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:  # Empty cell
                # Temporarily place opponent's mark
                board_np[i, j] = -1
                if check_winner(board_np) == -1:
                    board_np[i, j] = 0
                    return (i, j)
                board_np[i, j] = 0
    
    # If no immediate win/block, choose center cells if available
    center_positions = [(1,1), (1,2), (2,1), (2,2)]
    available_centers = [(i,j) for i,j in center_positions if board[i][j] == 0]
    
    if available_centers:
        # Prefer positions that lead to more winning opportunities
        best_move = available_centers[0]
        best_score = -1
        
        for i, j in available_centers:
            score = 0
            # Score based on how many potential lines we could complete
            # Count number of 1s in row, column, and diagonals
            row_count = sum(1 for val in board_np[i, :] if val == 1)
            col_count = sum(1 for val in board_np[:, j] if val == 1)
            if i == j:  # Main diagonal
                diag_count = sum(1 for k in range(4) if board_np[k,k] == 1)
                score += diag_count
            if i + j == 3:  # Anti-diagonal
                anti_diag_count = sum(1 for k in range(4) if board_np[k, 3-k] == 1)
                score += anti_diag_count
            score += row_count + col_count
            
            if score > best_score:
                best_score = score
                best_move = (i, j)
        
        return best_move
    
    # If no central positions, find the best available move
    # Prefer corners
    corner_positions = [(0,0), (0,3), (3,0), (3,3)]
    available_corners = [(i,j) for i,j in corner_positions if board[i][j] == 0]
    
    if available_corners:
        return available_corners[0]
    
    # Otherwise just pick the first available slot
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                return (i, j)

def check_winner(board: np.ndarray) -> int:
    # Check rows
    for i in range(4):
        if abs(sum(board[i, :])) == 4:
            return board[i, 0]
    
    # Check columns
    for j in range(4):
        if abs(sum(board[:, j])) == 4:
            return board[0, j]
    
    # Check main diagonal
    diag_sum = sum(board[i, i] for i in range(4))
    if abs(diag_sum) == 4:
        return board[0, 0]
    
    # Check anti-diagonal
    anti_diag_sum = sum(board[i, 3-i] for i in range(4))
    if abs(anti_diag_sum) == 4:
        return board[0, 3]
    
    return 0  # No winner

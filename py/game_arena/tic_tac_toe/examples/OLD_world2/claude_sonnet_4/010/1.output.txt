Looking at this 4x4 Tic Tac Toe problem, I need to create a strategy that can handle the larger board size effectively. I'll implement a minimax algorithm with alpha-beta pruning and include strategic heuristics.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(board):
        # Check rows
        for row in board:
            if abs(sum(row)) == 4:
                return row[0]
        
        # Check columns
        for col in range(4):
            col_sum = sum(board[row][col] for row in range(4))
            if abs(col_sum) == 4:
                return board[0][col]
        
        # Check diagonals
        main_diag = sum(board[i][i] for i in range(4))
        if abs(main_diag) == 4:
            return board[0][0]
        
        anti_diag = sum(board[i][3-i] for i in range(4))
        if abs(anti_diag) == 4:
            return board[0][3]
        
        return 0
    
    def evaluate_line(line):
        if 1 in line and -1 in line:
            return 0  # Mixed line, no threat
        
        count_1 = line.count(1)
        count_neg1 = line.count(-1)
        
        if count_1 == 4:
            return 1000
        elif count_1 == 3:
            return 50
        elif count_1 == 2:
            return 10
        elif count_1 == 1:
            return 1
        elif count_neg1 == 4:
            return -1000
        elif count_neg1 == 3:
            return -50
        elif count_neg1 == 2:
            return -10
        elif count_neg1 == 1:
            return -1
        
        return 0
    
    def evaluate(board):
        winner = check_winner(board)
        if winner == 1:
            return 1000
        elif winner == -1:
            return -1000
        
        score = 0
        
        # Evaluate rows
        for row in board:
            score += evaluate_line(row)
        
        # Evaluate columns
        for col in range(4):
            column = [board[row][col] for row in range(4)]
            score += evaluate_line(column)
        
        # Evaluate diagonals
        main_diag = [board[i][i] for i in range(4)]
        anti_diag = [board[i][3-i] for i in range(4)]
        score += evaluate_line(main_diag)
        score += evaluate_line(anti_diag)
        
        # Add positional bonus for center squares
        center_bonus = 0
        for i in [1, 2]:
            for j in [1, 2]:
                if board[i][j] == 1:
                    center_bonus += 5
                elif board[i][j] == -1:
                    center_bonus -= 5
        
        return score + center_bonus
    
    def get_moves(board):
        moves = []
        for i in range(4):
            for j in range(4):
                if board[i][j] == 0:
                    moves.append((i, j))
        return moves
    
    def minimax(board, depth, alpha, beta, maximizing_player):
        winner = check_winner(board)
        if winner != 0:
            return winner * 1000
        
        moves = get_moves(board)
        if not moves or depth == 0:
            return evaluate(board)
        
        if maximizing_player:
            max_eval = float('-inf')
            for move in moves:
                row, col = move
                board[row][col] = 1
                eval_score = minimax(board, depth - 1, alpha, beta, False)
                board[row][col] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for move in moves:
                row, col = move
                board[row][col] = -1
                eval_score = minimax(board, depth - 1, alpha, beta, True)
                board[row][col] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    # Check for immediate win
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                board[i][j] = 1
                if check_winner(board) == 1:
                    board[i][j] = 0
                    return (i, j)
                board[i][j] = 0
    
    # Check for blocking opponent's win
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                board[i][j] = -1
                if check_winner(board) == -1:
                    board[i][j] = 0
                    return (i, j)
                board[i][j] = 0
    
    # Use minimax for best move
    moves = get_moves(board)
    if not moves:
        return (0, 0)
    
    # Determine search depth based on game state
    empty_squares = len(moves)
    if empty_squares <= 8:
        depth = 6
    elif empty_squares <= 12:
        depth = 4
    else:
        depth = 3
    
    best_move = moves[0]
    best_score = float('-inf')
    
    for move in moves:
        row, col = move
        board[row][col] = 1
        score = minimax(board, depth - 1, float('-inf'), float('inf'), False)
        board[row][col] = 0
        
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move
```

This solution implements:

1. **Immediate win detection** - Always takes a winning move if available
2. **Blocking** - Prevents opponent from winning on their next turn
3. **Minimax with alpha-beta pruning** - Searches ahead to find optimal moves
4. **Strategic evaluation** - Considers line threats and positional advantages
5. **Adaptive depth** - Searches deeper as the game progresses and fewer moves remain
6. **Center preference** - Gives bonus points for controlling center squares

The algorithm balances tactical play (wins/blocks) with strategic positioning to perform well in 4x4 Tic Tac Toe.

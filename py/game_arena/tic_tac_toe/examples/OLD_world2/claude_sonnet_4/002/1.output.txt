Looking at this 4x4 Tic Tac Toe problem, I need to create a policy that can make smart moves. I'll implement a minimax algorithm with alpha-beta pruning for optimal play, combined with some heuristics for efficiency.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(board):
        # Check rows
        for row in board:
            if abs(sum(row)) == 4:
                return 1 if sum(row) > 0 else -1
        
        # Check columns
        for col in range(4):
            col_sum = sum(board[row][col] for row in range(4))
            if abs(col_sum) == 4:
                return 1 if col_sum > 0 else -1
        
        # Check diagonals
        main_diag = sum(board[i][i] for i in range(4))
        if abs(main_diag) == 4:
            return 1 if main_diag > 0 else -1
        
        anti_diag = sum(board[i][3-i] for i in range(4))
        if abs(anti_diag) == 4:
            return 1 if anti_diag > 0 else -1
        
        return 0
    
    def get_empty_cells(board):
        return [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]
    
    def evaluate_position(board):
        winner = check_winner(board)
        if winner != 0:
            return winner * 1000
        
        score = 0
        
        # Evaluate all lines (rows, columns, diagonals)
        lines = []
        # Rows
        for row in range(4):
            lines.append([board[row][col] for col in range(4)])
        # Columns
        for col in range(4):
            lines.append([board[row][col] for row in range(4)])
        # Diagonals
        lines.append([board[i][i] for i in range(4)])
        lines.append([board[i][3-i] for i in range(4)])
        
        for line in lines:
            my_count = line.count(1)
            opp_count = line.count(-1)
            empty_count = line.count(0)
            
            if opp_count == 0:  # No opponent pieces
                if my_count == 3:
                    score += 50
                elif my_count == 2:
                    score += 10
                elif my_count == 1:
                    score += 1
            elif my_count == 0:  # No my pieces
                if opp_count == 3:
                    score -= 50
                elif opp_count == 2:
                    score -= 10
                elif opp_count == 1:
                    score -= 1
        
        return score
    
    def minimax(board, depth, is_maximizing, alpha, beta):
        winner = check_winner(board)
        if winner != 0:
            return winner * 1000
        
        empty_cells = get_empty_cells(board)
        if not empty_cells or depth == 0:
            return evaluate_position(board)
        
        if is_maximizing:
            max_eval = float('-inf')
            for row, col in empty_cells:
                board[row][col] = 1
                eval_score = minimax(board, depth - 1, False, alpha, beta)
                board[row][col] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for row, col in empty_cells:
                board[row][col] = -1
                eval_score = minimax(board, depth - 1, True, alpha, beta)
                board[row][col] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    def get_best_move(board):
        empty_cells = get_empty_cells(board)
        
        if not empty_cells:
            return (0, 0)  # Should not happen
        
        # Check for immediate win
        for row, col in empty_cells:
            board[row][col] = 1
            if check_winner(board) == 1:
                board[row][col] = 0
                return (row, col)
            board[row][col] = 0
        
        # Check for immediate block
        for row, col in empty_cells:
            board[row][col] = -1
            if check_winner(board) == -1:
                board[row][col] = 0
                return (row, col)
            board[row][col] = 0
        
        # Use minimax for best move
        best_score = float('-inf')
        best_move = empty_cells[0]
        
        # Adaptive depth based on game state
        num_empty = len(empty_cells)
        if num_empty <= 6:
            depth = 8
        elif num_empty <= 10:
            depth = 6
        else:
            depth = 4
        
        for row, col in empty_cells:
            board[row][col] = 1
            score = minimax(board, depth, False, float('-inf'), float('inf'))
            board[row][col] = 0
            
            if score > best_score:
                best_score = score
                best_move = (row, col)
        
        return best_move
    
    return get_best_move(board)
```

This solution implements:

1. **Win/Loss Detection**: Checks all rows, columns, and diagonals for 4-in-a-row
2. **Minimax with Alpha-Beta Pruning**: Searches the game tree efficiently to find optimal moves
3. **Position Evaluation**: Scores positions based on potential winning lines
4. **Immediate Win/Block**: Prioritizes winning moves and blocking opponent wins
5. **Adaptive Depth**: Uses deeper search when fewer moves remain
6. **Performance Optimization**: Alpha-beta pruning ensures it runs within the 1-second limit

The algorithm will make tactically sound moves by looking ahead several moves and choosing the path that maximizes its winning chances while minimizing the opponent's opportunities.

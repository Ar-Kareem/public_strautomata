Looking at this 4x4 Tic Tac Toe problem, I need to create a strategy that can handle the increased complexity compared to 3x3. I'll implement a minimax algorithm with alpha-beta pruning and strategic heuristics.

```python
def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_winner(board):
        # Check rows
        for row in board:
            if abs(sum(row)) == 4:
                return row[0]
        
        # Check columns
        for col in range(4):
            col_sum = sum(board[row][col] for row in range(4))
            if abs(col_sum) == 4:
                return board[0][col]
        
        # Check diagonals
        diag1_sum = sum(board[i][i] for i in range(4))
        if abs(diag1_sum) == 4:
            return board[0][0]
        
        diag2_sum = sum(board[i][3-i] for i in range(4))
        if abs(diag2_sum) == 4:
            return board[0][3]
        
        return 0
    
    def is_terminal(board):
        if check_winner(board) != 0:
            return True
        return all(board[i][j] != 0 for i in range(4) for j in range(4))
    
    def get_empty_cells(board):
        return [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]
    
    def evaluate_board(board):
        winner = check_winner(board)
        if winner == 1:
            return 1000
        elif winner == -1:
            return -1000
        
        score = 0
        
        # Evaluate all lines (rows, columns, diagonals)
        lines = []
        
        # Rows and columns
        for i in range(4):
            lines.append([board[i][j] for j in range(4)])  # row
            lines.append([board[j][i] for j in range(4)])  # column
        
        # Diagonals
        lines.append([board[i][i] for i in range(4)])
        lines.append([board[i][3-i] for i in range(4)])
        
        for line in lines:
            line_score = evaluate_line(line)
            score += line_score
        
        # Add positional bonus for center squares
        center_positions = [(1, 1), (1, 2), (2, 1), (2, 2)]
        for i, j in center_positions:
            if board[i][j] == 1:
                score += 5
            elif board[i][j] == -1:
                score -= 5
        
        return score
    
    def evaluate_line(line):
        my_count = line.count(1)
        opp_count = line.count(-1)
        
        if opp_count > 0 and my_count > 0:
            return 0  # Mixed line has no value
        
        if my_count == 4:
            return 1000
        elif my_count == 3:
            return 50
        elif my_count == 2:
            return 10
        elif my_count == 1:
            return 2
        elif opp_count == 4:
            return -1000
        elif opp_count == 3:
            return -50
        elif opp_count == 2:
            return -10
        elif opp_count == 1:
            return -2
        
        return 0
    
    def minimax(board, depth, alpha, beta, is_maximizing):
        if depth == 0 or is_terminal(board):
            return evaluate_board(board)
        
        empty_cells = get_empty_cells(board)
        
        if is_maximizing:
            max_eval = float('-inf')
            for i, j in empty_cells:
                board[i][j] = 1
                eval_score = minimax(board, depth - 1, alpha, beta, False)
                board[i][j] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = float('inf')
            for i, j in empty_cells:
                board[i][j] = -1
                eval_score = minimax(board, depth - 1, alpha, beta, True)
                board[i][j] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    # Check for immediate winning move
    empty_cells = get_empty_cells(board)
    for i, j in empty_cells:
        board[i][j] = 1
        if check_winner(board) == 1:
            board[i][j] = 0
            return (i, j)
        board[i][j] = 0
    
    # Check for blocking opponent's winning move
    for i, j in empty_cells:
        board[i][j] = -1
        if check_winner(board) == -1:
            board[i][j] = 0
            return (i, j)
        board[i][j] = 0
    
    # Use minimax for strategic move
    best_move = None
    best_score = float('-inf')
    
    # Adjust depth based on game state
    num_empty = len(empty_cells)
    if num_empty > 12:
        depth = 3
    elif num_empty > 8:
        depth = 4
    elif num_empty > 4:
        depth = 5
    else:
        depth = min(6, num_empty)
    
    for i, j in empty_cells:
        board[i][j] = 1
        score = minimax(board, depth - 1, float('-inf'), float('inf'), False)
        board[i][j] = 0
        
        if score > best_score:
            best_score = score
            best_move = (i, j)
    
    return best_move if best_move else empty_cells[0]
```

This solution implements:

1. **Immediate win detection** - Always takes a winning move if available
2. **Threat blocking** - Prevents opponent from winning on their next turn
3. **Minimax with alpha-beta pruning** - Strategic lookahead with efficient pruning
4. **Dynamic depth adjustment** - Deeper search in endgame, shallower in opening
5. **Line evaluation** - Scores based on potential to complete 4-in-a-row
6. **Positional scoring** - Slight preference for center squares
7. **Multiple win condition checking** - Handles all rows, columns, and diagonals

The algorithm balances tactical accuracy with computational efficiency to stay within the 1-second time limit while making strong strategic moves in 4x4 Tic Tac Toe.

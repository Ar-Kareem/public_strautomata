import random

def check_win(board: list[list[int]], player: int) -> bool:
    """Check if the given player has won on the board."""
    # Check rows
    for row in board:
        if all(cell == player for cell in row):
            return True
    # Check columns
    for j in range(4):
        if all(board[i][j] == player for i in range(4)):
            return True
    # Check main diagonal
    if all(board[i][i] == player for i in range(4)):
        return True
    # Check anti-diagonal
    if all(board[i][3 - i] == player for i in range(4)):
        return True
    return False

def policy(board: list[list[int]]) -> tuple[int, int]:
    """Return the next move as a (row, col) tuple based on a simple strategy."""
    possible_moves = [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]
    
    # First, check for immediate winning moves
    for move in possible_moves:
        temp_board = [row[:] for row in board]
        temp_board[move[0]][move[1]] = 1
        if check_win(temp_board, 1):
            return move
    
    # Second, check for blocking moves (prevent opponent win)
    for move in possible_moves:
        temp_board = [row[:] for row in board]
        temp_board[move[0]][move[1]] = -1
        if check_win(temp_board, -1):
            return move
    
    # Third, if possible, play in the center(ish) areas (1,1), (1,2), (2,1), (2,2)
    centers = [(1, 1), (1, 2), (2, 1), (2, 2)]
    for center in centers:
        if center in possible_moves:
            return center
    
    # Fourth, play in corners if available
    corners = [(0, 0), (0, 3), (3, 0), (3, 3)]
    for corner in corners:
        if corner in possible_moves:
            return corner
    
    # Finally, if nothing else, just pick a random remaining move
    return random.choice(possible_moves)

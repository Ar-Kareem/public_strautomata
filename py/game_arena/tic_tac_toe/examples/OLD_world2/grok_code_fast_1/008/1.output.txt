import numpy as np
import copy

def is_winner(board, player):
    # Check rows
    for i in range(4):
        for j in range(1):
            if all(board[i][j + k] == player for k in range(4)):
                return True
    
    # Check columns
    for j in range(4):
        for i in range(1):
            if all(board[i + k][j] == player for k in range(4)):
                return True
    
    # Check main diagonals
    if all(board[i][i] == player for i in range(4)):
        return True
    if all(board[i][3 - i] == player for i in range(4)):
        return True
    
    return False

def get_empty_cells(board):
    empty = []
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                empty.append((i, j))
    return empty

def evaluate(board):
    if is_winner(board, 1):
        return 1000
    if is_winner(board, -1):
        return -1000
    if len(get_empty_cells(board)) == 0:
        return 0
    return 0  # Heuristic could be added, but simplistic for now

def minimax(board, depth, is_maximizing, alpha=-np.inf, beta=np.inf):
    score = evaluate(board)
    if score != 0 or depth == 0 or len(get_empty_cells(board)) == 0:
        return score
    
    if is_maximizing:
        max_eval = -np.inf
        for i, j in get_empty_cells(board):
            board_copy = copy.deepcopy(board)
            board_copy[i][j] = 1
            eval = minimax(board_copy, depth - 1, False, alpha, beta)
            max_eval = max(max_eval, eval)
            alpha = max(alpha, eval)
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = np.inf
        for i, j in get_empty_cells(board):
            board_copy = copy.deepcopy(board)
            board_copy[i][j] = -1
            eval = minimax(board_copy, depth - 1, True, alpha, beta)
            min_eval = min(min_eval, eval)
            beta = min(beta, eval)
            if beta <= alpha:
                break
        return min_eval

def policy(board):
    board = np.array(board)  # Convert to numpy for easier handling, but keep as list in input
    best_move = None
    best_value = -np.inf
    
    for i, j in get_empty_cells(board):
        board_copy = copy.deepcopy(board)
        board_copy[i][j] = 1
        move_value = minimax(board_copy, depth=5, is_maximizing=False)  # Depth limited to 5 for efficiency
        if move_value > best_value:
            best_value = move_value
            best_move = (i, j)
    
    return best_move if best_move is not None else get_empty_cells(board)[0]  # Fallback if no best found

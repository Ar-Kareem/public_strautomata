def policy(board: list[list[int]]) -> tuple[int, int]:
    def is_winning_move(b, r, c, player):
        # Check row
        if all(b[r][j] == player for j in range(4)):
            return True
        # Check column
        if all(b[i][c] == player for i in range(4)):
            return True
        # Check main diagonal
        if r == c:
            if all(b[i][i] == player for i in range(4)):
                return True
        # Check anti-diagonal
        if r + c == 3:
            if all(b[i][3-i] == player for i in range(4)):
                return True
        return False
    
    # Check for immediate win
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                # Simulate our move
                board[r][c] = 1
                if is_winning_move(board, r, c, 1):
                    board[r][c] = 0
                    return (r, c)
                board[r][c] = 0
    
    # Check for opponent's immediate win
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                # Simulate opponent's move
                board[r][c] = -1
                if is_winning_move(board, r, c, -1):
                    board[r][c] = 0
                    return (r, c)
                board[r][c] = 0
    
    # Evaluate heuristic for all empty cells
    def calculate_score(r, c):
        our_score = 0
        
        lines_with_indices = []
        # Row
        row_line = tuple(board[r][j] for j in range(4))
        lines_with_indices.append( (row_line, c) )
        # Column
        col_line = tuple(board[i][c] for i in range(4))
        lines_with_indices.append( (col_line, r) )
        # Diagonals
        if r == c:
            diag_line = tuple(board[i][i] for i in range(4))
            lines_with_indices.append( (diag_line, r) )
        if r + c == 3:
            anti_diag_line = tuple(board[i][3 - i] for i in range(4))
            lines_with_indices.append( (anti_diag_line, r) )
        
        # Calculate our_score
        for (line, index) in lines_with_indices:
            line_list = list(line)
            line_list[index] = 1
            ones = 0
            minus_ones = 0
            for val in line_list:
                if val == 1:
                    ones += 1
                elif val == -1:
                    minus_ones += 1
            if minus_ones == 0:
                our_score += 2 ** (ones - 1)
        
        # Calculate opponent_score
        opponent_score = 0
        lines_with_indices = []
        # Rebuild the lines_with_indices for the opponent's potential
        row_line = tuple(board[r][j] for j in range(4))
        lines_with_indices.append( (row_line, c) )
        col_line = tuple(board[i][c] for i in range(4))
        lines_with_indices.append( (col_line, r) )
        if r == c:
            diag_line = tuple(board[i][i] for i in range(4))
            lines_with_indices.append( (diag_line, r) )
        if r + c == 3:
            anti_diag_line = tuple(board[i][3 - i] for i in range(4))
            lines_with_indices.append( (anti_diag_line, r) )
        
        for (line, index) in lines_with_indices:
            line_list = list(line)
            line_list[index] = -1
            ones = 0
            minus_ones = 0
            for val in line_list:
                if val == 1:
                    ones += 1
                elif val == -1:
                    minus_ones += 1
            if ones == 0:
                opponent_score += 2 ** (minus_ones - 1)
        
        return our_score + opponent_score
    
    best_score = -1
    best_move = (0, 0)
    # Iterate all cells
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                score = calculate_score(r, c)
                if score > best_score:
                    best_score = score
                    best_move = (r, c)
    return best_move

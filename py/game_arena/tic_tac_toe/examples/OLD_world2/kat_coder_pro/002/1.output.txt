import numpy as np
from typing import Tuple, List, Optional
import random

# Game utilities for 4x4 Tic-Tac-Toe
def check_winner(board: np.ndarray) -> int:
    """Return 1 if player 1 wins, -1 if player -1 wins, 0 if no winner yet."""
    # Check rows, columns, and diagonals
    # Rows
    for i in range(4):
        row_sum = np.sum(board[i, :])
        if row_sum == 4:
            return 1
        if row_sum == -4:
            return -1
    # Columns
    for j in range(4):
        col_sum = np.sum(board[:, j])
        if col_sum == 4:
            return 1
        if col_sum == -4:
            return -1
    # Main diagonal
    diag_sum = np.sum(np.diag(board))
    if diag_sum == 4:
        return 1
    if diag_sum == -4:
        return -1
    # Anti-diagonal
    anti_diag_sum = np.sum(np.diag(np.fliplr(board)))
    if anti_diag_sum == 4:
        return 1
    if anti_diag_sum == -4:
        return -1
    return 0

def is_full(board: np.ndarray) -> bool:
    """Check if the board is full (no empty cells)."""
    return np.sum(board == 0) == 0

def get_valid_moves(board: np.ndarray) -> List[Tuple[int, int]]:
    """Return a list of empty cell coordinates."""
    return [(i, j) for i in range(4) for j in range(4) if board[i, j] == 0]

def simulate_random_game(board: np.ndarray, player: int) -> int:
    """
    Simulate a random playout from the current board state.
    Returns the result: 1 (win), -1 (loss), 0 (draw) from the player's perspective.
    """
    current_board = board.copy()
    current_player = player

    # Check for immediate win/loss
    winner = check_winner(current_board)
    if winner != 0:
        return winner * player  # Convert to player's perspective

    # Random playout
    while True:
        valid_moves = get_valid_moves(current_board)
        if not valid_moves:
            # Draw
            return 0

        # Choose a random move
        move = random.choice(valid_moves)
        current_board[move[0], move[1]] = current_player

        # Check for winner
        winner = check_winner(current_board)
        if winner != 0:
            return winner * player  # Convert to player's perspective

        # Switch player
        current_player *= -1

def policy(board: list[list[int]]) -> Tuple[int, int]:
    """
    Monte Carlo Tree Search (MCTS) policy for 4x4 Tic-Tac-Toe.
    Returns the best move based on simulations.
    """
    board_np = np.array(board, dtype=int)
    valid_moves = get_valid_moves(board_np)

    # If only one move available, take it
    if len(valid_moves) == 1:
        return valid_moves[0]

    # Quick win check: if there's a winning move, take it
    for move in valid_moves:
        test_board = board_np.copy()
        test_board[move[0], move[1]] = 1
        if check_winner(test_board) == 1:
            return move

    # Block opponent's winning move
    for move in valid_moves:
        test_board = board_np.copy()
        test_board[move[0], move[1]] = -1
        if check_winner(test_board) == -1:
            return move

    # MCTS parameters
    # We'll do a fixed number of simulations for speed
    num_simulations = 2000

    # Initialize scores
    move_scores = np.zeros(len(valid_moves), dtype=float)
    move_visits = np.zeros(len(valid_moves), dtype=int)

    # MCTS loop
    for _ in range(num_simulations):
        # Choose move with UCB1 or random for exploration
        # Since we don't have a full tree, we'll use a simple exploration strategy
        # Choose a move based on a mix of current score and exploration term
        if all(move_visits > 0):
            # UCB1 formula
            ucb_scores = move_scores / move_visits + np.sqrt(2 * np.log(num_simulations) / move_visits)
            move_idx = np.argmax(ucb_scores)
        else:
            # Random exploration until all moves have been visited
            unvisited = np.where(move_visits == 0)[0]
            move_idx = np.random.choice(unvisited)

        move = valid_moves[move_idx]
        move_visits[move_idx] += 1

        # Simulate game from this move
        test_board = board_np.copy()
        test_board[move[0], move[1]] = 1
        result = simulate_random_game(test_board, -1)  # Opponent plays next
        result = -result  # Convert to our perspective

        # Update score
        move_scores[move_idx] += result

    # Choose the move with the highest average score
    best_move_idx = np.argmax(move_scores / np.maximum(move_visits, 1))
    return valid_moves[best_move_idx]

from typing import Tuple
import math

def policy(board: list[list[int]]) -> Tuple[int, int]:
    """
    Choose the next move for a 4x4 Tic Tac Toe game.
    
    Args:
        board: 4x4 list of lists containing 0 (empty), 1 (you), -1 (opponent)
    
    Returns:
        (row, col) tuple where 0 <= row, col <= 3
    """
    # Find available moves
    available_moves = []
    for row in range(4):
        for col in range(4):
            if board[row][col] == 0:
                available_moves.append((row, col))
    
    if not available_moves:
        return (0, 0)  # Should not happen in valid games
    
    # Heuristic scoring function
    def evaluate(board, player):
        score = 0
        
        # Check all possible lines: rows, columns, diagonals
        lines = []
        
        # Rows
        for r in range(4):
            lines.append([(r, c) for c in range(4)])
        
        # Columns
        for c in range(4):
            lines.append([(r, c) for r in range(4)])
        
        # Main diagonal
        lines.append([(i, i) for i in range(4)])
        
        # Anti-diagonal
        lines.append([(i, 3-i) for i in range(4)])
        
        # Evaluate each line
        for line in lines:
            my_count = 0
            opp_count = 0
            empty_count = 0
            
            for r, c in line:
                if board[r][c] == player:
                    my_count += 1
                elif board[r][c] == -player:
                    opp_count += 1
                else:
                    empty_count += 1
            
            # Score this line
            if opp_count == 0:
                # We can potentially win this line
                if my_count == 4:
                    score += 10000  # Win
                elif my_count == 3:
                    score += 1000   # Almost winning
                elif my_count == 2:
                    score += 100    # Good position
                elif my_count == 1:
                    score += 10     # Some presence
            elif my_count == 0:
                # Opponent can potentially win this line
                if opp_count == 4:
                    score -= 10000  # Loss
                elif opp_count == 3:
                    score -= 5000   # Block opponent's win
                elif opp_count == 2:
                    score -= 200    # Opponent has good position
        
        return score
    
    # Minimax with alpha-beta pruning
    def minimax(board, depth, alpha, beta, maximizing_player):
        # Check if game is over or depth limit reached
        if depth == 0:
            return evaluate(board, 1)
        
        # Check for terminal states
        current_score = evaluate(board, 1)
        if abs(current_score) >= 10000:  # Win/loss detected
            return current_score
        
        player = 1 if maximizing_player else -1
        
        # Get available moves
        moves = []
        for row in range(4):
            for col in range(4):
                if board[row][col] == 0:
                    moves.append((row, col))
        
        if not moves:
            return 0  # Draw
        
        if maximizing_player:
            max_eval = -math.inf
            for row, col in moves:
                # Make move
                board[row][col] = 1
                eval_score = minimax(board, depth - 1, alpha, beta, False)
                # Undo move
                board[row][col] = 0
                
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = math.inf
            for row, col in moves:
                # Make move
                board[row][col] = -1
                eval_score = minimax(board, depth - 1, alpha, beta, True)
                # Undo move
                board[row][col] = 0
                
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    # Try to find immediate winning move
    for row, col in available_moves:
        board[row][col] = 1
        if evaluate(board, 1) >= 10000:
            board[row][col] = 0
            return (row, col)
        board[row][col] = 0
    
    # Try to block opponent's immediate winning move
    for row, col in available_moves:
        board[row][col] = -1
        if evaluate(board, 1) <= -10000:
            board[row][col] = 0
            return (row, col)
        board[row][col] = 0
    
    # Use minimax for deeper search
    best_move = available_moves[0]
    best_score = -math.inf
    
    # Search depth depends on available moves
    depth = 4 if len(available_moves) <= 8 else 3
    
    for row, col in available_moves:
        board[row][col] = 1
        score = minimax(board, depth - 1, -math.inf, math.inf, False)
        board[row][col] = 0
        
        if score > best_score:
            best_score = score
            best_move = (row, col)
    
    return best_move

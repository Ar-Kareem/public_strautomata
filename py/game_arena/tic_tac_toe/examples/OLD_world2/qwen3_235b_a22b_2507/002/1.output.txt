import numpy as np
from typing import Tuple, List

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Determines the optimal move for a 4x4 Tic Tac Toe game using minimax with alpha-beta pruning.
    Player 1 (me) is maximizing, player -1 (opponent) is minimizing.
    """
    board_array = np.array(board)
    
    def check_winner(board_state):
        # Check rows, columns, and diagonals for a winner
        # Check rows
        for i in range(4):
            row_sum = np.sum(board_state[i, :])
            if row_sum == 4:
                return 1
            elif row_sum == -4:
                return -1
                
        # Check columns
        for j in range(4):
            col_sum = np.sum(board_state[:, j])
            if col_sum == 4:
                return 1
            elif col_sum == -4:
                return -1
        
        # Check main diagonal
        diag_sum = np.sum(board_state.diagonal())
        if diag_sum == 4:
            return 1
        elif diag_sum == -4:
            return -1
            
        # Check anti-diagonal
        anti_diag_sum = np.sum(np.fliplr(board_state).diagonal())
        if anti_diag_sum == 4:
            return 1
        elif anti_diag_sum == -4:
            return -1
            
        # Check for draw (board full)
        if np.all(board_state != 0):
            return 0
            
        # Game continues
        return None
    
    def evaluate(board_state):
        """Heuristic evaluation of board position"""
        score = 0
        
        # Check all rows
        for i in range(4):
            line = board_state[i, :]
            score += evaluate_line(line)
        
        # Check all columns
        for j in range(4):
            line = board_state[:, j]
            score += evaluate_line(line)
        
        # Check main diagonal
        score += evaluate_line(board_state.diagonal())
        
        # Check anti-diagonal
        score += evaluate_line(np.fliplr(board_state).diagonal())
        
        return score
    
    def evaluate_line(line):
        """Evaluate a single line (row, column, or diagonal)"""
        player_count = np.sum(line == 1)
        opponent_count = np.sum(line == -1)
        
        # If both players have marks in this line, it's blocked
        if player_count > 0 and opponent_count > 0:
            return 0
            
        # Score based on number of marks (potential to win)
        if player_count == 3:
            return 100  # One move from winning
        elif player_count == 2:
            return 10
        elif player_count == 1:
            return 1
        elif opponent_count == 3:
            return -100  # Opponent one move from winning
        elif opponent_count == 2:
            return -10
        elif opponent_count == 1:
            return -1
            
        return 0
    
    def minimax(board_state, depth, alpha, beta, maximizing_player):
        winner = check_winner(board_state)
        if winner is not None:
            if winner == 1:
                return 1000 - depth  # Prefer quicker wins
            elif winner == -1:
                return -1000 + depth  # Prefer slower losses
            else:
                return 0  # Draw
        
        if depth == 0:
            return evaluate(board_state)
        
        if maximizing_player:
            max_eval = float('-inf')
            for i in range(4):
                for j in range(4):
                    if board_state[i, j] == 0:
                        board_state[i, j] = 1
                        eval_score = minimax(board_state, depth - 1, alpha, beta, False)
                        board_state[i, j] = 0
                        max_eval = max(max_eval, eval_score)
                        alpha = max(alpha, eval_score)
                        if beta <= alpha:
                            break  # Alpha-beta pruning
            return max_eval
        else:
            min_eval = float('inf')
            for i in range(4):
                for j in range(4):
                    if board_state[i, j] == 0:
                        board_state[i, j] = -1
                        eval_score = minimax(board_state, depth - 1, alpha, beta, True)
                        board_state[i, j] = 0
                        min_eval = min(min_eval, eval_score)
                        beta = min(beta, eval_score)
                        if beta <= alpha:
                            break  # Alpha-beta pruning
            return min_eval
    
    # Find all valid moves
    valid_moves = []
    for i in range(4):
        for j in range(4):
            if board_array[i, j] == 0:
                valid_moves.append((i, j))
    
    # If no valid moves (shouldn't happen), return center
    if not valid_moves:
        return (1, 1)
    
    # If it's the first move, take center position
    if len(valid_moves) == 16:
        return (1, 1)
    
    # Determine search depth based on remaining moves
    remaining_moves = len(valid_moves)
    if remaining_moves >= 12:
        max_depth = 4
    elif remaining_moves >= 8:
        max_depth = 5
    elif remaining_moves >= 5:
        max_depth = 6
    else:
        max_depth = 7  # More depth when fewer moves remain
    
    # Search for the best move
    best_score = float('-inf')
    best_move = valid_moves[0]  # Default to first valid move
    
    for move in valid_moves:
        i, j = move
        board_array[i, j] = 1  # Make the move
        score = minimax(board_array, max_depth, float('-inf'), float('inf'), False)
        board_array[i, j] = 0  # Undo the move
        
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the next move for a 4x4 Tic Tac Toe game.

    Args:
        board: A 4x4 list of lists representing the game board.
               0: empty, 1: AI (you), -1: opponent.

    Returns:
        A tuple of two integers (row, col) indicating the chosen move.
    """

    # Helper function to check for a winner
    def check_winner(current_board, player):
        # Check rows
        for row in current_board:
            if all(cell == player for cell in row):
                return True
        # Check columns
        for col_idx in range(4):
            if all(current_board[row_idx][col_idx] == player for row_idx in range(4)):
                return True
        # Check diagonals
        if all(current_board[i][i] == player for i in range(4)):
            return True
        if all(current_board[i][3 - i] == player for i in range(4)):
            return True
        return False

    # 1. Check for an immediate win
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                board[r][c] = 1  # Try placing our piece
                if check_winner(board, 1):
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # 2. Block opponent's immediate win
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                board[r][c] = -1  # Try placing opponent's piece
                if check_winner(board, -1):
                    return (r, c)
                board[r][c] = 0  # Undo the move

    # 3. Take the center (or near center) if available
    center_moves = [(1, 1), (1, 2), (2, 1), (2, 2)]
    random.shuffle(center_moves) # Add some variability
    for r, c in center_moves:
        if board[r][c] == 0:
            return (r, c)

    # 4. Take corners if available
    corners = [(0, 0), (0, 3), (3, 0), (3, 3)]
    random.shuffle(corners) # Add some variability
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)

    # 5. Take any available empty cell (prioritize edges, then other random spots)
    available_moves = []
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                available_moves.append((r, c))

    if available_moves:
        return random.choice(available_moves)

    # Should not happen in a valid game state, but as a fallback
    return (0, 0)

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    4x4 Tic Tac Toe policy that tries to win, block opponent, or make strategic moves.
    """
    
    def check_line(line):
        """Check a line and return (score, empty_positions)"""
        player_count = sum(1 for x in line if x == 1)
        opponent_count = sum(1 for x in line if x == -1)
        empty_count = sum(1 for x in line if x == 0)
        
        if player_count > 0 and opponent_count > 0:
            return 0, []  # Mixed line, no value
        elif player_count == 3 and empty_count == 1:
            return 1000, []  # Win immediately
        elif opponent_count == 3 and empty_count == 1:
            return 500, []  # Block opponent win
        elif player_count == 2 and empty_count == 2:
            return 50, []  # Good offensive position
        elif opponent_count == 2 and empty_count == 2:
            return 40, []  # Block opponent threat
        elif player_count == 1 and empty_count == 3:
            return 10, []
        elif opponent_count == 1 and empty_count == 3:
            return 5, []
        else:
            return 1, []
    
    def get_move_score(r, c):
        """Evaluate the score of placing a piece at position (r, c)"""
        if board[r][c] != 0:
            return -1
        
        score = 0
        
        # Check row
        row = board[r]
        row_copy = row.copy()
        row_copy[c] = 1
        if sum(1 for x in row_copy if x == 1) == 4:
            return 10000  # Winning move
        if sum(1 for x in row if x == -1) == 3 and row[c] == 0:
            return 5000  # Block opponent win
        score += check_line(row)[0]
        
        # Check column
        col = [board[i][c] for i in range(4)]
        col_copy = col.copy()
        col_copy[r] = 1
        if sum(1 for x in col_copy if x == 1) == 4:
            return 10000  # Winning move
        if sum(1 for x in col if x == -1) == 3 and col[r] == 0:
            return 5000  # Block opponent win
        score += check_line(col)[0]
        
        # Check main diagonal
        if r == c:
            diag = [board[i][i] for i in range(4)]
            diag_copy = diag.copy()
            diag_copy[r] = 1
            if sum(1 for x in diag_copy if x == 1) == 4:
                return 10000
            if sum(1 for x in diag if x == -1) == 3 and diag[r] == 0:
                return 5000
            score += check_line(diag)[0]
        
        # Check anti-diagonal
        if r + c == 3:
            anti_diag = [board[i][3-i] for i in range(4)]
            anti_diag_copy = anti_diag.copy()
            anti_diag_copy[r] = 1
            if sum(1 for x in anti_diag_copy if x == 1) == 4:
                return 10000
            if sum(1 for x in anti_diag if x == -1) == 3 and anti_diag[r] == 0:
                return 5000
            score += check_line(anti_diag)[0]
        
        # Bonus for center positions
        if (r, c) in [(1, 1), (1, 2), (2, 1), (2, 2)]:
            score += 15
        
        # Small bonus for corners
        if (r, c) in [(0, 0), (0, 3), (3, 0), (3, 3)]:
            score += 8
        
        return score
    
    # Find best move
    best_score = -1
    best_moves = []
    
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                score = get_move_score(r, c)
                if score > best_score:
                    best_score = score
                    best_moves = [(r, c)]
                elif score == best_score:
                    best_moves.append((r, c))
    
    # Return the best move (or first if multiple)
    if best_moves:
        return best_moves[0]
    
    # Fallback: return first empty cell
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                return (r, c)
    
    return (0, 0)

The strategy for this Tic Tac Toe policy is to always play optimally by:
1. Winning immediately if possible by placing a '1' that completes a row, column, or diagonal
2. Blocking the opponent's immediate win by placing a '1' where they would win next
3. Using minimax with alpha-beta pruning to evaluate future game states and choose the move that maximizes the chances of winning while minimizing losses

The minimax algorithm evaluates all possible game states by:
- Returning +10 for a win, -10 for a loss, and 0 for a tie
- Preferring wins that happen sooner (depth-based scoring)
- Using memoization to avoid recalculating the same board states
- Considering both our moves (maximizer) and opponent moves (minimizer)

This approach ensures the policy plays optimally in all game situations.

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determine the next move for a 3x3 Tic Tac Toe game.
    
    Args:
        board: 3x3 list of lists where 0=empty, 1=our move, -1=opponent's move
        
    Returns:
        tuple[int, int]: (row, col) coordinates for the next move
    """
    
    def check_win(board: list[list[int]], player: int) -> bool:
        """Check if the given player has won."""
        # Check rows
        for i in range(3):
            if board[i][0] == board[i][1] == board[i][2] == player:
                return True
        # Check columns
        for j in range(3):
            if board[0][j] == board[1][j] == board[2][j] == player:
                return True
        # Check diagonals
        if board[0][0] == board[1][1] == board[2][2] == player:
            return True
        if board[0][2] == board[1][1] == board[2][0] == player:
            return True
        return False
    
    def is_full(board: list[list[int]]) -> bool:
        """Check if the board is full."""
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    return False
        return True
    
    # Step 1: Check for immediate win
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                if check_win(board, 1):
                    board[i][j] = 0  # Undo move
                    return (i, j)
                board[i][j] = 0  # Undo move
    
    # Step 2: Check for immediate opponent win to block
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = -1
                if check_win(board, -1):
                    board[i][j] = 0  # Undo move
                    return (i, j)  # Block the opponent's winning move
                board[i][j] = 0  # Undo move
    
    # Step 3: Use minimax algorithm to find the best move
    memo = {}
    
    def minimax(board: list[list[int]], depth: int, is_maximizing: bool) -> int:
        """Minimax algorithm with memoization for Tic Tac Toe."""
        # Create a hashable key for the current board state
        key = (tuple(tuple(row) for row in board), is_maximizing)
        
        # Check memoization
        if key in memo:
            return memo[key]
        
        # Check terminal states
        if check_win(board, 1):
            return 10 - depth  # Win: prefer sooner wins
        if check_win(board, -1):
            return depth - 10  # Loss: prefer later losses
        if is_full(board):
            return 0  # Tie
        
        if is_maximizing:
            best_score = -10**9
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = 1
                        score = minimax(board, depth + 1, False)
                        board[i][j] = 0  # Undo move
                        best_score = max(best_score, score)
            memo[key] = best_score
            return best_score
        else:
            best_score = 10**9
            for i in range(3):
                for j in range(3):
                    if board[i][j] == 0:
                        board[i][j] = -1
                        score = minimax(board, depth + 1, True)
                        board[i][j] = 0  # Undo move
                        best_score = min(best_score, score)
            memo[key] = best_score
            return best_score
    
    # Find the best move using minimax
    best_score = -10**9
    best_move = (0, 0)
    
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                score = minimax(board, 1, False)  # Start with depth 1, opponent's turn
                board[i][j] = 0  # Undo move
                if score > best_score:
                    best_score = score
                    best_move = (i, j)
    
    return best_move



The strategy for this policy is to prioritize immediate wins, block opponent's winning moves, and use the minimax algorithm with alpha-beta pruning to determine the optimal move. It follows a preferred order (center, corners, edges) to break ties among moves with the same minimax score, ensuring strong play.

import math

def is_win(board, player):
    # Check rows, columns, and diagonals for a win
    for i in range(3):
        if all(board[i][j] == player for j in range(3)):
            return True
        if all(board[j][i] == player for j in range(3)):
            return True
    if board[0][0] == board[1][1] == board[2][2] == player:
        return True
    if board[0][2] == board[1][1] == board[2][0] == player:
        return True
    return False

def is_full(board):
    for row in board:
        if 0 in row:
            return False
    return True

def minimax(board, alpha, beta, is_maximizing):
    if is_win(board, 1):
        return 1
    if is_win(board, -1):
        return -1
    if is_full(board):
        return 0

    if is_maximizing:
        max_eval = -math.inf
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = 1
                    eval = minimax(board, alpha, beta, False)
                    board[i][j] = 0
                    max_eval = max(max_eval, eval)
                    alpha = max(alpha, eval)
                    if beta <= alpha:
                        break
        return max_eval
    else:
        min_eval = math.inf
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = -1
                    eval = minimax(board, alpha, beta, True)
                    board[i][j] = 0
                    min_eval = min(min_eval, eval)
                    beta = min(beta, eval)
                    if beta <= alpha:
                        break
        return min_eval

def policy(board):
    # Check for immediate win
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                if is_win(board, 1):
                    board[i][j] = 0
                    return (i, j)
                board[i][j] = 0
    # Check for opponent's immediate win
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = -1
                if is_win(board, -1):
                    board[i][j] = 0
                    return (i, j)
                board[i][j] = 0
    # Prefer center on first move
    if all(cell == 0 for row in board for cell in row):
        return (1, 1)
    # Use minimax to evaluate moves
    best_score = -math.inf
    candidates = []
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1
                score = minimax(board, -math.inf, math.inf, False)
                board[i][j] = 0
                if score > best_score:
                    best_score = score
                    candidates = [(i, j)]
                elif score == best_score:
                    candidates.append((i, j))
    # Select best candidate based on preferred order
    preferred_order = [(1, 1), (0, 0), (0, 2), (2, 0), (2, 2), 
                       (0, 1), (1, 0), (1, 2), (2, 1)]
    for move in preferred_order:
        if move in candidates:
            return move
    # Fallback to first available move (should not occur with valid input)
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                return (i, j)
    return (0, 0)

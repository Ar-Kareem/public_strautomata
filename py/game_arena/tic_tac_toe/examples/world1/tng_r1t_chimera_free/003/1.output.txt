def policy(board: list[list[int]]) -> tuple[int, int]:
    def find_winning_move(b, player):
        # Check rows
        for row in range(3):
            if b[row][0] == b[row][1] == player and b[row][2] == 0:
                return (row, 2)
            if b[row][0] == b[row][2] == player and b[row][1] == 0:
                return (row, 1)
            if b[row][1] == b[row][2] == player and b[row][0] == 0:
                return (row, 0)
        # Check columns
        for col in range(3):
            if b[0][col] == b[1][col] == player and b[2][col] == 0:
                return (2, col)
            if b[0][col] == b[2][col] == player and b[1][col] == 0:
                return (1, col)
            if b[1][col] == b[2][col] == player and b[0][col] == 0:
                return (0, col)
        # Check diagonals
        if b[0][0] == b[1][1] == player and b[2][2] == 0:
            return (2, 2)
        if b[0][0] == b[2][2] == player and b[1][1] == 0:
            return (1, 1)
        if b[1][1] == b[2][2] == player and b[0][0] == 0:
            return (0, 0)

            return (2, 0)
        if b[0][2] == b[2][0] == player and b[1][1] == 0:
            return (1, 1)
        if b[1][1] == b[2][0] == player and b[0][2] == 0:
            return (0, 2)
        return None

    # Check for winning move
    win_move = find_winning_move(board, 1)
    if win_move:
        return win_move

    # Check for opponent's winning move and block it
    block_move = find_winning_move(board, -1)
    if block_move:
        return block_move

    # Take center if available
    if board[1][1] == 0:
        return (1, 1)

    # Take any corner
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    for r, c in corners:
        if board[r][c] == 0:
            return (r, c)

    edges = [(0, 1), (1, 0), (1, 2), (2, 1)]
    for r, c in edges:

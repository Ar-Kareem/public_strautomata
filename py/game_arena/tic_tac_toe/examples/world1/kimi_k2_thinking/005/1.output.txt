The strategy implements optimal Tic-Tac-Toe play using minimax with alpha-beta pruning. It first checks for immediate wins or required blocks, then evaluates all possible moves to find the one with the best outcome assuming perfect play from both sides. Move ordering prioritizes the center and corners to improve pruning efficiency.

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Returns the best move for player 1 using minimax with alpha-beta pruning.
    Implements optimal Tic-Tac-Toe strategy.
    """
    player = 1
    opponent = -player
    
    # Get all legal moves
    legal_moves = [(i, j) for i in range(3) for j in range(3) if board[i][j] == 0]
    
    if not legal_moves:
        raise ValueError("No legal moves available")
    
    # 1. Check for immediate win
    for move in legal_moves:
        board[move[0]][move[1]] = player
        if check_winner(board) == player:
            board[move[0]][move[1]] = 0
            return move
        board[move[0]][move[1]] = 0
    
    # 2. Check for immediate block (opponent can win)
    for move in legal_moves:
        board[move[0]][move[1]] = opponent
        if check_winner(board) == opponent:
            board[move[0]][move[1]] = 0
            return move
        board[move[0]][move[1]] = 0
    
    # 3. Order moves for better alpha-beta pruning
    # Prioritize center, then corners, then edges
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    edges = [(0, 1), (1, 0), (1, 2), (2, 1)]
    
    ordered_moves = []
    if (1, 1) in legal_moves:
        ordered_moves.append((1, 1))
    for corner in corners:
        if corner in legal_moves:
            ordered_moves.append(corner)
    for edge in edges:
        if edge in legal_moves:
            ordered_moves.append(edge)
    
    # Add any remaining moves (shouldn't be needed)
    for move in legal_moves:
        if move not in ordered_moves:
            ordered_moves.append(move)
    
    # 4. Use minimax to find best move
    best_value = -float('inf')
    best_move = ordered_moves[0]
    
    for move in ordered_moves:
        # Make move
        board[move[0]][move[1]] = player
        
        # Evaluate position
        value = minimax(board, 0, False, -float('inf'), float('inf'), player)
        
        # Undo move
        board[move[0]][move[1]] = 0
        
        # Update best move
        if value > best_value:
            best_value = value
            best_move = move
    
    return best_move

def minimax(board, depth, is_maximizing, alpha, beta, player):
    """Minimax with alpha-beta pruning."""
    opponent = -player
    
    # Check terminal states
    winner = check_winner(board)
    if winner == player:
        return 10 - depth  # Prefer quicker wins
    elif winner == opponent:
        return depth - 10  # Prefer slower losses
    elif is_full(board):
        return 0
    
    if is_maximizing:
        max_eval = -float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = player
                    eval_val = minimax(board, depth + 1, False, alpha, beta, player)
                    board[i][j] = 0
                    max_eval = max(max_eval, eval_val)
                    alpha = max(alpha, eval_val)
                    if beta <= alpha:
                        break
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = opponent
                    eval_val = minimax(board, depth + 1, True, alpha, beta, player)
                    board[i][j] = 0
                    min_eval = min(min_eval, eval_val)
                    beta = min(beta, eval_val)
                    if beta <= alpha:
                        break
            if beta <= alpha:
                break
        return min_eval

def check_winner(board):
    """Check if there's a winner. Returns 1, -1, or 0."""
    # Check rows
    for row in board:
        if row[0] == row[1] == row[2] != 0:
            return row[0]
    
    # Check columns
    for col in range(3):
        if board[0][col] == board[1][col] == board[2][col] != 0:
            return board[0][col]
    
    # Check diagonals
    if board[0][0] == board[1][1] == board[2][2] != 0:
        return board[0][0]
    if board[0][2] == board[1][1] == board[2][0] != 0:
        return board[0][2]
    
    return 0

def is_full(board):
    """Check if the board is full."""
    return all(cell != 0 for row in board for cell in row)

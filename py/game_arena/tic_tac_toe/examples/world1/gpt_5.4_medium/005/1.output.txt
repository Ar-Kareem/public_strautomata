The strategy for this policy is to play perfect Tic Tac Toe using a full minimax search with memoization.

Key ideas:
- It evaluates the entire game tree, so it never misses a forced win, block, or draw.
- It scores terminal states to prefer:
  - faster wins,
  - slower losses,
  - and draws over losses.
- Among equally good moves, it uses a strong deterministic priority:
  - center,
  - corners,
  - edges.
- If given an unusual already-finished board, it still returns an empty cell if one exists, so it remains legal whenever possible.

from functools import lru_cache
from typing import List, Tuple

# Winning lines on a flattened 3x3 board.
_WIN_LINES = (
    (0, 1, 2),
    (3, 4, 5),
    (6, 7, 8),
    (0, 3, 6),
    (1, 4, 7),
    (2, 5, 8),
    (0, 4, 8),
    (2, 4, 6),
)

# Strong tie-breaking order: center, corners, then edges.
_MOVE_ORDER = (4, 0, 2, 6, 8, 1, 3, 5, 7)


def _winner(flat: Tuple[int, ...]) -> int:
    """Return 1 if we have won, -1 if opponent has won, else 0."""
    for a, b, c in _WIN_LINES:
        s = flat[a] + flat[b] + flat[c]
        if s == 3:
            return 1
        if s == -3:
            return -1
    return 0


@lru_cache(maxsize=None)
def _solve(flat: Tuple[int, ...], player: int) -> int:
    """
    Minimax value from the perspective of player 1 (us).
    player = 1 means it's our turn
    player = -1 means it's opponent's turn
    """
    w = _winner(flat)
    empties = flat.count(0)

    if w == 1:
        return 10 + empties   # prefer faster wins
    if w == -1:
        return -10 - empties  # prefer slower losses
    if empties == 0:
        return 0

    if player == 1:
        best = -10**9
        for idx in _MOVE_ORDER:
            if flat[idx] == 0:
                nxt = list(flat)
                nxt[idx] = 1
                score = _solve(tuple(nxt), -1)
                if score > best:
                    best = score
        return best
    else:
        best = 10**9
        for idx in _MOVE_ORDER:
            if flat[idx] == 0:
                nxt = list(flat)
                nxt[idx] = -1
                score = _solve(tuple(nxt), 1)
                if score < best:
                    best = score
        return best


def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Return a legal move (row, col) for a 3x3 Tic Tac Toe board.
    0 = empty, 1 = our mark, -1 = opponent mark.
    """
    flat = tuple(board[r][c] for r in range(3) for c in range(3))

    # Find a fallback legal move first, to ensure legality whenever possible.
    fallback = None
    for idx in _MOVE_ORDER:
        if flat[idx] == 0:
            fallback = idx
            break

    # No legal moves exist; this should not normally happen in a valid arena call.
    if fallback is None:
        return (0, 0)

    # If the board is already terminal, just return any legal move.
    if _winner(flat) != 0:
        return divmod(fallback, 3)

    best_move = fallback
    best_score = -10**9

    for idx in _MOVE_ORDER:
        if flat[idx] == 0:
            nxt = list(flat)
            nxt[idx] = 1
            score = _solve(tuple(nxt), -1)
            if score > best_score:
                best_score = score
                best_move = idx

    return divmod(best_move, 3)

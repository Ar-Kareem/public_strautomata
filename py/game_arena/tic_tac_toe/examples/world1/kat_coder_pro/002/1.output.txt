import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert to numpy array for easier manipulation
    board_np = np.array(board)
    
    # Helper function to check if a move leads to a win
    def is_winning_move(row, col, player):
        # Make the move temporarily
        temp_board = board_np.copy()
        temp_board[row, col] = player
        
        # Check rows, columns, and diagonals for win
        # Check rows
        if np.any(np.all(temp_board == player, axis=1)):
            return True
        # Check columns
        if np.any(np.all(temp_board == player, axis=0)):
            return True
        # Check main diagonal
        if np.all(np.diag(temp_board) == player):
            return True
        # Check anti-diagonal
        if np.all(np.diag(np.fliplr(temp_board)) == player):
            return True
        return False
    
    # Get all empty positions
    empty_positions = []
    for i in range(3):
        for j in range(3):
            if board_np[i, j] == 0:
                empty_positions.append((i, j))
    
    if not empty_positions:
        return (0, 0)  # Should not happen in normal play
    
    # 1. Try to win (player = 1)
    for row, col in empty_positions:
        if is_winning_move(row, col, 1):
            return (row, col)
    
    # 2. Block opponent from winning (opponent = -1)
    for row, col in empty_positions:
        if is_winning_move(row, col, -1):
            return (row, col)
    
    # 3. Take center if available
    if (1, 1) in empty_positions:
        return (1, 1)
    
    # 4. Take corners (prioritize corners)
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    available_corners = [pos for pos in corners if pos in empty_positions]
    if available_corners:
        # Prefer corners that are not adjacent to opponent moves
        # For simplicity, just take the first available corner
        return available_corners[0]
    
    # 5. Take any remaining edge
    edges = [(0, 1), (1, 0), (1, 2), (2, 1)]
    available_edges = [pos for pos in edges if pos in empty_positions]
    if available_edges:
        return available_edges[0]
    
    # Fallback (should not reach here)
    return empty_positions[0]

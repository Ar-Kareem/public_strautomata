import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Chooses the next move for a Tic Tac Toe game based on the given board.

    Args:
        board (list[list[int]]): A 3x3 list of lists representing the Tic Tac Toe board.
                                 0 represents an empty cell, 1 represents the AI player, and -1 represents the opponent.

    Returns:
        tuple[int, int]: A tuple containing the row and column indices (0-2) of the chosen move.
    """

    board_np = np.array(board)  # Convert to numpy array for easier manipulation

    for row in range(3):
        for col in range(3):
            if board_np[row, col] == 0:

    # 2. Check for a blocking move

            if board_np[row, col] == 0:
                board_np[row, col] = -1
                if check_win(board_np, -1):

    # 4. Take a corner if available

            if board_np[row, col] == 0:

    return (0, 0) # Should never get here in a valid game scenario


def check_win(board: np.ndarray, player: int) -> bool:
    """
    Checks if the specified player has won the Tic Tac Toe game.

    Args:
        board (np.ndarray): A 3x3 numpy array representing the Tic Tac Toe board.
        player (int): The player to check for a win (1 or -1).

    Returns:
        bool: True if the player has won, False otherwise.
    """
    # Check rows
    for row in range(3):
        if np.all(board[row, :] == player):
            return True

    # Check columns
    for col in range(3):
        if np.all(board[:, col] == player):
            return True

    # Check diagonals
    if np.all(np.diag(board) == player):
        return True
    if np.all(np.diag(np.fliplr(board)) == player):
        return True

    return False

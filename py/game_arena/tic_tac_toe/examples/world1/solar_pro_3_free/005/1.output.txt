A strong minimax strategy with alpha-beta pruning and a prioritized move order ensures optimal play. It first checks for immediate winning or blocking moves, then evaluates all legal moves using depth-first search with pruning to select the best outcome. The move order prioritizes the center, corners, then edges to exploit symmetry and maximize tactical advantages.

import sys
from functools import lru_cache

# Helper functions for game state evaluation
def is_win_for(board, player):
    # Check rows
    for i in range(3):
        if all(board[i][j] == player for j in range(3)):
            return True
    # Check columns
    for j in range(3):
        if all(board[i][j] == player for i in range(3)):
            return True
    # Check diagonals
    if board[0][0] == player and board[1][1] == player and board[2][2] == player:
        return True
    if board[0][2] == player and board[1][1] == player and board[2][0] == player:
        return True
    return False

def is_draw(board):
    # Check if board is full with no winner
    return 0 not in (cell for row in board for cell in row)

# Priority order for tie-breaking: center, corners, edges
move_order = [
    (1, 1),   # center
    (0, 0), (0, 2), (2, 0), (2, 2),   # corners
    (0, 1), (1, 0), (1, 2), (2, 1)    # edges
]

# Global memoization cache for minimax calculations
@lru_cache(maxsize=1000)
def minimax(board, is_maximizing, alpha, beta):
    """
    Returns the minimax score of the board:
    1 = win for player, -1 = loss, 0 = draw
    """
    # Convert board to tuple for caching
    board_tuple = tuple(tuple(row) for row in board)
    
    # Terminal states
    if is_win_for(board, 1):
        return 1
    if is_win_for(board, -1):
        return -1
    if is_draw(board):
        return 0
    
    # Minimize opponent's gain or maximize our score
    if is_maximizing:
        best_score = -float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    new_board = list(map(list, board))  # Copy the board
                    new_board[i][j] = 1  # Our move
                    score = minimax(new_board, False, alpha, beta)
                    best_score = max(best_score, score)
                    alpha = max(alpha, best_score)
                    if beta <= alpha:
                        break  # Alpha-beta pruning
        return best_score
    else:
        best_score = float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    new_board = list(map(list, board))
                    new_board[i][j] = -1  # Opponent's move
                    score = minimax(new_board, True, alpha, beta)
                    best_score = min(best_score, score)
                    beta = min(beta, best_score)
                    if beta <= alpha:
                        break  # Alpha-beta pruning
        return best_score

def policy(board):
    """
    Determines the optimal next move for a 3x3 Tic Tac Toe game.
    Returns (row, col) tuple representing the legal move.
    """
    # Check for immediate winning moves
    for move in move_order:
        if board[move[0]][move[1]] != 0:
            continue
        temp_board = list(map(list, board))
        temp_board[move[0]][move[1]] = 1
        if is_win_for(temp_board, 1):
            return move
    
    # Check for immediate blocking moves (prevent opponent win)
    for move in move_order:
        if board[move[0]][move[1]] != 0:
            continue
        temp_board = list(map(list, board))
        temp_board[move[0]][move[1]] = -1
        if is_win_for(temp_board, -1):
            return move
    
    # Evaluate all legal moves using minimax with alpha-beta pruning
    best_score = -float('inf')
    best_move = None
    alpha = -float('inf')
    beta = float('inf')
    
    for move in move_order:
        if board[move[0]][move[1]] != 0:
            continue
        
        temp_board = list(map(list, board))
        temp_board[move[0]][move[1]] = 1
        score = minimax(temp_board, False, alpha, beta)
        
        if score > best_score:
            best_score = score
            best_move = move
            alpha = max(alpha, best_score)
            if beta <= alpha:
                break  # Early exit if no better moves found
    
    return best_move

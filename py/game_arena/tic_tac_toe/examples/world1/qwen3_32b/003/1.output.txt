def policy(board: list[list[int]]) -> tuple[int, int]:
    def is_win(b, player):
        for i in range(3):
            if all(b[i][j] == player for j in range(3)):
                return True
        for j in range(3):
            if all(b[i][j] == player for i in range(3)):
                return True
        if all(b[i][i] == player for i in range(3)) or all(b[i][2 - i] == player for i in range(3)):
            return True
        return False

    # Check for winning move
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                new_board = [row[:] for row in board]
                new_board[i][j] = 1
                if is_win(new_board, 1):
                    return (i, j)

    # Check for opponent's winning move
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                new_board = [row[:] for row in board]
                new_board[i][j] = -1
                if is_win(new_board, -1):
                    return (i, j)

    # Default strategy: center, corners, edges
    if board[1][1] == 0:
        return (1, 1)
    for x, y in [(0, 0), (0, 2), (2, 0), (2, 2)]:
        if board[x][y] == 0:
            return (x, y)
    for x, y in [(0, 1), (1, 0), (1, 2), (2, 1)]:
        if board[x][y] == 0:
            return (x, y)

    return (0, 0)

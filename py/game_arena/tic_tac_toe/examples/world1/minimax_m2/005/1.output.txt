def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determine the next move for a 3x3 Tic Tac Toe game.
    
    Args:
        board: 3x3 list containing 0 (empty), 1 (AI piece), -1 (opponent piece)
    
    Returns:
        tuple: (row, col) for the chosen move
    """
    
    def win_condition(player_board, player):
        """Check if the given player has won."""
        # Check rows
        for i in range(3):
            if all(player_board[i][j] == player for j in range(3)):
                return True
        
        # Check columns
        for j in range(3):
            if all(player_board[i][j] == player for i in range(3)):
                return True
        
        # Check diagonals
        if all(player_board[i][i] == player for i in range(3)):
            return True
        if all(player_board[i][2-i] == player for i in range(3)):
            return True
        
        return False
    
    # Get all empty positions
    empty_positions = []
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                empty_positions.append((i, j))
    
    # Step 1: Check for winning move for AI (player 1)
    for i, j in empty_positions:
        temp_board = [row[:] for row in board]
        temp_board[i][j] = 1
        if win_condition(temp_board, 1):
            return (i, j)
    
    # Step 2: Check for blocking move (if opponent can win)
    for i, j in empty_positions:
        temp_board = [row[:] for row in board]
        temp_board[i][j] = -1
        if win_condition(temp_board, -1):
            return (i, j)
    
    # Step 3: Take center if available
    if board[1][1] == 0:
        return (1, 1)
    
    # Step 4: Take opposite corner if opponent has a corner
    corners = [(0, 0), (0, 2), (2, 0), (2, 2)]
    for corner in corners:
        i, j = corner
        if board[i][j] == -1:  # Opponent has this corner
            opposite = (2-i, 2-j)
            if board[opposite[0]][opposite[1]] == 0:
                return opposite
    
    # Step 5: Take any available corner
    for corner in corners:
        i, j = corner
        if board[i][j] == 0:
            return (i, j)
    
    # Step 6: Take any available side
    sides = [(0, 1), (1, 0), (1, 2), (2, 1)]
    for side in sides:
        i, j = side
        if board[i][j] == 0:
            return (i, j)
    
    # Should not reach here if game isn't over
    return (0, 0)

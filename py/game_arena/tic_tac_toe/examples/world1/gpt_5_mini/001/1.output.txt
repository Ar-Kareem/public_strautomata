from typing import List, Tuple, Optional

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Choose the next move for player 1 on a 3x3 Tic-Tac-Toe board.

    board: 3x3 list of lists with values:
        0  -> empty
        1  -> our pieces
        -1 -> opponent pieces

    Returns (row, col) with 0 <= row,col <= 2.
    """
    # Helper functions
    def winner(b: List[List[int]]) -> int:
        # Return 1 if player 1 wins, -1 if player -1 wins, 0 otherwise
        lines = []
        # rows and cols
        for i in range(3):
            lines.append(b[i][0] + b[i][1] + b[i][2])
            lines.append(b[0][i] + b[1][i] + b[2][i])
        # diagonals
        lines.append(b[0][0] + b[1][1] + b[2][2])
        lines.append(b[0][2] + b[1][1] + b[2][0])
        if 3 in lines:
            return 1
        if -3 in lines:
            return -1
        return 0

    def legal_moves(b: List[List[int]]) -> List[Tuple[int, int]]:
        moves = []
        for i in range(3):
            for j in range(3):
                if b[i][j] == 0:
                    moves.append((i, j))
        return moves

    # Count pieces to determine whose turn it is expected to be.
    my_count = sum(1 for i in range(3) for j in range(3) if board[i][j] == 1)
    opp_count = sum(1 for i in range(3) for j in range(3) if board[i][j] == -1)
    # In a normal game, player 1 moves when my_count <= opp_count
    next_player = 1 if my_count <= opp_count else -1

    moves = legal_moves(board)
    if not moves:
        return (0, 0)  # board full; return a default valid coordinate

    # If it's not our turn as given, return a sensible legal move (first available).
    # Tests typically call policy when it's our turn; otherwise we avoid simulating opponent.
    if next_player != 1:
        # Prefer center, then corners, then any
        if board[1][1] == 0:
            return (1, 1)
        for (i, j) in [(0,0),(0,2),(2,0),(2,2)]:
            if board[i][j] == 0:
                return (i, j)
        return moves[0]

    # Immediate win: take any move that makes us win
    for (i, j) in moves:
        board[i][j] = 1
        if winner(board) == 1:
            board[i][j] = 0
            return (i, j)
        board[i][j] = 0

    # Immediate block: block opponent's winning move
    for (i, j) in moves:
        board[i][j] = -1
        if winner(board) == -1:
            board[i][j] = 0
            return (i, j)
        board[i][j] = 0

    # Prefer center then corners if no immediate win/block
    if board[1][1] == 0:
        return (1, 1)
    for (i, j) in [(0,0),(0,2),(2,0),(2,2)]:
        if board[i][j] == 0:
            return (i, j)

    # Minimax with alpha-beta to choose optimal move
    def minimax(b: List[List[int]], player: int, alpha: int, beta: int) -> int:
        w = winner(b)
        if w != 0:
            return w  # 1 if player 1 wins, -1 if player -1 wins
        ms = legal_moves(b)
        if not ms:
            return 0  # draw

        if player == 1:
            value = -2
            for (i, j) in ms:
                b[i][j] = 1
                score = minimax(b, -1, alpha, beta)
                b[i][j] = 0
                if score > value:
                    value = score
                if value > alpha:
                    alpha = value
                if alpha >= beta:
                    break
            return value
        else:
            value = 2
            for (i, j) in ms:
                b[i][j] = -1
                score = minimax(b, 1, alpha, beta)
                b[i][j] = 0
                if score < value:
                    value = score
                if value < beta:
                    beta = value
                if alpha >= beta:
                    break
            return value

    best_score = -2
    best_move: Optional[Tuple[int, int]] = None
    for (i, j) in moves:
        board[i][j] = 1
        score = minimax(board, -1, -2, 2)
        board[i][j] = 0
        if score > best_score:
            best_score = score
            best_move = (i, j)
        # Prefer earlier winning choices; if find immediate win we would have returned earlier.

    if best_move is None:
        return moves[0]
    return best_move


# For quick local testing (not executed on import)
if __name__ == "__main__":
    # Empty board -> choose center
    b = [[0,0,0],[0,0,0],[0,0,0]]
    print(policy(b))  # Expect (1,1)

from copy import deepcopy
from typing import List, Tuple, Optional

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Return the best move for player 1 on a 3x3 Tic-Tac-Toe board.
    board: 3x3 list of lists with 0 empty, 1 (you), -1 (opponent)
    Returns (row, col) with 0 <= row,col <= 2
    """
    def check_winner(b: List[List[int]]) -> int:
        # returns 1 if player 1 wins, -1 if player -1 wins, 0 otherwise
        for i in range(3):
            s = b[i][0] + b[i][1] + b[i][2]
            if s == 3: return 1
            if s == -3: return -1
            s = b[0][i] + b[1][i] + b[2][i]
            if s == 3: return 1
            if s == -3: return -1
        s = b[0][0] + b[1][1] + b[2][2]
        if s == 3: return 1
        if s == -3: return -1
        s = b[0][2] + b[1][1] + b[2][0]
        if s == 3: return 1
        if s == -3: return -1
        return 0

    def available_moves(b: List[List[int]]) -> List[Tuple[int,int]]:
        moves = []
        for r in range(3):
            for c in range(3):
                if b[r][c] == 0:
                    moves.append((r,c))
        return moves

    def is_terminal(b: List[List[int]]) -> bool:
        if check_winner(b) != 0:
            return True
        for r in range(3):
            for c in range(3):
                if b[r][c] == 0:
                    return False
        return True

    # minimax with alpha-beta pruning
    def minimax(b: List[List[int]], player: int, depth: int, alpha: int, beta: int) -> Tuple[int, Optional[Tuple[int,int]]]:
        winner = check_winner(b)
        if winner != 0:
            # prefer faster wins and slower losses
            return (winner * (10 - depth), None)
        moves = available_moves(b)
        if not moves:
            return (0, None)  # draw

        best_move: Optional[Tuple[int,int]] = None
        if player == 1:
            max_eval = -10**9
            for (r,c) in moves:
                b[r][c] = 1
                eval_score, _ = minimax(b, -1, depth + 1, alpha, beta)
                b[r][c] = 0
                if eval_score > max_eval:
                    max_eval = eval_score
                    best_move = (r,c)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return (max_eval, best_move)
        else:
            min_eval = 10**9
            for (r,c) in moves:
                b[r][c] = -1
                eval_score, _ = minimax(b, 1, depth + 1, alpha, beta)
                b[r][c] = 0
                if eval_score < min_eval:
                    min_eval = eval_score
                    best_move = (r,c)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return (min_eval, best_move)

    # Quick checks: if center is free, take it (helps speed), else use minimax
    if board[1][1] == 0:
        # but verify no immediate winning/blocking move is required
        # check for immediate winning move
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    board[r][c] = 1
                    if check_winner(board) == 1:
                        board[r][c] = 0
                        return (r, c)
                    board[r][c] = 0
        # check for immediate block
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    board[r][c] = -1
                    if check_winner(board) == -1:
                        board[r][c] = 0
                        # we must block, don't take center automatically
                        break
                    board[r][c] = 0
        else:
            # no immediate block needed, take center
            return (1,1)

    _, move = minimax(board, 1, 0, -10**9, 10**9)
    if move is None:
        # fallback: choose first empty cell
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    return (r,c)
    return move

# Example usage when run as script (not required by arena)
if __name__ == "__main__":
    # empty board example
    b = [[0,0,0],[0,0,0],[0,0,0]]
    print(policy(b))  # should prefer center (1,1)

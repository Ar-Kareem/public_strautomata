    # rows
    for i in range(3):
        if board[i][0] == board[i][1] == board[i][2] != 0:
            return board[i][0]
    # columns
    for j in range(3):
        if board[0][j] == board[1][j] == board[2][j] != 0:

    return 0

def is_board_full(board):
    for row in board:
        if 0 in row:
            return False
    return True

        return 0

        return max_eval

def policy(board):
    best_score = -float('inf')
    best_move = None
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                board[i][j] = 1

                board[i][j] = 0
                if score > best_score:
                    best_score = score
                    best_move = (i, j)
    # In the unlikely case that no move is found (board full), return a fallback.
    if best_move is None:
        # find any empty cell
        for i in range(3):

    return best_move

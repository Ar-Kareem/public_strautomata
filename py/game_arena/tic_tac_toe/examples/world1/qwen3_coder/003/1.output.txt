def check_winner(board: np.ndarray) -> int:
    # Check rows
    for i in range(3):
        if abs(board[i, 0] + board[i, 1] + board[i, 2]) == 3:
            return board[i, 0]

    for j in range(3):
        if abs(board[0, j] + board[1, j] + board[2, j]) == 3:

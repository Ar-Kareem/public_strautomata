

The strategy for this policyis to prioritize winning moves, then blocking the opponent's winning move, then taking the center, then corners, and finally edges.

def is_winning_move(board, move, player):
    i, j = move
    # Check row i
    if board[i][0] == player and board[i][1] == player and board[i][2] == player:
        return True
    # Check column j
    if board[0][j] == player and board[1][j] == player and board[2][j] == player:
        return True
    # Check diagonals
    if i == j:
        if board[0][0] == player and board[1][1] == player and board[2][2] == player:
            return True
    if i + j == 2:
        if board[0][2] == player and board[1][1] == player and board[2][0] == player:
            return True
    return False

def policy(board):
    # Count the number of 1's and -1's
    count1 = 0
    count_opponent = 0
    for i in range(3):
        for j in range(3):
            if board[i][j] == 1:
                count1 += 1
            elif board[i][j] == -1:
                count_opponent += 1

    current_player = 1 if count1 > count_opponent else -1

    # Check for winning move
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                if is_winning_move(board, (i,j), current_player):
                    return (i,j)

    # Check for blocking opponent's winning move
    opponent = -current_player
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                if is_winning_move(board, (i,j), opponent):
                    return (i,j)

    # Take the center
    if board[1][1] == 0:
        return (1,1)

    # Take a corner
    corners = [(0,0), (0,2), (2,0), (2,2)]
    for corner in corners:
        if board[corner[0]][corner[0]] == 0:
            return corner

    # Take any edge
    edges = [(0,1), (1,0), (1,2), (2,1)]
    for edge in edges:
        if board[edge[0]][edge[1]] == 0:
            return edge

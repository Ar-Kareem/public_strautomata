The strategy for this policy is to use the Minimax algorithm to play perfectly. Tic Tac Toe is a solved game, and Minimax guarantees an optimal outcome: the AI will win if possible and will never lose. To optimize performance and decision-making, the algorithm searches through all possible game states, assigning higher scores to faster wins and lower scores to slower losses. It prioritizes winning moves and blocks opponent wins automatically through the game tree search.

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Calculates the optimal next move for Tic Tac Toe using the Minimax algorithm.
    
    Args:
        board: A 3x3 list of lists where 0 is empty, 1 is the AI, and -1 is the opponent.
        
    Returns:
        A tuple (row, col) representing the best move.
    """
    
    # Check for a winner.
    # Returns 1 if AI wins, -1 if Opponent wins, 0 otherwise.
    def check_winner(b):
        # Check rows
        for r in range(3):
            if sum(b[r]) == 3: return 1
            if sum(b[r]) == -3: return -1
        # Check columns
        for c in range(3):
            if b[0][c] + b[1][c] + b[2][c] == 3: return 1
            if b[0][c] + b[1][c] + b[2][c] == -3: return -1
        # Check diagonals
        if b[0][0] + b[1][1] + b[2][2] == 3: return 1
        if b[0][0] + b[1][1] + b[2][2] == -3: return -1
        if b[0][2] + b[1][1] + b[2][0] == 3: return 1
        if b[0][2] + b[1][1] + b[2][0] == -3: return -1
        return 0

    # Check if the board is full
    def is_full(b):
        for r in range(3):
            for c in range(3):
                if b[r][c] == 0:
                    return False
        return True

    # Minimax recursive algorithm
    def minimax(b, depth, is_maximizing):
        score = check_winner(b)
        
        # Terminal states
        if score == 1: return 10 - depth  # AI wins (prefer faster wins)
        if score == -1: return depth - 10 # Opponent wins (delay loss)
        if is_full(b): return 0           # Draw

        if is_maximizing:
            best_val = -float('inf')
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = 1
                        value = minimax(b, depth + 1, False)
                        b[r][c] = 0
                        best_val = max(best_val, value)
            return best_val
        else:
            best_val = float('inf')
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = -1
                        value = minimax(b, depth + 1, True)
                        b[r][c] = 0
                        best_val = min(best_val, value)
            return best_val

    best_score = -float('inf')
    best_move = (-1, -1)
    
    # Iterate through all available spots to find the optimal move
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                # Make the move
                board[r][c] = 1
                move_score = minimax(board, 0, False)
                # Undo the move
                board[r][c] = 0
                
                if move_score > best_score:
                    best_score = move_score
                    best_move = (r, c)
                    
    return best_move

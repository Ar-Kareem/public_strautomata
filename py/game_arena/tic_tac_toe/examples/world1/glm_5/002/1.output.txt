The strategy for this policy is to implement the Minimax algorithm with Alpha-Beta pruning. This approach explores the entire game tree of Tic Tac Toe to determine the mathematically optimal move. Since Tic Tac Toe is a solved game, this policy guarantees at least a draw against any opponent and will secure a win if the opponent makes a mistake. Alpha-Beta pruning is used to optimize the search process, ensuring the execution is well within the time limit. Additionally, the move search order is prioritized (Center > Corners > Edges) to maximize pruning efficiency and play stronger moves against sub-optimal opponents.

import math

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the optimal move for Tic Tac Toe using Minimax with Alpha-Beta pruning.
    
    Args:
        board: A 3x3 list of lists where 0 is empty, 1 is AI, -1 is opponent.
        
    Returns:
        A tuple (row, col) representing the chosen move.
    """
    
    # Check for a winner on the board.
    # Returns 1 if AI wins, -1 if Opponent wins, 0 if no winner yet.
    def check_winner(b):
        # Check rows
        for r in range(3):
            if b[r][0] != 0 and b[r][0] == b[r][1] == b[r][2]:
                return b[r][0]
        # Check columns
        for c in range(3):
            if b[0][c] != 0 and b[0][c] == b[1][c] == b[2][c]:
                return b[0][c]
        # Check diagonals
        if b[0][0] != 0 and b[0][0] == b[1][1] == b[2][2]:
            return b[0][0]
        if b[0][2] != 0 and b[0][2] == b[1][1] == b[2][0]:
            return b[0][2]
        return 0

    # Check if the board is full
    def is_full(b):
        for r in range(3):
            for c in range(3):
                if b[r][c] == 0:
                    return False
        return True

    # Minimax algorithm with Alpha-Beta pruning
    def minimax(b, depth, is_maximizing, alpha, beta):
        winner = check_winner(b)
        
        # Terminal states
        if winner == 1:
            return 10 - depth  # AI wins, prefer faster wins
        if winner == -1:
            return depth - 10  # Opponent wins, delay loss
        if is_full(b):
            return 0  # Draw

        if is_maximizing:
            best_val = -math.inf
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = 1
                        value = minimax(b, depth + 1, False, alpha, beta)
                        b[r][c] = 0
                        best_val = max(best_val, value)
                        alpha = max(alpha, best_val)
                        if beta <= alpha:
                            break
            return best_val
        else:
            best_val = math.inf
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = -1
                        value = minimax(b, depth + 1, True, alpha, beta)
                        b[r][c] = 0
                        best_val = min(best_val, value)
                        beta = min(beta, best_val)
                        if beta <= alpha:
                            break
            return best_val

    best_move = (-1, -1)
    best_value = -math.inf
    
    # Identify available moves
    moves = []
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                moves.append((r, c))
    
    # Heuristic move ordering: Center -> Corners -> Edges
    # This improves alpha-beta pruning efficiency and plays 'stronger' opening moves.
    def get_move_priority(move):
        r, c = move
        if (r, c) == (1, 1):
            return 0  # Center
        if r != 1 and c != 1:
            return 1  # Corners
        return 2  # Edges
        
    moves.sort(key=get_move_priority)
    
    # Evaluate each move
    for r, c in moves:
        board[r][c] = 1
        move_value = minimax(board, 0, False, -math.inf, math.inf)
        board[r][c] = 0
        
        if move_value > best_value:
            best_value = move_value
            best_move = (r, c)
            
    return best_move

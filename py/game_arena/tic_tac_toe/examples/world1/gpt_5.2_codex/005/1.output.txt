The strategy for this policy is to play perfect Tic Tac Toe using minimax with memoization. It evaluates terminal positions (win/loss/draw) and chooses a move that maximizes the outcome for player 1. This guarantees at least a draw and will beat any non-optimal opponent.

<code>
from functools import lru_cache

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Flatten board to tuple for caching
    flat = tuple(cell for row in board for cell in row)

    def winner(b):
        lines = [
            (0,1,2),(3,4,5),(6,7,8),  # rows
            (0,3,6),(1,4,7),(2,5,8),  # cols
            (0,4,8),(2,4,6)           # diags
        ]
        for a,b1,c in lines:
            if b[a] != 0 and b[a] == b[b1] == b[c]:
                return b[a]
        return 0

    @lru_cache(None)
    def minimax(b, player):
        w = winner(b)
        if w != 0:
            return w, None  # winner is 1 or -1
        if all(x != 0 for x in b):
            return 0, None  # draw
        best_move = None
        if player == 1:
            best_val = -2
            for i in range(9):
                if b[i] == 0:
                    nb = list(b)
                    nb[i] = player
                    val, _ = minimax(tuple(nb), -player)
                    if val > best_val:
                        best_val = val
                        best_move = i
                        if best_val == 1:
                            break
            return best_val, best_move
        else:
            best_val = 2
            for i in range(9):
                if b[i] == 0:
                    nb = list(b)
                    nb[i] = player
                    val, _ = minimax(tuple(nb), -player)
                    if val < best_val:
                        best_val = val
                        best_move = i
                        if best_val == -1:
                            break
            return best_val, best_move

    _, move = minimax(flat, 1)
    if move is None:
        # fallback: choose first empty
        for i in range(9):
            if flat[i] == 0:
                move = i
                break
    return divmod(move, 3)

The strategy for this policy is to play perfect Tic Tac Toe using minimax with memoization. It evaluates all possible outcomes and selects a move that guarantees at least a draw and takes a win whenever possible. This makes it unbeatable against any policy.

from functools import lru_cache

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert board to tuple for hashing
    b = tuple(tuple(row) for row in board)

    def winner(bd):
        # Check rows, cols, diagonals for win
        for i in range(3):
            s = sum(bd[i])
            if s == 3:
                return 1
            if s == -3:
                return -1
            s = bd[0][i] + bd[1][i] + bd[2][i]
            if s == 3:
                return 1
            if s == -3:
                return -1
        diag1 = bd[0][0] + bd[1][1] + bd[2][2]
        diag2 = bd[0][2] + bd[1][1] + bd[2][0]
        if diag1 == 3 or diag2 == 3:
            return 1
        if diag1 == -3 or diag2 == -3:
            return -1
        return 0

    @lru_cache(None)
    def minimax(bd, player):
        w = winner(bd)
        if w != 0:
            return w, None
        # Check for draw
        if all(cell != 0 for row in bd for cell in row):
            return 0, None

        best_move = None
        if player == 1:
            best_score = -2
            for r in range(3):
                for c in range(3):
                    if bd[r][c] == 0:
                        new_bd = [list(row) for row in bd]
                        new_bd[r][c] = player
                        score, _ = minimax(tuple(tuple(row) for row in new_bd), -player)
                        if score > best_score:
                            best_score = score
                            best_move = (r, c)
                            if best_score == 1:
                                return best_score, best_move
            return best_score, best_move
        else:
            best_score = 2
            for r in range(3):
                for c in range(3):
                    if bd[r][c] == 0:
                        new_bd = [list(row) for row in bd]
                        new_bd[r][c] = player
                        score, _ = minimax(tuple(tuple(row) for row in new_bd), -player)
                        if score < best_score:
                            best_score = score
                            best_move = (r, c)
                            if best_score == -1:
                                return best_score, best_move
            return best_score, best_move

    # Choose best move for player 1
    _, move = minimax(b, 1)
    if move is None:
        # Fallback: first empty cell
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    return (r, c)
    return move

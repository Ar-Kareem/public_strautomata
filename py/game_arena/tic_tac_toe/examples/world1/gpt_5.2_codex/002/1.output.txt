The strategy for this policy is to play perfectly using a minimax search with alpha-beta pruning. It evaluates terminal positions (win/loss/draw) and selects the move that maximizes the outcome for the current player (us), ensuring it never loses and wins when possible.

def policy(board):
    # Perfect play using minimax with alpha-beta pruning
    # board: 3x3 list of lists with 0 empty, 1 us, -1 opponent
    
    def check_winner(b):
        # returns 1 if us win, -1 if opponent win, 0 otherwise
        lines = []
        lines.extend(b)  # rows
        lines.extend([[b[0][c], b[1][c], b[2][c]] for c in range(3)])  # cols
        lines.append([b[0][0], b[1][1], b[2][2]])
        lines.append([b[0][2], b[1][1], b[2][0]])
        for line in lines:
            s = sum(line)
            if s == 3:
                return 1
            if s == -3:
                return -1
        return 0
    
    def is_full(b):
        return all(cell != 0 for row in b for cell in row)
    
    def minimax(b, player, alpha, beta):
        winner = check_winner(b)
        if winner != 0:
            return winner
        if is_full(b):
            return 0
        
        if player == 1:  # maximizing for us
            best = -2
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = 1
                        score = minimax(b, -1, alpha, beta)
                        b[r][c] = 0
                        if score > best:
                            best = score
                        if best > alpha:
                            alpha = best
                        if beta <= alpha:
                            return best
            return best
        else:  # minimizing for opponent
            best = 2
            for r in range(3):
                for c in range(3):
                    if b[r][c] == 0:
                        b[r][c] = -1
                        score = minimax(b, 1, alpha, beta)
                        b[r][c] = 0
                        if score < best:
                            best = score
                        if best < beta:
                            beta = best
                        if beta <= alpha:
                            return best
            return best
    
    # Choose best move
    best_score = -2
    best_move = None
    for r in range(3):
        for c in range(3):
            if board[r][c] == 0:
                board[r][c] = 1
                score = minimax(board, -1, -2, 2)
                board[r][c] = 0
                if score > best_score:
                    best_score = score
                    best_move = (r, c)
    
    # Fallback in case something goes wrong
    if best_move is None:
        for r in range(3):
            for c in range(3):
                if board[r][c] == 0:
                    return (r, c)
    
    return best_move

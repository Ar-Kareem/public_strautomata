The strategy for this policy is to utilize the **Minimax algorithm**. As Tic Tac Toe is a deterministic, zero-sum game with a small state space, Minimax can perfectly solve the game. The algorithm recursively explores all possible future game states, assuming the opponent plays optimally to minimize our score. This ensures that the policy never loses a game; it will always win if the opponent makes a mistake, or force a draw if the opponent plays perfectly. Given the 3x3 grid size, this computation is instantaneous and well within the 1-second time limit.

import sys

# Increase recursion depth to be safe, though depth won't exceed 9 for 3x3 Tic Tac Toe
sys.setrecursionlimit(2000)

def check_winner(board: list[list[int]]):
    """
    Checks the current board state for a winner or a draw.
    Returns:
        1 if Player 1 wins
        -1 if Player -1 wins
        0 if the game is a draw (board full, no winner)
        None if the game is still ongoing
    """
    # Check rows and columns
    for i in range(3):
        # Row check
        if board[i][0] == board[i][1] == board[i][2] and board[i][0] != 0:
            return board[i][0]
        # Column check
        if board[0][i] == board[1][i] == board[2][i] and board[0][i] != 0:
            return board[0][i]
    
    # Check diagonals
    if board[0][0] == board[1][1] == board[2][2] and board[0][0] != 0:
        return board[0][0]
    if board[0][2] == board[1][1] == board[2][0] and board[0][2] != 0:
        return board[0][2]
    
    # Check for empty cells (Game ongoing)
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                return None
                
    # If no winner and no empty cells, it's a draw
    return 0

def minimax(board: list[list[int]], depth: int, is_maximizing: bool):
    """
    Recursive Minimax algorithm to determine the value of the board.
    """
    result = check_winner(board)
    if result is not None:
        return result
    
    if is_maximizing:
        best_score = -float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = 1
                    score = minimax(board, depth + 1, False)
                    board[i][j] = 0
                    best_score = max(score, best_score)
                    # Optimization: If we found a win, no need to check further
                    if best_score == 1:
                        return 1
        return best_score
    else:
        best_score = float('inf')
        for i in range(3):
            for j in range(3):
                if board[i][j] == 0:
                    board[i][j] = -1
                    score = minimax(board, depth + 1, True)
                    board[i][j] = 0
                    best_score = min(score, best_score)
                    # Optimization: If opponent found a win, no need to check further
                    if best_score == -1:
                        return -1
        return best_score

def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Determines the best move for the current board state.
    """
    best_score = -float('inf')
    best_move = None
    
    # Iterate through all available cells
    for i in range(3):
        for j in range(3):
            if board[i][j] == 0:
                # Try the move
                board[i][j] = 1
                # Calculate score using minimax (next turn is opponent, so False)
                score = minimax(board, 0, False)
                # Undo the move
                board[i][j] = 0
                
                if score > best_score:
                    best_score = score
                    best_move = (i, j)
                    
    return best_move

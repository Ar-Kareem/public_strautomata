The strategy for this policy is minimax with alpha-beta pruning, which makes the policy play perfectly in 3x3 Tic Tac Toe. It always checks for immediate wins, blocks immediate losses, and otherwise searches the full game tree to choose an optimal move. Since Tic Tac Toe is small, full-depth search is fast and guarantees at least a draw against any opponent, while exploiting mistakes to win. As a safety measure, if something unexpected happens, the policy still returns a legal empty cell.

from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    def legal_moves(b):
        return [(r, c) for r in range(3) for c in range(3) if b[r][c] == 0]

    def winner(b):
        lines = []

        for i in range(3):
            lines.append(b[i])  # rows
            lines.append([b[0][i], b[1][i], b[2][i]])  # cols

        lines.append([b[0][0], b[1][1], b[2][2]])
        lines.append([b[0][2], b[1][1], b[2][0]])

        for line in lines:
            s = sum(line)
            if s == 3:
                return 1
            if s == -3:
                return -1
        return 0

    def is_full(b):
        return all(b[r][c] != 0 for r in range(3) for c in range(3))

    def minimax(b, maximizing, alpha, beta):
        w = winner(b)
        if w == 1:
            return 1
        if w == -1:
            return -1
        if is_full(b):
            return 0

        moves = legal_moves(b)

        if maximizing:
            value = -2
            for r, c in moves:
                b[r][c] = 1
                value = max(value, minimax(b, False, alpha, beta))
                b[r][c] = 0
                alpha = max(alpha, value)
                if alpha >= beta:
                    break
            return value
        else:
            value = 2
            for r, c in moves:
                b[r][c] = -1
                value = min(value, minimax(b, True, alpha, beta))
                b[r][c] = 0
                beta = min(beta, value)
                if alpha >= beta:
                    break
            return value

    moves = legal_moves(board)
    if not moves:
        return (0, 0)

    # 1) Immediate winning move
    for r, c in moves:
        board[r][c] = 1
        if winner(board) == 1:
            board[r][c] = 0
            return (r, c)
        board[r][c] = 0

    # 2) Immediate block of opponent winning move
    for r, c in moves:
        board[r][c] = -1
        if winner(board) == -1:
            board[r][c] = 0
            return (r, c)
        board[r][c] = 0

    # 3) Full minimax search
    best_score = -2
    best_move = moves[0]

    # Good move ordering improves pruning and gives nice play
    preferred_order = [(1, 1), (0, 0), (0, 2), (2, 0), (2, 2), (0, 1), (1, 0), (1, 2), (2, 1)]
    ordered_moves = [m for m in preferred_order if m in moves]

    for r, c in ordered_moves:
        board[r][c] = 1
        score = minimax(board, False, -2, 2)
        board[r][c] = 0

        if score > best_score:
            best_score = score
            best_move = (r, c)

    # Safety: always return a legal move
    if board[best_move[0]][best_move[1]] == 0:
        return best_move

    for move in moves:
        if board[move[0]][move[1]] == 0:
            return move

    return (0, 0)

def check_winner(board: np.ndarray) -> int:
    """Return 1 if player 1 wins, -1 if player -1 wins, 0 otherwise."""
    # Check rows
    for r in range(4):
        if all(board[r, c] == 1 for c in range(4)):
            return 1
        if all(board[r, c] == -1 for c in range(4)):
            return -1

    def check_winner(b):

        for i in range(4):

                return np.sign(sum(b[i, :]))

        for j in range(4):
            if abs(sum(b[:, j])) == 4:
                return np.sign(sum(b[:, j]))

        if abs(sum(b.diagonal())) == 4:
            return np.sign(sum(b.diagonal()))

        return [(i, j) for i in range(4) for j in range(4) if b[i][j] == 0]

        for i in range(4):
            if b[i][0] != 0 and b[i][0] == b[i][1] == b[i][2] == b[i][3]:
                return b[i][0]
        # Check columns
        for j in range(4):

import numpy as np
from typing import List, Tuple

def policy(board: List[List[int]]) -> Tuple[int, int]:
    """
    Minimax policy for 4x4 Tic Tac Toe with alpha-beta pruning.
    
    Args:
        board: 4x4 list of lists where 0=empty, 1=agent's move, -1=opponent's move
        
    Returns:
        Tuple (row, col) representing the best move
    """
    
    def evaluate(board_state: np.ndarray) -> int:
        """Evaluate the current board state."""
        # Check rows
        for row in range(4):
            if np.all(board_state[row, :] == 1):
                return 10
            elif np.all(board_state[row, :] == -1):
                return -10
        
        # Check columns
        for col in range(4):
            if np.all(board_state[:, col] == 1):
                return 10
            elif np.all(board_state[:, col] == -1):
                return -10
        
        # Check diagonals
        if np.all(np.diag(board_state) == 1) or np.all(np.diag(np.fliplr(board_state)) == 1):
            return 10
        elif np.all(np.diag(board_state) == -1) or np.all(np.diag(np.fliplr(board_state)) == -1):
            return -10
        
        return 0  # No winner yet
    
    def is_moves_left(board_state: np.ndarray) -> bool:
        """Check if there are any empty cells left."""
        return 0 in board_state
        
    def minimax(board_state: np.ndarray, depth: int, alpha: float, beta: float, is_maximizing: bool) -> int:
        """Minimax algorithm with alpha-beta pruning."""
        score = evaluate(board_state)
        
        # If maximizer has won
        if score == 10:
            return score - depth
        
        # If minimizer has won
        if score == -10:
            return score + depth
        
        # If no moves left, it's a tie
        if not is_moves_left(board_state):
            return 0
        
        if is_maximizing:
            best = -float('inf')
            
            for i in range(4):
                for j in range(4):
                    if board_state[i, j] == 0:
                        board_state[i, j] = 1  # Make move
                        best = max(best, minimax(board_state, depth + 1, alpha, beta, not is_maximizing))
                        board_state[i, j] = 0  # Undo move
                        
                        alpha = max(alpha, best)
                        if beta <= alpha:
                            break
            
            return best
        else:
            best = float('inf')
            
            for i in range(4):
                for j in range(4):
                    if board_state[i, j] == 0:
                        board_state[i, j] = -1  # Make move
                        best = min(best, minimax(board_state, depth + 1, alpha, beta, not is_maximizing))
                        board_state[i, j] = 0  # Undo move
                        
                        beta = min(beta, best)
                        if beta <= alpha:
                            break
            
            return best
    
    # Convert to numpy array for easier manipulation
    board_state = np.array(board)
    
    # Find all empty cells
    empty_cells = []
    for i in range(4):
        for j in range(4):
            if board_state[i, j] == 0:
                empty_cells.append((i, j))
    
    # If board is empty, take center position
    if len(empty_cells) == 16:
        return (1, 1)
    
    # If only one move available, take it
    if len(empty_cells) == 1:
        return empty_cells[0]
    
    # Find the best move
    best_value = -float('inf')
    best_move = empty_cells[0]  # Default to first available move
    
    for i, j in empty_cells:
        board_state[i, j] = 1  # Make move
        move_value = minimax(board_state, 0, -float('inf'), float('inf'), False)
        board_state[i, j] = 0  # Undo move
        
        if move_value > best_value:
            best_value = move_value
            best_move = (i, j)
    
    return best_move

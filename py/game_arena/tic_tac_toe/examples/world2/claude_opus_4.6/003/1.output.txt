## Strategy

For 4x4 Tic Tac Toe, I'll implement a minimax algorithm with alpha-beta pruning and several optimizations:

1. **Win condition**: 4 in a row horizontally, vertically, or diagonally.
2. **Minimax with alpha-beta pruning**: Search the game tree efficiently to find optimal moves.
3. **Move ordering**: Prioritize center/corner moves first to improve pruning efficiency.
4. **Depth limiting with evaluation function**: If the search takes too long, use a heuristic evaluation that counts threats (lines with 3 in a row, 2 in a row, etc.).
5. **Early termination**: Detect immediate wins/blocks first.

The 4x4 board has 16 cells, and with alpha-beta pruning and good move ordering, we should be able to search quite deep within the time limit. I'll use iterative deepening to ensure we always have a move ready.

import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    b = np.array(board, dtype=int)
    
    lines = []
    for r in range(4):
        lines.append([(r, c) for c in range(4)])
    for c in range(4):
        lines.append([(r, c) for r in range(4)])
    lines.append([(i, i) for i in range(4)])
    lines.append([(i, 3-i) for i in range(4)])
    
    def check_win(b, player):
        for line in lines:
            if all(b[r][c] == player for r, c in line):
                return True
        return False
    
    def get_moves(b):
        # Order: center, corners, edges
        preferred = [(1,1),(1,2),(2,1),(2,2),(0,0),(0,3),(3,0),(3,3),
                     (0,1),(0,2),(1,0),(1,3),(2,0),(2,3),(3,1),(3,2)]
        return [m for m in preferred if b[m[0]][m[1]] == 0]
    
    def evaluate(b):
        score = 0
        for line in lines:
            vals = [b[r][c] for r, c in line]
            p1 = vals.count(1)
            p2 = vals.count(-1)
            if p2 == 0 and p1 > 0:
                if p1 == 4: return 100000
                elif p1 == 3: score += 100
                elif p1 == 2: score += 10
                elif p1 == 1: score += 1
            elif p1 == 0 and p2 > 0:
                if p2 == 4: return -100000
                elif p2 == 3: score -= 100
                elif p2 == 2: score -= 10
                elif p2 == 1: score -= 1
        return score
    
    def minimax(b, depth, alpha, beta, maximizing):
        if check_win(b, 1):
            return 100000 + depth
        if check_win(b, -1):
            return -100000 - depth
        
        moves = get_moves(b)
        if not moves:
            return 0
        if depth == 0:
            return evaluate(b)
        
        if maximizing:
            val = -float('inf')
            for r, c in moves:
                b[r][c] = 1
                val = max(val, minimax(b, depth-1, alpha, beta, False))
                b[r][c] = 0
                alpha = max(alpha, val)
                if alpha >= beta:
                    break
            return val
        else:
            val = float('inf')
            for r, c in moves:
                b[r][c] = -1
                val = min(val, minimax(b, depth-1, alpha, beta, True))
                b[r][c] = 0
                beta = min(beta, val)
                if alpha >= beta:
                    break
            return val
    
    moves = get_moves(b)
    if not moves:
        return (0, 0)
    
    empty_count = int(np.sum(b == 0))
    # Use full depth if few moves remain, else limit
    max_depth = min(empty_count, 12)
    
    best_move = moves[0]
    best_val = -float('inf')
    
    for r, c in moves:
        b[r][c] = 1
        if check_win(b, 1):
            b[r][c] = 0
            return (r, c)
        b[r][c] = 0
    
    for r, c in moves:
        b[r][c] = -1
        if check_win(b, -1):
            b[r][c] = 0
            return (r, c)
        b[r][c] = 0
    
    for r, c in moves:
        b[r][c] = 1
        val = minimax(b, max_depth - 1, -float('inf'), float('inf'), False)
        b[r][c] = 0
        if val > best_val:
            best_val = val
            best_move = (r, c)
    
    return best_move

def check_winning_move(board, player):
    """Check if the current player can win with their next move."""
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:  # Empty spot
                board[r][c] = player  # Simulate move
                if check_winner(board, player):
                    board[r][c] = 0  # Undo move
                    return (r, c)
                board[r][c] = 0  # Undo move
    return None

def check_winner(board, player):
    """Check if the current player has won."""
    # Check rows and columns
    for i in range(4):
        if all(board[i][j] == player for j in range(4)) or all(board[j][i] == player for j in range(4)):
            return True
    # Check diagonals
    if all(board[i][i] == player for i in range(4)) or all(board[i][3 - i] == player for i in range(4)):
        return True
    return False

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Check if we have a winning move
    move = check_winning_move(board, 1)
    if move is not None:
        return move

    # Check if we need to block the opponent's winning move
    move = check_winning_move(board, -1)
    if move is not None:
        return move

    # If no immediate threats, we can try to play in the center or corners
    center = (1, 1)

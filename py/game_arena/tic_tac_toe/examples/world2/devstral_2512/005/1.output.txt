# Strategy for the 4x4 Tic Tac Toe Policy

This policy implements a smart AI player for 4x4 Tic Tac Toe using a combination of:
1. **Immediate Win Check**: First checks if there's a move that would win the game immediately
2. **Immediate Block**: Then checks if the opponent has a winning move that needs to be blocked
3. **Fork Creation**: Looks for opportunities to create multiple winning threats
4. **Center Control**: Prioritizes center positions for better control of the board
5. **Corner Control**: Prefers corners over edges when available
6. **Random Move**: Falls back to a random legal move if no strategic move is found

The policy uses a scoring system to evaluate potential moves, with higher priority given to moves that lead to immediate wins or blocks. The evaluation considers all possible winning lines (rows, columns, and both diagonals) in the 4x4 grid.

import random
import numpy as np

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Convert to numpy array for easier manipulation
    board = np.array(board)
    empty_cells = [(i, j) for i in range(4) for j in range(4) if board[i][j] == 0]

    # Check for immediate winning move
    for move in empty_cells:
        board[move[0], move[1]] = 1
        if is_winner(board, 1):
            return move
        board[move[0], move[1]] = 0

    # Check for opponent's immediate winning move to block
    for move in empty_cells:
        board[move[0], move[1]] = -1
        if is_winner(board, -1):
            return move
        board[move[0], move[1]] = 0

    # Try to create a fork (two winning threats)
    best_fork_score = -1
    best_fork_move = None
    for move in empty_cells:
        board[move[0], move[1]] = 1
        fork_score = count_potential_wins(board, 1)
        if fork_score > best_fork_score:
            best_fork_score = fork_score
            best_fork_move = move
        board[move[0], move[1]] = 0

    if best_fork_score >= 2:  # If we can create at least 2 winning threats
        return best_fork_move

    # Center control - prioritize center positions
    center_moves = [(1,1), (1,2), (2,1), (2,2)]
    for move in center_moves:
        if move in empty_cells:
            return move

    # Corner control - prioritize corners
    corner_moves = [(0,0), (0,3), (3,0), (3,3)]
    for move in corner_moves:
        if move in empty_cells:
            return move

    # If no strategic move found, return a random move
    return random.choice(empty_cells)

def is_winner(board: np.ndarray, player: int) -> bool:
    # Check rows
    for row in board:
        if np.all(row == player):
            return True

    # Check columns
    for col in range(4):
        if np.all(board[:, col] == player):
            return True

    # Check main diagonal
    if np.all(np.diag(board) == player):
        return True

    # Check anti-diagonal
    if np.all(np.diag(np.fliplr(board)) == player):
        return True

    return False

def count_potential_wins(board: np.ndarray, player: int) -> int:
    count = 0

    # Check rows
    for row in board:
        if np.sum(row == player) == 3 and np.sum(row == 0) == 1:
            count += 1

    # Check columns
    for col in range(4):
        column = board[:, col]
        if np.sum(column == player) == 3 and np.sum(column == 0) == 1:
            count += 1

    # Check main diagonal
    diag = np.diag(board)
    if np.sum(diag == player) == 3 and np.sum(diag == 0) == 1:
        count += 1

    # Check anti-diagonal
    anti_diag = np.diag(np.fliplr(board))
    if np.sum(anti_diag == player) == 3 and np.sum(anti_diag == 0) == 1:
        count += 1

    return count

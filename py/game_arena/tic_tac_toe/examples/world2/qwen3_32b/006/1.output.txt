The strategy for this policy is to first check for any immediate winning moves. If none, it checks for and blocks any opponent's potential winning moves. When neither is applicable, it selects the move that maximizes a heuristic score. This score is based on the number of 1s formed in rows, columns, and diagonals after the move, with higher values for more 1s. Additionally, positional bonuses are added for corners and center positions, prioritizing strategic locations. This approach balances offensive and defensive actions while favoring moves with high potential for future wins.

def policy(board: list[list[int]]) -> tuple[int, int]:
    def check_win(current_board, player):
        # Check rows
        for row in current_board:
            if all(cell == player for cell in row):
                return True
        # Check columns
        for c in range(4):
            if all(current_board[r][c] == player for r in range(4)):
                return True
        # Check main diagonals
        if all(current_board[i][i] == player for i in range(4)):
            return True
        if all(current_board[i][3 - i] == player for i in range(4)):
            return True
        return False

    moves = [(r, c) for r in range(4) for c in range(4) if board[r][c] == 0]

    # Check for winning move
    for r, c in moves:
        new_board = [row[:] for row in board]
        new_board[r][c] = 1
        if check_win(new_board, 1):
            return (r, c)

    # Check for blocking moves
    blocking_candidates = []
    for r, c in moves:
        new_board = [row[:] for row in board]
        new_board[r][c] = -1
        if check_win(new_board, -1):
            # Calculate score if we take (r,c)
            temp_for_score = [row[:] for row in board]
            temp_for_score[r][c] = 1
            score = 0
            # Evaluate row
            row_count = sum(1 for cell in temp_for_score[r] if cell == 1)
            score += row_count ** 2
            # Evaluate column
            col_count = sum(1 for i in range(4) if temp_for_score[i][c] == 1)
            score += col_count ** 2
            # Evaluate main diagonal
            if r == c:
                diag = [temp_for_score[i][i] for i in range(4)]
                diag_count = sum(1 for cell in diag if cell == 1)
                score += diag_count ** 2
            # Evaluate anti-diagonal
            if r + c == 3:
                diag = [temp_for_score[i][3 - i] for i in range(4)]
                diag_count = sum(1 for cell in diag if cell == 1)
                score += diag_count ** 2
            # Add positional bonus
            if (r, c) in [(0, 0), (0, 3), (3, 0), (3, 3)]:
                score += 5
            elif (r, c) in [(1, 1), (1, 2), (2, 1), (2, 2)]:
                score += 3
            blocking_candidates.append((-score, r, c))
    if blocking_candidates:
        blocking_candidates.sort()
        return (blocking_candidates[0][1], blocking_candidates[0][2])

    # Evaluate all moves with heuristic and choose best
    scored_moves = []
    for r, c in moves:
        temp_for_score = [row[:] for row in board]
        temp_for_score[r][c] = 1
        score = 0
        # Evaluate row
        row_count = sum(1 for cell in temp_for_score[r] if cell == 1)
        score += row_count ** 2
        # Evaluate column
        col_count = sum(1 for i in range(4) if temp_for_score[i][c] == 1)
        score += col_count ** 2
        # Evaluate main diagonal
        if r == c:
            diag = [temp_for_score[i][i] for i in range(4)]
            diag_count = sum(1 for cell in diag if cell == 1)
            score += diag_count ** 2
        # Evaluate anti-diagonal
        if r + c == 3:
            diag = [temp_for_score[i][3 - i] for i in range(4)]

            score += diag_count ** 2

        if (r, c) in [(0, 0), (0, 3), (3, 0), (3, 3)]:
            score += 5
        elif (r, c) in [(1, 1), (1, 2), (2, 1), (2, 2)]:
            score += 3
        scored_moves.append((-score, r, c))
    scored_moves.sort()
    return (scored_moves[0][1], scored_moves[0][2])

def policy(board: list[list[int]]) -> tuple[int, int]:
    # Check for immediate win
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                board[r][c] = 1
                if check_win(board, 1):
                    board[r][c] = 0  # undo
                    return (r, c)
                board[r][c] = 0  # undo
    # Check for opponent's immediate win and block
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                board[r][c] = -1
                if check_win(board, -1):
                    board[r][c] = 0  # undo
                    return (r, c)
                board[r][c] = 0  # undo
    # No immediate win or block, choose best move based on heuristic
    moves = [(r, c) for r in range(4) for c in range(4) if board[r][c] == 0]
    best_score = -float('inf')
    best_moves = []
    for move in moves:
        score = evaluate_move(board, move)
        if score > best_score:
            best_score = score
            best_moves = [move]
        elif score == best_score:
            best_moves.append(move)
    # Prioritize center, then corners, then edges
    def prioritize(move):
        r, c = move
        if (r in [1,2]) and (c in [1,2]):
            return 0
        if (r, c) in [(0,0), (0,3), (3,0), (3,3)]:
            return 1
        return 2
    best_moves.sort(key=prioritize)
    return best_moves[0]

def check_win(board, player):
    # Check rows
    for row in board:
        if all(x == player for x in row):
            return True
    # Check columns
    for c in range(4):
        if all(board[r][c] == player for r in range(4)):
            return True
    # Check diagonals
    if all(board[i][i] == player for i in range(4)):
        return True
    if all(board[i][3-i] == player for i in range(4)):
        return True
    return False

def evaluate_move(board, move):
    r, c = move
    board[r][c] = 1
    score = evaluate_board(board)
    board[r][c] = 0
    return score

def evaluate_board(board):
    my_score = 0
    opponent_score = 0
    # Check rows
    for row in board:
        my_score += calculate_line_score(row, 1)
        opponent_score += calculate_line_score(row, -1)
    # Check columns
    for c in range(4):
        column = [board[r][c] for r in range(4)]
        my_score += calculate_line_score(column, 1)
        opponent_score += calculate_line_score(column, -1)
    # Check main diagonal
    diag = [board[i][i] for i in range(4)]
    my_score += calculate_line_score(diag, 1)
    opponent_score += calculate_line_score(diag, -1)
    # Check anti-diagonal
    diag = [board[i][3-i] for i in range(4)]
    my_score += calculate_line_score(diag, 1)
    opponent_score += calculate_line_score(diag, -1)
    return my_score - opponent_score

def calculate_line_score(line, player):
    count = line.count(player)
    if count == 4:
        return 1000
    elif count == 3:
        return 100
    elif count == 2:
        return 10
    elif count == 1:
        return 1
    else:
        return 0

def get_all_lines(board: list[list[int]]) -> list[list[tuple[int, int]]]:
    # Get all rows
    rows = [[(row, col) for col in range(4)] for row in range(4)]
    # Get all columns
    columns = [[(row, col) for row in range(4)] for col in range(4)]
    # Get main diagonals (top-left to bottom-right and top-right to bottom-left)
    diagonal1 = [(i, i) for i in range(4)]
    diagonal2 = [(i, 3-i) for i in range(4)]
    # Get diagonals of length 3
    diagonals_length3 = [
        [(0,1), (1,2), (2,3)],  # offset 1
        [(0,2), (1,3)],         # offset 2
        [(1,0), (2,1), (3,2)],  # offset 1, other direction
        [(2,0), (3,1)],         # offset 2, other direction
        [(0,2), (1,1), (2,0)],  # offset 1, other direction (top-right to bottom-left)
        [(1,3), (2,2), (3,1)],  # offset 1, other direction
        [(0,1), (1,0)],         # offset 1, other direction (but length 2)
        [(1,2), (2,1)],         # offset 1, other direction
        [(2,3), (3,2)],         # offset 1, other direction
        [(0,3), (1,2), (2,1), (3,0)]  # This is the same as diagonal2, already included
    ]
    # Filter to only include diagonals of length 3 or 4
    # Actually, the above includes some length 2, so let's correct:
    diagonals_length3 = [
        [(0,1), (1,2), (2,3)],  # length 3
        [(1,0), (2,1), (3,2)],  # length 3
        [(0,2), (1,3)],         # length 2 (but we need length 3 or 4)
        [(2,0), (3,1)],         # length 2
        [(0,2), (1,1), (2,0)],  # length 3
        [(1,3), (2,2), (3,1)],  # length 3
        [(0,3), (1,2), (2,1), (3,0)]  # length 4 (same as diagonal2)
    ]
    # Actually, the diagonals of length 3 are:
    # For top-left to bottom-right direction:
    # (0,1)-(1,2)-(2,3)
    # (1,0)-(2,1)-(3,2)
    # For top-right to bottom-left direction:
    # (0,2)-(1,1)-(2,0)
    # (1,3)-(2,2)-(3,1)
    diagonals_length3 = [
        [(0,1), (1,2), (2,3)],  # length 3, top-left to bottom-right
        [(1,0), (2,1), (3,2)],  # length 3, top-left to bottom-right
        [(0,2), (1,1), (2,0)],  # length 3, top-right to bottom-left
        [(1,3), (2,2), (3,1)]   # length 3, top-right to bottom-left
    ]
    # Also, diagonals of length 4 are already included as main diagonals
    # So total lines: rows + columns + main diagonals + diagonals of length 3
    all_lines = rows + columns + [diagonal1, diagonal2] + diagonals_length3
    return all_lines

def would_win_or_block(board: list[list[int]], lines: list[list[tuple[int, int]]]) -> bool:
    # Check if placing 1 would complete a line of four or block opponent's line of four
    # For winning: any line has three 1s and one 0 where we placed the 1
    # For blocking: any line has three -1s and one 0 where we placed the 1
    for line in lines:
        count_1 = sum(1 for (row, col) in line if board[row][col] == 1)
        count_neg1 = sum(1 for (row, col) in line if board[row][col] == -1)
        count_0 = sum(1 for (row, col) in line if board[row][col] == 0)
        if count_0 == 1 and count_1 == 3:
            return True  # would win
        if count_0 == 1 and count_neg1 == 3:
            return True  # would block
    return False

### Explanation
1. **`policy` function**:
   - Gets all possible lines (rows, columns, diagonals) on the board.
   - Checks each empty spot to see if placing a `1` there would complete a line of four or block the opponent.
   - If no such move is found, it picks a random empty spot.

2. **`get_all_lines` function**:
   - Returns a list of all possible lines (rows, columns, diagonals) on the board.
   - Includes rows, columns, main diagonals, and diagonals of length 3.

3. **`would_win_or_block` function**:
   - Checks if placing a `1` in any of the lines would result in three `1`s and one `0` (winning move) or three `-1`s and one `0` (blocking move).
   - Returns `True` if such a move is found.

This approach should be efficient enough to run within the 1-second time limit while providing a reasonable strategy for the game. If the opponent has more complex strategies, a minimax algorithm might be better, but given the time constraints, this greedy approach is a good start.

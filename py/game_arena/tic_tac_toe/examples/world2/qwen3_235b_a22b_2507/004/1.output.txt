def is_winner(board: np.ndarray, player: int) -> bool:
    # Check rows
    for i in range(4):
        if np.all(board[i, :] == player):
            return True
    
    # Check columns
    for j in range(4):
        if np.all(board[:, j] == player):
            return True
    
    # Check diagonals
    if np.all(np.diag(board) == player):
        return True
    if np.all(np.diag(np.fliplr(board)) == player):
        return True
    
    return False

def evaluate(board: np.ndarray) -> int:
    score = 0
    # Check all rows
    for i in range(4):
        score += evaluate_line(board[i, :])
    
    # Check all columns
    for j in range(4):
        score += evaluate_line(board[:, j])
    
    # Check diagonals
    score += evaluate_line(np.diag(board))
    score += evaluate_line(np.diag(np.fliplr(board)))
    
    return score

def evaluate_line(line: np.ndarray) -> int:
    if 1 in line and -1 in line:  # Both players in line, can't win
        return 0
    
    count = np.sum(line == 1)  # Our marks
    opp_count = np.sum(line == -1)  # Opponent marks
    
    if count > 0:
        # More marks = higher value
        return 10 ** count
    elif opp_count > 0:
        # Opponent's potential line
        return -(10 ** opp_count)
    
    return 0

def minimax(board: np.ndarray, depth: int, is_maximizing: bool, alpha: float, beta: float) -> float:

    if len(empty_positions) == 0:

            eval_score = minimax(board, depth + 1, False, alpha, beta)
            board[i, j] = 0
            max_eval = max(max_eval, eval_score)
            alpha = max(alpha, eval_score)

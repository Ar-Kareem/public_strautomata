To win a 4x4 Tic Tac Toe game (standard 4-in-a-row), I need a strategy that balances offensive and defensive play. Since a 4x4 board has many states, a full minimax might be tight on time depending on the depth, but a depth-limited search with a strong heuristic or a simplified win/block check is highly effective. 

Strategy:
1. Check if I can win in one move.
2. Check if the opponent can win in one move and block them.
3. Check for "forks" or strategic center/power positions.
4. Use a Negamax algorithm with Alpha-Beta pruning to a depth of 4 or 5 to ensure quick response and solid play.

import random

def policy(board: list[list[int]]) -> tuple[int, int]:
    my_id = 1
    opp_id = -1
    size = 4

    def get_lines(b):
        lines = []
        # Rows
        for r in range(size):
            lines.append(b[r])
        # Columns
        for c in range(size):
            lines.append([b[r][c] for r in range(size)])
        # Diagonals
        lines.append([b[i][i] for i in range(size)])
        lines.append([b[i][size - 1 - i] for i in range(size)])
        return lines

    def evaluate_line(line):
        my_count = line.count(my_id)
        opp_count = line.count(opp_id)
        if my_count == 4: return 1000
        if opp_count == 4: return -1000
        if my_count > 0 and opp_count > 0: return 0
        if my_count > 0: return 10 ** (my_count - 1)
        if opp_count > 0: return -(10 ** (opp_count - 1))
        return 0

    def heuristic(b):
        score = 0
        for line in get_lines(b):
            score += evaluate_line(line)
        return score

    def get_moves(b):
        moves = []
        # Prioritize center for better branching
        order = [(1,1), (1,2), (2,1), (2,2), (0,0), (0,3), (3,0), (3,3),
                 (0,1), (0,2), (1,0), (1,3), (2,0), (2,3), (3,1), (3,2)]
        for r, c in order:
            if b[r][c] == 0:
                moves.append((r, c))
        return moves

    def is_win(b, p):
        for line in get_lines(b):
            if line.count(p) == 4:
                return True
        return False

    # 1. Immediate Win
    moves = get_moves(board)
    for r, c in moves:
        board[r][c] = my_id
        if is_win(board, my_id):
            return (r, c)
        board[r][c] = 0

    # 2. Immediate Block
    for r, c in moves:
        board[r][c] = opp_id
        if is_win(board, opp_id):
            return (r, c)
        board[r][c] = 0

    # 3. Simple Minimax with depth 3 to stay within 1s
    def minimax(b, depth, alpha, beta, maximizing):
        if is_win(b, my_id): return 1000 + depth
        if is_win(b, opp_id): return -1000 - depth
        if depth == 0 or not get_moves(b):
            return heuristic(b)

        if maximizing:
            max_eval = -float('inf')
            for r, c in get_moves(b):
                b[r][c] = my_id
                eval = minimax(b, depth - 1, alpha, beta, False)
                b[r][c] = 0
                max_eval = max(max_eval, eval)
                alpha = max(alpha, eval)
                if beta <= alpha: break
            return max_eval
        else:
            min_eval = float('inf')
            for r, c in get_moves(b):
                b[r][c] = opp_id
                eval = minimax(b, depth - 1, alpha, beta, True)
                b[r][c] = 0
                min_eval = min(min_eval, eval)
                beta = min(beta, eval)
                if beta <= alpha: break
            return min_eval

    best_val = -float('inf')
    best_move = moves[0]
    
    # Use depth 3 for 4x4 to ensure speed
    for r, c in moves:
        board[r][c] = my_id
        val = minimax(board, 3, -float('inf'), float('inf'), False)
        board[r][c] = 0
        if val > best_val:
            best_val = val
            best_move = (r, c)

    return best_move

def check_win(board, player):
    # Check rows
    for i in range(4):
        if all(board[i][j] == player for j in range(4)):
            return True
    
    # Check columns
    for j in range(4):
        if all(board[i][j] == player for i in range(4)):

        return True

    if all(board[i][3-i] == player for i in range(4)):

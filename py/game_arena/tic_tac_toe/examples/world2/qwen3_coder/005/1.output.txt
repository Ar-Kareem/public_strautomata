def check_winner(board: List[List[int]]) -> Optional[int]:
    """Check if there's a winner. Returns 1 for player, -1 for opponent, 0 for tie, None for no winner."""
    # Check rows
    for row in board:
        if abs(sum(row)) == 4 and abs(sum(row)) != 0:
            if sum(row) == 4:
                return 1
            elif sum(row) == -4:
                return -1
    
    # Check columns
    for col in range(4):
        col_sum = sum(board[row][col] for row in range(4))
        if abs(col_sum) == 4:
            if col_sum == 4:
                return 1
            elif col_sum == -4:
                return -1
    
    # Check diagonals
    diag1_sum = sum(board[i][i] for i in range(4))
    if abs(diag1_sum) == 4:
        if diag1_sum == 4:
            return 1
        elif diag1_sum == -4:
            return -1

    if abs(diag2_sum) == 4:
        if diag2_sum == 4:
            return 1

            return -1
    
    # Check for tie (board full)
    if all(board[i][j] != 0 for i in range(4) for j in range(4)):
        return 0
    
    return None

def get_empty_positions(board: List[List[int]]) -> List[Tuple[int, int]]:
    """Get all empty positions on the board."""
    empty = []
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                empty.append((i, j))
    return empty

    """Evaluate board position for a player."""

def minimax(board: List[List[int]], depth: int, is_maximizing: bool, alpha: int, beta: int) -> int:
    """Minimax algorithm with alpha-beta pruning."""
    winner = check_winner(board)
    if winner is not None:
        if winner == 1:  # Player wins
            return 10000
        elif winner == -1:  # Opponent wins
            return -10000
        else:  # Tie
            return 0
    
    if depth == 0:
        return evaluate_position(board, 1)
    
    empty_positions = get_empty_positions(board)
    if not empty_positions:
        return 0
    
    if is_maximizing:
        max_eval = -float('inf')

            board[i][j] = 1
            eval_score = minimax(board, depth - 1, False, alpha, beta)
            board[i][j] = 0
            max_eval = max(max_eval, eval_score)

        return min_eval

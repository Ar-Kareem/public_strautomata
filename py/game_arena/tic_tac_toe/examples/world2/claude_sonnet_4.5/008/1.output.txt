def policy(board: list[list[int]]) -> tuple[int, int]:
    """
    Policy for 4x4 Tic Tac Toe
    Returns the best move as (row, col)
    """
    
    def check_line(line, player):
        """Check if a line has potential for player to win"""
        count = sum(1 for x in line if x == player)
        empty = sum(1 for x in line if x == 0)
        opponent = sum(1 for x in line if x == -player)
        return count, empty, opponent
    
    def get_all_lines():
        """Get all possible winning lines (rows, cols, diagonals)"""
        lines = []
        
        # Rows
        for i in range(4):
            lines.append([(i, j) for j in range(4)])
        
        # Columns
        for j in range(4):
            lines.append([(i, j) for i in range(4)])
        
        # Diagonals
        lines.append([(i, i) for i in range(4)])
        lines.append([(i, 3-i) for i in range(4)])
        
        return lines
    
    def evaluate_move(row, col):
        """Evaluate a move's strategic value"""
        if board[row][col] != 0:
            return -1000
        
        score = 0
        temp_board = [r[:] for r in board]
        temp_board[row][col] = 1
        
        lines = get_all_lines()
        
        for line in lines:
            if (row, col) not in line:
                continue
            
            line_vals = [temp_board[r][c] for r, c in line]
            my_count = sum(1 for x in line_vals if x == 1)
            opp_count = sum(1 for x in line_vals if x == -1)
            
            # Win immediately
            if my_count == 4:
                score += 100000
            # Three in a row with no opponent pieces
            elif my_count == 3 and opp_count == 0:
                score += 5000
            # Two in a row with no opponent pieces
            elif my_count == 2 and opp_count == 0:
                score += 100
            # One with potential
            elif my_count == 1 and opp_count == 0:
                score += 10
        
        # Check center positions (more strategic)
        if (row, col) in [(1, 1), (1, 2), (2, 1), (2, 2)]:
            score += 50
        
        return score
    
    def find_winning_move(player):
        """Find if there's a winning move for player"""
        lines = get_all_lines()
        
        for line in lines:
            line_vals = [board[r][c] for r, c in line]
            count = sum(1 for x in line_vals if x == player)
            empty_count = sum(1 for x in line_vals if x == 0)
            opponent = sum(1 for x in line_vals if x == -player)
            
            # 3 in a row with 1 empty and no opponent
            if count == 3 and empty_count == 1 and opponent == 0:
                for r, c in line:
                    if board[r][c] == 0:
                        return (r, c)
        
        return None
    
    # 1. Check if we can win
    win_move = find_winning_move(1)
    if win_move:
        return win_move
    
    # 2. Check if we need to block opponent's win
    block_move = find_winning_move(-1)
    if block_move:
        return block_move
    
    # 3. Find best strategic move
    best_score = -float('inf')
    best_move = None
    
    for i in range(4):
        for j in range(4):
            if board[i][j] == 0:
                score = evaluate_move(i, j)
                if score > best_score:
                    best_score = score
                    best_move = (i, j)
    
    # Fallback: return first empty cell
    if best_move is None:
        for i in range(4):
            for j in range(4):
                if board[i][j] == 0:
                    return (i, j)
    
    return best_move

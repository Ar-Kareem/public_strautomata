

The strategy for this policy is toprioritize winning moves, blocking opponent's winning moves, then placing in the center, corners, and any available cell. This approach ensures optimal play by capitalizing on immediate opportunities while maintaining strategic positioning.

def policy(board):
    # Define all possible lines of three cells
    lines = []
    # Rows
    for i in range(4):
        lines.append([(i,0), (i,1), (i,2)])
        lines.append([(i,1), (i,2), (i,3)])
    # Columns
    for j in range(4):
        lines.append([(0,j), (1,j), (2,j)])
        lines.append([(1,j), (2,j), (3,j)])
    # Diagonals (main direction: top-left to bottom-right)
    for i in range(2):
        for j in range(2):
            lines.append([(i,j), (i+1,j+1), (i+2,j+2)])
    # Diagonals (other direction: top-right to bottom-left)
    for i in range(2):
        for j in range(2,4):
            lines.append([(i,j), (i+1,j-1), (i+2,j-2)])
    
    # Check for winning move or blocking move
    for line in lines:
        count_my = 0
        count_opp = 0
        empty_cell = None
        for cell in line:
            r, c = cell
            if board[r][c] == 1:
                count_my += 1
            elif board[r][c] == -1:
                count_opp += 1
            else:
                empty_cell = cell
        if count_my == 2 and empty_cell is not None:
            return empty_cell
        if count_opp == 2 and empty_cell is not None:
            return empty_cell
    
    # Check center cells
    center_cells = [(1,1), (1,2), (2,1), (2,2)]
    for cell in center_cells:
        r, c = cell
        if board[r][c] == 0:
            return cell
    
    # Check corners
    corners = [(0,0), (0,3), (3,0), (3,3)]
    for cell in corners:
        r, c = cell
        if board[r][c] == 0:
            return cell
    
    # Check any other cell
    for r in range(4):
        for c in range(4):
            if board[r][c] == 0:
                return (r, c)

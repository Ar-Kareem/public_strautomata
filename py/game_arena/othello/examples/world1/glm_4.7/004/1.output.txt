The strategy for this policy is based on an Alpha-Beta pruning Minimax algorithm with Iterative Deepening. It uses bitboard representation for high-performance move generation and state manipulation. The evaluation function combines a weighted positional matrix (prioritizing corners and avoiding X-squares) with disc count parity. Moves are ordered by their static positional value to maximize pruning efficiency. The search iterates deeper until the time limit (approx. 1 second) is reached, ensuring the best move found within the allotted time is returned.

```python
import numpy as np
import time

# --- Constants & Precomputed Data ---

# Directions for bitboard shifts: E, SE, S, SW, W, NW, N, NE
# Indices: 0=A1, 7=H1, 8=A2, ..., 63=H8 (Row-major: r*8 + c)
DIRECTIONS = [1, 9, 8, 7, -1, -9, -8, -7]

# Masks to prevent board wrap-around
MASKS = {
    1:  0x7F7F7F7F7F7F7F7F,  # E (exclude H file / col 7)
    9:  0x7F7F7F7F7F7F7F7F,  # SE (exclude H file)
    8:  0xFFFFFFFFFFFFFFFF,  # S (all)
    7:  0xFEFEFEFEFEFEFEFE,  # SW (exclude A file / col 0)
    -1: 0xFEFEFEFEFEFEFEFE,  # W (exclude A file)
    -9: 0xFEFEFEFEFEFEFEFE,  # NW (exclude A file)
    -8: 0xFFFFFFFFFFFFFFFF,  # N (all)
    -7: 0x7F7F7F7F7F7F7F7F   # NE (exclude H file)
}

# Positional weights (flattened 8x8). 
# Corners (0, 7, 56, 63) are high value.
# Squares adjacent to corners (C-squares and X-squares) are negative.
WEIGHTS = np.array([
    100, -20,  10,   5,   5,  10, -20, 100, # Row 0 (Rank 1)
    -20, -50,  -2,  -2,  -2,  -2, -50, -20, # Row 1
     10,  -2,  -1,  -1,  -1,  -1,  -2,  10, # Row 2
      5,  -2,  -1,  -1,  -1,  -1,  -2,   5, # Row 3
      5,  -2,  -1,  -1,  -1,  -1,  -2,   5, # Row 4
     10,  -2,  -1,  -1,  -1,  -1,  -2,  10, # Row 5
    -20, -50,  -2,  -2,  -2,  -2, -50, -20, # Row 6
    100, -20,  10,   5,   5,  10, -20, 100  # Row 7 (Rank 8)
], dtype=np.int32)

# Precompute powers of 2 for fast board conversion
# 2**0 ... 2**63
POWERS_OF_2 = 1 << np.arange(64, dtype=np.uint64)

# --- Helper Functions ---

def board_to_int(board: np.ndarray) -> int:
    """Convert 8x8 numpy board to 64-bit integer."""
    return int(np.dot(board.flatten(), POWERS_OF_2))

def shift_mask(board: int, direction: int) -> int:
    """Shift board bit representation with wrap-around masking."""
    if direction > 0:
        return (board << direction) & MASKS[direction]
    else:
        return (board >> -direction) & MASKS[direction]

def get_moves(me: int, opp: int) -> int:
    """Generate a bitboard of legal moves for `me` against `opp`."""
    moves = 0
    empty = ~(me | opp) & 0xFFFFFFFFFFFFFFFF
    
    for d in DIRECTIONS:
        # Find opponent pieces adjacent to me in direction d
        candidates = opp & shift_mask(me, d)
        
        # Extend the line of opponent pieces
        if candidates:
            for _ in range(6): # Max line length on 8x8 is 7
                # If the square after the line of opponents is empty, it's a move
                moves |= empty & shift_mask(candidates, d)
                # Extend candidates further
                candidates = opp & shift_mask(candidates, d)
                if candidates == 0:
                    break
    return moves

def do_move(me: int, opp: int, move_sq: int) -> tuple[int, int]:
    """Apply move. Returns (new_me, new_opp)."""
    new_me = me | move_sq
    flips = 0
    
    for d in DIRECTIONS:
        # Check direction for captures
        line = opp & shift_mask(move_sq, d)
        if not line:
            continue
        
        # Propagate to see if line is bracketed by another `me` piece
        temp_line = line
        while True:
            shifted = shift_mask(temp_line, d)
            if shifted & me:
                flips |= line # Valid bracket
                break
            if shifted & opp:
                line |= shifted
                temp_line = line
            else:
                break # Hit empty square, invalid
    
    new_opp = opp ^ flips
    new_me |= flips # Add flipped discs to me
    return new_me, new_opp

def evaluate(me: int, opp: int) -> int:
    """
    Static board evaluation.
    Returns a positive score if `me` is winning.
    """
    # 1. Disc count difference
    # Using built-in bit_count (Python 3.8+)
    score = (me.bit_count() - opp.bit_count()) 
    
    # 2. Positional weights
    # Iterate over set bits is faster than looping 64 times
    pos_score = 0
    m = me
    while m:
        lsb = m & -m
        idx = (lsb.bit_length() - 1)
        pos_score += WEIGHTS[idx]
        m ^= lsb
        
    m = opp
    while m:
        lsb = m & -m
        idx = (lsb.bit_length() - 1)
        pos_score -= WEIGHTS[idx]
        m ^= lsb
        
    # Weight disc count lower than positional weights early/mid game, 
    # but positional weights handle the logic implicitly.
    # However, we combine them.
    return pos_score + score

# --- Search Algorithm ---

def solve(me: int, opp: int, depth: int, alpha: int, beta: int) -> int:
    """
    Minimax with Alpha-Beta pruning.
    Returns the evaluation score from the perspective of `me`.
    """
    # Generate moves
    moves = get_moves(me, opp)
    
    # Terminal condition or depth limit
    if depth == 0:
        return evaluate(me, opp)
        
    if moves == 0:
        # If I have no moves, check if opponent has moves
        if get_moves(opp, me) == 0:
            # Game over
            return evaluate(me, opp)
        else:
            # Pass turn
            return -solve(opp, me, depth - 1, -beta, -alpha)

    # Move Ordering: prioritize corners and edges implicitly via weights
    # Sort moves by a quick static eval of just the position (no recursion)
    move_list = []
    m = moves
    while m:
        sq = m & -m
        idx = sq.bit_length() - 1
        move_list.append(sq)
        m ^= sq
        
    # Simple sort by weight of the square (corners first)
    move_list.sort(key=lambda sq: -WEIGHTS[sq.bit_length() - 1])

    max_eval = -100000
    for sq in move_list:
        n_me, n_opp = do_move(me, opp, sq)
        eval_score = -solve(n_opp, n_me, depth - 1, -beta, -alpha)
        
        if eval_score > max_eval:
            max_eval = eval_score
            
        if max_eval >= beta:
            return max_eval # Beta cutoff
            
        if max_eval > alpha:
            alpha = max_eval
            
    return max_eval

# --- Policy API ---

def policy(you: np.ndarray, opponent: np.ndarray) -> str:
    start_time = time.time()
    
    # Convert to bitboards
    me = board_to_int(you)
    opp = board_to_int(opponent)
    
    # Get all legal moves
    legal_moves = get_moves(me, opp)
    
    if legal_moves == 0:
        return "pass"
    
    # Convert moves to list for iteration
    move_candidates = []
    m = legal_moves
    while m:
        sq = m & -m
        move_candidates.append(sq)
        m ^= m & -m # m ^= lsb is same
        
    # Sort candidates for the root move ordering (consistent with solve)
    move_candidates.sort(key=lambda sq: -WEIGHTS[sq.bit_length() - 1])
    
    best_move = move_candidates[0] # Default to first valid move
    best_score = -100000
    
    # Iterative Deepening with Time Management
    # We aim to use up to 0.95 seconds to ensure we return a result
    limit = 0.95 
    
    for depth in range(1, 10): # Search deeper and deeper
        if time.time() - start_time > limit:
            break
            
        alpha = -100000
        beta = 100000
        
        for sq in move_candidates:
            n_me, n_opp = do_move(me, opp, sq)
            
            # Use fail-soft alpha-beta
            val = -solve(n_opp, n_me, depth - 1, -beta, -alpha)
            
            if val > best_score:
                best_score = val
                best_move = sq
                
            if val > alpha:
                alpha = val
                
            # Optimization: if we found a winning move (corner capture), we can stop early?
            # Not always safe, but if score is huge, maybe.
            # But we continue IDDFS to be safe.
            
    # Convert best_move bitboard square to algebraic notation
    # index 0 -> a1, index 7 -> h1, index 8 -> a2
    # col = index % 8, row = index // 8
    idx = best_move.bit_length() - 1
    c = idx % 8
    r = idx // 8
    
    file_char = chr(ord('a') + c)
    rank_char = str(r + 1)
    
    return f"{file_char}{rank_char}"
```

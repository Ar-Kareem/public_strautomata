The strategy for this policy is based on an **Iterative Deepening Minimax** search with **Alpha-Beta Pruning**, optimized using **Bitboards** for extremely fast state manipulation and legal move generation.

Key components of the strategy include:
1.  **State Representation**: The board is converted from numpy arrays to two 64-bit integers. This allows calculating legal moves and applying updates using bitwise operations rather than slow loops, enabling the agent to search much deeper within the 1-second limit.
2.  **Move Generation**: A custom bitwise algorithm identifies legal moves by simulating disc flips in parallel across all 8 directions.
3.  **Evaluation Function**: The board state is evaluated using a combination of:
    *   **Positional Weights**: Prioritizing corners (stable discs) and penalizing "X-squares" (squares diagonally adjacent to corners) which often surrender corners to the opponent.
    *   **Mobility**: Incentivizing states where the agent has many legal moves and limiting the opponent's options.
    *   **Frontier**: Minimizing the number of discs adjacent to empty squares to prevent the opponent from finding easy capture lines.
4.  **Time Management**: The search uses iterative deepening, starting at depth 1 and increasing, stopping immediately if the 1-second time allocation is nearly exhausted to ensure a move is always returned.

import numpy as np
import time

def policy(you: np.ndarray, opponent: np.ndarray) -> str:
    """
    Othello policy that uses iterative deepening minimax with alpha-beta pruning
    and bitboard optimizations.
    """
    
    # --- BITBOARD CONSTANTS & HELPERS ---
    MASK_64 = 0xFFFFFFFFFFFFFFFF
    NOT_A_COL = 0xFEFEFEFEFEFEFEFE  # Mask excluding column A (0)
    NOT_H_COL = 0x7F7F7F7F7F7F7F7F  # Mask excluding column H (7)
    
    # Direction offsets and their required boundary masks
    # (Shift Amount, Source Mask)
    # Right (+1), Left (-1), Down (+8), Up (-8)...
    # When moving Right (+1), we must mask out H col to prevent wrap to A (next row)
    DIRECTIONS = [
        (1, NOT_H_COL),   # East
        (-1, NOT_A_COL),  # West
        (8, MASK_64),     # South
        (-8, MASK_64),    # North
        (9, NOT_H_COL),   # South-East
        (-9, NOT_A_COL),  # North-West
        (7, NOT_A_COL),   # South-West
        (-7, NOT_H_COL)   # North-East
    ]
    
    # Static weights for positional evaluation
    # Shape 8x8 flattened
    WEIGHTS = [
        100, -20,  10,   5,   5,  10, -20, 100,
        -20, -50,  -2,  -2,  -2,  -2, -50, -20,
         10,  -2,  -1,  -1,  -1,  -1,  -2,  10,
          5,  -2,  -1,  -1,  -1,  -1,  -2,   5,
          5,  -2,  -1,  -1,  -1,  -1,  -2,   5,
         10,  -2,  -1,  -1,  -1,  -1,  -2,  10,
        -20, -50,  -2,  -2,  -2,  -2, -50, -20,
        100, -20,  10,   5,   5,  10, -20, 100,
    ]

    def np_to_bitboard(arr):
        """Convert 8x8 numpy array to 64-bit integer."""
        flattened = arr.flatten()
        # Create powers of 2 array: [1, 2, 4, ..., 2^63]
        powers = (1 << np.arange(64, dtype=object))
        # Dot product to get integer
        return int(np.dot(flattened, powers))

    def get_legal_moves_mask(my_discs, op_discs):
        """Returns a bitmask where 1s are legal move positions."""
        empty = ~(my_discs | op_discs) & MASK_64
        legal_moves = 0
        
        for shift, mask in DIRECTIONS:
            # We want to bracket opp discs: Own + Opp + ... + Empty
            # First step: Shift own discs into opponent territory
            if shift > 0:
                candidates = (my_discs & mask) << shift
            else:
                candidates = (my_discs & mask) >> abs(shift)
            
            candidates &= op_discs # Must land on opponent
            
            # Slide through contiguous opponent pieces (max 6 times)
            while candidates:
                if shift > 0:
                    next_step = (candidates & mask) << shift
                else:
                    next_step = (candidates & mask) >> abs(shift)
                
                # Check which candidates found an empty spot (valid move)
                potential_moves = next_step & empty
                legal_moves |= potential_moves
                
                # Setup for next iteration: stay on opponent discs
                candidates = next_step & op_discs
                
        return legal_moves

    def make_move(my_discs, op_discs, move_idx):
        """Applies a move at move_idx (0-63). Returns (new_my_discs, new_op_discs)."""
        move_bit = 1 << move_idx
        flipped_discs = 0
        
        for shift, mask in DIRECTIONS:
            captured_in_dir = 0
            candidate_bit = move_bit
            
            # Scan in direction
            while True:
                if shift > 0:
                    candidate_bit = (candidate_bit & mask) << shift
                else:
                    candidate_bit = (candidate_bit & mask) >> abs(shift)
                
                if not (candidate_bit & op_discs):
                    break
                captured_in_dir |= candidate_bit
            
            # If line ends with my disc, valid capture
            if candidate_bit & my_discs:
                flipped_discs |= captured_in_dir

        new_my = my_discs | move_bit | flipped_discs
        new_op = op_discs & ~flipped_discs
        return new_my, new_op

    def count_bits(n):
        """Counts set bits in integer."""
        return bin(n).count('1')

    def evaluate(my_discs, op_discs):
        """Heuristic evaluation function."""
        my_pop = count_bits(my_discs)
        op_pop = count_bits(op_discs)
        
        # 1. Mobility (Number of legal moves)
        my_moves = get_legal_moves_mask(my_discs, op_discs)
        op_moves = get_legal_moves_mask(op_discs, my_discs)
        my_mob = count_bits(my_moves)
        op_mob = count_bits(op_moves)
        
        if my_pop + op_pop == 64:  # Exact end game
            return (my_pop - op_pop) * 1000
        
        # 2. Position Weights
        score = 0
        
        # Simple bit iteration for position weights
        # (Optimized: we iterate 0..63 and check bits)
        # To make this fast, we can't loop in Python too much.
        # But looping 64 times is faster than complex logic in python.
        # Let's assess occupied squares.
        
        # A faster way for weights:
        # We can iterate only the set bits if population is low, 
        # or iterate 64 and branch. 
        # Given Python overhead, checking all 64 is acceptable.
        
        # Actually, let's just do weighted sum using bitmask logic?
        # No, iterating is safer for arbitrary weights list.
        # Or... (my_discs >> i) & 1.
        
        current_my = my_discs
        current_op = op_discs
        idx = 0
        while (current_my | current_op) and idx < 64:
            w = WEIGHTS[idx]
            if (current_my & 1):
                score += w
            elif (current_op & 1):
                score -= w
            current_my >>= 1
            current_op >>= 1
            idx += 1
            
        # 3. Mobility Score
        # Early game: Mobility is king.
        # Late game: Disc count matters.
        if (my_pop + op_pop) < 20:
            score += (my_mob - op_mob) * 10
        else:
            score += (my_mob - op_mob) * 5
            
        # 4. Corner Closeness Adjustment (Heuristic Fixes)
        # If corner is empty, penalize X-squares heavily
        # Corners: 0, 7, 56, 63
        # X-squares: 9, 14, 49, 54
        corners_indices = [0, 7, 56, 63]
        x_indices = [9, 14, 49, 54]
        
        # This is partially handled by weights, but conditional penalties are better
        # If corner not taken...
        # corner 0 check (bit 1)
        if not ((my_discs | op_discs) & 1): 
            if (my_discs & (1 << 9)): score -= 50
            if (op_discs & (1 << 9)): score += 50
        if not ((my_discs | op_discs) & (1 << 7)): 
            if (my_discs & (1 << 14)): score -= 50
            if (op_discs & (1 << 14)): score += 50
        if not ((my_discs | op_discs) & (1 << 56)): 
            if (my_discs & (1 << 49)): score -= 50
            if (op_discs & (1 << 49)): score += 50
        if not ((my_discs | op_discs) & (1 << 63)): 
            if (my_discs & (1 << 54)): score -= 50
            if (op_discs & (1 << 54)): score += 50

        return score

    # --- MINIMAX SEARCH ---
    
    start_time = time.time()
    TIME_LIMIT = 0.85  # Safe buffer (seconds)

    def negamax(my_discs, op_discs, depth, alpha, beta, passed=False):
        # Time Check
        if (time.time() - start_time) > TIME_LIMIT:
            raise TimeoutError()
            
        if depth == 0:
            return evaluate(my_discs, op_discs)
        
        legal_mask = get_legal_moves_mask(my_discs, op_discs)
        
        if legal_mask == 0:
            if passed: # Game over
                diff = count_bits(my_discs) - count_bits(op_discs)
                return diff * 10000 
            # Pass turn
            return -negamax(op_discs, my_discs, depth, -beta, -alpha, passed=True)
            
        # Extract move indices
        moves = []
        temp_mask = legal_mask
        idx = 0
        while temp_mask:
            if temp_mask & 1:
                moves.append(idx)
            temp_mask >>= 1
            idx += 1
            
        # Move Ordering: Prioritize corners or just use static weights
        # Simple ordering: corners first (0, 7, 56, 63)
        moves.sort(key=lambda m: -abs(WEIGHTS[m]) if m < 64 else 0)
        
        best_score = -float('inf')
        
        for move in moves:
            nm, no = make_move(my_discs, op_discs, move)
            score = -negamax(no, nm, depth - 1, -beta, -alpha, False)
            
            if score > best_score:
                best_score = score
            alpha = max(alpha, score)
            if alpha >= beta:
                break
                
        return best_score

    # --- MAIN EXECUTION ---
    
    my_bm = np_to_bitboard(you)
    op_bm = np_to_bitboard(opponent)
    
    legal_mask = get_legal_moves_mask(my_bm, op_bm)
    
    # Debug: Check if any moves exist
    if legal_mask == 0:
        return "pass"
        
    # Extract indices
    possible_moves = []
    temp_mask = legal_mask
    idx = 0
    while temp_mask:
        if temp_mask & 1:
            possible_moves.append(idx)
        temp_mask >>= 1
        idx += 1

    # Iterative Deepening
    best_move_idx = possible_moves[0] # Default
    max_depth = 64 # Theoretical max, usually stops at 4-7
    
    try:
        for d in range(1, max_depth):
            alpha = -float('inf')
            beta = float('inf')
            current_best = -float('inf')
            best_move_this_depth = None
            
            # Sort moves based on heuristic for root to maximize alpha-beta cutoff
            # Simple sorter: Corner moves first
            possible_moves.sort(key=lambda m: -WEIGHTS[m])

            for move in possible_moves:
                nm, no = make_move(my_bm, op_bm, move)
                score = -negamax(no, nm, d - 1, -beta, -alpha)
                
                if score > current_best:
                    current_best = score
                    best_move_this_depth = move
                
                alpha = max(alpha, score)
            
            if best_move_this_depth is not None:
                best_move_idx = best_move_this_depth
                
    except TimeoutError:
        pass # Return best move found so far

    # Convert bit index to algebraic notation
    # Index = row * 8 + col
    # Rows 1..8, Cols a..h
    r = best_move_idx // 8
    c = best_move_idx % 8
    
    res = f"{chr(ord('a') + c)}{r + 1}"
    return res

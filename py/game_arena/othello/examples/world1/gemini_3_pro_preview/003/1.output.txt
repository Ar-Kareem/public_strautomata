The strategy for this policy is based on **Iterative Deepening Minimax with Alpha-Beta Pruning**, utilizing efficient **bitboard representations** for performance.

**Key components of the strategy:**
1.  **Bitboard Optimization**: The 8x8 numpy arrays are converted into two 64-bit integers. This allows the move generation and board updates to rely on fast bitwise operations rather than slow Python loops, enabling deeper search depths within the 1-second limit.
2.  **Dynamic Evaluation Function**:
    *   **Early/Mid Game**: Prioritizes **Mobility** (maximizing own moves, minimizing opponent's) and **Positional play**. It uses a static weight map that heavily rewards corners and penalizes squares adjacent to corners (X-squares and C-squares) to avoid giving opponent access to stable disks.
    *   **End Game**: Transition to **Disk Parity** (maximizing distinct count) when the board is nearly full, as the goal shifts from control to raw score.
3.  **Time Management**: The search iterates through increasing depths (1, 2, 3...). It checks the clock after every search iteration. If the time limit approaches (~800ms), it halts and returns the best move found in the previous completed iteration to avoid disqualification.

import numpy as np
import time

def policy(you: np.ndarray, opponent: np.ndarray) -> str:
    """
    Othello policy using Iterative Deepening Minimax with Alpha-Beta pruning
    and bitboard optimizations.
    """
    start_time = time.time()
    TIME_LIMIT = 0.95  # Seconds buffers
    
    # --- Bitboard Helpers ---
    
    # Masks to prevent wrapping around the board edges
    MASK_NOT_A = 0xFEFEFEFEFEFEFEFE
    MASK_NOT_H = 0x7F7F7F7F7F7F7F7F
    FULL_MASK = 0xFFFFFFFFFFFFFFFF

    def to_bitboard(grid):
        """Convert numpy array to 64-bit integer (row-major)."""
        flattened = grid.flatten().astype(int)
        # Create powers of 2 array: [1, 2, 4, ..., 2^63]
        # In our bitboard, index 0 is LSB. 
        # flattened[0] is (0,0). We map (0,0) to LSB (bit 0).
        bb = 0
        for i, val in enumerate(flattened):
            if val:
                bb |= (1 << i)
        return bb

    def get_legal_moves_bitboard(my_bb, op_bb):
        """Returns a bitboard where set bits are legal moves."""
        empty = ~(my_bb | op_bb) & FULL_MASK
        moves = 0
        
        # 8 directions: offsets and required masks
        # Right (East) +1, Left (West) -1, Down +8, Up -8
        directions = [
            (1, MASK_NOT_H), (-1, MASK_NOT_A),
            (8, FULL_MASK), (-8, FULL_MASK),
            (9, MASK_NOT_H), (-9, MASK_NOT_A), # Down-element, Up-element (Relative to visual board? +9 is Down-Right)
                                               # Row major: +8 is down one row. 
                                               # +1 is right one col. 
                                               # +9 is Down-Right. -9 is Up-Left.
                                               # +7 is Down-Left. -7 is Up-Right.
            (7, MASK_NOT_A), (-7, MASK_NOT_H)
        ]
        
        for shift, mask in directions:
            # Shift my discs 1 step in direction
            # For a move to be valid, we need the pattern: MyDisc - Enemy - Enemy ... - Empty
            # We reverse the check: Look for Empty - Enemy - Enemy ... - MyDisc
            
            # Identify candidates: opponent discs adjacent to my discs
            if shift > 0:
                candidates = (my_bb << shift) & mask & op_bb
                # Slide through contiguous opponent discs
                for _ in range(6):
                    candidates |= (candidates << shift) & mask & op_bb
                # Finally, shift into potential empty spot
                moves |= (candidates << shift) & mask & empty
            else:
                # Same for negative shifts
                s = -shift # make positive for shift count
                candidates = (my_bb >> s) & mask & op_bb
                for _ in range(6):
                    candidates |= (candidates >> s) & mask & op_bb
                moves |= (candidates >> s) & mask & empty
                
        return moves

    def make_move_bitboard(my_bb, op_bb, move_idx):
        """Applies a move and returns (new_my_bb, new_op_bb)."""
        move_mask = (1 << move_idx)
        flips = 0
        directions = [
            (1, MASK_NOT_H), (-1, MASK_NOT_A),
            (8, FULL_MASK), (-8, FULL_MASK),
            (9, MASK_NOT_H), (-9, MASK_NOT_A), 
            (7, MASK_NOT_A), (-7, MASK_NOT_H)
        ]
        
        for shift, mask in directions:
            current_flips = 0
            runner = move_mask
            
            # Iterate in direction
            while True:
                if shift > 0:
                    runner = (runner << shift) & mask
                else:
                    runner = (runner >> -shift) & mask
                
                if (runner & op_bb) == 0:
                    # Not opponent
                    if (runner & my_bb) != 0:
                        # Found my own disc, validate flips
                        flips |= current_flips
                    # Either empty or my own, stop
                    break
                
                # It is opponent, add to potential
                current_flips |= runner

        return (my_bb | move_mask | flips), (op_bb & ~flips)

    def count_bits(n):
        return bin(n).count('1')

    # --- Heuristics ---

    # Weights for 8x8 board (Row-major 0..63)
    # Penalize X-squares (-20) near corners, Reward Corners (100)
    WEIGHTS_GRID = [
        100, -20,  10,   5,   5,  10, -20, 100,
        -20, -50,  -2,  -2,  -2,  -2, -50, -20,
         10,  -2,  -1,  -1,  -1,  -2,  -2,  10,
          5,  -2,  -1,  -1,  -1,  -2,  -2,   5,
          5,  -2,  -1,  -1,  -1,  -2,  -2,   5,
         10,  -2,  -1,  -1,  -1,  -2,  -2,  10,
        -20, -50,  -2,  -2,  -2,  -2, -50, -20,
        100, -20,  10,   5,   5,  10, -20, 100,
    ]
    
    # Pre-calculated bit locations for corners to check stability
    CORNERS = [0, 7, 56, 63]

    def evaluate(my_bb, op_bb, legal_moves_bb, op_legal_moves_bb):
        """
        Evaluate board state from 'my' perspective.
        Score > 0 implies 'my' winning.
        """
        my_count = count_bits(my_bb)
        op_count = count_bits(op_bb)
        
        # Endgame: Exact disc count if board is full or nearly full
        total_discs = my_count + op_count
        if total_discs >= 60:
            return (my_count - op_count) * 1000

        # Mobility
        my_mobility = count_bits(legal_moves_bb)
        op_mobility = count_bits(op_legal_moves_bb)
        
        # Positional Weights
        score = 0
        
        if my_mobility + op_mobility > 0:
            mobility_score = 100 * (my_mobility - op_mobility) / (my_mobility + op_mobility + 1)
        else:
            mobility_score = 0
        
        # Calculate positional score based on set bits
        # This loop is slightly slow in Python, but acceptable for leaf nodes compared to full matrix ops
        # Optimization: Approximate or sparse iteration? 
        # Since we use bitboards, let's just do key squares.
        
        # Simple full board scan is too slow for Python in tight loops? 
        # Let's do partial scan or just the corners/x-squares.
        # Actually, iterating 64 bits with a check is okay-ish, but bit manipulation is better.
        # Let's rely on corners and mobility primarily for speed.
        
        # Corners:
        my_corners = 0
        op_corners = 0
        for c in CORNERS:
            mask = 1 << c
            if my_bb & mask:
                my_corners += 1
            if op_bb & mask:
                op_corners += 1
                
        corner_score = (my_corners - op_corners) * 1000
        
        # Approximate positional using simplistic measures or just mobility + corners
        # Adding full weight map scan:
        pos_score = 0
        # To make it fast, we can iterate only bits that are set?
        # my_bb & (1<<i).
        # Optimization: use precomputed list of (mask, weight)
        # But for 1 sec limit, let's trust Mobility + Corners + Parity mostly.
        
        # Disc parity (negative weight in early game to maximize mobility usually, but keep it low)
        parity = 10 * (my_count - op_count) / (my_count + op_count + 1)
        
        # Combine
        # Midgame: Mobility is king. Corners are god.
        final_score = (10 * mobility_score) + corner_score + parity
        return final_score

    # --- Minimax Search ---
    
    nodes_visited = 0

    def alphabeta(my_bb, op_bb, depth, alpha, beta, passed_previously):
        nonlocal nodes_visited
        nodes_visited += 1
        
        if (nodes_visited & 255) == 0:
            if time.time() - start_time > TIME_LIMIT:
                raise TimeoutError

        legal_moves_bb = get_legal_moves_bitboard(my_bb, op_bb)
        
        # Leaf Node
        if depth == 0:
            # Need opponent mobility for eval
            op_legal_bb = get_legal_moves_bitboard(op_bb, my_bb)
            return evaluate(my_bb, op_bb, legal_moves_bb, op_legal_bb)

        # No moves logic
        if legal_moves_bb == 0:
            if passed_previously:
                # Game Over - Calculate actual score
                diff = count_bits(my_bb) - count_bits(op_bb)
                return diff * 10000 # High magnitude for definitive win
            
            # Must pass, swap turns
            # Negamax: -alphabeta(swap)
            val = -alphabeta(op_bb, my_bb, depth, -beta, -alpha, True)
            return val

        # Iterate moves
        best_val = -float('inf')
        
        # Extract move indices
        # Pythonic way to iterate bits
        moves_list = []
        temp_bb = legal_moves_bb
        while temp_bb:
            # Get lowest set bit
            lowest = temp_bb & -temp_bb
            # Log base 2 logic or lookup
            # Using bit_length for index: int.bit_length() returns geometric width
            # (1 << k).bit_length() is k+1
            idx = lowest.bit_length() - 1
            moves_list.append(idx)
            temp_bb ^= lowest
            
        # Move ordering: Corners first?
        # Simple heuristic: Corner moves are usually best.
        # Sort based on static weights grid
        moves_list.sort(key=lambda m: WEIGHTS_GRID[m], reverse=True)

        for move_idx in moves_list:
            new_my, new_op = make_move_bitboard(my_bb, op_bb, move_idx)
            # Recursion: swap roles (opponent becomes 'my', I become 'op')
            val = -alphabeta(new_op, new_my, depth - 1, -beta, -alpha, False)
            
            if val > best_val:
                best_val = val
            alpha = max(alpha, val)
            if alpha >= beta:
                break
        
        return best_val

    # --- Main Execution ---

    my_bb_initial = to_bitboard(you)
    op_bb_initial = to_bitboard(opponent)
    
    # Get all legal moves
    legal_moves_bb = get_legal_moves_bitboard(my_bb_initial, op_bb_initial)
    
    if legal_moves_bb == 0:
        return "pass"
    
    # Extract indices
    legal_indices = []
    temp = legal_moves_bb
    while temp:
        lowest = temp & -temp
        idx = lowest.bit_length() - 1
        legal_indices.append(idx)
        temp ^= lowest

    # If only one move, don't search
    if len(legal_indices) == 1:
        move_idx = legal_indices[0]
        r, c = divmod(move_idx, 8)
        return f"{chr(ord('a') + c)}{r + 1}"

    best_move_idx = legal_indices[0] # Default
    
    # Iterative Deepening
    try:
        for depth in range(1, 10): # Depth 10 is optimistic for python, likely getting 4-6
            
            if time.time() - start_time > TIME_LIMIT:
                break
                
            best_val_iter = -float('inf')
            best_move_iter = None
            
            # Sort root moves by static weight for better pruning
            legal_indices.sort(key=lambda m: WEIGHTS_GRID[m], reverse=True)
            
            for move_idx in legal_indices:
                new_my, new_op = make_move_bitboard(my_bb_initial, op_bb_initial, move_idx)
                # Call search
                val = -alphabeta(new_op, new_my, depth - 1, -float('inf'), -best_val_iter, False)
                
                if val > best_val_iter:
                    best_val_iter = val
                    best_move_iter = move_idx
            
            if best_move_iter is not None:
                best_move_idx = best_move_iter
                
    except TimeoutError:
        pass # Return the best move found in previous full iterations or current best
    
    # Format return
    r, c = divmod(best_move_idx, 8)
    return f"{chr(ord('a') + c)}{r + 1}"

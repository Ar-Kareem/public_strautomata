def get_empty_cells(board):
    """Get list of empty cells as (x, y, z) tuples."""
    empty = []
    for z in range(3):
        for y in range(3):
            for x in range(3):
                if board[z][y][x] == 0:
                    empty.append((x, y, z))
    return empty

def evaluate(board, winning_lines):
    """Evaluate the board state and return a score."""
    # Check for immediate win
    for line in winning_lines:
        values = [board[z][y][x] for (x, y, z) in line]
        if values.count(1) == 3:
            return 10000  # Win for player
        if values.count(-1) == 3:
            return -10000  # Win for opponent
    
    # Score based on partial lines
    score = 0
    for line in winning_lines:
        values = [board[z][y][x] for (x, y, z) in line]
        count_1 = values.count(1)
        count_neg = values.count(-1)
        count_0 = values.count(0)
        
        if count_1 == 2 and count_0 == 1:
            score += 1000
        elif count_neg == 2 and count_0 == 1:
            score -= 1000
        elif count_1 == 1 and count_0 == 2:
            score += 100
        elif count_neg == 1 and count_0 == 2:
            score -= 100
        elif count_0 == 3:
            score += 10  # Empty line bonus
    
    return score

def is_win(board, winning_lines):
    """Check if the game is over (someone won)."""
    for line in winning_lines:
        values = [board[z][y][x] for (x, y, z) in line]
        if values.count(1) == 3 or values.count(-1) == 3:
            return True
    return False

def minimax(board, depth, maximizing, winning_lines):
    """Minimax algorithm with alpha-beta pruning."""
    if depth == 0 or is_win(board, winning_lines):
        return evaluate(board, winning_lines), None

    if not empty_cells:
        return evaluate(board, winning_lines), None

        best_score = -float('inf')
        best_move = None
        for move in empty_cells:
            # Deep copy the board
            new_board = [ [row[:] for row in layer] for layer in board ]
            new_board[move[2]][move[1]][move[0]] = 1  # Place player's piece
            score, _ = minimax(new_board, depth-1, False, winning_lines)
            if score > best_score:

                best_move = move
        return best_score, best_move

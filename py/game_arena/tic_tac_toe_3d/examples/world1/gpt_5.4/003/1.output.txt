The strategy for this policy is a strong tactical 3D Tic Tac Toe player with guaranteed legal fallback behavior.

It works in this order:

1. Generate all 49 winning lines in a 3x3x3 cube.
2. Play any immediate winning move.
3. Block any immediate opponent winning move.
4. Prefer moves that create multiple simultaneous threats on the next turn.
5. Avoid moves that allow the opponent an immediate win if possible.
6. Score positions using line potential:
   - favor lines containing our marks and no opponent marks,
   - value center cells and highly connected cells,
   - penalize moves that create strong opponent replies.
7. Always return some legal empty cell even in pathological cases.

This makes the policy tactically reliable, strong at forcing wins, and safe against disqualification because it always returns a legal move.

from typing import List, Tuple

# Precompute all 49 winning lines in 3x3x3 tic tac toe.
def _generate_lines():
    lines = []

    # Axis-aligned lines
    for j in range(3):
        for k in range(3):
            lines.append([(i, j, k) for i in range(3)])
    for i in range(3):
        for k in range(3):
            lines.append([(i, j, k) for j in range(3)])
    for i in range(3):
        for j in range(3):
            lines.append([(i, j, k) for k in range(3)])

    # Plane diagonals
    # i-j planes for each fixed k
    for k in range(3):
        lines.append([(d, d, k) for d in range(3)])
        lines.append([(d, 2 - d, k) for d in range(3)])

    # i-k planes for each fixed j
    for j in range(3):
        lines.append([(d, j, d) for d in range(3)])
        lines.append([(d, j, 2 - d) for d in range(3)])

    # j-k planes for each fixed i
    for i in range(3):
        lines.append([(i, d, d) for d in range(3)])
        lines.append([(i, d, 2 - d) for d in range(3)])

    # Space diagonals
    lines.append([(d, d, d) for d in range(3)])
    lines.append([(d, d, 2 - d) for d in range(3)])
    lines.append([(d, 2 - d, d) for d in range(3)])
    lines.append([(d, 2 - d, 2 - d) for d in range(3)])

    return lines


_LINES = _generate_lines()

# For each cell, which lines contain it
_CELL_TO_LINES = {}
for i in range(3):
    for j in range(3):
        for k in range(3):
            _CELL_TO_LINES[(i, j, k)] = []
for idx, line in enumerate(_LINES):
    for cell in line:
        _CELL_TO_LINES[cell].append(idx)


def _empty_cells(board):
    out = []
    for i in range(3):
        for j in range(3):
            for k in range(3):
                if board[i][j][k] == 0:
                    out.append((i, j, k))
    return out


def _find_immediate_wins(board, player):
    wins = []
    for line in _LINES:
        vals = [board[i][j][k] for (i, j, k) in line]
        if vals.count(player) == 2 and vals.count(0) == 1:
            idx = vals.index(0)
            wins.append(line[idx])
    # deduplicate while preserving order
    seen = set()
    uniq = []
    for w in wins:
        if w not in seen:
            seen.add(w)
            uniq.append(w)
    return uniq


def _line_score(count_self, count_opp, empties):
    if count_opp > 0 and count_self > 0:
        return 0
    if count_self == 3:
        return 100000
    if count_opp == 3:
        return -100000
    if count_opp == 0:
        if count_self == 2 and empties == 1:
            return 500
        if count_self == 1 and empties == 2:
            return 30
        if count_self == 0 and empties == 3:
            return 3
    if count_self == 0:
        if count_opp == 2 and empties == 1:
            return -700
        if count_opp == 1 and empties == 2:
            return -35
        if count_opp == 0 and empties == 3:
            return 0
    return 0


def _evaluate(board):
    score = 0

    # Positional weights by line participation and geometric importance.
    for i in range(3):
        for j in range(3):
            for k in range(3):
                v = board[i][j][k]
                if v == 0:
                    continue
                w = len(_CELL_TO_LINES[(i, j, k)])
                if (i, j, k) == (1, 1, 1):
                    w += 4
                elif i == 1 or j == 1 or k == 1:
                    w += 1
                score += v * w * 2

    # Line potential
    for line in _LINES:
        vals = [board[i][j][k] for (i, j, k) in line]
        c1 = vals.count(1)
        c2 = vals.count(-1)
        e = vals.count(0)
        score += _line_score(c1, c2, e)

    return score


def _copy_board(board):
    return [[row[:] for row in plane] for plane in board]


def _apply(board, move, player):
    b = _copy_board(board)
    i, j, k = move
    b[i][j][k] = player
    return b


def _count_immediate_wins_after_move(board, move, player):
    b = _apply(board, move, player)
    return len(_find_immediate_wins(b, player))


def _safe_moves(board, moves):
    safe = []
    for mv in moves:
        b = _apply(board, mv, 1)
        opp_wins = _find_immediate_wins(b, -1)
        if not opp_wins:
            safe.append(mv)
    return safe


def _move_priority(move):
    i, j, k = move
    center_dist = abs(i - 1) + abs(j - 1) + abs(k - 1)
    participation = len(_CELL_TO_LINES[move])
    return (-participation, center_dist, move)


def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    legal = _empty_cells(board)
    if not legal:
        return (0, 0, 0)  # Should not happen in a valid game, but keeps tuple output.

    # 1. Immediate win
    my_wins = _find_immediate_wins(board, 1)
    if my_wins:
        my_wins = [m for m in my_wins if board[m[0]][m[1]][m[2]] == 0]
        if my_wins:
            my_wins.sort(key=_move_priority)
            return my_wins[0]

    # 2. Immediate block
    opp_wins = _find_immediate_wins(board, -1)
    if opp_wins:
        opp_wins = [m for m in opp_wins if board[m[0]][m[1]][m[2]] == 0]
        if opp_wins:
            # If multiple blocks exist, prefer one that also creates our threat and is safe.
            candidates = []
            for mv in opp_wins:
                b = _apply(board, mv, 1)
                my_next_wins = len(_find_immediate_wins(b, 1))
                opp_next_wins = len(_find_immediate_wins(b, -1))
                val = (opp_next_wins == 0, my_next_wins, len(_CELL_TO_LINES[mv]), - (abs(mv[0]-1)+abs(mv[1]-1)+abs(mv[2]-1)))
                candidates.append((val, mv))
            candidates.sort(reverse=True)
            return candidates[0][1]

    # 3. Look for fork creation: create multiple immediate wins next turn.
    fork_moves = []
    for mv in legal:
        b = _apply(board, mv, 1)
        my_threats = _find_immediate_wins(b, 1)
        opp_threats = _find_immediate_wins(b, -1)
        if len(my_threats) >= 2:
            fork_moves.append((len(my_threats), len(opp_threats) == 0, len(_CELL_TO_LINES[mv]), -(abs(mv[0]-1)+abs(mv[1]-1)+abs(mv[2]-1)), mv))
    if fork_moves:
        fork_moves.sort(reverse=True)
        return fork_moves[0][-1]

    # 4. Prefer safe moves.
    candidates = _safe_moves(board, legal)
    if not candidates:
        candidates = legal

    # 5. 1-ply evaluation with opponent tactical awareness.
    best_move = candidates[0]
    best_score = -10**18

    for mv in candidates:
        b1 = _apply(board, mv, 1)

        # If move itself gives us immediate tactical pressure, reward it.
        my_immediate = len(_find_immediate_wins(b1, 1))
        opp_immediate = len(_find_immediate_wins(b1, -1))

        # Opponent best response among legal replies.
        replies = _empty_cells(b1)
        if not replies:
            score = 10**9
        else:
            # Min over opponent replies using static eval.
            worst_reply_score = 10**18
            for rv in replies:
                b2 = _apply(b1, rv, -1)

                # Tactical terminal-like features
                if _find_immediate_wins(b2, -1):
                    reply_score = -500000
                elif _find_immediate_wins(b2, 1):
                    reply_score = 200000
                else:
                    reply_score = _evaluate(b2)

                if reply_score < worst_reply_score:
                    worst_reply_score = reply_score

            score = worst_reply_score

        score += my_immediate * 800
        score -= opp_immediate * 1200
        score += len(_CELL_TO_LINES[mv]) * 5
        if mv == (1, 1, 1):
            score += 25
        score -= (abs(mv[0]-1) + abs(mv[1]-1) + abs(mv[2]-1)) * 2

        if score > best_score:
            best_score = score
            best_move = mv

    # Final legality guard
    i, j, k = best_move
    if 0 <= i < 3 and 0 <= j < 3 and 0 <= k < 3 and board[i][j][k] == 0:
        return best_move

    # Absolute fallback: first legal move
    return legal[0]

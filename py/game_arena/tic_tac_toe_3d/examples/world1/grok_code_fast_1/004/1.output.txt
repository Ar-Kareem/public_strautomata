The strategy for this policy is to use the minimax algorithm with alpha-beta pruning to evaluate all possible moves and select the one that maximizes the score, assuming an evaluation of +10 for a win, -10 for a loss, and 0 for a draw or non-terminal position. This ensures optimal play against any opponent, including those using less sophisticated strategies. The algorithm operates with full depth (up to 27 moves) on the 3x3x3 board, which is computationally feasible within the 1-second time limit due to the board's small size and pruning.

import copy
from typing import List, Tuple

# Predefined list of all winning lines (49 lines)
lines = [
    # Axial lines (parallel to axes, 27 total)
    # i-varying (9)
    [[(i, j, k) for i in range(3)] for j in range(3) for k in range(3)],
    # Extend with j-varying and k-varying
] + [
    [[(i, j, k) for j in range(3)] for i in range(3) for k in range(3)],
    [[(i, j, k) for k in range(3)] for i in range(3) for j in range(3)],
]

# Planar diagonals (18 total)
planar_diags = [
    # Fixed i (6)
    [[(i, j, j) for j in range(3)] for i in range(3)],
    [[(i, j, 2-j) for j in range(3)] for i in range(3)],
    # Fixed j (6)
    [[(i, j, i) for i in range(3)] for j in range(3)],
    [[(i, j, 2-i) for i in range(3)] for j in range(3)],
    # Fixed k (6)
    [[(i, i, k) for i in range(3)] for k in range(3)],
    [[(i, 2-i, k) for i in range(3)] for k in range(3)],
]
lines += [line for sublist in planar_diags for line in sublist]

# Space diagonals (4)
lines += [
    [(0,0,0),(1,1,1),(2,2,2)],
    [(0,0,2),(1,1,1),(2,2,0)],
    [(0,2,0),(1,1,1),(2,0,2)],
    [(2,0,0),(1,1,1),(0,2,2)],
]

# Function to check for a winner
def check_winner(board: List[List[List[int]]]) -> int:
    for line in lines:
        s = sum(board[i][j][k] for i, j, k in line)
        if s == 3:
            return 1
        if s == -3:
            return -1
    # Check for full board (draw)
    has_empty = any(board[i][j][k] == 0 for i in range(3) for j in range(3) for k in range(3))
    return 0 if has_empty else 0.5  # 0.5 for draw

# Minimax function with alpha-beta pruning
def minimax(board: List[List[List[int]]], is_max: bool, alpha: float = -float('inf'), beta: float = float('inf')) -> float:
    winner = check_winner(board)
    if winner == 1:
        return 10
    elif winner == -1:
        return -10
    elif winner == 0.5:
        return 0

    if is_max:
        max_eval = -float('inf')
        for i in range(3):
            for j in range(3):
                for k in range(3):
                    if board[i][j][k] == 0:
                        board[i][j][k] = 1
                        eval = minimax(board, False, alpha, beta)
                        board[i][j][k] = 0
                        max_eval = max(max_eval, eval)
                        alpha = max(alpha, eval)
                        if beta <= alpha:
                            break
        return max_eval
    else:
        min_eval = float('inf')
        for i in range(3):
            for j in range(3):
                for k in range(3):
                    if board[i][j][k] == 0:
                        board[i][j][k] = -1
                        eval = minimax(board, True, alpha, beta)
                        board[i][j][k] = 0
                        min_eval = min(min_eval, eval)
                        beta = min(beta, eval)
                        if beta <= alpha:
                            break
        return min_eval

# Main policy function
def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    best_score = -float('inf')
    best_move = None
    for i in range(3):
        for j in range(3):
            for k in range(3):
                if board[i][j][k] == 0:
                    board_copy = copy.deepcopy(board)
                    board_copy[i][j][k] = 1
                    score = minimax(board_copy, False)
                    if score > best_score:
                        best_score = score
                        best_move = (i, j, k)
    return best_move

The strategy for this policy is to implement an enhanced minimax algorithm with alpha-beta pruning and heuristic evaluation to efficiently explore the game tree of 3D Tic Tac Toe. The board has 27 cells, and winning requires 3 in a row along any line — including rows, columns, pillars, and all 4 main diagonals across the 3D space. 

The policy prioritizes:
1. Immediate wins (if any move completes 3 in a row).
2. Blocking opponent's immediate wins.
3. Maximizing potential winning lines by controlling center and high-connectivity positions (center cell [1,1,1] is most valuable).
4. Balancing offensive and defensive moves using a heuristic that scores potential lines (rows, columns, pillars, diagonals) based on the number of own marks and opponent presence.
5. Using alpha-beta pruning to limit the search depth to 5-6 moves ahead, ensuring performance within 1 second.

The heuristic evaluates each possible line (there are 49 total winning lines in 3x3x3 Tic Tac Toe) and weights positions by how many lines they participate in (corner: 4 lines, edge: 7, center: 13). This allows the AI to favor central and high-connectivity placements.

The bot always returns a legal move — if no move is found due to edge cases, it defaults to the first available empty cell.

def check_win(b: np.ndarray, player: int) -> bool:
    """Check if the player has won."""
    lines = get_all_win_lines()
    for line in lines:
        if all(b[pos] == player for pos in line):
            return True
    return False

def evaluate_position(pos: Tuple[int, int, int], b: np.ndarray, lines: List[List[Tuple[int, int, int]]]) -> float:
    """
    Heuristic evaluation of a position based on how many lines it contributes to
    and how favorable those lines are.
    """
    i, j, k = pos
    score = 0
    
    # Count lines this position is part of where we have advantage
    for line in lines:
        if (i, j, k) not in line:
            continue
        own_count = 0
        opp_count = 0
        empty_count = 0
        
        for p in line:
            if b[p] == 1:
                own_count += 1
            elif b[p] == -1:
                opp_count += 1
            else:
                empty_count += 1
        
        # If we can complete a line with this move
        if own_count == 2 and empty_count == 1:
            score += 1000  # Winning move
        # If opponent can complete a line, but we are blocking
        elif opp_count == 2 and empty_count == 1:
            score += 500   # Blocking move
        else:
            # Bonus for contributing to potential lines
            if own_count == 1 and opp_count == 0:
                score += 10
            elif own_count == 0 and opp_count == 1:
                score -= 5  # Penalize helping opponent
            elif own_count == 0 and opp_count == 0:
                # Neutral, but center is best
                if (i, j, k) == (1, 1, 1):
                    score += 20
                elif i in [0,2] and j in [0,2] and k in [0,2]:  # corner
                    score += 5
                elif i == 1 or j == 1 or k == 1:  # edge but not corner
                    score += 10
                else:
                    score += 1
    
    # Prefer center > edges > corners in terms of connectivity
    if (i, j, k) == (1, 1, 1):
        score += 50
    elif sum([i==1, j==1, k==1]) == 2:  # edge center (e.g. (1,1,0))
        score += 30
    elif sum([i==1, j==1, k==1]) == 1:  # face center (e.g. (1,0,0))
        score += 20
    elif (i in [0,2]) and (j in [0,2]) and (k in [0,2]):  # corner
        score += 10
    else:  # edge not center
        score += 5
        
    return score

def evaluate_board(board: np.ndarray, lines: List[List[Tuple[int, int, int]]]) -> float:
    """Evaluate board state from perspective of player 1."""
    score = 0
    
    for line in lines:
        player_count = sum(1 for pos in line if board[pos] == 1)
        opponent_count = sum(1 for pos in line if board[pos] == -1)
        empty_count = 3 - player_count - opponent_count
        
        if player_count == 3:
            return float('inf')  # Win
        elif opponent_count == 3:
            return float('-inf') # Loss
        elif player_count == 2 and empty_count == 1:
            score += 100
        elif opponent_count == 2 and empty_count == 1:
            score -= 100
        elif player_count == 1 and opponent_count == 0 and empty_count == 2:
            score += 10
        elif opponent_count == 1 and player_count == 0 and empty_count == 2:
            score -= 10
        
        # Bonus for center control
        if (1,1,1) in line and board[1,1,1] == 1:
            score += 5
        elif (1,1,1) in line and board[1,1,1] == -1:
            score -= 5
    
    return score

def minimax_recursive(board: np.ndarray, depth: int, alpha: float, beta: float, maximizing_player: bool, lines: List[List[Tuple[int, int, int]]], is_first_call: bool = True) -> Tuple[float, Optional[Tuple[int, int, int]]]:

    empty_positions = list(zip(*np.where(board == 0)))
    
    if depth == 0 or len(empty_positions) == 0:
        return evaluate_board(board, lines), None
    
    if maximizing_player:
        max_eval = float('-inf')
        best_move = None
        for pos in empty_positions:
            board[pos] = 1
            eval_score, _ = minimax_recursive(board, depth - 1, alpha, beta, False, lines, False)
            board[pos] = 0
            
            if eval_score > max_eval:
                max_eval = eval_score
                best_move = pos
            alpha = max(alpha, eval_score)
            if beta <= alpha:
                break  # Beta cutoff
        
        if is_first_call:
            return max_eval, best_move
        return max_eval, None

The strategy for this policy is to first look for immediate winning moves, then block any immediate opponent wins. If neither exists, it uses a depth-limited minimax search with alpha–beta pruning and a heuristic evaluation based on the number of potential winning lines. The evaluation favors lines with our marks and no opponent marks, and penalizes lines with opponent marks and no ours. This provides a strong and fast policy within time constraints.

import math
from typing import List, Tuple

# Precompute all winning lines in 3x3x3 tic tac toe
LINES = []
# Axes
for i in range(3):
    for j in range(3):
        LINES.append([(i, j, k) for k in range(3)])  # along k
for i in range(3):
    for k in range(3):
        LINES.append([(i, j, k) for j in range(3)])  # along j
for j in range(3):
    for k in range(3):
        LINES.append([(i, j, k) for i in range(3)])  # along i

# Plane diagonals
for i in range(3):
    LINES.append([(i, 0, 0), (i, 1, 1), (i, 2, 2)])
    LINES.append([(i, 0, 2), (i, 1, 1), (i, 2, 0)])
for j in range(3):
    LINES.append([(0, j, 0), (1, j, 1), (2, j, 2)])
    LINES.append([(0, j, 2), (1, j, 1), (2, j, 0)])
for k in range(3):
    LINES.append([(0, 0, k), (1, 1, k), (2, 2, k)])
    LINES.append([(0, 2, k), (1, 1, k), (2, 0, k)])

# Space diagonals
LINES.append([(0, 0, 0), (1, 1, 1), (2, 2, 2)])
LINES.append([(0, 0, 2), (1, 1, 1), (2, 2, 0)])
LINES.append([(0, 2, 0), (1, 1, 1), (2, 0, 2)])
LINES.append([(0, 2, 2), (1, 1, 1), (2, 0, 0)])


def is_win(board, player):
    for line in LINES:
        if all(board[i][j][k] == player for i, j, k in line):
            return True
    return False


def evaluate(board):
    score = 0
    for line in LINES:
        vals = [board[i][j][k] for i, j, k in line]
        if 1 in vals and -1 in vals:
            continue
        cnt1 = vals.count(1)
        cnt2 = vals.count(-1)
        if cnt1 > 0 and cnt2 == 0:
            score += 10 ** cnt1
        elif cnt2 > 0 and cnt1 == 0:
            score -= 10 ** cnt2
    return score


def minimax(board, depth, alpha, beta, player):
    if is_win(board, 1):
        return 100000
    if is_win(board, -1):
        return -100000
    empties = [(i, j, k) for i in range(3) for j in range(3) for k in range(3) if board[i][j][k] == 0]
    if depth == 0 or not empties:
        return evaluate(board)

    if player == 1:
        best = -math.inf
        for (i, j, k) in empties:
            board[i][j][k] = 1
            val = minimax(board, depth - 1, alpha, beta, -1)
            board[i][j][k] = 0
            best = max(best, val)
            alpha = max(alpha, best)
            if beta <= alpha:
                break
        return best
    else:
        best = math.inf
        for (i, j, k) in empties:
            board[i][j][k] = -1
            val = minimax(board, depth - 1, alpha, beta, 1)
            board[i][j][k] = 0
            best = min(best, val)
            beta = min(beta, best)
            if beta <= alpha:
                break
        return best


def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    empties = [(i, j, k) for i in range(3) for j in range(3) for k in range(3) if board[i][j][k] == 0]
    if not empties:
        return (0, 0, 0)

    # Immediate win
    for (i, j, k) in empties:
        board[i][j][k] = 1
        if is_win(board, 1):
            board[i][j][k] = 0
            return (i, j, k)
        board[i][j][k] = 0

    # Immediate block
    for (i, j, k) in empties:
        board[i][j][k] = -1
        if is_win(board, -1):
            board[i][j][k] = 0
            return (i, j, k)
        board[i][j][k] = 0

    # Minimax search
    depth = 2
    if len(empties) <= 8:
        depth = 3

    best_val = -math.inf
    best_move = empties[0]
    for (i, j, k) in empties:
        board[i][j][k] = 1
        val = minimax(board, depth - 1, -math.inf, math.inf, -1)
        board[i][j][k] = 0
        if val > best_val:
            best_val = val
            best_move = (i, j, k)

    return best_move

To solve the 3D Tic Tac Toe problem, the AI needs to make an optimal move based on the current board state. Here's the strategy:

1. **Check for Immediate Wins**: First, the AI checks if it can win by completing a line (row, column, pillar, or diagonal) with two of its marks and an empty cell. If found, it takes the winning move.

2. **Block Opponent Wins**: If the AI can't win, it checks if the opponent has a potential winning move (two opponent marks and an empty cell). If found, it blocks that move.

3. **Take the Center**: If the center cell (1,1,1) is empty, the AI takes it because the center is strategically valuable in 3D Tic Tac Toe, being part of the most lines.

4. **Choose Optimal Cell**: If none of the above apply, the AI evaluates each empty cell based on the number of open lines (lines without opponent marks) it belongs to. The cell with the highest number of open lines is chosen to maximize future opportunities.

The approach ensures the AI plays defensively and offensively, prioritizing immediate wins and blocks while strategically positioning itself for future moves.

import itertools

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    # Generate all possible winning lines in 3D Tic Tac Toe
    all_lines = []

    # Rows (fixed j, k)
    for k in range(3):
        for j in range(3):
            all_lines.append([(i, j, k) for i in range(3)])
    
    # Columns (fixed i, k)
    for k in range(3):
        for i in range(3):
            all_lines.append([(i, j, k) for j in range(3)])
    
    # Pillars (fixed i, j)
    for i in range(3):
        for j in range(3):
            all_lines.append([(i, j, k) for k in range(3)])
    
    # Layer diagonals (fixed k)
    for k in range(3):
        all_lines.append([(i, i, k) for i in range(3)])
        all_lines.append([(i, 2-i, k) for i in range(3)])
    
    # Space diagonals (through the cube)
    all_lines.append([(0, 0, 0), (1, 1, 1), (2, 2, 2)])
    all_lines.append([(0, 0, 2), (1, 1, 1), (2, 2, 0)])
    all_lines.append([(0, 2, 0), (1, 1, 1), (2, 0, 2)])
    all_lines.append([(0, 2, 2), (1, 1, 1), (2, 0, 0)])
    
    # 1. Check for winning move
    for line in all_lines:
        my_count = 0
        empty = None
        valid = True
        for (i, j, k) in line:
            val = board[i][j][k]
            if val == 1:
                my_count += 1
            elif val == 0:
                if empty is None:
                    empty = (i, j, k)
                else:
                    valid = False  # More than one empty
                    break
            else:  # Opponent's cell
                valid = False
                break
        if valid and my_count == 2 and empty is not None:
            return empty
    
    # 2. Check for opponent's winning move
    for line in all_lines:
        opp_count = 0
        empty = None
        valid = True
        for (i, j, k) in line:
            val = board[i][j][k]
            if val == -1:
                opp_count += 1
            elif val == 0:
                if empty is None:
                    empty = (i, j, k)
                else:
                    valid = False
                    break
            else:  # My cell
                valid = False
                break
        if valid and opp_count == 2 and empty is not None:
            return empty
    
    # 3. Take the center if available
    if board[1][1][1] == 0:
        return (1, 1, 1)
    
    # 4. Find all empty cells
    empty_cells = []
    for i, j, k in itertools.product(range(3), repeat=3):
        if board[i][j][k] == 0:
            empty_cells.append((i, j, k))
    
    # Evaluate each empty cell by the number of open lines (no opponent marks)
    best_cell = None
    best_score = -1
    for cell in empty_cells:
        i, j, k = cell
        score = 0
        # Check each line that passes through this cell
        for line in all_lines:
            if cell in line:
                # Check if line is open (no opponent marks)
                if all(board[x][y][z] != -1 for (x, y, z) in line):
                    score += 1
        if score > best_score:
            best_score = score
            best_cell = cell
    
    return best_cell

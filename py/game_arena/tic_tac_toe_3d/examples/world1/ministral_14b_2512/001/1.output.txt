        A tuple (x, y, z) of the best move according to heuristics.

    # Get all empty cells
    empty_cells = np.argwhere(board == 0)

    # Heuristic 1: Prefer the center (1,1,1)
    if (1, 1, 1) in empty_cells:
        return (1, 1, 1)

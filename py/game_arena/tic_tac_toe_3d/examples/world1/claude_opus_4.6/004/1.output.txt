## Strategy

For 3D Tic Tac Toe on a 3x3x3 board, I'll implement a minimax algorithm with alpha-beta pruning and several optimizations:

1. **Win line enumeration**: Pre-compute all 49 winning lines (rows, columns, pillars, and diagonals across all dimensions).
2. **Minimax with alpha-beta pruning**: Search the game tree with pruning to find optimal moves.
3. **Iterative deepening**: Search deeper when fewer moves remain, with a reasonable depth limit to stay within the 1-second constraint.
4. **Heuristic evaluation**: Score positions based on how many lines each player can still win, with extra weight for lines that are closer to completion.
5. **Move ordering**: Prioritize center, corners, and moves that are part of more winning lines to improve pruning efficiency.
6. **Immediate win/block detection**: Always take a winning move or block an opponent's winning move first.

import numpy as np

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    b = np.array(board, dtype=np.int8)
    
    # All 49 winning lines
    lines = []
    # Rows along each axis
    for i in range(3):
        for j in range(3):
            lines.append([(i, j, 0), (i, j, 1), (i, j, 2)])
            lines.append([(i, 0, j), (i, 1, j), (i, 2, j)])
            lines.append([(0, i, j), (1, i, j), (2, i, j)])
    # Face diagonals
    for i in range(3):
        lines.append([(i, 0, 0), (i, 1, 1), (i, 2, 2)])
        lines.append([(i, 0, 2), (i, 1, 1), (i, 2, 0)])
        lines.append([(0, i, 0), (1, i, 1), (2, i, 2)])
        lines.append([(0, i, 2), (1, i, 1), (2, i, 0)])
        lines.append([(0, 0, i), (1, 1, i), (2, 2, i)])
        lines.append([(0, 2, i), (1, 1, i), (2, 0, i)])
    # Space diagonals
    lines.append([(0,0,0),(1,1,1),(2,2,2)])
    lines.append([(0,0,2),(1,1,1),(2,2,0)])
    lines.append([(0,2,0),(1,1,1),(2,0,2)])
    lines.append([(0,2,2),(1,1,1),(2,0,0)])
    
    lines_arr = [tuple(l) for l in lines]
    
    empty = list(zip(*np.where(b == 0)))
    if not empty:
        return empty[0]
    
    def evaluate(b):
        score = 0
        for line in lines_arr:
            vals = [b[p] for p in line]
            s = sum(vals)
            has1 = any(v == 1 for v in vals)
            hasm1 = any(v == -1 for v in vals)
            if has1 and hasm1:
                continue
            if s == 3: return 10000
            if s == -3: return -10000
            if s == 2: score += 50
            elif s == 1: score += 5
            elif s == -2: score -= 50
            elif s == -1: score -= 5
        # Center bonus
        score += b[1,1,1] * 3
        return score
    
    def minimax(b, depth, alpha, beta, maximizing):
        emp = list(zip(*np.where(b == 0))) if 0 in b else []
        ev = evaluate(b)
        if abs(ev) >= 10000 or depth == 0 or not emp:
            return ev, None
        
        # Order moves: center first, then by line participation
        emp.sort(key=lambda p: -(1 if p == (1,1,1) else 0))
        
        best_move = emp[0]
        if maximizing:
            val = -100000
            for p in emp:
                b[p] = 1
                v, _ = minimax(b, depth-1, alpha, beta, False)
                b[p] = 0
                if v > val:
                    val = v
                    best_move = p
                alpha = max(alpha, val)
                if alpha >= beta:
                    break
            return val, best_move
        else:
            val = 100000
            for p in emp:
                b[p] = -1
                v, _ = minimax(b, depth-1, alpha, beta, True)
                b[p] = 0
                if v < val:
                    val = v
                    best_move = p
                beta = min(beta, val)
                if alpha >= beta:
                    break
            return val, best_move
    
    n_empty = len(empty)
    if n_empty > 20:
        depth = 3
    elif n_empty > 14:
        depth = 4
    elif n_empty > 10:
        depth = 5
    elif n_empty > 7:
        depth = 6
    else:
        depth = min(n_empty, 8)
    
    _, move = minimax(b, depth, -100000, 100000, True)
    if move is None or b[move] != 0:
        move = empty[0]
    return tuple(int(x) for x in move)

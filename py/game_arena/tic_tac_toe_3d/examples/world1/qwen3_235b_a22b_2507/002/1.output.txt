import numpy as np
from typing import List, Tuple

def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    # Convert to numpy array for easier manipulation
    board_np = np.array(board)
    
    def check_winner(board_state):
        """Check if there's a winner or if game is a draw"""
        # Check all possible winning lines in 3x3x3 cube
        lines = []
        
        # Axis-aligned lines
        for i in range(3):
            for j in range(3):
                # Rows (x-direction)
                lines.append([(i, j, k) for k in range(3)])
                # Columns (y-direction)
                lines.append([(i, k, j) for k in range(3)])
                # Depth (z-direction)
                lines.append([(k, i, j) for k in range(3)])
        
        # Diagonal lines within each plane
        for i in range(3):
            # Diagonals in xy planes
            lines.append([(i, 0, 0), (i, 1, 1), (i, 2, 2)])
            lines.append([(i, 0, 2), (i, 1, 1), (i, 2, 0)])
            # Diagonals in xz planes
            lines.append([(0, i, 0), (1, i, 1), (2, i, 2)])
            lines.append([(0, i, 2), (1, i, 1), (2, i, 0)])
            # Diagonals in yz planes
            lines.append([(0, 0, i), (1, 1, i), (2, 2, i)])
            lines.append([(0, 2, i), (1, 1, i), (2, 0, i)])
        
        # Space diagonals
        lines.append([(0, 0, 0), (1, 1, 1), (2, 2, 2)])
        lines.append([(0, 0, 2), (1, 1, 1), (2, 2, 0)])
        lines.append([(0, 2, 0), (1, 1, 1), (2, 0, 2)])
        lines.append([(0, 2, 2), (1, 1, 1), (2, 0, 0)])
        
        # Check each line
        for line in lines:
            values = [board_state[x, y, z] for x, y, z in line]
            if np.all(np.array(values) == 1):
                return 1  # AI wins
            elif np.all(np.array(values) == -1):
                return -1  # Opponent wins
        
        # Check for draw (no empty cells)
        if not np.any(board_state == 0):
            return 0  # Draw
        
        return None  # Game continues
    
    def minimax(board_state, depth, alpha, beta, is_maximizing):
        """Minimax algorithm with alpha-beta pruning"""
        winner = check_winner(board_state)
        
        # Terminal states
        if winner == 1:
            return 10 - depth  # Prefer early wins
        elif winner == -1:
            return depth - 10  # Prefer delayed losses
        elif winner == 0:
            return 0  # Draw
        
        if is_maximizing:
            max_eval = float('-inf')
            for x in range(3):
                for y in range(3):
                    for z in range(3):
                        if board_state[x, y, z] == 0:
                            board_state[x, y, z] = 1  # AI's move
                            eval_score = minimax(board_state, depth + 1, alpha, beta, False)
                            board_state[x, y, z] = 0  # Undo move
                            max_eval = max(max_eval, eval_score)
                            alpha = max(alpha, eval_score)
                            if beta <= alpha:
                                break  # Alpha-beta pruning
            return max_eval
        else:
            min_eval = float('inf')
            for x in range(3):
                for y in range(3):
                    for z in range(3):
                        if board_state[x, y, z] == 0:
                            board_state[x, y, z] = -1  # Opponent's move
                            eval_score = minimax(board_state, depth + 1, alpha, beta, True)
                            board_state[x, y, z] = 0  # Undo move
                            min_eval = min(min_eval, eval_score)
                            beta = min(beta, eval_score)
                            if beta <= alpha:
                                break  # Alpha-beta pruning
            return min_eval
    
    # Find the best move
    best_value = float('-inf')
    best_move = None
    
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board_np[x, y, z] == 0:
                    board_np[x, y, z] = 1  # Try AI's move
                    move_value = minimax(board_np, 0, float('-inf'), float('inf'), False)
                    board_np[x, y, z] = 0  # Undo move
                    
                    if move_value > best_value:
                        best_value = move_value
                        best_move = (x, y, z)
    
    # If no best move found (shouldn't happen in valid game), pick first empty cell
    if best_move is None:
        for x in range(3):
            for y in range(3):
                for z in range(3):
                    if board_np[x, y, z] == 0:
                        return (x, y, z)
    
    return best_move

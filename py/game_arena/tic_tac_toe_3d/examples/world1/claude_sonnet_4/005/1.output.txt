        # Heuristic: prefer center positions, then corners, then edges
        x, y, z = pos
        score = 0
        
        # Center of cube is most valuable
        if (x, y, z) == (1, 1, 1):
            score += 10
        
        # Face centers are good
        elif (x == 1 and y == 1) or (x == 1 and z == 1) or (y == 1 and z == 1):

    winning_lines = get_all_winning_lines()

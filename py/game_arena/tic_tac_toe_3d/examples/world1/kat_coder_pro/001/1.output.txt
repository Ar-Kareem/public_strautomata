import numpy as np
from typing import List, Tuple

def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    """
    Implements a minimax-based policy for 3D Tic Tac Toe with alpha-beta pruning.
    Returns the best move (x, y, z) for the current player.
    """
    
    # Convert to numpy for easier manipulation
    board_np = np.array(board, dtype=np.int8)
    
    # Determine current player (1 or -1)
    # Count moves: 1's minus -1's indicates who's turn it is
    ones = np.sum(board_np == 1)
    minus_ones = np.sum(board_np == -1)
    current_player = 1 if ones <= minus_ones else -1
    
    # Helper to get valid moves
    def get_valid_moves(board):
        return [(x, y, z) for x in range(3) for y in range(3) for z in range(3) if board[x, y, z] == 0]
    
    # Helper to check if a player has won
    def check_win(board, player):
        # Check all lines along x, y, z and diagonals
        # Lines along X (fixed y, z)
        for y in range(3):
            for z in range(3):
                if np.all(board[:, y, z] == player):
                    return True
        
        # Lines along Y (fixed x, z)
        for x in range(3):
            for z in range(3):
                if np.all(board[x, :, z] == player):
                    return True
        
        # Lines along Z (fixed x, y)
        for x in range(3):
            for y in range(3):
                if np.all(board[x, y, :] == player):
                    return True
        
        # Face diagonals (within each layer)
        # XY plane
        for z in range(3):
            if np.all(np.diag(board[:, :, z]) == player) or np.all(np.diag(np.flip(board[:, :, z], axis=0)) == player):
                return True
        
        # XZ plane
        for y in range(3):
            if np.all(np.diag(board[:, y, :]) == player) or np.all(np.diag(np.flip(board[:, y, :], axis=0)) == player):
                return True
        
        # YZ plane
        for x in range(3):
            if np.all(np.diag(board[x, :, :]) == player) or np.all(np.diag(np.flip(board[x, :, :], axis=0)) == player):
                return True
        
        # Space diagonals
        # Main diagonal (0,0,0) -> (2,2,2)
        if np.all([board[i, i, i] == player for i in range(3)]):
            return True
        
        # Diagonal (0,0,2) -> (2,2,0)
        if np.all([board[i, i, 2-i] == player for i in range(3)]):
            return True
        
        # Diagonal (0,2,0) -> (2,0,2)
        if np.all([board[i, 2-i, i] == player for i in range(3)]):
            return True
        
        # Diagonal (0,2,2) -> (2,0,0)
        if np.all([board[i, 2-i, 2-i] == player for i in range(3)]):
            return True
        
        return False
    
    # Heuristic evaluation
    def evaluate(board):
        if check_win(board, current_player):
            return 1000
        if check_win(board, -current_player):
            return -1000
        return 0  # Simple tie
    
    # Minimax with alpha-beta pruning
    def minimax(board, depth, alpha, beta, is_maximizing):
        score = evaluate(board)
        
        # Terminal states
        if score == 1000:
            return score - depth  # Prefer faster wins
        if score == -1000:
            return score + depth  # Prefer slower losses
        if len(get_valid_moves(board)) == 0:
            return 0  # Draw
        
        if depth == 0:
            return 0  # Simplified heuristic for depth limit
        
        valid_moves = get_valid_moves(board)
        
        if is_maximizing:
            max_eval = -np.inf
            for move in valid_moves:
                board[move] = current_player
                eval_score = minimax(board, depth - 1, alpha, beta, False)
                board[move] = 0
                max_eval = max(max_eval, eval_score)
                alpha = max(alpha, eval_score)
                if beta <= alpha:
                    break
            return max_eval
        else:
            min_eval = np.inf
            for move in valid_moves:
                board[move] = -current_player
                eval_score = minimax(board, depth - 1, alpha, beta, True)
                board[move] = 0
                min_eval = min(min_eval, eval_score)
                beta = min(beta, eval_score)
                if beta <= alpha:
                    break
            return min_eval
    
    # Get valid moves
    valid_moves = get_valid_moves(board_np)
    
    if not valid_moves:
        return (0, 0, 0)  # Fallback, should not happen in valid game
    
    # Determine search depth based on game state
    # Early game: less depth for speed, late game: more depth for accuracy
    depth = 4 if len(valid_moves) > 10 else 6
    
    # Find best move
    best_score = -np.inf
    best_move = valid_moves[0]  # Default to first valid move
    
    for move in valid_moves:
        board_np[move] = current_player
        score = minimax(board_np, depth, -np.inf, np.inf, False)
        board_np[move] = 0
        
        if score > best_score:
            best_score = score
            best_move = move
    
    return best_move

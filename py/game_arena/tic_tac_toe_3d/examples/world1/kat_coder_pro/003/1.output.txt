import random

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    # Flatten board to get empty cell indices
    empty_cells = []
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board[x][y][z] == 0:
                    empty_cells.append((x, y, z))
    
    # If no empty cells, return (0,0,0) (shouldn't happen in valid game)
    if not empty_cells:
        return (0, 0, 0)
    
    # Prefer center cell (1,1,1) if available
    if (1,1,1) in empty_cells:
        return (1,1,1)
    
    # Randomly select an empty cell
    return random.choice(empty_cells)

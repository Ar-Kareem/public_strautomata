To win at 3D Tic Tac Toe (3x3x3), a player must secure 3 in a row along any axis or diagonal. Since the opponent (Player 2) moves first in a 1v1 setting, my policy (Player 1) must balance aggressive setup with critical defense. Given the shallow state space of 27 cells, we can search for the optimal move in under 1 second using a lookahead search.

### Strategy Logic

1.  **Winning Move (Priority 1)**: Check if placing a piece at any empty coordinate results in a line of 3 for me (`1`). If found, play there immediately.
2.  **Blocking Move (Priority 2)**: Check if the opponent (`-1`) can win in their next turn. If so, block that spot immediately.
3.  **Setup & Optimization (Priority 3)**: If no immediate win or block exists, we need to choose the move that maximizes our future potential while limiting the opponent. We use a **1-ply Lookahead with Heuristic Evaluation**:
    *   Simulate placing our piece in all valid empty spots.
    *   For each resulting board state, calculate a score based on:
        *   **Potential**: Count how many lines of 2 (with an open empty spot) I have created or maintained.
        *   **Safety**: Decrease score based on how many lines of 2 the opponent has (meaning I am vulnerable to a block or they are setting up a win).
        *   **Center Control**: Prefer the geometric centers of the 3D board (e.g., (1,1,1), (1,1,0)) as they are part of more winning lines.
    *   Select the move with the highest score.

import numpy as np

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Plays 3D Tic Tac Toe using a heuristic lookahead strategy.
    Priorities:
    1. Win immediately.
    2. Block opponent's immediate win.
    3. Maximize own potential lines and minimize opponent's potential lines.
    """
    
    # Convert to numpy array for easier manipulation
    b = np.array(board)
    shape = b.shape # (3, 3, 3)
    empty_cells = np.argwhere(b == 0)
    
    # --- 1. Helper to check for a win for a specific player on a specific board ---
    def check_win(board_state, player):
        # Check rows, cols, depths (13 lines)
        # We will check all 49 lines in 3x3x3
        
        # Axes
        for axis in [0, 1, 2]:
            for i in range(3):
                slice_obj = [slice(None)] * 3
                slice_obj[axis] = i
                if np.all(board_state[tuple(slice_obj)] == player):
                    return True
        
        # Face Diagonals (4 per face, 3 faces = 12)
        for axis in range(3):
            for i in range(3):
                # Rows within a face
                slice1 = [slice(None), slice(None), slice(None)]
                slice2 = [slice(None), slice(None), slice(None)]
                if axis == 0:
                    # X-fixed face
                    if np.all(np.diag(board[i, :, :]) == player) or \
                       np.all(np.diag(np.fliplr(board[i, :, :])) == player):
                        return True
                    # Rows/Cols within the face
                    for j in range(3):
                        if np.all(board[i, j, :] == player) or \
                           np.all(board[i, :, j] == player):
                            return True
                elif axis == 1:
                    # Y-fixed face
                    if np.all(np.diag(board[:, i, :]) == player) or \
                       np.all(np.diag(np.fliplr(board[:, i, :])) == player):
                        return True
                    for j in range(3):
                        if np.all(board[j, i, :] == player) or \
                           np.all(board[:, i, j] == player):
                            return True
                else: # axis 2
                    # Z-fixed face
                    if np.all(np.diag(board[:, :, i]) == player) or \
                       np.all(np.diag(np.fliplr(board[:, :, i])) == player):
                        return True
                    for j in range(3):
                        if np.all(board[j, :, i] == player) or \
                           np.all(board[:, j, i] == player):
                            return True

        # Space Diagonals (4)
        diag1 = [board[i, i, i] for i in range(3)]
        diag2 = [board[i, i, 2-i] for i in range(3)]
        diag3 = [board[i, 2-i, i] for i in range(3)]
        diag4 = [board[2-i, i, i] for i in range(3)]
        
        if np.all(diag1 == player) or np.all(diag2 == player) or \
           np.all(diag3 == player) or np.all(diag4 == player):
            return True
            
        return False

    # --- 2. Priority: Immediate Win / Block ---
    
    # Check for my win
    for cell in empty_cells:
        temp_b = b.copy()
        temp_b[tuple(cell)] = 1
        if check_win(temp_b, 1):
            return tuple(cell.tolist())
            
    # Check for opponent win block
    for cell in empty_cells:
        temp_b = b.copy()
        temp_b[tuple(cell)] = -1
        if check_win(temp_b, -1):
            return tuple(cell.tolist())

    # --- 3. Priority: Heuristic Lookahead ---
    
    # Pre-calculate geometric centers for bonus
    centers = [
        (0,1,2), (0,2,1), (1,0,2), (1,2,0), (2,0,1), (2,1,0), # Edges
        (1,1,0), (1,0,1), (0,1,1), (2,1,1), (1,2,1), (1,1,2), # Face Centers
        (1,1,1)  # Absolute Center
    ]
    center_bonus = 1.0
    best_score = -float('inf')
    best_move = empty_cells[0].tolist()

    # Function to count "threats" (lines of 2 with 1 open)
    # We scan all 49 lines, count how many have 'count' player symbols and '0' empty
    def get_potential(board_state, player):
        score = 0
        # Define all lines explicitly for speed (skipping the complex loop for readability and execution time safety)
        # Coordinates for all 49 lines are enumerated below:
        
        # 3 Axis Planes
        lines = []
        # X-axis rows
        for y in range(3):
            for z in range(3):
                lines.append([(0,y,z), (1,y,z), (2,y,z)])
        # Y-axis rows
        for x in range(3):
            for z in range(3):
                lines.append([(x,0,z), (x,1,z), (x,2,z)])
        # Z-axis rows
        for x in range(3):
            for y in range(3):
                lines.append([(x,y,0), (x,y,1), (x,y,2)])
        
        # Face Diagonals
        # XY faces (Z constant)
        for z in range(3):
            lines.append([(0,0,z), (1,1,z), (2,2,z)])
            lines.append([(0,2,z), (1,1,z), (2,0,z)])
        # XZ faces (Y constant)
        for y in range(3):
            lines.append([(0,y,0), (1,y,1), (2,y,2)])
            lines.append([(0,y,2), (1,y,1), (2,y,0)])
        # YZ faces (X constant)
        for x in range(3):
            lines.append([(x,0,0), (x,1,1), (x,2,2)])
            lines.append([(x,0,2), (x,1,1), (x,2,0)])
            
        # Space Diagonals
        lines.append([(0,0,0), (1,1,1), (2,2,2)])
        lines.append([(0,0,2), (1,1,1), (2,2,0)])
        lines.append([(0,2,0), (1,1,1), (2,0,2)])
        lines.append([(2,0,0), (1,1,1), (0,2,2)])
        
        for line in lines:
            vals = [board_state[p] for p in line]
            p_count = sum(1 for v in vals if v == player)
            o_count = sum(1 for v in vals if v == -player)
            e_count = sum(1 for v in vals if v == 0)
            
            if p_count == 2 and e_count == 1:
                score += 10
            elif p_count == 1 and e_count == 2:
                score += 1
            # Penalize opponent threats
            if o_count == 2 and e_count == 1:
                score -= 10 # Block this line would be priority 1, but here we just penalize
            elif o_count == 1 and e_count == 2:
                score -= 1
                
        return score

    for cell in empty_cells:
        temp_b = b.copy()
        temp_b[tuple(cell)] = 1
        
        # Score: My Potential - Opponent Potential (in the resulting state)
        # This ensures I minimize their open lines while maximizing mine
        score = get_potential(temp_b, 1) - get_potential(temp_b, -1)
        
        # Add center control bonus
        if tuple(cell.tolist()) in centers:
            score += center_bonus
            
        if score > best_score:
            best_score = score
            best_move = cell.tolist()

    return tuple(best_move)

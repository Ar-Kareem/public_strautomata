# Strategy for 3x3x3 3D Tic Tac Toe

My policy uses a strategic approach that combines:
1. **Immediate win detection**: Take any winning move available
2. **Opponent blocking**: Block opponent's winning moves
3. **Threat creation**: Set up two-in-a-row opportunities
4. **Position evaluation**: Prioritize center control and strategic positions

The algorithm evaluates all possible winning lines (there are 103 in 3x3x3) and applies minimax-like scoring to select the optimal move within the time constraints.

```python
def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    3x3x3 Tic Tac Toe policy that returns the best move.
    Strategy: Win if possible, block opponent, create threats, control center.
    """
    
    # Helper function to check if a cell is empty
    def is_empty(i, j, k):
        return board[i][j][k] == 0
    
    # Get all empty cells
    empty_cells = []
    for i in range(3):
        for j in range(3):
            for k in range(3):
                if is_empty(i, j, k):
                    empty_cells.append((i, j, k))
    
    # Generate all possible winning lines (103 total in 3x3x3)
    winning_lines = []
    
    # Rows in each layer
    for layer in range(3):
        for row in range(3):
            winning_lines.append([(layer, row, col) for col in range(3)])
    
    # Columns in each layer
    for layer in range(3):
        for col in range(3):
            winning_lines.append([(layer, row, col) for row in range(3)])
    
    # Diagonals in each layer
    for layer in range(3):
        winning_lines.append([(layer, i, i) for i in range(3)])
        winning_lines.append([(layer, i, 2-i) for i in range(3)])
    
    # Vertical columns (through layers)
    for row in range(3):
        for col in range(3):
            winning_lines.append([(layer, row, col) for layer in range(3)])
    
    # Space diagonals
    winning_lines.append([(i, i, i) for i in range(3)])
    winning_lines.append([(i, i, 2-i) for i in range(3)])
    winning_lines.append([(i, 2-i, i) for i in range(3)])
    winning_lines.append([(i, 2-i, 2-i) for i in range(3)])
    
    # Function to evaluate a winning line
    def evaluate_line(line):
        my_count = 0
        opp_count = 0
        empty_count = 0
        
        for (i, j, k) in line:
            if board[i][j][k] == 1:
                my_count += 1
            elif board[i][j][k] == -1:
                opp_count += 1
            else:
                empty_count += 1
        
        return my_count, opp_count, empty_count
    
    # Check for immediate winning moves
    for (i, j, k) in empty_cells:
        for line in winning_lines:
            if (i, j, k) in line:
                my_count, opp_count, empty_count = evaluate_line(line)
                # If I have 2 and can win with this move
                if my_count == 2 and empty_count == 1:
                    return (i, j, k)
    
    # Check for blocking opponent's winning moves
    for (i, j, k) in empty_cells:
        for line in winning_lines:
            if (i, j, k) in line:
                my_count, opp_count, empty_count = evaluate_line(line)
                # If opponent has 2 and can win with this move, block it
                if opp_count == 2 and empty_count == 1:
                    return (i, j, k)
    
    # Score each empty cell based on strategic value
    best_score = -float('inf')
    best_move = empty_cells[0] if empty_cells else (0, 0, 0)
    
    for (i, j, k) in empty_cells:
        score = 0
        
        # Priority: Center cell is extremely valuable
        if (i, j, k) == (1, 1, 1):
            score += 100
        
        # Priority: Corners of the cube
        corners = [(0, 0, 0), (0, 0, 2), (0, 2, 0), (0, 2, 2),
                   (2, 0, 0), (2, 0, 2), (2, 2, 0), (2, 2, 2)]
        if (i, j, k) in corners:
            score += 50
        
        # Evaluate each winning line this cell is part of
        for line in winning_lines:
            if (i, j, k) in line:
                my_count, opp_count, empty_count = evaluate_line(line)
                
                # Strong bonus for creating a threat (1 in row with 2 empty)
                if my_count == 1 and empty_count == 2:
                    score += 10
                
                # Bonus for being in a line where opponent has 1
                if opp_count == 1 and empty_count == 2:
                    score += 5
                
                # Small bonus for any strategic position
                if my_count == 0 and empty_count == 3:
                    score += 1
        
        # Slight randomness to avoid deterministic patterns
        import random
        score += random.uniform(0, 0.5)
        
        if score > best_score:
            best_score = score
            best_move = (i, j, k)
    
    return best_move
```

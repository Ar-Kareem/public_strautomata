def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Determines the next move for a 3x3x3 Tic-Tac-Toe game.
    
    Args:
        board: A 3x3x3 list of lists where each element is 0 (empty), 1 (player), or -1 (opponent).
    
    Returns:
        A tuple (x, y, z) representing the coordinates of the chosen move.
    """
    # Precompute all winning lines for 3x3x3 Tic-Tac-Toe
    all_lines = []
    
    # Axis-aligned lines
    for z in range(3):
        for y in range(3):
            all_lines.append([(x, y, z) for x in range(3)])
    for z in range(3):
        for x in range(3):
            all_lines.append([(x, y, z) for y in range(3)])
    for x in range(3):
        for y in range(3):
            all_lines.append([(x, y, z) for z in range(3)])
    
    # Layer diagonals (fixed z)
    for z in range(3):
        all_lines.append([(0, 0, z), (1, 1, z), (2, 2, z)])
        all_lines.append([(0, 2, z), (1, 1, z), (2, 0, z)])
    
    # Plane xz diagonals (fixed y)
    for y in range(3):
        all_lines.append([(0, y, 0), (1, y, 1), (2, y, 2)])
        all_lines.append([(0, y, 2), (1, y, 1), (2, y, 0)])
    
    # Plane yz diagonals (fixed x)
    for x in range(3):
        all_lines.append([(x, 0, 0), (x, 1, 1), (x, 2, 2)])
        all_lines.append([(x, 0, 2), (x, 1, 1), (x, 2, 0)])
    
    # Space diagonals
    all_lines.append([(0, 0, 0), (1, 1, 1), (2, 2, 2)])
    all_lines.append([(0, 0, 2), (1, 1, 1), (2, 2, 0)])
    all_lines.append([(0, 2, 0), (1, 1, 1), (2, 0, 2)])
    all_lines.append([(0, 2, 2), (1, 1, 1), (2, 0, 0)])
    
    # Find all empty cells
    empty_cells = []
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board[z][y][x] == 0:
                    empty_cells.append((x, y, z))
    
    # Step 1: Check for winning moves
    for (x, y, z) in empty_cells:
        for line in all_lines:
            if (x, y, z) in line:
                # Count our markers in the other two cells of the line
                count_ones = 0
                for (x1, y1, z1) in line:
                    if (x1, y1, z1) != (x, y, z):
                        if board[z1][y1][x1] == 1:
                            count_ones += 1
                if count_ones == 2:
                    return (x, y, z)
    
    # Step 2: Check for blocking moves
    candidate_moves = {}
    for line in all_lines:
        values = [board[z][y][x] for (x, y, z) in line]
        if values.count(-1) == 2 and values.count(0) == 1:
            for (x, y, z) in line:
                if board[z][y][x] == 0:
                    candidate_moves[(x, y, z)] = candidate_moves.get((x, y, z), 0) + 1
                    break
    
    if candidate_moves:
        # Choose the move that blocks the most threats
        best_count = max(candidate_moves.values())
        best_moves = [move for move, count in candidate_moves.items() if count == best_count]
        return best_moves[0]
    
    # Step 3: Use priority-based selection
    priority_order = [
        (1, 1, 1),    # Center
        (0, 0, 0), (0, 0, 2), (0, 2, 0), (0, 2, 2),    # Corners
        (2, 0, 0), (2, 0, 2), (2, 2, 0), (2, 2, 2),
        (0, 1, 0), (0, 1, 2), (2, 1, 0), (2, 1, 2),    # Edge centers (front/back)
        (1, 0, 0), (1, 0, 2), (1, 2, 0), (1, 2, 2),    # Edge centers (left/right)
        (0, 0, 1), (0, 2, 1), (2, 0, 1), (2, 2, 1),    # Edge centers (top/bottom)
        (0, 1, 1), (2, 1, 1), (1, 0, 1), (1, 2, 1),    # Face centers (front/back)
        (1, 1, 0), (1, 1, 2)                           # Face centers (top/bottom)
    ]
    
    for move in priority_order:
        x, y, z = move
        if board[z][y][x] == 0:
            return move
    
    # Fallback: return the first empty cell
    return empty_cells[0]

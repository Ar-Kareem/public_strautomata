import random

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Implements a simple random policy for 3D Tic-Tac-Toe.
    Selects a random empty cell from the board.
    """
    empties = []
    for i in range(3):
        for j in range(3):
            for k in range(3):
                if board[i][j][k] == 0:
                    empties.append((i, j, k))
    if empties:
        return random.choice(empties)
    else:
        # This should not happen in a valid game
        raise ValueError("No empty cells available.")

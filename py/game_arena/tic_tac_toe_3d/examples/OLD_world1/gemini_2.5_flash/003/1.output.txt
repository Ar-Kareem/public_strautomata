import random

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Determines the next optimal move for a 3x3x3 3D Tic Tac Toe game.

    Args:
        board: A 3x3x3 list of lists representing the game board.
               0: empty, 1: AI's piece, -1: Opponent's piece.

    Returns:
        A tuple of three integers (z, y, x) representing the chosen move.
    """
    n = len(board)
    me = 1
    opponent = -1

    # 1. Check for a winning move
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board[z][y][x] == 0:
                    board[z][y][x] = me
                    if check_win(board, me):
                        return (z, y, x)
                    board[z][y][x] = 0  # Undo the temporary move

    # 2. Check for opponent's winning move and block it
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board[z][y][x] == 0:
                    board[z][y][x] = opponent
                    if check_win(board, opponent):
                        return (z, y, x)
                    board[z][y][x] = 0  # Undo the temporary move

    # 3. Try to establish a winning threat (fork) - more complex for 3D, simplified here
    # This part is highly heuristic and can be expanded for better play
    # For now, it prioritizes center cells or corners if available.

    # Center cell is often strategic
    if n % 2 == 1: # Only for odd sized boards
        center_coord = n // 2
        if board[center_coord][center_coord][center_coord] == 0:
            return (center_coord, center_coord, center_coord)

    # Prioritize corners
    corners = []
    for z_c in [0, n - 1]:
        for y_c in [0, n - 1]:
            for x_c in [0, n - 1]:
                corners.append((z_c, y_c, x_c))
    
    random.shuffle(corners) # Randomize corner preference if multiple are empty
    for z_c, y_c, x_c in corners:
        if board[z_c][y_c][x_c] == 0:
            return (z_c, y_c, x_c)

    # 4. If no immediate win, block, or strategic position, make a random move
    empty_cells = []
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board[z][y][x] == 0:
                    empty_cells.append((z, y, x))

    if empty_cells:
        return random.choice(empty_cells)
    
    # Should not happen in a valid game, but return a default/error if no moves are possible
    return (0, 0, 0)


def check_win(board: list[list[list[int]]], player: int) -> bool:
    """
    Checks if the given player has won the 3x3x3 Tic Tac Toe game.
    """
    n = len(board)

    # Check all rows, columns, and depth lines
    for i in range(n):
        for j in range(n):
            # Rows (fixed z, y, varying x)
            if all(board[i][j][k] == player for k in range(n)): return True
            # Columns (fixed z, x, varying y)
            if all(board[i][k][j] == player for k in range(n)): return True
            # Depth (fixed y, x, varying z)
            if all(board[k][i][j] == player for k in range(n)): return True

    # Check all 2D diagonals on each slice
    for i in range(n):
        # Slice Z (fixed z, varying y, x)
        if all(board[i][k][k] == player for k in range(n)): return True  # Main diagonal
        if all(board[i][k][n - 1 - k] == player for k in range(n)): return True  # Anti-diagonal
        # Slice Y (fixed y, varying z, x)
        if all(board[k][i][k] == player for k in range(n)): return True  # Main diagonal
        if all(board[k][i][n - 1 - k] == player for k in range(n)): return True  # Anti-diagonal
        # Slice X (fixed x, varying z, y)
        if all(board[k][k][i] == player for k in range(n)): return True  # Main diagonal
        if all(board[k][n - 1 - k][i] == player for k in range(n)): return True  # Anti-diagonal
            
    # Check all 4 3D diagonals
    # Main 3D diagonal
    if all(board[i][i][i] == player for i in range(n)): return True
    # Anti-3D diagonal (z, y, N-1-x)
    if all(board[i][i][n - 1 - i] == player for i in range(n)): return True
    # Anti-3D diagonal (z, N-1-y, x)
    if all(board[i][n - 1 - i][i] == player for i in range(n)): return True
    # Anti-3D diagonal (z, N-1-y, N-1-x)
    if all(board[i][n - 1 - i][n - 1 - i] == player for i in range(n)): return True

    return False

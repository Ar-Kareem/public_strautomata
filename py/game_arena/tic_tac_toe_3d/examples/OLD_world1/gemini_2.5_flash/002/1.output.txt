import numpy as np

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    n = len(board)
    board_np = np.array(board)

    # 1. Check for a winning move
    for x in range(n):
        for y in range(n):
            for z in range(n):
                if board_np[x, y, z] == 0:
                    # Temporarily make the move and check if it's a win
                    board_np[x, y, z] = 1
                    if check_win(board_np, 1, n):
                        return (x, y, z)
                    board_np[x, y, z] = 0  # Undo the temporary move

    # 2. Block opponent's winning move
    for x in range(n):
        for y in range(n):
            for z in range(n):
                if board_np[x, y, z] == 0:
                    # Temporarily make the opponent's move and check if it's a win for them
                    board_np[x, y, z] = -1
                    if check_win(board_np, -1, n):
                        return (x, y, z)
                    board_np[x, y, z] = 0  # Undo the temporary move

    # 3. Prefer center moves if available
    center_coords = []
    if n % 2 != 0:
        center = n // 2
        center_coords = [(center, center, center)]
        if n == 3: # For n=3, also consider faces around center
            # Edges of the center slice
            for i in range(n):
                for j in range(n):
                    if board_np[center, i, j] == 0:
                        center_coords.append((center, i, j))
                    if board_np[i, center, j] == 0:
                        center_coords.append((i, center, j))
                    if board_np[i, j, center] == 0:
                        center_coords.append((i, j, center))

    for coord in center_coords:
        x, y, z = coord
        if 0 <= x < n and 0 <= y < n and 0 <= z < n and board_np[x, y, z] == 0:
            return (x, y, z)

    # 4. Take any available corner
    corners = []
    for i in [0, n - 1]:
        for j in [0, n - 1]:
            for k in [0, n - 1]:
                corners.append((i, j, k))

    for x, y, z in corners:
        if board_np[x, y, z] == 0:
            return (x, y, z)

    # 5. Take any available edge
    edges = []
    for x in range(n):
        for y in range(n):
            for z in range(n):
                is_corner = (x in (0, n-1) and y in (0, n-1) and z in (0, n-1))
                is_face_center = (
                    (x in (0, n-1) and y == n//2 and z == n//2) or
                    (y in (0, n-1) and x == n//2 and z == n//2) or
                    (z in (0, n-1) and x == n//2 and y == n//2)
                )
                if not is_corner and not is_face_center: # A simple definition of edge, can be refined
                    # For n=3, this is spots like (0,1,0), (1,0,0) etc.
                    # It's any cell that's not a corner or a face center (for n=3, the center (1,1,1) is handled separately)
                    if not (n % 2 != 0 and x == n // 2 and y == n // 2 and z == n // 2): # Avoid counting center as edge
                         edges.append((x,y,z))

    for x, y, z in edges:
        if board_np[x, y, z] == 0:
            return (x, y, z)

    # 6. Fallback: Take the first available empty cell (should not be reached if game is not full)
    for x in range(n):
        for y in range(n):
            for z in range(n):
                if board_np[x, y, z] == 0:
                    return (x, y, z)

    # Should not happen in a valid game unless the board is full
    raise ValueError("No empty cells left on the board.")

def check_win(board: np.ndarray, player: int, n: int) -> bool:
    """Checks if the given player has won on the board."""

    # Check rows, columns, and stacks
    for i in range(n):
        for j in range(n):
            if np.all(board[i, j, :] == player) or \
               np.all(board[i, :, j] == player) or \
               np.all(board[:, i, j] == player):
                return True

    # Check 2D diagonals on each slice
    for i in range(n):
        if np.all(np.diag(board[i, :, :]) == player) or \
           np.all(np.diag(np.fliplr(board[i, :, :])) == player) or \
           np.all(np.diag(board[:, i, :]) == player) or \
           np.all(np.diag(np.fliplr(board[:, i, :])) == player) or \
           np.all(np.diag(board[:, :, i]) == player) or \
           np.all(np.diag(np.fliplr(board[:, :, i])) == player):
            return True

    # Check 3D diagonals
    if np.all(np.array([board[i, i, i] for i in range(n)]) == player) or \
       np.all(np.array([board[i, i, n - 1 - i] for i in range(n)]) == player) or \
       np.all(np.array([board[i, n - 1 - i, i] for i in range(n)]) == player) or \
       np.all(np.array([board[n - 1 - i, i, i] for i in range(n)]) == player):
        return True

    return False

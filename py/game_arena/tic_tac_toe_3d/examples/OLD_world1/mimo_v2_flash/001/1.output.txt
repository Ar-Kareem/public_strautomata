```python
import random
import sys
import numpy as np

# Increase recursion depth just in case, though iterative approach is used.
sys.setrecursionlimit(2000)

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    AI Policy for 3x3x3 3D Tic Tac Toe.
    
    Strategy:
    1. Check for an immediate winning move and take it.
    2. Check if the opponent has an immediate winning move and block it.
    3. Try to fork: Create two lines of 2 (with a 0) such that they intersect at an empty cell.
       - Specifically, look for a line pattern: [1, 1, 0] or [1, 0, 1].
       - If a move creates a second such line (including diagonals), it's a fork.
    4. Block opponent forks (lines of 2) or simply defend against threats if no fork found.
    5. Take the center (1,1,1).
    6. Take corners if available.
    7. Take any other available edge/side cell.
    """
    
    n = len(board)
    size_range = range(n)
    me = 1
    opp = -1
    empty = 0
    
    # Helper to get all lines in the board
    # Returns a list of lines, where each line is a list of (x,y,z) coordinates
    def get_all_lines():
        lines = []
        
        # Rows, Cols, Pillars (along x, y, z axes)
        for i in size_range:
            # Fixed y, z -> varying x
            lines.append([(i, j, k) for i in size_range])
            # Fixed x, z -> varying y
            lines.append([(i, j, k) for j in size_range])
            # Fixed x, y -> varying z
            lines.append([(i, j, k) for k in size_range])
            
        # Face Diagonals (2D diagonals within each plane)
        for i in size_range:
            # Diagonals on x-y planes (vary z)
            lines.append([(x, x, i) for x in size_range])
            lines.append([(x, n-1-x, i) for x in size_range])
            
            # Diagonals on x-z planes (vary y)
            lines.append([(x, i, x) for x in size_range])
            lines.append([(x, i, n-1-x) for x in size_range])
            
            # Diagonals on y-z planes (vary x)
            lines.append([(i, x, x) for x in size_range])
            lines.append([(i, x, n-1-x) for x in size_range])
            
        # Space Diagonals
        lines.append([(x, x, x) for x in size_range])
        lines.append([(x, x, n-1-x) for x in size_range])
        lines.append([(x, n-1-x, x) for x in size_range])
        lines.append([(x, n-1-x, n-1-x) for x in size_range])
        
        return lines

    ALL_LINES = get_all_lines()

    # --- 1. Immediate Win ---
    # Check if I can complete a line of 2
    for line in ALL_LINES:
        line_vals = [board[x][y][z] for x, y, z in line]
        if line_vals.count(opp) == 0 and line_vals.count(me) == 2:
            for x, y, z in line:
                if board[x][y][z] == empty:
                    return (x, y, z)

    # --- 2. Block Opponent Win ---
    # Check if opponent has 2 in a line
    for line in ALL_LINES:
        line_vals = [board[x][y][z] for x, y, z in line]
        if line_vals.count(me) == 0 and line_vals.count(opp) == 2:
            for x, y, z in line:
                if board[x][y][z] == empty:
                    return (x, y, z)

    # --- 3. Create Fork ---
    # A fork is a move that creates two threats simultaneously.
    # We look for patterns like [1, 1, 0], [1, 0, 1] on lines.
    # If a move extends two such patterns, it's a fork.
    def count_threats(x, y, z):
        threats = 0
        # Identify all lines passing through (x,y,z)
        # Filter ALL_LINES for those containing the coordinate
        for line in ALL_LINES:
            if (x, y, z) in line:
                line_vals = [board[a][b][c] for a, b, c in line]
                # Check if this line is a "potential line" for me (2 blanks, 1 mine? No, I want 2 mines)
                # Wait, I want to ADD to an existing 1 or 2.
                # More robust: Check if the line is currently [1, 1, 0] where 0 is the move?
                # No, the move hasn't happened yet. We simulate.
                
                # Let's look for "2-in-line" patterns that include the empty spot.
                # A line is a threat if it has [1, 1, 0] or [1, 0, 1].
                # If we place at 0, it becomes a win.
                # So we want to place at 0 such that we have *two* such lines.
                
                my_count = line_vals.count(me)
                empty_count = line_vals.count(empty)
                
                # A "1-line" means we have 1 mine and 2 blanks. Placing a mine here creates a "2-line".
                # A "2-line" means we have 2 mines and 1 blank. This is a winning threat.
                
                # We want to count how many "2-lines" we create.
                if my_count == 2 and empty_count == 1:
                    # Check if the empty cell is (x,y,z)
                    for (cx, cy, cz), val in zip(line, line_vals):
                        if val == empty and (cx, cy, cz) == (x, y, z):
                            return True
        return False

    # Find all empty spots
    empty_spots = []
    for x in size_range:
        for y in size_range:
            for z in size_range:
                if board[x][y][z] == empty:
                    empty_spots.append((x, y, z))

    # Check for fork moves
    # Optimization: Only check lines connected to the move
    for x, y, z in empty_spots:
        threat_count = 0
        for line in ALL_LINES:
            if (x, y, z) in line:
                # Simulate the move
                board[x][y][z] = me
                # Check if this line is now winning (3 me) or 2 me + 1 empty?
                # Wait, we want exactly 2 me and 1 empty on the *original* line?
                # No, we want to count how many lines become "2 me 1 empty" (threats).
                line_vals = [board[a][b][c] for a, b, c in line]
                if line_vals.count(me) == 2 and line_vals.count(empty) == 1:
                    threat_count += 1
                # Undo simulation
                board[x][y][z] = empty
                
        # If we create 2 threats (one is the move itself, the other is a side effect)
        # Actually, the move creates 1 threat (the line with 2 mines).
        # To be a fork, we need *at least one additional* threat created.
        # So threat_count >= 2 means we create 2 threats simultaneously.
        # Wait, if we place in a line [1, 0, 0], we get [1, 1, 0]. That's not a threat yet (needs 2).
        # If we place in [1, 1, 0], we win. That's handled in step 1.
        # A fork move is placing in [1, 0, 0] such that we end up with two lines of [1, 1, 0] (or better).
        # So we should check lines *before* placing. If a line has 1 mine and 2 blanks, placing here makes it 2 mines.
        # We need this to happen on two different lines.
        
        pre_threats = 0
        for line in ALL_LINES:
            if (x, y, z) in line:
                line_vals = [board[a][b][c] for a, b, c in line]
                my_count = line_vals.count(me)
                # If we place here, my_count increases by 1.
                if my_count == 1 and (line_vals.count(empty) == 2): # The spot itself is empty
                    pre_threats += 1
                elif my_count == 2 and (line_vals.count(empty) == 1): # The spot itself is empty
                    # This line is already [1, 1, 0] (assuming the 0 is the current spot)
                    # Wait, if I'm at (x,y,z), the spot is empty, so yes.
                    # This means I win immediately if I place here. Should have been caught above.
                    pass
        
        # If we convert 2 lines from [1, 0, 0] (or similar) to [1, 1, 0] -> That's a fork.
        # Actually, if pre_threats >= 2, we create 2 lines of "1 mine becoming 2 mines".
        # This is the standard fork definition.
        if pre_threats >= 2:
            return (x, y, z)

    # --- 4. Defend / Block Potential Forks ---
    # If opponent has 1 mine and 2 blanks on multiple lines intersecting, they might fork.
    # We block by taking the intersection if possible.
    # Simple heuristic: Find a move that blocks the most opponent threats.
    best_defense = -1
    best_def_move = None
    for x, y, z in empty_spots:
        opp_threats = 0
        for line in ALL_LINES:
            if (x, y, z) in line:
                line_vals = [board[a][b][c] for a, b, c in line]
                if line_vals.count(opp) == 1 and line_vals.count(empty) == 2:
                    opp_threats += 1
        if opp_threats > best_defense:
            best_defense = opp_threats
            best_def_move = (x, y, z)
    
    # --- 5. Center ---
    if board[1][1][1] == empty:
        return (1, 1, 1)

    # --- 6. Corners ---
    corners = [
        (0,0,0), (0,0,2), (0,2,0), (0,2,2),
        (2,0,0), (2,0,2), (2,2,0), (2,2,2)
    ]
    random.shuffle(corners)
    for x, y, z in corners:
        if board[x][y][z] == empty:
            return (x, y, z)

    # --- 7. Other ---
    # We already populated empty_spots in step 3.
    # If we didn't return by now, just pick a random empty spot.
    if empty_spots:
        return random.choice(empty_spots)

    # Should not happen unless board is full
    return (0, 0, 0)
```

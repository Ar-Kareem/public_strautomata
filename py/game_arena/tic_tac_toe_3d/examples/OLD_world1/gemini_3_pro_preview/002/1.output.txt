import numpy as np

# Precompute all 49 winning lines for a 3x3x3 board
def _get_winning_lines():
    lines = []
    domain = range(3)
    
    # Orthogonal lines (3 directions * 9 positions = 27)
    # Along X-axis (fix Z, Y)
    for z in domain:
        for y in domain:
            lines.append([(z, y, x) for x in domain])
    # Along Y-axis (fix Z, X)
    for z in domain:
        for x in domain:
            lines.append([(z, y, x) for y in domain])
    # Along Z-axis (fix Y, X)
    for y in domain:
        for x in domain:
            lines.append([(z, y, x) for z in domain])
            
    # Planar diagonals (3 orientations * 3 slices * 2 diags = 18)
    # XY planes (fix Z)
    for z in domain:
        lines.append([(z, i, i) for i in domain])
        lines.append([(z, i, 2 - i) for i in domain])
    # XZ planes (fix Y)
    for y in domain:
        lines.append([(i, y, i) for i in domain])
        lines.append([(i, y, 2 - i) for i in domain])
    # YZ planes (fix X)
    for x in domain:
        lines.append([(i, i, x) for i in domain])
        lines.append([(i, 2 - i, x) for i in domain])
        
    # Spatial diagonals (4 lines connecting opposite corners)
    lines.append([(i, i, i) for i in domain])
    lines.append([(i, i, 2 - i) for i in domain])
    lines.append([(i, 2 - i, i) for i in domain])
    lines.append([(i, 2 - i, 2 - i) for i in domain])
    
    return lines

_WINNING_LINES = _get_winning_lines()

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Returns the coordinates (z, y, x) of the chosen move.
    1 = AI (me), -1 = Opponent, 0 = Empty.
    Strategies:
    1. Win immediately.
    2. Block immediate defeat.
    3. Heuristic scoring (create threats, block potential threats, centralization).
    """
    grid = np.array(board)
    
    # 1. Identify valid moves
    empty_indices = np.argwhere(grid == 0)
    if len(empty_indices) == 0:
        return (0, 0, 0) # Fallback, game should be over
    
    empty_cells = [tuple(idx) for idx in empty_indices]
    
    # 2. Check for Immediate Win (Two 1s and a 0 in a line)
    for line in _WINNING_LINES:
        vals = [grid[c] for c in line]
        if sum(vals) == 2 and 0 in vals:
            target = line[vals.index(0)]
            return int(target[0]), int(target[1]), int(target[2])
            
    # 3. Check for Immediate Block (Two -1s and a 0 in a line)
    for line in _WINNING_LINES:
        vals = [grid[c] for c in line]
        if sum(vals) == -2 and 0 in vals:
            target = line[vals.index(0)]
            return int(target[0]), int(target[1]), int(target[2])
            
    # 4. Heuristic Evaluation
    # Score each empty cell based on its contribution to winning lines
    best_score = -float('inf')
    best_move = empty_cells[0]
    
    # Weights
    W_CREATE_THREAT = 50      # Value of turning a line with one '1' into two '1's
    W_PREVENT_THREAT = 40     # Value of turning a line with one '-1' into a blocked line
    W_BUILD = 10              # Value of placing a mark in an empty line
    W_BLOCK_EMPTY = 5         # Value of placing a mark in a line opponent could use
    W_CENTER = 25             # Bias for the center cell (1,1,1)
    
    for cell in empty_cells:
        score = 0
        
        # Positional Biases
        if cell == (1, 1, 1):
            score += W_CENTER
        elif cell in [(0,0,0), (0,0,2), (0,2,0), (0,2,2), (2,0,0), (2,0,2), (2,2,0), (2,2,2)]:
            # Slight preference for corners over edges/faces if center taken
            score += 2
            
        # Analyze lines passing through this cell
        for line in _WINNING_LINES:
            if cell in line:
                # Extract content of the line
                line_vals = [grid[c] for c in line]
                
                my_count = line_vals.count(1)
                opp_count = line_vals.count(-1)
                
                # Attack Potential (Line has no opponent marks)
                if opp_count == 0:
                    if my_count == 1:
                        # [1, 0, 0] -> Move makes it [1, 1, 0] (Threat created)
                        score += W_CREATE_THREAT
                    elif my_count == 0:
                        # [0, 0, 0] -> Move makes it [1, 0, 0] (Building)
                        score += W_BUILD
                        
                # Defense Potential (Line has no my marks)
                if my_count == 0:
                    if opp_count == 1:
                        # [-1, 0, 0] -> Move prevents opponent from making [-1, -1, 0]
                        score += W_PREVENT_THREAT
                    elif opp_count == 0:
                        # [0, 0, 0] -> Denies opponent an empty line (overlaps with W_BUILD)
                        score += W_BLOCK_EMPTY

        if score > best_score:
            best_score = score
            best_move = cell
            
    return int(best_move[0]), int(best_move[1]), int(best_move[2])

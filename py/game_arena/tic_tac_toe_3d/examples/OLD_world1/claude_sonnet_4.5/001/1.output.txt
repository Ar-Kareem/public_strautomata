import numpy as np
from typing import Optional

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Policy for 3x3x3 3D Tic Tac Toe.
    Returns the best move as (z, y, x) coordinates.
    """
    board_np = np.array(board)
    n = 3
    
    # First, check if we can win in one move
    win_move = find_winning_move(board_np, 1)
    if win_move is not None:
        return win_move
    
    # Second, block opponent from winning
    block_move = find_winning_move(board_np, -1)
    if block_move is not None:
        return block_move
    
    # Third, try to take center if available
    center = n // 2
    if board_np[center, center, center] == 0:
        return (center, center, center)
    
    # Fourth, take corners if available
    corners = [
        (0, 0, 0), (0, 0, 2), (0, 2, 0), (0, 2, 2),
        (2, 0, 0), (2, 0, 2), (2, 2, 0), (2, 2, 2)
    ]
    for corner in corners:
        if board_np[corner] == 0:
            return corner
    
    # Fifth, find the move that creates the most threats
    best_move = find_best_strategic_move(board_np)
    if best_move is not None:
        return best_move
    
    # Finally, take any available move
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board_np[z, y, x] == 0:
                    return (z, y, x)
    
    return (0, 0, 0)


def find_winning_move(board: np.ndarray, player: int) -> Optional[tuple[int, int, int]]:
    """Find a move that wins the game for the given player."""
    n = 3
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board[z, y, x] == 0:
                    # Try this move
                    board[z, y, x] = player
                    if check_win(board, player):
                        board[z, y, x] = 0
                        return (z, y, x)
                    board[z, y, x] = 0
    return None


def check_win(board: np.ndarray, player: int) -> bool:
    """Check if the given player has won."""
    n = 3
    
    # Check rows (along x-axis)
    for z in range(n):
        for y in range(n):
            if np.all(board[z, y, :] == player):
                return True
    
    # Check columns (along y-axis)
    for z in range(n):
        for x in range(n):
            if np.all(board[z, :, x] == player):
                return True
    
    # Check pillars (along z-axis)
    for y in range(n):
        for x in range(n):
            if np.all(board[:, y, x] == player):
                return True
    
    # Check diagonals in xy planes
    for z in range(n):
        if np.all(board[z, range(n), range(n)] == player):
            return True
        if np.all(board[z, range(n), range(n-1, -1, -1)] == player):
            return True
    
    # Check diagonals in xz planes
    for y in range(n):
        if np.all(board[range(n), y, range(n)] == player):
            return True
        if np.all(board[range(n), y, range(n-1, -1, -1)] == player):
            return True
    
    # Check diagonals in yz planes
    for x in range(n):
        if np.all(board[range(n), range(n), x] == player):
            return True
        if np.all(board[range(n), range(n-1, -1, -1), x] == player):
            return True
    
    # Check 4 main 3D diagonals
    if np.all(board[range(n), range(n), range(n)] == player):
        return True
    if np.all(board[range(n), range(n), range(n-1, -1, -1)] == player):
        return True
    if np.all(board[range(n), range(n-1, -1, -1), range(n)] == player):
        return True
    if np.all(board[range(n), range(n-1, -1, -1), range(n-1, -1, -1)] == player):
        return True
    
    return False


def find_best_strategic_move(board: np.ndarray) -> Optional[tuple[int, int, int]]:
    """Find the move that creates the most winning opportunities."""
    n = 3
    best_score = -1
    best_move = None
    
    for z in range(n):
        for y in range(n):
            for x in range(n):
                if board[z, y, x] == 0:
                    score = evaluate_move(board, (z, y, x), 1)
                    if score > best_score:
                        best_score = score
                        best_move = (z, y, x)
    
    return best_move


def evaluate_move(board: np.ndarray, move: tuple[int, int, int], player: int) -> int:
    """Evaluate how good a move is based on threat creation."""
    board[move] = player
    score = count_threats(board, player)
    board[move] = 0
    return score


def count_threats(board: np.ndarray, player: int) -> int:
    """Count the number of lines with 2 of our pieces and 1 empty."""
    # This is a simplified heuristic
    return np.sum(board == player)

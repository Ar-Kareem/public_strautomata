import random
from typing import Optional

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    """
    Strategy for 3x3x3 3D Tic Tac Toe
    1. Win if possible
    2. Block opponent from winning
    3. Take center if available
    4. Take strategic positions
    5. Random valid move
    """
    n = len(board)
    
    # Get all winning lines in 3D tic tac toe
    def get_all_lines():
        lines = []
        
        # Lines parallel to axes
        for i in range(n):
            for j in range(n):
                # Lines parallel to x-axis
                lines.append([(i, j, k) for k in range(n)])
                # Lines parallel to y-axis
                lines.append([(i, k, j) for k in range(n)])
                # Lines parallel to z-axis
                lines.append([(k, i, j) for k in range(n)])
        
        # Diagonals in planes parallel to xy
        for i in range(n):
            lines.append([(i, j, j) for j in range(n)])
            lines.append([(i, j, n-1-j) for j in range(n)])
        
        # Diagonals in planes parallel to xz
        for j in range(n):
            lines.append([(i, j, i) for i in range(n)])
            lines.append([(i, j, n-1-i) for i in range(n)])
        
        # Diagonals in planes parallel to yz
        for k in range(n):
            lines.append([(i, i, k) for i in range(n)])
            lines.append([(i, n-1-i, k) for i in range(n)])
        
        # 4 main space diagonals
        lines.append([(i, i, i) for i in range(n)])
        lines.append([(i, i, n-1-i) for i in range(n)])
        lines.append([(i, n-1-i, i) for i in range(n)])
        lines.append([(n-1-i, i, i) for i in range(n)])
        
        return lines
    
    def check_line_for_win(line, player):
        """Check if player can win on this line, return the winning move"""
        values = [board[x][y][z] for x, y, z in line]
        if values.count(player) == n - 1 and values.count(0) == 1:
            # Find the empty spot
            for i, (x, y, z) in enumerate(line):
                if board[x][y][z] == 0:
                    return (x, y, z)
        return None
    
    lines = get_all_lines()
    
    # 1. Try to win
    for line in lines:
        move = check_line_for_win(line, 1)
        if move:
            return move
    
    # 2. Block opponent from winning
    for line in lines:
        move = check_line_for_win(line, -1)
        if move:
            return move
    
    # 3. Take center if available
    center = n // 2
    if board[center][center][center] == 0:
        return (center, center, center)
    
    # 4. Take corners (strategic positions)
    corners = [
        (0, 0, 0), (0, 0, n-1), (0, n-1, 0), (0, n-1, n-1),
        (n-1, 0, 0), (n-1, 0, n-1), (n-1, n-1, 0), (n-1, n-1, n-1)
    ]
    available_corners = [c for c in corners if board[c[0]][c[1]][c[2]] == 0]
    if available_corners:
        return random.choice(available_corners)
    
    # 5. Take any valid move
    empty_cells = []
    for i in range(n):
        for j in range(n):
            for k in range(n):
                if board[i][j][k] == 0:
                    empty_cells.append((i, j, k))
    
    if empty_cells:
        return random.choice(empty_cells)
    
    # Fallback (should never reach here in a valid game)
    return (0, 0, 0)

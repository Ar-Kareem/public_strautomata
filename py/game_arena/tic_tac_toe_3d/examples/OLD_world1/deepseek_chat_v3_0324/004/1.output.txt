Here's a Python implementation for a 3D Tic Tac Toe AI that uses a simple heuristic-based approach to make moves. The AI will prioritize winning moves and blocking opponent winning moves, with fallback to strategic positions:

```python
import numpy as np
from itertools import product
from collections import defaultdict

def policy(board: list[list[list[int]]]) -> tuple[int, int, int]:
    size = len(board)
    board_np = np.array(board)
    
    # Generate all possible lines (rows, columns, depths, and diagonals)
    def get_all_lines():
        lines = []
        
        # Add all straight lines (rows, columns, depths)
        for dim in range(3):
            for i in range(size):
                for j in range(size):
                    lines.append([(i, j, k) if dim == 0 else 
                                 (k, i, j) if dim == 1 else 
                                 (j, k, i) for k in range(size)])
        
        # Add all possible 3D diagonals
        diags = [
            [(i, i, i) for i in range(size)],  # Main space diagonal
            [(i, i, size-1-i) for i in range(size)],  # Other diagonals
            [(i, size-1-i, i) for i in range(size)],
            [(size-1-i, i, i) for i in range(size)],
            [(i, size-1-i, size-1-i) for i in range(size)],
            [(size-1-i, i, size-1-i) for i in range(size)],
            [(size-1-i, size-1-i, i) for i in range(size)]
        ]
        lines.extend(diags)
        
        return lines
    
    all_lines = get_all_lines()
    
    # Check if a line is a winning opportunity
    def evaluate_line(line):
        values = [board_np[x][y][z] for (x, y, z) in line]
        count_1 = values.count(1)
        count_neg1 = values.count(-1)
        empty_positions = [line[i] for i in range(size) if values[i] == 0]
        
        if count_neg1 == 0 and count_1 == size - 1:
            return (1, empty_positions[0])  # Winning move
        if count_1 == 0 and count_neg1 == size - 1:
            return (-1, empty_positions[0])  # Blocking move
        return (0, None)
    
    # Priority 1: Check for immediate winning moves
    for line in all_lines:
        score, pos = evaluate_line(line)
        if score == 1 and pos:
            return pos
    
    # Priority 2: Block opponent's winning moves
    for line in all_lines:
        score, pos = evaluate_line(line)
        if score == -1 and pos:
            return pos
    
    # Priority 3: Choose strategic positions (center, edges, corners)
    center = size // 2
    strategic_positions = []
    
    # Add center position (most valuable in 3D)
    if board_np[center][center][center] == 0:
        strategic_positions.append((center, center, center))
    
    # Add face centers
    for dim in range(3):
        for i in [0, center, size-1]:
            for j in [0, center, size-1]:
                pos = [i, j, center]
                pos[dim], pos[2] = pos[2], pos[dim]  # rotate to cover all faces
                x, y, z = pos
                if board_np[x][y][z] == 0 and (x, y, z) not in strategic_positions:
                    strategic_positions.append((x, y, z))
    
    # Add corners
    corners = list(product([0, size-1], repeat=3))
    for corner in corners:
        x, y, z = corner
        if board_np[x][y][z] == 0 and (x, y, z) not in strategic_positions:
            strategic_positions.append((x, y, z))
    
    # Add edges
    edges = []
    for i in range(size):
        for j in range(size):
            for k in range(size):
                if (i in [0, size-1] and j in [0, size-1]) or \
                   (i in [0, size-1] and k in [0, size-1]) or \
                   (j in [0, size-1] and k in [0, size-1]):
                    if board_np[i][j][k] == 0 and (i, j, k) not in strategic_positions:
                        edges.append((i, j, k))
    strategic_positions.extend(edges)
    
    # Return the first available strategic position
    for pos in strategic_positions:
        x, y, z = pos
        if board_np[x][y][z] == 0:
            return (x, y, z)
    
    # Fallback: Return the first available position if no strategic positions found
    for x in range(size):
        for y in range(size):
            for z in range(size):
                if board_np[x][y][z] == 0:
                    return (x, y, z)
    
    # This should never happen if the game isn't already over
    return (0, 0, 0)
```

This implementation:
1. First checks for immediate winning moves (where the AI can win in one move)
2. Then checks for opponent winning moves to block them
3. Falls back to strategic positions, prioritizing:
   - The center cube (most valuable position in 3D Tic Tac Toe)
   - Face centers
   - Corners
   - Edges
4. Finally falls back to the first available position if no better moves are found

The algorithm generates all possible winning lines in a 3D space (including diagonals) and evaluates each to find immediate threats or opportunities. This provides a good balance between defense and offense while maintaining strategic positioning.

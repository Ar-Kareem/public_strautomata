import numpy as np
from typing import List, Tuple

def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    """
    Determines the next move for 3x3x3 3D Tic Tac Toe using a minimax algorithm with alpha-beta pruning.
    """
    # Convert to numpy array for easier manipulation
    board_array = np.array(board)
    
    # Find all empty positions
    empty_positions = []
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board_array[x, y, z] == 0:
                    empty_positions.append((x, y, z))
    
    # If board is empty, take the center (1,1,1)
    if len(empty_positions) == 27:
        return (1, 1, 1)
    
    # If only one empty position, take it
    if len(empty_positions) == 1:
        return empty_positions[0]
    
    # Evaluate utility of each possible move
    best_move = empty_positions[0]
    best_score = -float('inf')
    
    for pos in empty_positions:
        # Try the move
        board_array[pos] = 1  # Place my mark
        
        # Calculate score using minimax
        score = minimax(board_array, False, -float('inf'), float('inf'), 0)
        
        # Undo the move
        board_array[pos] = 0
        
        if score > best_score:
            best_score = score
            best_move = pos
    
    return best_move

def minimax(board: np.ndarray, is_maximizing: bool, alpha: float, beta: float, depth: int) -> float:
    """
    Minimax algorithm with alpha-beta pruning for 3D Tic Tac Toe.
    """
    # Check for terminal states
    winner = check_winner(board)
    if winner == 1:  # I win
        return 10 - depth  # Prefer quicker wins
    elif winner == -1:  # Opponent wins
        return depth - 10  # Prefer longer losses
    elif not any((board == 0).flatten()):  # Board is full (draw)
        return 0
    
    if depth >= 4:  # Depth limit to prevent excessive computation
        return evaluate_board(board)
    
    empty_positions = [(x, y, z) for x in range(3) for y in range(3) for z in range(3) if board[x, y, z] == 0]
    
    if is_maximizing:
        max_eval = -float('inf')
        for pos in empty_positions:
            board[pos] = 1  # My move
            eval_score = minimax(board, False, alpha, beta, depth + 1)
            board[pos] = 0  # Undo move
            max_eval = max(max_eval, eval_score)
            alpha = max(alpha, eval_score)
            if beta <= alpha:
                break  # Alpha-beta pruning
        return max_eval
    else:
        min_eval = float('inf')
        for pos in empty_positions:
            board[pos] = -1  # Opponent's move
            eval_score = minimax(board, True, alpha, beta, depth + 1)
            board[pos] = 0  # Undo move
            min_eval = min(min_eval, eval_score)
            beta = min(beta, eval_score)
            if beta <= alpha:
                break  # Alpha-beta pruning
        return min_eval

def check_winner(board: np.ndarray) -> int:
    """
    Check if there's a winner on the board.
    Returns 1 if I win, -1 if opponent wins, 0 otherwise.
    """
    # Check all possible lines in 3D
    lines = []
    
    # Axes: x, y, z
    for i in range(3):
        for j in range(3):
            # Lines along x-axis
            lines.append([(k, i, j) for k in range(3)])
            # Lines along y-axis
            lines.append([(i, k, j) for k in range(3)])
            # Lines along z-axis
            lines.append([(i, j, k) for k in range(3)])
    
    # Diagonals on each face (xy planes)
    for z in range(3):
        lines.append([(i, i, z) for i in range(3)])
        lines.append([(i, 2-i, z) for i in range(3)])
    
    # Diagonals on each face (xz planes)
    for y in range(3):
        lines.append([(i, y, i) for i in range(3)])
        lines.append([(i, y, 2-i) for i in range(3)])
    
    # Diagonals on each face (yz planes)
    for x in range(3):
        lines.append([(x, i, i) for i in range(3)])
        lines.append([(x, i, 2-i) for i in range(3)])
    
    # Main 3D diagonals
    lines.append([(i, i, i) for i in range(3)])
    lines.append([(i, i, 2-i) for i in range(3)])
    lines.append([(i, 2-i, i) for i in range(3)])
    lines.append([(2-i, i, i) for i in range(3)])
    
    # Check each line
    for line in lines:
        values = [board[pos] for pos in line]
        if all(v == 1 for v in values):
            return 1
        elif all(v == -1 for v in values):
            return -1
    
    return 0

def evaluate_board(board: np.ndarray) -> float:
    """
    Heuristic evaluation of the board position.
    """
    score = 0
    center = (1, 1, 1)
    
    # Bonus for center control
    if board[center] == 1:
        score += 0.5
    elif board[center] == -1:
        score -= 0.5
    
    # Evaluate potential winning lines
    lines = []
    
    # Generate all possible lines
    for i in range(3):
        for j in range(3):
            lines.append([(k, i, j) for k in range(3)])
            lines.append([(i, k, j) for k in range(3)])
            lines.append([(i, j, k) for k in range(3)])
    
    for z in range(3):
        lines.append([(i, i, z) for i in range(3)])
        lines.append([(i, 2-i, z) for i in range(3)])
    
    for y in range(3):
        lines.append([(i, y, i) for i in range(3)])
        lines.append([(i, y, 2-i) for i in range(3)])
    
    for x in range(3):
        lines.append([(x, i, i) for i in range(3)])
        lines.append([(x, i, 2-i) for i in range(3)])
    
    lines.append([(i, i, i) for i in range(3)])
    lines.append([(i, i, 2-i) for i in range(3)])
    lines.append([(i, 2-i, i) for i in range(3)])
    lines.append([(2-i, i, i) for i in range(3)])
    
    for line in lines:
        my_count = sum(1 for pos in line if board[pos] == 1)
        opp_count = sum(1 for pos in line if board[pos] == -1)
        
        if my_count > 0 and opp_count == 0:
            # Potential win for me
            if my_count == 2:
                score += 0.8  # Close to winning
            else:
                score += 0.1  # On the way to winning
        elif opp_count > 0 and my_count == 0:
            # Potential win for opponent
            if opp_count == 2:
                score -= 0.7  # Close to opponent winning
            else:
                score -= 0.1  # Opponent progressing
    
    return score

import numpy as np
from typing import Tuple, List

def policy(board: List[List[List[int]]]) -> Tuple[int, int, int]:
    """
    Determine the next move for 3D Tic Tac Toe using a minimax algorithm with alpha-beta pruning.
    
    Args:
        board: 3x3x3 list of lists representing the game board
               0 = empty, 1 = player, -1 = opponent
    
    Returns:
        Tuple of (x, y, z) coordinates for the next move
    """
    # Convert to numpy array for easier manipulation
    board_np = np.array(board)
    
    # First, check if we can win immediately
    win_move = find_winning_move(board_np, 1)
    if win_move:
        return win_move
    
    # Second, block opponent's winning move
    block_move = find_winning_move(board_np, -1)
    if block_move:
        return block_move
    
    # Otherwise, use minimax to find the best move
    best_move = minimax_best_move(board_np)
    return best_move

def find_winning_move(board: np.ndarray, player: int) -> Tuple[int, int, int]:
    """Check if there's a winning move for the given player."""
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board[x, y, z] == 0:
                    # Try placing the player's piece here
                    board[x, y, z] = player
                    if check_win(board, player):
                        # Revert the board
                        board[x, y, z] = 0
                        return (x, y, z)
                    # Revert the board
                    board[x, y, z] = 0
    return None

def minimax_best_move(board: np.ndarray) -> Tuple[int, int, int]:
    """Find the best move using minimax with alpha-beta pruning."""
    best_score = -float('inf')
    best_move = None
    
    for x in range(3):
        for y in range(3):
            for z in range(3):
                if board[x, y, z] == 0:
                    board[x, y, z] = 1  # Our move
                    score = minimax(board, depth=0, alpha=-float('inf'), beta=float('inf'), 
                                   maximizing_player=False)
                    board[x, y, z] = 0  # Undo move
                    
                    if score > best_score:
                        best_score = score
                        best_move = (x, y, z)
    
    return best_move

def minimax(board: np.ndarray, depth: int, alpha: float, beta: float, 
           maximizing_player: bool) -> int:
    """Minimax algorithm with alpha-beta pruning."""
    # Check for terminal states
    if check_win(board, 1):
        return 10 - depth  # Prefer quicker wins
    if check_win(board, -1):
        return depth - 10  # Prefer slower losses
    if is_board_full(board):
        return 0  # Draw
    
    # Limit depth to prevent excessive computation
    if depth > 4:
        return 0  # Heuristic evaluation could be added here
    
    if maximizing_player:
        max_eval = -float('inf')
        for x in range(3):
            for y in range(3):
                for z in range(3):
                    if board[x, y, z] == 0:
                        board[x, y, z] = 1
                        eval_score = minimax(board, depth + 1, alpha, beta, False)
                        board[x, y, z] = 0
                        max_eval = max(max_eval, eval_score)
                        alpha = max(alpha, eval_score)
                        if beta <= alpha:
                            break  # Alpha-beta pruning
                if beta <= alpha:
                    break
            if beta <= alpha:
                break
        return max_eval
    else:
        min_eval = float('inf')
        for x in range(3):
            for y in range(3):
                for z in range(3):
                    if board[x, y, z] == 0:
                        board[x, y, z] = -1
                        eval_score = minimax(board, depth + 1, alpha, beta, True)
                        board[x, y, z] = 0
                        min_eval = min(min_eval, eval_score)
                        beta = min(beta, eval_score)
                        if beta <= alpha:
                            break  # Alpha-beta pruning
                if beta <= alpha:
                    break
            if beta <= alpha:
                break
        return min_eval

def check_win(board: np.ndarray, player: int) -> bool:
    """Check if the given player has won."""
    # Check all lines (rows, columns, pillars, and diagonals)
    
    # Check rows in each layer (x-y planes)
    for z in range(3):
        for x in range(3):
            if all(board[x, y, z] == player for y in range(3)):
                return True
        for y in range(3):
            if all(board[x, y, z] == player for x in range(3)):
                return True
    
    # Check pillars (z-axis)
    for x in range(3):
        for y in range(3):
            if all(board[x, y, z] == player for z in range(3)):
                return True
    
    # Check layer diagonals (in each z-layer)
    for z in range(3):
        # Main diagonal
        if all(board[i, i, z] == player for i in range(3)):
            return True
        # Anti-diagonal
        if all(board[i, 2-i, z] == player for i in range(3)):
            return True
    
    # Check vertical diagonals (through layers)
    # Diagonals in x-z planes
    for y in range(3):
        # Main diagonal
        if all(board[i, y, i] == player for i in range(3)):
            return True
        # Anti-diagonal
        if all(board[i, y, 2-i] == player for i in range(3)):
            return True
    
    # Diagonals in y-z planes
    for x in range(3):
        # Main diagonal
        if all(board[x, i, i] == player for i in range(3)):
            return True
        # Anti-diagonal
        if all(board[x, i, 2-i] == player for i in range(3)):
            return True
    
    # Check space diagonals
    # Main space diagonal
    if all(board[i, i, i] == player for i in range(3)):
        return True
    # Other space diagonals
    if all(board[i, i, 2-i] == player for i in range(3)):
        return True
    if all(board[i, 2-i, i] == player for i in range(3)):
        return True
    if all(board[2-i, i, i] == player for i in range(3)):
        return True
    
    return False

def is_board_full(board: np.ndarray) -> bool:
    """Check if the board is full."""
    return not np.any(board == 0)
